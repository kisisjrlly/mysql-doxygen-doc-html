53 /**
54  Base class for parse-time Item objects
55 
56  Parse-time Item objects are placeholders for real Item objects: in some
57  cases it is not easy or even possible to decide what exact Item class object
58  we need to allocate in the parser. Parse-time Item objects are intended
59  to defer real Item object allocation to the contextualization phase (see
60  the Item::itemize() function).
61 
62  This wrapper class overrides abstract virtual functions of the parent
63  class with dummy wrappers to make C++ compiler happy.
64 */
65 class Parse_tree_item : public Item {
66  public:
67  explicit Parse_tree_item(const POS &pos) : Item(pos) {}
68 
69  enum Type type() const override { return INVALID_ITEM; }
70  double val_real() override {
71  DBUG_ASSERT(0);
72  return 0;
73  }
74  longlong val_int() override {
75  DBUG_ASSERT(0);
76  return 0;
77  }
78  String *val_str(String *) override {
79  DBUG_ASSERT(0);
80  return nullptr;
81  }
82  my_decimal *val_decimal(my_decimal *) override {
83  DBUG_ASSERT(0);
84  return nullptr;
85  }
86  bool get_date(MYSQL_TIME *, uint) override {
87  DBUG_ASSERT(0);
88  return false;
89  }

91  DBUG_ASSERT(0);

93  }

96 /**
97  Wrapper class for an Item list head, used to allocate Item lists in the parser
98  in a context-independent way
99 */
100 class PT_item_list : public Parse_tree_node {
101  typedef Parse_tree_node super;
102 
103  public:
104  PT_item_list() : value(*THR_MALLOC) {}
105 
106  mem_root_deque<Item *> value;
107 
108  bool contextualize(Parse_context *pc) override {
109  if (super::contextualize(pc)) return true;
110  for (Item *&item : value) {
111  if (item->itemize(pc, &item)) return true;
112  }
113  return false;
114  }
115 
116  bool is_empty() const { return value.empty(); }
117  uint elements() const { return value.size(); }
118 
119  bool push_back(Item *item) {
120  /*
121  Item may be NULL in case of OOM: just ignore it and check thd->is_error()
122  in the caller code.
123  */
124  if (item == nullptr) return true;
125  value.push_back(item);
126  return false;
127  }
128 
129  bool push_front(Item *item) {
130  /*
131  Item may be NULL in case of OOM: just ignore it and check thd->is_error()
132  in the caller code.
133  */
134  if (item == nullptr) return true;
135  value.push_front(item);
136  return false;
137  }
138 
139  Item *pop_front() {
140  DBUG_ASSERT(!is_empty());
141  Item *ret = value.front();
142  value.pop_front();
143  return ret;
144  }
145 
146  Item *operator[](uint index) const { return value[index]; }
147 };

168 /**
169  Helper function to imitate dynamic_cast for Item_cond hierarchy.
170 
171  Template parameter @p To is the destination type (@c Item_cond_and etc.)
172  Template parameter @p Tag is the Functype tag to compare from->functype() with
173 
174  @param from source item
175 
176  @return typecasted item of the type To or NULL
177 */
178 template <class To, Item_func::Functype Tag>
179 To *item_cond_cast(Item *const from) {
180  return ((from->type() == Item::COND_ITEM &&
181  static_cast<Item_func *>(from)->functype() == Tag)
182  ? static_cast<To *>(from)
183  : nullptr);
184 }

186 /**
187  Flatten associative operators at parse time
188 
189  This function flattens AND and OR operators at parse time if applicable,
190  otherwise it creates new Item_cond_and or Item_cond_or respectively.
191 
192  Template parameter @p Class is @c Item_cond_and or @c Item_cond_or
193  Template parameter @p Tag is @c COND_AND_FUNC (for @c Item_cond_and) or @c
194  COND_OR_FUNC otherwise
195 
196  @param mem_root MEM_ROOT
197  @param pos parse location
198  @param left left argument of the operator
199  @param right right argument of the operator
200 
201  @return resulting parse tree Item
202 */
203 template <class Class, Item_func::Functype Tag>
204 Item *flatten_associative_operator(MEM_ROOT *mem_root, const POS &pos,
205  Item *left, Item *right) {
206  if (left == nullptr || right == nullptr) return nullptr;
207  Class *left_func = item_cond_cast<Class, Tag>(left);
208  Class *right_func = item_cond_cast<Class, Tag>(right);
209  if (left_func) {
210  if (right_func) {
211  // (X1 op X2) op (Y1 op Y2) ==> op (X1, X2, Y1, Y2)
212  right_func->add_at_head(left_func->argument_list());
213  return right;
214  } else {
215  // (X1 op X2) op Y ==> op (X1, X2, Y)
216  left_func->add(right);
217  return left;
218  }
219  } else if (right_func) {
220  // X op (Y1 op Y2) ==> op (X, Y1, Y2)
221  right_func->add_at_head(left);
222  return right;
223  } else {
224  /* X op Y */
225  return new (mem_root) Class(pos, left, right);
226  }
227 }
