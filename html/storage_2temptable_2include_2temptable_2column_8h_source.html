238 inline void Column::write_is_null(bool is_null, unsigned char *mysql_row,
239  size_t mysql_row_length
240  TEMPTABLE_UNUSED_NODBUG) const {
241  if (is_nullable()) {
242  unsigned char *b = mysql_row + m_null_byte_offset;
243 
244  DBUG_ASSERT(buf_is_inside_another(b, 1, mysql_row, mysql_row_length));
245 
246  if (is_null) {
247  *b |= m_null_bitmask;
248  } else {
249  *b &= ~m_null_bitmask;
250  }
251  } else {
252  DBUG_ASSERT(!is_null);
253  }
254 }

258 inline uint32_t Column::read_user_data_length(
259  const unsigned char *mysql_row) const {
260  if (m_length_bytes_size == 0) {
261  /* Fixed size cell. */
262  return m_length;
263  }
264 
265  // data_length may contain junk value while particular cell is NULL
266  if (read_is_null(mysql_row)) {
267  return 0;
268  }
269 
270  const unsigned char *p = mysql_row + m_offset;
271  switch (m_length_bytes_size) {
272  case 1:
273  return *p;
274  case 2:
275  return *p | (*(p + 1) << 8);
276  case 3:
277  return *p | (*(p + 1) << 8) | (*(p + 2) << 16);
278  case 4:
279  return *p | (*(p + 1) << 8) | (*(p + 2) << 16) | (*(p + 3) << 24);
280  }
281 
282  abort();
283  return 0;
284 }

286 inline void Column::write_user_data_length(
287  uint32_t data_length, unsigned char *mysql_row,
288  size_t mysql_row_length TEMPTABLE_UNUSED_NODBUG) const {
289  unsigned char *p = mysql_row + m_offset;
290 
291  DBUG_ASSERT((m_length_bytes_size == 0) ||
292  (buf_is_inside_another(p, m_length_bytes_size, mysql_row,
293  mysql_row_length)));
294 
295  switch (m_length_bytes_size) {
296  case 0:
297  /* Fixed size cell. */
298  break;
299  case 1:
300  DBUG_ASSERT(data_length <= 0xFF);
301  p[0] = data_length;
302  break;
303  case 2:
304  DBUG_ASSERT(data_length <= 0xFFFF);
305  p[0] = (data_length & 0x000000FF);
306  p[1] = (data_length & 0x0000FF00) >> 8;
307  break;
308  case 3:
309  DBUG_ASSERT(data_length <= 0xFFFFFF);
310  p[0] = (data_length & 0x000000FF);
311  p[1] = (data_length & 0x0000FF00) >> 8;
312  p[2] = (data_length & 0x00FF0000) >> 16;
313  break;
314  case 4:
315  /* DBUG_ASSERT(data_length <= 0xFFFFFFFF). */
316  p[0] = (data_length & 0x000000FF);
317  p[1] = (data_length & 0x0000FF00) >> 8;
318  p[2] = (data_length & 0x00FF0000) >> 16;
319  p[3] = (data_length & 0xFF000000) >> 24;
320  break;
321  default:
322  DBUG_ABORT();
323  }

380 inline void Column::read_blob_user_data(
381  unsigned char *data, uint32_t data_length, const unsigned char *mysql_row,
382  size_t mysql_row_length TEMPTABLE_UNUSED_NODBUG) const {
383  DBUG_ASSERT(is_blob());
384 
385  const unsigned char *p = mysql_row + m_user_data_offset;
386 
387  const unsigned char *ptr_to_data;
388 
389  DBUG_ASSERT(buf_is_inside_another(p, sizeof(ptr_to_data), mysql_row,
390  mysql_row_length));
391 
392  /* read the address */
393  memcpy(&ptr_to_data, p, sizeof(ptr_to_data));
394 
395  DBUG_ASSERT((ptr_to_data) || (data_length == 0));
396 
397  memcpy(data, ptr_to_data, data_length);
398 }

400 inline void Column::write_blob_user_data(
401  const unsigned char *data, uint32_t data_length TEMPTABLE_UNUSED,
402  unsigned char *mysql_row,
403  size_t mysql_row_length TEMPTABLE_UNUSED_NODBUG) const {
404  DBUG_ASSERT(is_blob());
405 
406  unsigned char *p = mysql_row + m_user_data_offset;
407 
408  /* Note1: data could be NULL.
409  * Note2: shallow copy - pointer to original data is stored. */
410  const unsigned char *ptr_to_data = data;
411 
412  DBUG_ASSERT(buf_is_inside_another(p, sizeof(ptr_to_data), mysql_row,
413  mysql_row_length));
414 
415  memcpy(p, &ptr_to_data, sizeof(ptr_to_data));
416 }
