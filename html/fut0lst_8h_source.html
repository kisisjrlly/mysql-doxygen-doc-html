150 /** In-memory representation of flst_base_node_t */
151 struct flst_bnode_t {
152  ulint len;
153  fil_addr_t first;
154  fil_addr_t last;
155 
156  flst_bnode_t() : len(0) {}
157 
158  flst_bnode_t(const flst_base_node_t *base, mtr_t *mtr)
159  : len(flst_get_len(base)),
160  first(flst_get_first(base, mtr)),
161  last(flst_get_last(base, mtr)) {}
162 
163  void set(const flst_base_node_t *base, mtr_t *mtr) {
164  len = flst_get_len(base);
165  first = flst_get_first(base, mtr);
166  last = flst_get_last(base, mtr);
167  }
168 
169  void reset() {
170  len = 0;
171  first = fil_addr_null;
172  last = fil_addr_null;
173  }
174 
175  std::ostream &print(std::ostream &out) const {
176  out << "[flst_base_node_t: len=" << len << ", first=" << first
177  << ", last=" << last << "]";
178  return (out);
179  }
180 };
181 
182 inline std::ostream &operator<<(std::ostream &out, const flst_bnode_t &obj) {
183  return (obj.print(out));
184 }
