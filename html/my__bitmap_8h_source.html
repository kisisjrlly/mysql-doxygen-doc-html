75 static inline void bitmap_set_bit(MY_BITMAP *map, uint bit) {
76  DBUG_ASSERT(bit < map->n_bits);
77  ((uchar *)map->bitmap)[bit / 8] |= (1 << (bit & 7));

95 /**
96  Quite unlike other C comparison functions ending with 'cmp', e.g. memcmp(),
97  strcmp(), this function returns true if the bitmaps are equal, and false
98  otherwise.
99 
100  @retval true The bitmaps are equal.
101  @retval false The bitmaps differ.
102  */
103 static inline bool bitmap_cmp(const MY_BITMAP *map1, const MY_BITMAP *map2) {
104  DBUG_ASSERT(map1->n_bits > 0);
105  DBUG_ASSERT(map2->n_bits > 0);
106 
107  if (memcmp(map1->bitmap, map2->bitmap, 4 * (no_words_in_map(map1) - 1)) != 0)
108  return false;
109  return ((*map1->last_word_ptr | map1->last_word_mask) ==
110  (*map2->last_word_ptr | map2->last_word_mask));
111 }
