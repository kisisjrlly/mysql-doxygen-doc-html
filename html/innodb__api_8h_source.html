40 /** Macros to lock/unlock the engine connection mutex */

42  if (!(has_lock)) { \
43  pthread_mutex_lock(&(engine)->conn_mutex); \
44  }

46 #define UNLOCK_CONN_IF_NOT_LOCKED(has_lock, engine) \

66 /** We would need to fetch 5 column values from each key value rows if they
67 are available. They are the "key", "value", "cas", "exp" and "flag" */
68 #define MCI_COL_TO_GET 5

70 typedef enum mci_col {
71  MCI_COL_KEY, /*!< key */
72  MCI_COL_VALUE, /*!< value */
73  MCI_COL_FLAG, /*!< flag */
74  MCI_COL_CAS, /*!< check and set value */
75  MCI_COL_EXP /*!< expiration */
76 } mci_col_t;

78 /** mci_column is the structure that stores and describes a column info
79 in InnoDB Memcached. The supported column types are either character
80 type or integer type (see above "enum mci_col" for columns
81 supporting memcached) */
82 typedef struct mci_column {
83  char *value_str; /*!< char value of the column */
84  int value_len; /*!< char value length in bytes */
85  uint64_t value_int; /*!< integer value */
86  bool is_str; /*!< whether the value is char or int */
87  bool is_unsigned; /*!< whether the value is signed or not */
88  bool is_valid; /*!< this structure contains valid
89  or stale column value */
90  bool is_null; /*!< whether it is a NULL value */
91  bool allocated; /*!< whether memory allocated to store
92  the value */
93 } mci_column_t;

95 /** "mci_item_t" represents values we read from a table row, and enough
96 to assemble to an memcached response. As described in above mci_col,
97 we must have "MCI_COL_TO_GET" (5) column values to read. In addition,
98 the user table could have multiple "value" columns, and it is possible
99 to map such multiple "value" columns to a single memcached key,
100 such value is separated by "separator" as defined in the "config_option"
101 table. And we will assemble and disassemble the memcached value from these
102 column values. And "extra_col_value" and "n_extra_col" is used to support
103 multiple value columns */
104 typedef struct mci_item {
105  mci_column_t col_value[MCI_COL_TO_GET]; /*!< columns in a row */
106  mci_column_t *extra_col_value; /*!< whether there will be
107  additional/multiple "values"
108  to be stored */
109  int n_extra_col; /*!< number of additional
110  "value" columns */
111 } mci_item_t;
