39 /** Information of a group required for parsing information from the archived
40 file. */
41 struct Arch_Recv_Group_Info {
42  /** Number of archived files belonging to the group. */
43  uint m_num_files{0};
44 
45  /** Group is active or not. */
46  bool m_active{false};
47 
48  /** True if group is from durable archiving, false if left over from a crash
49  * during clone operation. */
50  bool m_durable{false};
51 
52  /** The file index which is part of the file name may not necessarily
53  be 0 always. It's possible that purge might have purged files in the
54  group leading to the file index of the first file in the group being
55  greater than 0. So we need this info to know the index of the first
56  file in the group. */
57  uint m_file_start_index{std::numeric_limits<uint>::max()};
58 };
