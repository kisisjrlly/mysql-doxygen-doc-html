54 /** Rw-lock with very fast, highly concurrent s-lock but slower x-lock.
55 It's basically array of rw-locks. When s-lock is being acquired, single
56 rw-lock from array is selected randomly and s-locked. Therefore, all
57 rw-locks from array has to be x-locked when x-lock is being acquired.
58 
59 Purpose of this data structure is to reduce contention on single atomic
60 in single rw-lock when a lot of threads need to acquire s-lock very often,
61 but x-lock is very rare. */
62 class Sharded_rw_lock {
63  public:
64  void create(
65 #ifdef UNIV_PFS_RWLOCK
66  mysql_pfs_key_t pfs_key,
67 #endif
68  latch_level_t latch_level, size_t n_shards) {
69  m_n_shards = n_shards;
70 
71  m_shards = static_cast<Shard *>(ut_zalloc_nokey(sizeof(Shard) * n_shards));
72 
73  for_each([
74 #ifdef UNIV_PFS_RWLOCK
75  pfs_key,
76 #endif
77  latch_level](rw_lock_t &lock) {
78  static_cast<void>(latch_level); // clang -Wunused-lambda-capture
79  rw_lock_create(pfs_key, &lock, latch_level);
80  });
81  }
82 
83  void free() {
84  ut_a(m_shards != nullptr);
85 
86  for_each([](rw_lock_t &lock) { rw_lock_free(&lock); });
87 
88  ut_free(m_shards);
89  m_shards = nullptr;
90  m_n_shards = 0;
91  }
92 
93  size_t s_lock() {
94  const size_t shard_no = ut_rnd_interval(0, m_n_shards - 1);
95  rw_lock_s_lock(&m_shards[shard_no]);
96  return shard_no;
97  }
98 
99  ibool s_lock_nowait(size_t &shard_no, const char *file, ulint line) {
100  shard_no = ut_rnd_interval(0, m_n_shards - 1);
101  return rw_lock_s_lock_nowait(&m_shards[shard_no], file, line);
102  }
103 
104  void s_unlock(size_t shard_no) {
105  ut_a(shard_no < m_n_shards);
106  rw_lock_s_unlock(&m_shards[shard_no]);
107  }
108 
109  /**
110  Tries to obtain exclusive latch - similar to x_lock(), but non-blocking, and
111  thus can fail.
112  @return true iff succeeded to acquire the exclusive latch
113  */
114  bool try_x_lock() {
115  for (size_t shard_no = 0; shard_no < m_n_shards; ++shard_no) {
116  if (!rw_lock_x_lock_nowait(&m_shards[shard_no])) {
117  while (0 < shard_no--) {
118  rw_lock_x_unlock(&m_shards[shard_no]);
119  }
120  return (false);
121  }
122  }
123  return (true);
124  }
125 
126  void x_lock() {
127  for_each([](rw_lock_t &lock) { rw_lock_x_lock(&lock); });
128  }
129 
130  void x_unlock() {
131  for_each([](rw_lock_t &lock) { rw_lock_x_unlock(&lock); });
132  }
133 
134 #ifdef UNIV_DEBUG
135  bool s_own(size_t shard_no) const {
136  return rw_lock_own(&m_shards[shard_no], RW_LOCK_S);
137  }
138 
139  bool x_own() const { return rw_lock_own(&m_shards[0], RW_LOCK_X); }
140 #endif /* !UNIV_DEBUG */
141 
142  private:
143  using Shard = ut::Cacheline_padded<rw_lock_t>;
144 
145  template <typename F>
146  void for_each(F f) {
147  std::for_each(m_shards, m_shards + m_n_shards, f);
148  }
149 
150  Shard *m_shards = nullptr;
151 
152  size_t m_n_shards = 0;
153 };
