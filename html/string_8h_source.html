66 template <class Container>
67 struct Join<Container, std::string> {
68  static std::string impl(Container cont, const std::string &delim) {
69  if (cont.begin() == cont.end()) return {};
70 
71  std::string o(*(cont.begin()));
72 
73  // if T::value_type has a .size() method reallocs can be avoided
74  // when joining the strings by calculating the size upfront
75  {
76  const size_t delim_size = delim.size();
77  size_t space =
78  std::accumulate(std::next(cont.begin()), cont.end(), o.size(),
79  [delim_size](size_t sum, const std::string &b) {
80  return sum + delim_size + b.size();
81  });
82  o.reserve(space);
83  }
84 
85 #if 0
86  // once benchmarked that this is equivalent ot the hand-rolled version
87  // (number of allocs, ...) this implmentation could be used.
88  return std::accumulate(std::next(cont.begin()), cont.end(), o,
89  [&delim](std::string a, const std::string &b) {
90  return a.append(delim).append(b);
91  });
92 #else
93  // add the first element directly
94  auto it = std::next(cont.begin());
95  const auto last = cont.end();
96 
97  // all other elements with delim
98  for (; it != last; ++it) {
99  o += delim;
100 
101  o += *it;
102  }
103 
104  return o;
105 #endif
106  }
107 };

109 template <class Container>
110 struct Join<Container, const char *> {
111  static std::string impl(Container cont, const std::string &delim) {
112  if (cont.begin() == cont.end()) return {};
113 
114  return std::accumulate(std::next(cont.begin()), cont.end(),
115  std::string(*(cont.begin())),
116  [&delim](const std::string &a, const char *b) {
117  return a + delim + b;
118  });
119  }

124 /**
125  * join elements of an container into a string seperated by a delimiter.
126  *
127  * Container MUST:
128  *
129  * - have .begin() and end()
130  * - ::iterator must be ForwardIterator + InputIterator
131  * - ::value_type must be appendable to std::string
132  *
133  * should work with:
134  *
135  * - std::vector<const char *|std::string>
136  * - std::array<const char *|std::string, N>
137  * - std::list<const char *|std::string>
138  *
139  * @param cont a container
140  * @param delim delimiter
141  * @returns string of elements of container seperated by delim
142  */
143 template <class Container>
144 std::string join(Container cont, const std::string &delim) {
145  return detail::Join<Container, typename Container::value_type>::impl(cont,
146  delim);
147 }
