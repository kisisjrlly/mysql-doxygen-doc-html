52 class poll_io_service : public IoServiceBase {
53  public:
54  ~poll_io_service() override {
55  if (wakeup_fds_.first != impl::socket::kInvalidSocket)
56  impl::socket::close(wakeup_fds_.first);
57  if (wakeup_fds_.second != impl::socket::kInvalidSocket)
58  impl::socket::close(wakeup_fds_.second);
59  }
60 
61  bool is_open() const noexcept {
62  return wakeup_fds_.first != impl::socket::kInvalidSocket &&
63  wakeup_fds_.second != impl::socket::kInvalidSocket;
64  }
65 
66  stdx::expected<void, std::error_code> open() noexcept override {
67  if (is_open()) {
68  return stdx::make_unexpected(
69  make_error_code(net::socket_errc::already_open));
70  }
71 
72 #if defined(_WIN32)
73  // on windows we build a AF_INET socketpair()
74  auto pipe_res = impl::socket::socketpair(AF_INET, SOCK_STREAM, 0);
75 #else
76  auto pipe_res = impl::socket::socketpair(AF_UNIX, SOCK_STREAM, 0);
77 #endif
78  if (!pipe_res) return stdx::make_unexpected(pipe_res.error());
79 
80  wakeup_fds_ = *pipe_res;
81 
82  // set both ends of the pipe non-blocking as
83  //
84  // - read() shouldn't block if pipe is empty
85  // - write() shouldn't block if pipe is full as it only matters there is
86  // something in the pipe to wakeup the poll_one()
87  auto non_block_wakeup_0_res =
88  impl::socket::native_non_blocking(wakeup_fds_.first, true);
89  if (!non_block_wakeup_0_res) return non_block_wakeup_0_res;
90  auto non_block_wakeup_1_res =
91  impl::socket::native_non_blocking(wakeup_fds_.second, true);
92  if (!non_block_wakeup_1_res) return non_block_wakeup_1_res;
93 
94  add_fd_interest(wakeup_fds_.first, impl::socket::wait_type::wait_read);
95 
96  return {};
97  }

99  stdx::expected<short, std::error_code> poll_event_from_wait_type(
100  impl::socket::wait_type event) {
101  switch (event) {
102  case impl::socket::wait_type::wait_read:
103  return POLLIN;
104  case impl::socket::wait_type::wait_write:
105  return POLLOUT;
106  case impl::socket::wait_type::wait_error:
107  return POLLERR;
108  default:
109  return stdx::make_unexpected(
110  make_error_code(std::errc::invalid_argument));
111  }
112  }

114  // split fd-interest
115  //
116  // internally splits the fds into multiple buckets to reduce
117  // the search-space and lead to lower resize cost
118  class FdInterests {
119  public:
120  using element_type = fd_event;
121 
122  // we could use list, deque, vector, ... here
123  //
124  // container_type | concurrency | mem-usage | tps
125  // ---------------+-------------+-----------+------
126  // list: | 8000 | 137M | 56000
127  // vector | 8000 | 145M | 58000
128  using container_type = std::vector<element_type>;
129 
130  void push_back(element_type &&t) {
131  auto &b = bucket(t.fd);
132 
133  std::lock_guard<std::mutex> lk(mtx_);
134  b.push_back(std::move(t));
135  }

137  element_type erase_all(native_handle_type fd) {
138  auto &b = bucket(fd);
139 
140  std::lock_guard<std::mutex> lk(mtx_);
141  const auto end = b.end();
142 
143  for (auto cur = b.begin(); cur != end;) {
144  if (cur->fd == fd) {
145  auto op = std::move(*cur);
146  cur = b.erase(cur);
147 
148  return op;
149  } else {
150  ++cur;
151  }
152  }
153 
154  // not found
155  return {};
156  }
157 
158  std::vector<pollfd> poll_fds() const {
159  std::vector<pollfd> fds;
160  {
161  std::lock_guard<std::mutex> lk(mtx_);
162  size_t count{};
163 
164  for (const auto &b : buckets_) {
165  count += b.size();
166  }
167 
168  fds.reserve(count);
169 
170  for (const auto &b : buckets_) {
171  for (auto const &fd_int : b) {
172  fds.push_back({fd_int.fd, fd_int.event, 0});
173  }
174  }
175  }
176 
177  return fds;
178  }
179 
180  void erase_fd_event(native_handle_type fd, short event) {
181  auto &b = bucket(fd);
182 
183  std::lock_guard<std::mutex> lk(mtx_);
184  b.erase(std::remove_if(
185  b.begin(), b.end(),
186  [=](fd_event v) { return v.fd == fd && (v.event & event); }),
187  b.end());
188  }
189 
190  private:
191  container_type &bucket(native_handle_type fd) {
192  size_t ndx = fd % buckets_.size();
193 
194  return buckets_[ndx];
195  }
196 
197  // tps @8000 client connections
198  //
199  // cnt : tps
200  // ----:------
201  // 1: 32000
202  // 3: 45000
203  // 7: 54000
204  // 13: 56000
205  // 23: 57000
206  // 47: 58000
207  // 101: 58000
208  // 1009: 57000
209  static constexpr const size_t bucket_count_{101};
210 
211  mutable std::mutex mtx_;
212  std::array<container_type, bucket_count_> buckets_;
213  };

215  stdx::expected<void, std::error_code> add_fd_interest(
216  native_handle_type fd, impl::socket::wait_type event) override {
217  if (fd == impl::socket::kInvalidSocket) {
218  return stdx::make_unexpected(
219  make_error_code(std::errc::invalid_argument));
220  }
221 
222  auto event_res = poll_event_from_wait_type(event);
223  if (!event_res) return stdx::make_unexpected(event_res.error());
224 
225  fd_interests_.push_back({fd, event_res.value()});
226 
227  return {};
228  }

236  return stdx::make_unexpected(

245  stdx::expected<fd_event, std::error_code> poll_one(
246  std::chrono::milliseconds timeout) override {
247  if (!is_open()) {
248  return stdx::make_unexpected(
249  make_error_code(std::errc::invalid_argument));
250  }
251 
252  fd_event ev;
253  {
254  std::lock_guard<std::mutex> lk(mtx_);
255 
256  if (triggered_events_.empty()) {
257  // build fds for poll() from fd-interest
258 
259  auto poll_fds = fd_interests_.poll_fds();
260  auto res = impl::poll::poll(poll_fds.data(), poll_fds.size(), timeout);
261  if (!res) return stdx::make_unexpected(res.error());
262 
263  size_t num_revents = res.value();
264 
265  // translate poll()'s revents into triggered events.
266  for (auto ev : poll_fds) {
267  if (ev.revents != 0) {
268  --num_revents;
269  // in case of connection close at "wait_read" without any data
270  //
271  // windows: POLLERR|POLLHUP
272  // linux: POLLIN|POLLHUP
273  //
274  // as the uppoer layers expect that the waited-for event appears
275  // in the output, we hve to merge it in.
276  if (ev.events & POLLIN) {
277  ev.revents |= POLLIN;
278  }
279  if (ev.events & POLLOUT) {
280  ev.revents |= POLLOUT;
281  }
282  triggered_events_.push_back({ev.fd, ev.revents});
283  if (ev.fd != wakeup_fds_.first) {
284  // don't remove interest in the wakeup file-descriptors
285  remove_fd_interest(ev.fd, ev.revents);
286  }
287  }
288 
289  if (0 == num_revents) break;
290  }
291  }
292 
293  if (triggered_events_.empty()) {
294  return stdx::make_unexpected(std::error_code{});
295  }
296 
297  ev = triggered_events_.front();
298  triggered_events_.pop_front();
299  }
300 
301  // we could drop mutex here, right?
302 
303  if (ev.fd == wakeup_fds_.first) {
304  on_notify();
305  return stdx::make_unexpected(make_error_code(std::errc::interrupted));
306  }
307 
308  return ev;
309  }
310 
311  void on_notify() {
312  // 256 seems to be a nice sweetspot between "not run too many rounds" and
313  // "copy user space to kernel space"
314  std::array<uint8_t, 256> buf;
315  stdx::expected<size_t, std::error_code> res;
316  do {
317  res = impl::socket::recv(wakeup_fds_.first, buf.data(), buf.size(), 0);
318  } while (res ||
319  res.error() == make_error_condition(std::errc::interrupted));
320  }
321 
322  void notify() override {
323  // don't notify if there is noone listening
324  if (!is_open()) return;
325 
326  stdx::expected<size_t, std::error_code> res;
327  do {
328  std::array<uint8_t, 1> buf = {{'.'}};
329  res = impl::socket::send(wakeup_fds_.second, buf.data(), buf.size(), 0);
330  // retry if interrupted
331  } while (res ==
332  stdx::make_unexpected(make_error_code(std::errc::interrupted)));
333  }
334 
335  private:
336  /**
337  * remove interest of event from fd.
338  *
339  * mtx_ must be held, when called.
340  */
341  stdx::expected<void, std::error_code> remove_fd_interest(
342  native_handle_type fd, short event) {
343  if (fd == impl::socket::kInvalidSocket) {
344  return stdx::make_unexpected(
345  make_error_code(std::errc::invalid_argument));
346  }
347 
348  fd_interests_.erase_fd_event(fd, event);
349 
350  return {};
351  }
352 
353  std::pair<impl::socket::native_handle_type, impl::socket::native_handle_type>
354  wakeup_fds_{impl::socket::kInvalidSocket, impl::socket::kInvalidSocket};
355 
356  FdInterests fd_interests_;
357 
358  std::list<fd_event> triggered_events_;
359 
360  // mutex for triggered_events
361  std::mutex mtx_;
362 };
