130 /** Delays execution for at most max_wait_us microseconds or returns earlier
131  if cond becomes true.

133  @param max_wait_us in: maximum delay to wait, in microseconds */

135  do { \
136  const auto start_us = ut_time_monotonic_us(); \
137  while (!(cond)) { \
138  const auto diff = ut_time_monotonic_us() - start_us; \

355 /** Finds the first format specifier in `fmt` format string
356 @param[in] fmt The format string
357 @return Either the longest suffix of `fmt` which starts with format specifier,
358 or `nullptr` if could not find any format specifier inside `fmt`.
359 */
360 static inline const char *get_first_format(const char *fmt) {
361  const char *pos = strchr(fmt, '%');
362  if (pos != nullptr && pos[1] == '%') {
363  return (get_first_format(pos + 2));
364  }
365  return (pos);
366 }

368 /** Verifies that the `fmt` format string does not require any arguments
369 @param[in] fmt The format string
370 @return true if and only if there is no format specifier inside `fmt` which
371 requires passing an argument */
372 static inline bool verify_fmt_match(const char *fmt) {
373  return (get_first_format(fmt) == nullptr);
374 }
375 
376 /** Verifies that the `fmt` format string contains format specifiers which match
377 the type and order of the arguments
378 @param[in] fmt The format string
379 @param[in] head The first argument
380 @param[in] tail Others (perhaps none) arguments
381 @return true if and only if the format specifiers found in `fmt` correspond to
382 types of head, tail...
383 */
384 template <typename Head, typename... Tail>
385 static bool verify_fmt_match(const char *fmt, Head &&head, Tail &&... tail) {
386  using H =
387  typename std::remove_cv<typename std::remove_reference<Head>::type>::type;
388  const char *pos = get_first_format(fmt);
389  if (pos == nullptr) {
390  return (false);
391  }
392  /* We currently only handle :
393  %[-0-9.*]*(d|ld|lld|u|lu|llu|zu|zx|zd|s|x|i|f|c|X|p|lx|llx|lf)
394  Feel free to extend the parser, if you need something more, as the parser is
395  not intended to be any stricter than real printf-format parser, and if it does
396  not handle something, it is only to keep the code simpler. */
397  const std::string skipable("-+ #0123456789.*");
398 
399  pos++;
400  while (*pos != '\0' && skipable.find_first_of(*pos) != std::string::npos) {
401  pos++;
402  }
403  if (*pos == '\0') {
404  return (false);
405  }
406  bool is_ok = true;
407  if (pos[0] == 'l') {
408  if (pos[1] == 'l') {
409  if (pos[2] == 'd') {
410  is_ok = std::is_same<H, long long int>::value;
411  } else if (pos[2] == 'u' || pos[2] == 'x') {
412  is_ok = std::is_same<H, unsigned long long int>::value;
413  } else {
414  is_ok = false;
415  }
416  } else if (pos[1] == 'd') {
417  is_ok = std::is_same<H, long int>::value;
418  } else if (pos[1] == 'u') {
419  is_ok = std::is_same<H, unsigned long int>::value;
420  } else if (pos[1] == 'x') {
421  is_ok = std::is_same<H, unsigned long int>::value;
422  } else if (pos[1] == 'f') {
423  is_ok = std::is_same<H, double>::value;
424  } else {
425  is_ok = false;
426  }
427  } else if (pos[0] == 'd') {
428  is_ok = std::is_same<H, int>::value;
429  } else if (pos[0] == 'u') {
430  is_ok = std::is_same<H, unsigned int>::value;
431  } else if (pos[0] == 'x') {
432  is_ok = std::is_same<H, unsigned int>::value;
433  } else if (pos[0] == 'X') {
434  is_ok = std::is_same<H, unsigned int>::value;
435  } else if (pos[0] == 'i') {
436  is_ok = std::is_same<H, int>::value;
437  } else if (pos[0] == 'f') {
438  is_ok = std::is_same<H, float>::value;
439  } else if (pos[0] == 'c') {
440  is_ok = std::is_same<H, char>::value;
441  } else if (pos[0] == 'p') {
442  is_ok = std::is_pointer<H>::value;
443  } else if (pos[0] == 's') {
444  is_ok = (std::is_same<H, char *>::value ||
445  std::is_same<H, char const *>::value ||
446  (std::is_array<H>::value &&
447  std::is_same<typename std::remove_cv<
448  typename std::remove_extent<H>::type>::type,
449  char>::value));
450  } else if (pos[0] == 'z') {
451  if (pos[1] == 'u') {
452  is_ok = std::is_same<H, size_t>::value;
453  } else if (pos[1] == 'x') {
454  is_ok = std::is_same<H, size_t>::value;
455  } else if (pos[1] == 'd') {
456  is_ok = std::is_same<H, ssize_t>::value;
457  } else {
458  is_ok = false;
459  }
460  } else {
461  is_ok = false;
462  }
463  return (is_ok && verify_fmt_match(pos + 1, std::forward<Tail>(tail)...));
464 }

555  protected:
556 #ifndef UNIV_NO_ERR_MSGS
557  /** Format an error message.
558  @param[in] err Error code from errmsg-*.txt.
559  @param[in] args Variable length argument list */
560  template <class... Args>
561  static std::string msg(int err, Args &&... args) {
562  const char *fmt = srv_get_server_errmsgs(err);
563 
564  int ret;
565  char buf[LOG_BUFF_MAX];
566 #ifdef UNIV_DEBUG
567  if (get_first_format(fmt) != nullptr) {
568  if (!verify_fmt_match(fmt, std::forward<Args>(args)...)) {
569  fprintf(stderr, "The format '%s' does not match arguments\n", fmt);
570  ut_error;
571  }
572  }
573 #endif
574  ret = snprintf(buf, sizeof(buf), fmt, std::forward<Args>(args)...);
575 
576  std::string str;
577

841 /** For measuring time elapsed. Since std::chrono::high_resolution_clock
