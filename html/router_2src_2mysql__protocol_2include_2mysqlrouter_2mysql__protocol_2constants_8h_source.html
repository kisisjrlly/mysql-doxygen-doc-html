34 /** Type used to pass capability bitset as one number */
35 typedef uint32_t AllFlags;
36 
37 /** Type used to pass high/low half of capability bitset as one number */
38 typedef uint16_t HalfFlags;
39 
40 class Flags {
41  public:
42  constexpr Flags() : flags_(0) {}
43  explicit constexpr Flags(AllFlags flags) : flags_(flags) {}
44 
45 // Developer Studio does not like this even if it's deprecated not to have it.
46 #ifndef __SUNPRO_CC
47  constexpr Flags(const Flags &) = default;
48 #endif
49 
50  Flags &operator=(const Flags &other) = default;
51 
52  constexpr bool operator==(const Flags &other) const {
53  return flags_ == other.flags_;
54  }
55 
56  constexpr bool operator!=(const Flags &other) const {
57  return flags_ != other.flags_;
58  }
59 
60  constexpr Flags operator|(const Flags &other) const {
61  return Flags(flags_ | other.flags_);
62  }
63 
64  constexpr Flags operator&(const Flags &other) const {
65  return Flags(flags_ & other.flags_);
66  }
67 
68  bool test(const Flags &want) const {
69  return (flags_ & want.flags_) == want.flags_;
70  }
71  Flags &set(const Flags &other) {
72  flags_ |= other.flags_;
73  return *this;
74  }
75  Flags &clear(const Flags &other) {
76  flags_ &= ~other.flags_;
77  return *this;
78  }
79  Flags &reset() {
80  flags_ = 0;
81  return *this;
82  }
83 
84  constexpr AllFlags bits() const { return flags_; }
85  constexpr HalfFlags high_16_bits() const {
86  return static_cast<HalfFlags>(flags_ >> 16);
87  }
88  constexpr HalfFlags low_16_bits() const {
89  return static_cast<HalfFlags>(flags_ & 0x0000ffff);
90  }
91 
92  Flags &clear_high_16_bits() {
93  flags_ &= 0x0000ffff;
94  return *this;
95  }
96  Flags &clear_low_16_bits() {
97  flags_ &= 0xffff0000;
98  return *this;
99  }
100 
101  private:
102  AllFlags flags_;
103 };
