59  bool is_open() const noexcept {
60  return epfd_ != impl::file::kInvalidHandle &&
61  (notify_fd_ != impl::file::kInvalidHandle ||
62  (wakeup_fds_.first != impl::file::kInvalidHandle &&
63  wakeup_fds_.second != impl::file::kInvalidHandle));

66  stdx::expected<void, std::error_code> open() noexcept override {
67  if (is_open()) {
68  return stdx::make_unexpected(
69  make_error_code(net::socket_errc::already_open));
70  }
71 
72  auto res = impl::epoll::create();
73  if (!res) return stdx::make_unexpected(res.error());
74 
75  epfd_ = *res;
76 #if defined(USE_EVENTFD)
77  notify_fd_ = eventfd(0, EFD_NONBLOCK);
78  if (notify_fd_ != impl::file::kInvalidHandle) {
79  add_fd_interest_permanent(notify_fd_, impl::socket::wait_type::wait_read);
80 
81  return {};
82  }
83 #endif
84  auto pipe_res = impl::file::pipe(O_NONBLOCK);
85  if (!pipe_res) return stdx::make_unexpected(pipe_res.error());
86 
87  wakeup_fds_ = *pipe_res;
88 
89  // set both ends of the pipe non-blocking as
90  //
91  // - read() shouldn't block is pipe is empty
92  // - write() shouldn't block is pipe is full as it only matters there is
93  // something in the pipe to wakeup the poll_one()
94  auto non_block_wakeup_0_res =
95  impl::socket::native_non_blocking(wakeup_fds_.first, true);
96  if (!non_block_wakeup_0_res) return non_block_wakeup_0_res;
97  auto non_block_wakeup_1_res =
98  impl::socket::native_non_blocking(wakeup_fds_.second, true);
99  if (!non_block_wakeup_1_res) return non_block_wakeup_1_res;
100 
101  add_fd_interest_permanent(wakeup_fds_.first,
102  impl::socket::wait_type::wait_read);
103 
104  return {};
105  }

107  void on_notify() {
108  if (notify_fd_ != impl::file::kInvalidHandle) {
109  uint64_t some{1};
110  ssize_t res;
111  do {
112  res = ::read(notify_fd_, &some, sizeof(some));
113  // in case of EINTR, loop again
114  // otherwise exit
115  //
116  // no need to loop again on success, as the read() will reset the
117  // counter to 0 anyway.
118  } while (res == -1 && errno == EINTR);
119  } else {
120  std::array<uint8_t, 256> buf;
121  ssize_t res;
122  do {
123  res = ::read(wakeup_fds_.first, buf.data(), buf.size());
124  // in case of EINTR, loop again
125  // in case of read > 0, loop again
126  // otherwise exist
127  } while (res > 0 || (res == -1 && errno == EINTR));
128  }
129  }

131  /**
132  * notify the poll_one() that something may have changed.
133  *
134  * can be called from another thread then poll_one().
135  */
136  void notify() override {
137  if (!is_open()) return;
138 
139  // optimization idea:
140  //
141  // if notify() runs in the same thread as poll_one() runs in,
142  // then there is no need to interrupt the poll_one() as it couldn't be
143  // running
144  //
145  // it would save the poll_one(), read(), write() call.
146 
147  if (notify_fd_ != impl::file::kInvalidHandle) {
148  ssize_t res;
149  do {
150  uint64_t one{1};
151  res = ::write(notify_fd_, &one, sizeof(one));
152  // retry if interrupted
153  } while ((res == -1) && (errno == EINTR));
154  } else {
155  ssize_t res;
156  do {
157  res = ::write(wakeup_fds_.second, ".", 1);
158  // retry if interrupted
159  } while ((res == -1) && (errno == EINTR));
160  }
161  }
162 
163  stdx::expected<void, std::error_code> close() {
164  remove_fd(wakeup_fds_.first);
165 
166  if (wakeup_fds_.first != impl::file::kInvalidHandle) {
167  impl::file::close(wakeup_fds_.first);
168  wakeup_fds_.first = impl::file::kInvalidHandle;
169  }
170 
171  if (wakeup_fds_.second != impl::file::kInvalidHandle) {
172  impl::file::close(wakeup_fds_.second);
173  wakeup_fds_.second = impl::file::kInvalidHandle;
174  }
175 
176  if (epfd_ != impl::file::kInvalidHandle) {
177  impl::file::close(epfd_);
178  epfd_ = impl::file::kInvalidHandle;
179  }
180 
181  return {};
182  }

184  class FdInterest {
185  public:
186  stdx::expected<void, std::error_code> merge(int epfd, native_handle_type fd,
187  impl::socket::wait_type wt,
188  bool oneshot) {
189  epoll_event ev{};
190  switch (wt) {
191  case impl::socket::wait_type::wait_read:
192  ev.events = EPOLLIN;
193  break;
194  case impl::socket::wait_type::wait_write:
195  ev.events = EPOLLOUT;
196  break;
197  case impl::socket::wait_type::wait_error:
198  ev.events = EPOLLERR;
199  break;
200  }
201 
202  ev.data.fd = fd;
203  ev.events |= EPOLLET;
204 
205  if (oneshot) {
206  ev.events |= EPOLLONESHOT;
207  }
208 
209  auto &b = bucket(fd);
210 
211  std::lock_guard<std::mutex> lk(b.mtx_);
212  const auto it = b.interest_.find(fd);
213  if (it != b.interest_.end()) {
214  // found, let's modify to handle the old and the new interest
215 
216  // std::cerr << __LINE__ << ": merge: " << fd << std::endl;
217 
218  ev.events |= it->second;
219  const auto ctl_res =
220  impl::epoll::ctl(epfd, impl::epoll::Cmd::mod, fd, &ev);
221  if (!ctl_res) return ctl_res;
222 
223  it->second = ev.events;
224  return {};
225  }
226  // std::cerr << __LINE__ << ": add: " << fd << " "
227  // << std::bitset<32>(ev.events) << std::endl;
228 
229  // not found, let's add it.
230 
231  const auto ctl_res =
232  impl::epoll::ctl(epfd, impl::epoll::Cmd::add, fd, &ev);
233  if (!ctl_res) return ctl_res;
234 
235  const uint32_t ev_events = ev.events;
236  b.interest_.emplace(fd, ev_events);
237 
238  return {};
239  }

241  stdx::expected<void, std::error_code> erase(int epfd,
242  native_handle_type fd) {
243  auto &b = bucket(fd);
244  // std::cerr << __LINE__ << ": del: " << fd << std::endl;
245 
246  std::lock_guard<std::mutex> lk(b.mtx_);
247 
248  // may be called from another thread through ->cancel()
249  const auto it = b.interest_.find(fd);
250  if (it != b.interest_.end()) {
251  auto epoll_ctl_res =
252  impl::epoll::ctl(epfd, impl::epoll::Cmd::del, fd, nullptr);
253  if (!epoll_ctl_res) return epoll_ctl_res;
254 
255  b.interest_.erase(it);
256  } else {
257  // return ENOENT as epoll_ctl() would do
258  return stdx::make_unexpected(
259  make_error_code(std::errc::no_such_file_or_directory));
260  }
261 
262  return {};

265  // remove interest for revent from file-descriptor.
266  stdx::expected<void, std::error_code> remove_fd_interest(
267  int epfd, native_handle_type fd, uint32_t revent) {
268  auto &b = bucket(fd);
269 
270  std::lock_guard<std::mutex> lk(b.mtx_);
271 
272  const auto it = b.interest_.find(fd);
273  if (it != b.interest_.end()) {
274  // fd is found
275 
276  // check if the register interest is aware of the interest we want to
277  // remove
278  auto const ev_mask = EPOLLIN | EPOLLOUT | EPOLLERR;
279  if (0 == ((it->second & ev_mask) & (revent & ev_mask))) {
280  return stdx::make_unexpected(
281  make_error_code(std::errc::argument_out_of_domain));
282  }
283 
284  it->second &= ~revent;
285 
286  epoll_event ev{};
287  ev.data.fd = fd;
288  ev.events = it->second;
289 
290  const auto ctl_res =
291  impl::epoll::ctl(epfd, impl::epoll::Cmd::mod, fd, &ev);
292  if (!ctl_res) return ctl_res;
293  } else {
294  // return ENOENT as epoll_ctl() would do
295  return stdx::make_unexpected(
296  make_error_code(std::errc::no_such_file_or_directory));
297  }
298 
299  return {};
300  }

302  /**
303  * update registered fd-interest after a oneshot event fired.
304  */
305  stdx::expected<void, std::error_code> after_event_fired(
306  native_handle_type fd, uint32_t revent) {
307  auto &b = bucket(fd);
308 
309  std::lock_guard<std::mutex> lk(b.mtx_);
310 
311  const auto it = b.interest_.find(fd);
312  if (it == b.interest_.end()) {
313  // return ENOENT as epoll_ctl() would do
314  return stdx::make_unexpected(
315  make_error_code(std::errc::no_such_file_or_directory));
316  }
317 
318  if (!(it->second & EPOLLONESHOT)) {
319  // not a oneshot event. The interest hasn't changed.
320  return {};
321  }
322 
323  // check the interest we remove is actually part of the requested set
324  auto const ev_mask = EPOLLIN | EPOLLOUT | EPOLLERR;
325  if (((it->second & ev_mask) & (revent & ev_mask)) == 0) {
326  std::cerr << "after_event_fired(" << fd << ", "
327  << std::bitset<32>(revent) << ") not in "
328  << std::bitset<32>(it->second) << std::endl;
329  return stdx::make_unexpected(
330  make_error_code(std::errc::argument_out_of_domain));
331  }
332 
333  // update the fd-interest
334  it->second &= ~revent;
335 
336  return {};
337  }

407  /**
408  * poll one event from the registered fd-interest.
409  *
410  * removes the interest of the event that fired
411  *
412  * @param timeout wait at most timeout milliseconds
413  *
414  * @returns fd_event which fired
415  * @retval std::errc::timed_out in case of timeout
416  */
417  stdx::expected<fd_event, std::error_code> poll_one(
418  std::chrono::milliseconds timeout) override {
419  if (!is_open()) {
420  return stdx::make_unexpected(
421  make_error_code(std::errc::invalid_argument));
422  }
423 
424  if (fd_events_processed_ == fd_events_size_) {
425  auto res = impl::epoll::wait(epfd_, fd_events_.data(), fd_events_.size(),
426  timeout);
427 
428  if (!res) return stdx::make_unexpected(res.error());
429 
430  fd_events_processed_ = 0;
431  fd_events_size_ = *res;
432 
433  if (fd_events_size_ == 0) {
434  return stdx::make_unexpected(make_error_code(std::errc::timed_out));
435  }
436 
437  for (size_t ndx{}; ndx < fd_events_size_; ++ndx) {
438  const ::epoll_event ev = fd_events_[ndx];
439 
440  auto after_res = after_event_fired(ev.data.fd, ev.events);
441  if (!after_res) {
442  std::cerr << "after_event_fired(" << ev.data.fd << ", "
443  << std::bitset<32>(ev.events) << ") " << after_res.error()
444  << " " << after_res.error().message() << std::endl;
445  }
446  }
447  }
448 
449  // traverse the events in reverse order as the epoll::wait() returns the
450  // events in LIFO order which may result in higher average latency.
451 
452  size_t ndx = fd_events_size_ - fd_events_processed_ - 1;
453 
454  ::epoll_event ev = fd_events_[ndx];
455 
456  fd_events_processed_++;
457 
458  if ((notify_fd_ != impl::file::kInvalidHandle)
459  ? (ev.data.fd == notify_fd_)
460  : (ev.data.fd == wakeup_fds_.first)) {
461  // wakeup fd fired
462  //
463  // - don't remove the interest for it
464  // - report to the caller that we don't have an event yet by saying we got
465  // interrupted
466  on_notify();
467 
468  return stdx::make_unexpected(make_error_code(std::errc::interrupted));
469  }
470 
471  return fd_event{ev.data.fd, static_cast<short>(ev.events)};
472  }

475  FdInterest registered_events_;
476 
477  // the event-set should be large enough to get a full picture as we otherwise
478  // might starve connections because we fetch a hot set of fds instead of the
479  // full set
480  //
481  // ready-set = [ 1 2 3 4 5 6 ]
482  //
483  // epoll_wait(.., 4, ...) = [ 1 2 3 4 ]
484  // epoll_ctl(MOD, POLLIN, 1)
485  // epoll_ctl(MOD, POLLIN, 2)
486  // epoll_ctl(MOD, POLLIN, 3)
487  // epoll_ctl(MOD, POLLIN, 4)
488  //
489  // ... 1, 2, 3, 4 may become ready in the meantime
490  //
491  // epoll_wait(.., 4, ...) = [ 1 2 3 4 ]
492  //
493  // ... and 5, 6 never get processed.
