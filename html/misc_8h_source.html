43 /** Check if a given buffer is inside another buffer.
44 @return true if inside */
45 inline bool buf_is_inside_another(
46  /** [in] First buffer, that should be inside the other. */
47  const unsigned char *small,
48  /** [in] First buffer length in bytes. */
49  size_t small_length,
50  /** [in] Second buffer, that should contain the other. */
51  const unsigned char *big,
52  /** [in] Second buffer length in bytes. */
53  size_t big_length) {
54  const unsigned char *small_after_last = small + small_length;
55  const unsigned char *big_after_last = big + big_length;
56 
57  return small >= big && small_after_last <= big_after_last;
58 }
