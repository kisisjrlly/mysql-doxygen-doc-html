74 /**
75  * base class of all http request handler threads
76  *
77  * - HttpRequestMainThread opens the socket, and accepts and handles connections
78  * - HttpRequestWorkerThread accepts and handles connections, using the socket
79  * listened by the main-thread
80  *
81  * As all threads can accept in parallel this may lead to a thundering herd
82  * problem and quite likely it is better to let only one thread accept() and
83  * push the socket handling into async-deque and let all workers steal from the
84  * queue
85  */
86 class HttpRequestThread {
87  public:
88  HttpRequestThread()
89  : ev_base(event_base_new(), &event_base_free),
90  ev_http(evhttp_new(ev_base.get()), &evhttp_free),
91  ev_shutdown_timer(event_new(ev_base.get(), -1, EV_PERSIST,
92  stop_eventloop, ev_base.get()),
93  &event_free) {
94  // enable all methods to allow the higher layers to handle them
95  //
96  // CONNECT, TRACE and OPTIONS are disabled by default if not explicitly
97  // enabled.
98  evhttp_set_allowed_methods(
99  ev_http.get(), EVHTTP_REQ_CONNECT | EVHTTP_REQ_DELETE | EVHTTP_REQ_GET |
100  EVHTTP_REQ_HEAD | EVHTTP_REQ_OPTIONS |
101  EVHTTP_REQ_PATCH | EVHTTP_REQ_POST | EVHTTP_REQ_PUT |
102  EVHTTP_REQ_TRACE);
103  }
