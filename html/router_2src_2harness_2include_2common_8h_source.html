89 /**
90  * Emit a range of elements using the serial comma.
91  *
92  * This function can be used to output a range of elements using a
93  * serial comma (also known as the Oxford comma). To emit a list of
94  * the first five prime numbers as "The first five prime numbers are
95  * 2, 3, 5, 7, and 11":
96  *
97  * @code
98  * std::vector<int> primes{2, 3, 5, 7, 11};
99  * std::cout << "The first five prime numbers are ";
100  * serial_comma(std::cout, primes.begin(), primes.end());
101  * std::cout << std::endl;
102  * @endcode
103  *
104  * @param out Output stream
105  * @param start Input iterator to start of range.
106  * @param finish Input iterator to one-after-end of range.
107  * @param delim Delimiter to use. Defaults to "and".
108  */
109 template <class InputIt>
110 void serial_comma(std::ostream &out, InputIt start, InputIt finish,
111  const std::string &delim = "and") {
112  auto elements = std::distance(start, finish);
113  if (elements == 1) {
114  out << *start;
115  } else if (elements == 2) {
116  out << *start++;
117  out << " " << delim << " " << *start;
118  } else {
119  while (elements-- > 0) {
120  out << *start++;
121  if (elements > 0) out << ", ";
122  if (elements == 1) out << delim << " ";
123  }
124  }
125 }

127 /**
128  * Returns string containing list of the elements using the serial comma.
129  *
130  * This function can be used to output a range of elements using a
131  * serial comma (also known as the Oxford comma). To return a list of
132  * the first five prime numbers as "The first five prime numbers are
133  * 2, 3, 5, 7, and 11":

139  * @endcode
140  *

144  *

147 template <class InputIt>

156 /**
157  * Returns string containing list of the elements separated by selected
158  * delimiter.
159  *
160  * To return a list of the first five prime numbers as "The first five prime
161  * numbers are 2, 3, 5, 7, 11":
162  *
163  * @code
164  * std::vector<int> primes{2, 3, 5, 7, 11};
165  * std::cout << "The first five prime numbers are "
166  * << list_elements(primes.begin(), primes.end()) << std::endl;
167  * @endcode
168  *
169  * @param start Input iterator to start of range.
170  * @param finish Input iterator to one-after-end of range.
171  * @param delim Delimiter to use. Defaults to ",".
172  *
173  * @return string containing list of the elements
174  */
175 template <class InputIt>
176 std::string list_elements(InputIt start, InputIt finish,
177  const std::string &delim = ",") {
178  std::string result;
179  for (auto cur = start; cur != finish; ++cur) {
180  if (cur != start) result += delim;
181  result += *cur;
182  }
183 
184  return result;
185 }
