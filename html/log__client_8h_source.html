67 template <ldap_log_type::ldap_type type>
68 void Ldap_logger::log(std::string msg) {
69  std::stringstream log_stream;
70  switch (type) {
71  case ldap_log_type::LDAP_LOG_DBG:
72  if (LDAP_LOG_LEVEL_ALL > m_log_level) {
73  return;
74  }
75  log_stream << "[DBG] ";
76  break;
77  case ldap_log_type::LDAP_LOG_INFO:
78  if (LDAP_LOG_LEVEL_ERROR_WARNING_INFO > m_log_level) {
79  return;
80  }
81  log_stream << "[Note] ";
82  break;
83  case ldap_log_type::LDAP_LOG_WARNING:
84  if (LDAP_LOG_LEVEL_ERROR_WARNING > m_log_level) {
85  return;
86  }
87  log_stream << "[Warning] ";
88  break;
89  case ldap_log_type::LDAP_LOG_ERROR:
90  if (LDAP_LOG_LEVEL_NONE >= m_log_level) {
91  return;
92  }
93  log_stream << "[Error] ";
94  break;
95  };
96 
97  /** We can write debug messages also in error log file if logging level is set
98  to debug. For MySQL client this will come from environment variable */
99  if (m_log_writer) {
100  log_stream << ": " << msg;
101  m_log_writer->write(log_stream.str());
102  }
103 }
