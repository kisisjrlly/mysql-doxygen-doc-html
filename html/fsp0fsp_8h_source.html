122 /* SPACE HEADER
123  ============
124 
125 File space header data structure: this data structure is contained in the
126 first page of a space. The space for this header is reserved in every extent
127 descriptor page, but used only in the first. */
128 
129 /*-------------------------------------*/
130 #define FSP_SPACE_ID 0 /* space id */
131 #define FSP_NOT_USED \
132  4 /* this field contained a value up to \
133  which we know that the modifications \
134  in the database have been flushed to \
135  the file space; not used now */
136 #define FSP_SIZE \
137  8 /* Current size of the space in \
138  pages */
139 #define FSP_FREE_LIMIT \
140  12 /* Minimum page number for which the \
141  free list has not been initialized: \
142  the pages >= this limit are, by \
143  definition, free; note that in a \
144  single-table tablespace where size \
145  < 64 pages, this number is 64, i.e., \
146  we have initialized the space \
147  about the first extent, but have not \
148  physically allocated those pages to the \
149  file */
150 #define FSP_SPACE_FLAGS \
151  16 /* fsp_space_t.flags, similar to \
152  dict_table_t::flags */
153 #define FSP_FRAG_N_USED \
154  20 /* number of used pages in the \
155  FSP_FREE_FRAG list */
156 #define FSP_FREE 24 /* list of free extents */
157 #define FSP_FREE_FRAG (24 + FLST_BASE_NODE_SIZE)
158 /* list of partially free extents not
159 belonging to any segment */
160 #define FSP_FULL_FRAG (24 + 2 * FLST_BASE_NODE_SIZE)
161 /* list of full extents not belonging
162 to any segment */
163 #define FSP_SEG_ID (24 + 3 * FLST_BASE_NODE_SIZE)
164 /* 8 bytes which give the first unused
165 segment id */
166 #define FSP_SEG_INODES_FULL (32 + 3 * FLST_BASE_NODE_SIZE)
167 /* list of pages containing segment
168 headers, where all the segment inode
169 slots are reserved */
170 #define FSP_SEG_INODES_FREE (32 + 4 * FLST_BASE_NODE_SIZE)
171 /* list of pages containing segment
172 headers, where not all the segment
173 header slots are reserved */
174 /*-------------------------------------*/
175 /* File space header size */
176 #define FSP_HEADER_SIZE (32 + 5 * FLST_BASE_NODE_SIZE)
177 
178 #define FSP_FREE_ADD \
179  4 /* this many free extents are added \
180  to the free list from above \
181  FSP_FREE_LIMIT at a time */

189 
190 Segment inode which is created for each segment in a tablespace. NOTE: in

202 /*-------------------------------------*/

204  0 /* 8 bytes of segment id: if this is 0, \

270 /* EXTENT DESCRIPTOR
271  =================
272 
273 File extent descriptor data structure: contains bits to tell which pages in
274 the extent are free and which contain old tuple version to clean. */
275 
276 /*-------------------------------------*/
277 #define XDES_ID \
278  0 /* The identifier of the segment \
279  to which this extent belongs */
280 #define XDES_FLST_NODE \
281  8 /* The list node data structure \
282  for the descriptors */
283 #define XDES_STATE (FLST_NODE_SIZE + 8)
284 /* contains state information
285 of the extent */
286 #define XDES_BITMAP (FLST_NODE_SIZE + 12)
287 /* Descriptor bitmap of the pages
288 in the extent */
289 /*-------------------------------------*/
290 
291 #define XDES_BITS_PER_PAGE 2 /* How many bits are there per page */
292 #define XDES_FREE_BIT \
293  0 /* Index of the bit which tells if \
294  the page is free */
295 #define XDES_CLEAN_BIT \
296  1 /* NOTE: currently not used! \
297  Index of the bit which tells if \
298  there are old versions of tuples \
299  on the page */
300 /** States of a descriptor */
301 enum xdes_state_t {
302 
303  /** extent descriptor is not initialized */
304  XDES_NOT_INITED = 0,
305 
306  /** extent is in free list of space */
307  XDES_FREE = 1,
308 
309  /** extent is in free fragment list of space */
310  XDES_FREE_FRAG = 2,
311 
312  /** extent is in full fragment list of space */
313  XDES_FULL_FRAG = 3,
314 
315  /** extent belongs to a segment */
316  XDES_FSEG = 4,
317 
318  /** fragment extent leased to segment */
319  XDES_FSEG_FRAG = 5
320 };

586 operation which will probably result in more space usage, like an
587 insert in a B-tree; FSP_UNDO means allocation to undo logs: if we are

602 different rules in that special case, just ensuring that there are n_pages
603 free pages available.
