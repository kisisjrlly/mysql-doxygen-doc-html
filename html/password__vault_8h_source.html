31 class PasswordVault {
32  public:
33  /** @brief Create an instance of the vault.
34  *
35  * On creation the vault cache is initialized with the contents of the vault
36  * file at %APPDATA%/MySQL/MySQL Router/mysql_router_user_data.dat.
37  * The passwords are stored in the cache in cleartext.
38  */
39  PasswordVault();
40 
41  /** @brief wipes the contents of the vault cache.
42  */
43  ~PasswordVault();
44 
45  /** @brief Updates a pair (section name, password) in the vault cache.
46  *
47  * If the record for the given section name & password does not exits, it is
48  * created. If it exists it is just updated with the new password.
49  *
50  * @param section_name The name of the configuration section to store in the
51  * vault.
52  * @param password The password, in clear text, of the user in the
53  * configuration section to store in the vault.
54  */
55  void update_password(const std::string &section_name,
56  const std::string &password);
57 
58  /** @brief Retrieves the password, in clear text, for the given section as
59  * is stored in the vault.
60  *
61  * @param section_name The name of the configuration section for which to
62  * retrieve the password.
63  * @param out_password Output paramter. The password in clear text if the
64  * section name was found in the vault.
65  * @return true if a password was retrieved for the given section, false if
66  * the section name could not be found in the vault.
67  */
68  bool get_password(const std::string &section_name,
69  std::string &out_password) const;
70 
71  /** @brief Removes the password from the vault for the given section name.
72  *
73  * After executing this method for a fiven section name, the method
74  * get_password will return false for the same section name.
75  *
76  * @param section_name The name of the configuration section for which to
77  * remove the password.
78  */
79  void remove_password(const std::string &section_name);
80 
81  /** @brief Stores the vault cache into persistent storage in encrypted form.
82  *
83  * The vault location in persistent storage is
84  * %APPDATA%/MySQL/MySQL Router/mysql_router_user_data.dat.
85  */
86  void store_passwords();
87 
88  /** @brief Wipes the contents of the vault file.
89  *
90  * NOTE: The delete the vault cache (in memory) created for an instance of
91  * PasswordVault is done automatically in the destructor.
92  */
93  void clear_passwords();
94 
95  private:
96  void load_passwords();
97  std::string get_vault_path() const;
98  // Password cache as pairs <section_name, password>
99  std::map<std::string, std::string> _passwords;
100  void password_scrambler(std::string &pass);
101 };
