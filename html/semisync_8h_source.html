38 /**
39  This class is used to trace function calls and other process
40  information
41 */
42 class Trace {
43  public:
44  static const unsigned long kTraceFunction;
45  static const unsigned long kTraceGeneral;
46  static const unsigned long kTraceDetail;
47  static const unsigned long kTraceNetWait;
48 
49  unsigned long trace_level_; /* the level for tracing */
50 
51  inline void function_enter(const char *func_name) {
52  if ((trace_level_ & kTraceFunction) && log_bi)
53  LogErr(INFORMATION_LEVEL, ER_SEMISYNC_TRACE_ENTER_FUNC, func_name);
54  }
55 
56  inline int function_exit(const char *func_name, int exit_code) {
57  if ((trace_level_ & kTraceFunction) && log_bi)
58  LogErr(INFORMATION_LEVEL, ER_SEMISYNC_TRACE_EXIT_WITH_INT_EXIT_CODE,
59  func_name, exit_code);
60  return exit_code;
61  }
62 
63  inline bool function_exit(const char *func_name, bool exit_code) {
64  if ((trace_level_ & kTraceFunction) && log_bi)
65  LogErr(INFORMATION_LEVEL, ER_SEMISYNC_TRACE_EXIT_WITH_BOOL_EXIT_CODE,
66  func_name, exit_code ? "True" : "False");
67  return exit_code;
68  }
69 
70  inline void function_exit(const char *func_name) {
71  if ((trace_level_ & kTraceFunction) && log_bi)
72  LogErr(INFORMATION_LEVEL, ER_SEMISYNC_TRACE_EXIT, func_name);
73  }
74 
75  Trace() : trace_level_(0L) {}
76  Trace(unsigned long trace_level) : trace_level_(trace_level) {}
77 };
