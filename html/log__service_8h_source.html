42 /* service helpers */
43 typedef enum enum_log_service_chistics {
44  /** We do not have information about this service yet. */
45  LOG_SERVICE_UNSPECIFIED = 0,
46 
47  /** Service is read-only -- it guarantees it will not modify the log-event.
48  This information may later be used to e.g. run log-writers in parallel. */
49  LOG_SERVICE_READ_ONLY = 1,
50 
51  /** Service is a singleton -- it may occur in the log service pipeline
52  only once. */
53  LOG_SERVICE_SINGLETON = 2,
54 
55  /** Service is built-in (and can not be INSTALLed/UNINSTALLed */
56  LOG_SERVICE_BUILTIN = 4,
57 
58  // values from 8..64 are reserved for extensions
59 
60  /** Service is a source. It adds key/value pairs beyond those in the
61  statement that first created the log-event. Log-sources are not
62  normally READ_ONLY. */
63  LOG_SERVICE_SOURCE = 128,
64 
65  /** Service is a filter. A filter should not be the last service in
66  the log service pipeline. */
67  LOG_SERVICE_FILTER = 256,
68 
69  /** Service is a sink (usually a log-writer). Sinks will normally
70  not modify the log-event, but be READ_ONLY. */
71  LOG_SERVICE_SINK = 512,
72 
73  /** Service supports the performance_schema.error_log table.
74  If the caller provides a buffer, the service will write output
75  to be displayed in the performance-schema table there.
76  This can be the entirety of the log-entry, or a projection
77  thereof (usually omitting key/value pairs that are already
78  shown in other columns of said table).
79  Services flagged this must also be flagged LOG_SERVICE_SINK! */
80  LOG_SERVICE_PFS_SUPPORT = 1024,
81 
82  /** Service can parse lines in the format it outputs. Services flagged
83  this must also be flagged LOG_SERVICE_SINK | LOG_SERVICE_PFS_SUPPORT! */
84  LOG_SERVICE_LOG_PARSER = 2048,
85 
86  /** Service is a special sink used during start-up that buffers log-events
87  until the log service pipeline is fully set up, at which point we'll
88  flush (that is, filter and prints) the buffered events.
89  Services flagged this must also be flagged LOG_SERVICE_SINK! */
90  LOG_SERVICE_BUFFER = 8192
91 
92 } log_service_chistics;

94 /**
95  Error codes. These are grouped (general issues, invalid data,
96  file ops, etc.). Each group has a sparsely populated range so
97  we may add entries as needed without introducing incompatibility
98  by renumbering the existing ones.
99 */
100 typedef enum enum_log_service_error {
101  /// no error
102  LOG_SERVICE_SUCCESS = 0,
103 
104  /// error not otherwise specified
105  LOG_SERVICE_MISC_ERROR = -1,
106 
107  /// no error, but no effect either
108  LOG_SERVICE_NOTHING_DONE = -2,
109 
110  /**
111  arguments are valid, we just don't have the space (either pre-allocated
112  in this function, or passed to us by the caller)
113  */
114  LOG_SERVICE_BUFFER_SIZE_INSUFFICIENT = -10,
115 
116  /// we cannot allocate a (temporary or return) buffer of the required size
117  LOG_SERVICE_OUT_OF_MEMORY = -11,
118 
119  /// service uninavailable (bad internal state/underlying service unavailable)

121

124 
125  /// argument was invalid (out of range, malformed, etc.)
126  LOG_SERVICE_INVALID_ARGUMENT = -30,
127

129  argument too long (a special case of malformed).

134  */
135  LOG_SERVICE_ARGUMENT_TOO_LONG = -31,

137  /**

145 
146  /// lock error (could not init, or is not inited, etc.)

158 } log_service_error;

161 /**

163 
164  @param instance State-pointer that was returned on open.

173 /**
174  Flush any buffers. This function will be called by the server
175  on FLUSH ERROR LOGS. The service may write its buffers, close
176  and re-open any log files to work with log-rotation, etc.
177  The flush function MUST NOT itself log anything!
178  A service implementation may provide a nullptr if it does not
179  wish to provide a flush function.
180 
181  @param instance State-pointer that was returned on open.
182  Value may be changed in flush.
183 
184  @retval LOG_SERVICE_NOTHING_DONE no work was done
185  @retval LOG_SERVICE_SUCCESS flush completed without incident
186  @retval otherwise an error occurred
187 */
188 DECLARE_METHOD(log_service_error, flush, (void **instance));
189 
190 /**
191  Open a new instance.
192 
193  @param ll optional arguments
194  @param instance If state is needed, the service may allocate and
195  initialize it and return a pointer to it here.
196  (This of course is particularly pertinent to
197  components that may be opened multiple times,
198  such as the JSON log writer.)
199  This state is for use of the log-service component
200  in question only and can take any layout suitable
201  to that component's need. The state is opaque to
202  the server/logging framework. It must be released
203  on close.
204 
205  @retval LOG_SERVICE_SUCCESS success, returned hande is valid
206  @retval otherwise a new instance could not be created
207 */
208 DECLARE_METHOD(log_service_error, open, (log_line * ll, void **instance));
209 
210 /**
211  Close and release an instance. Flushes any buffers.
212 
213  @param instance State-pointer that was returned on open.

215  it should be released, and the pointer
216  set to nullptr.
217 
218  @retval LOG_SERVICE_SUCCESS success
219  @retval otherwise an error occurred
220 */
221 DECLARE_METHOD(log_service_error, close, (void **instance));

243 /**
244  Provide the name for a log file this service would access.
245 
246  @param instance instance info returned by open() if requesting
247  the file-name for a specific open instance.
248  nullptr to get the name of the default instance
249  (even if it that log is not open). This is used
250  to determine the name of the log-file to load on
251  start-up.
252  @param buf Address of a buffer allocated in the caller.
253  The callee may return an extension starting
254  with '.', in which case the path and file-name
255  will be the system's default, except with the
256  given extension.
257  Alternatively, the callee may return a file-name
258  which is assumed to be in the same directory
259  as the default log.
260  Values are C-strings.
261  @param bufsize The size of the allocation in the caller.
262 
263  @retval 0 Success
264  @retval -1 Mode not supported (only default / only instances supported)
265  @retval -2 Buffer not large enough
266  @retval -3 Misc. error
267 */
268 DECLARE_METHOD(log_service_error, get_log_name,
269  (void *instance, char *buf, size_t bufsize));
