157 /** Page identifier. */
158 class page_id_t {
159  public:
160  /**
161  This class does not have a default constructor, because there is no natural
162  choice for default values of m_space and m_page_no.
163 
164  If 0,0 were used, then it's not good as it doesn't match UINT32_UNDEFINED
165  used to denote impossible page_no_t in several places, and 0 is a legal
166  value for both space_id_t and page_id_t of a real page!
167 
168  If UINT32_UNDEFINED,UINT32_UNDEFINED were used, then it doesn't match the
169  most common usage where use use memset(parent,0,sizeof(parent_t)); on a
170  parent struct where one of the members has page_id_t type - which is ok
171  given that page_id_t is TriviallyCopyable, and that the field is not
172  used until it is assigned some real value. Such constructor would be
173  misleading to people reading the code, as they might expect UINT32_UNDEFINED
174  value, if they didn't notice the memset code burried somewhere in parent's
175  initialization routine.
176 
177  Therefore, please either be explicit by using (space,page_no) overload,
178  or continue to use memset at your own risk.
179  */
180  page_id_t() = delete;
181 
182  /** Constructor from (space, page_no).
183  @param[in] space tablespace id
184  @param[in] page_no page number */
185  page_id_t(space_id_t space, page_no_t page_no)
186  : m_space(space), m_page_no(page_no) {}
187 
188  /** Retrieve the tablespace id.
189  @return tablespace id */
190  inline space_id_t space() const { return (m_space); }
191 
192  /** Retrieve the page number.
193  @return page number */
194  inline page_no_t page_no() const { return (m_page_no); }
195 
196  /** Retrieve the fold value.
197  @return fold value */
198  inline uint32_t fold() const { return (m_space << 20) + m_space + m_page_no; }
199 
200  /** Reset the values from a (space, page_no).
201  @param[in] space tablespace id
202  @param[in] page_no page number */
203  inline void reset(space_id_t space, page_no_t page_no) {
204  m_space = space;
205  m_page_no = page_no;
206  }
207 
208  /** Reset the page number only.
209  @param[in] page_no page number */
210  inline void set_page_no(page_no_t page_no) { m_page_no = page_no; }
211 
212  /** Check if a given page_id_t object is equal to the current one.
213  @param[in] a page_id_t object to compare
214  @return true if equal */
215  inline bool operator==(const page_id_t &a) const {
216  return (a.space() == m_space && a.page_no() == m_page_no);
217  }
218 
219  /** Check if a given page_id_t object is not equal to the current one.
220  @param[in] a page_id_t object to compare
221  @return true if not equal */
222  inline bool operator!=(const page_id_t &a) const { return !(*this == a); }
223 
224  /** Provides a lexicographic ordering on <space_id,page_no> pairs
225  @param[in] other page_id_t object to compare
226  @return true if this is strictly smaller than other */
227  inline bool operator<(const page_id_t &other) const {
228  return m_space < other.space() ||
229  (m_space == other.space() && m_page_no < other.page_no());
230  }
231 
232  private:
233  /** Tablespace id. */
234  space_id_t m_space;
235 
236  /** Page number. */
237  page_no_t m_page_no;
238 
239  friend std::ostream &operator<<(std::ostream &out, const page_id_t &page_id);
240 };
