47 using AllowedNodes = std::vector<std::string>;
48 // first argument is the new set of the allowed nodes
49 // second argument is the description of the condition that triggered the change
50 // (like ' metadata change' etc.) can be used for logging purposes by the caller
51 using AllowedNodesChangedCallback =
52  std::function<void(const AllowedNodes &, const std::string &)>;
53 // NOTE: this has to be container like std::list that does not invalidate
54 // iterators when it is modified as we return the iterator to the insterted
55 // callback to the caller to allow unregistering
56 using AllowedNodesChangeCallbacksList = std::list<AllowedNodesChangedCallback>;
57 using AllowedNodesChangeCallbacksListIterator =
58  AllowedNodesChangeCallbacksList::iterator;

92 /** @class RouteDestination
93  * @brief Manage destinations for a Connection Routing
94  *
95  * This class manages destinations which are used in Connection Routing.
96  * A destination is usually a MySQL Server and is stored using the IP
97  * or hostname together with the TCP port (defaulting to 3306 for classic
98  * protocol or to 33060 for x protocol).
99  *
100  * RouteDestination is meant to be a base class and used to inherite and
101  * create class which change the behavior. For example, the `get_next()`
102  * method is usually changed to get the next server in the list.
103  */

105  public:

110  * @param protocol Protocol for the destination, defaults to value returned

113  * to "real" (not mock) implementation
114  * (mysql_harness::SocketOperations)
115  */
116  RouteDestination(Protocol::Type protocol = Protocol::get_default(),
117  mysql_harness::SocketOperationsBase *sock_ops =
118  mysql_harness::SocketOperations::instance())
