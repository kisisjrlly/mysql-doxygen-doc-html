75  set to String result object. We have to release them properly manually
76  since they won't be released when the String result is destroyed.

78 class BG_result_buf_mgr {
79  typedef Prealloced_array<void *, 64> Prealloced_buffers;
80 
81  public:
82  BG_result_buf_mgr() : bg_result_buf(nullptr), bg_results(PSI_INSTRUMENT_ME) {}
83 
84  ~BG_result_buf_mgr() {
85  free_intermediate_result_buffers();
86  free_result_buffer();
87  }
88 
89  void add_buffer(void *buf) { bg_results.insert_unique(buf); }
90 
91  void forget_buffer(void *buf) {
92  if (bg_result_buf == buf) bg_result_buf = nullptr;
93  bg_results.erase_unique(buf);
94  }
95 
96  /* Free intermediate result buffers accumulated during GIS calculation. */
97  void free_intermediate_result_buffers() {
98  bg_results.erase_unique(bg_result_buf);
99  for (Prealloced_buffers::iterator itr = bg_results.begin();
100  itr != bg_results.end(); ++itr)
101  gis_wkb_raw_free(*itr);
102  bg_results.clear();
103  }
104 
105  // Free the final result buffer, should be called after the result used.
106  void free_result_buffer() {
107  gis_wkb_raw_free(bg_result_buf);
108  bg_result_buf = nullptr;
109  }
110 
111  void set_result_buffer(void *buf) {
112  bg_result_buf = buf;
113  bg_results.erase_unique(bg_result_buf);
114  }
115 
116  private:
117  /*
118  Hold data buffer of this set operation's final result geometry which is
119  freed next time val_str is called since it can be used by upper Item nodes.
120  */
121  void *bg_result_buf;
122 
123  /*
124  Result buffers for intermediate set operation results, which are freed
125  before val_str returns.
126  */
127  Prealloced_buffers bg_results;
