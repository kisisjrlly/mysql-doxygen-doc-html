167  /**
168  * search entire line for match.

170  bool search(const std::string &line,
171  match_flag_type match_flags = match_default) const {
172 #ifdef USE_POSIX_RE_IMPL
173  if (0 != regexec(reg_.get(), line.c_str(), 0, NULL, match_flags)) {
174  return false;
175  }
176 
177  return true;
178 #else
179  return std::regex_search(line, reg_, match_flags);
180 #endif
181  }
