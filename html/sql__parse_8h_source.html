143 /* Bits in sql_command_flags */
144 
145 #define CF_CHANGES_DATA (1U << 0)
146 /* The 2nd bit is unused -- it used to be CF_HAS_ROW_COUNT. */
147 #define CF_STATUS_COMMAND (1U << 2)
148 #define CF_SHOW_TABLE_COMMAND (1U << 3)
149 #define CF_WRITE_LOGS_COMMAND (1U << 4)
150 /**
151  Must be set for SQL statements that may contain
152  Item expressions and/or use joins and tables.
153  Indicates that the parse tree of such statement may
154  contain rule-based optimizations that depend on metadata
155  (i.e. number of columns in a table), and consequently
156  that the statement must be re-prepared whenever
157  referenced metadata changes. Must not be set for
158  statements that themselves change metadata, e.g. RENAME,
159  ALTER and other DDL, since otherwise will trigger constant
160  reprepare. Consequently, complex item expressions and
161  joins are currently prohibited in these statements.
162 */
163 #define CF_REEXECUTION_FRAGILE (1U << 5)
164 /**
165  Implicitly commit before the SQL statement is executed.
166 
167  Statements marked with this flag will cause any active
168  transaction to end (commit) before proceeding with the
169  command execution.
170 
171  This flag should be set for statements that probably can't
172  be rolled back or that do not expect any previously metadata
173  locked tables.
174 */
175 #define CF_IMPLICIT_COMMIT_BEGIN (1U << 6)
176 /**
177  Implicitly commit after the SQL statement.
178 
179  Statements marked with this flag are automatically committed
180  at the end of the statement.
181 
182  This flag should be set for statements that will implicitly
183  open and take metadata locks on system tables that should not
184  be carried for the whole duration of a active transaction.
185 */
186 #define CF_IMPLICIT_COMMIT_END (1U << 7)
187 /**
188  CF_IMPLICIT_COMMIT_BEGIN and CF_IMPLICIT_COMMIT_END are used
189  to ensure that the active transaction is implicitly committed
190  before and after every DDL statement and any statement that
191  modifies our currently non-transactional system tables.
192 */
193 #define CF_AUTO_COMMIT_TRANS (CF_IMPLICIT_COMMIT_BEGIN | CF_IMPLICIT_COMMIT_END)
194 
195 /**
196  Diagnostic statement.
197  Diagnostic statements:
198  - SHOW WARNING
199  - SHOW ERROR
200  - GET DIAGNOSTICS (WL#2111)
201  do not modify the Diagnostics Area during execution.
202 */
203 #define CF_DIAGNOSTIC_STMT (1U << 8)
204 
205 /**
206  Identifies statements that may generate row events
207  and that may end up in the binary log.
208 */
209 #define CF_CAN_GENERATE_ROW_EVENTS (1U << 9)
210 
211 /**
212  Identifies statements which may deal with temporary tables and for which
213  temporary tables should be pre-opened to simplify privilege checks.
214 */
215 #define CF_PREOPEN_TMP_TABLES (1U << 10)
216 
217 /**
218  Identifies statements for which open handlers should be closed in the
219  beginning of the statement.
220 */
221 #define CF_HA_CLOSE (1U << 11)
222 
223 /**
224  Identifies statements that can be explained with EXPLAIN.
225 */
226 #define CF_CAN_BE_EXPLAINED (1U << 12)
227 
228 /** Identifies statements which may generate an optimizer trace */
229 #define CF_OPTIMIZER_TRACE (1U << 14)
230 
231 /**
232  Identifies statements that should always be disallowed in
233  read only transactions.
234 */
235 #define CF_DISALLOW_IN_RO_TRANS (1U << 15)
236 
237 /**
238  Identifies statements and commands that can be used with Protocol Plugin
239 */
240 #define CF_ALLOW_PROTOCOL_PLUGIN (1U << 16)
241 
242 /**
243  Identifies statements (typically DDL) which needs auto-commit mode
244  temporarily turned off.
245 
246  @note This is necessary to prevent InnoDB from automatically committing
247  InnoDB transaction each time data-dictionary tables are closed
248  after being updated.
249 
250  @note This is also necessary for ACL DDL, so the code which
251  saves GTID state or slave state in the system tables at the
252  commit time works correctly. This code does statement commit
253  on low-level (see System_table_access:: close_table()) and
254  thus can pre-maturely commit DDL if @@autocommit=1.
255 */
256 #define CF_NEEDS_AUTOCOMMIT_OFF (1U << 17)
257 
258 /**
259  Identifies statements which can return rows of data columns (SELECT, SHOW ...)
260 */
261 #define CF_HAS_RESULT_SET (1U << 18)
262 
263 /**
264  Identifies DDL statements which can be atomic.
265  Having the bit ON does not yet define an atomic.
266  The property is used both on the master and slave.
267  On the master atomicity infers the binlog and gtid_executed system table.
268  On the slave it more involves the slave info table.
269 
270  @note At the momemnt of declaration the covered DDL subset coincides
271  with the of CF_NEEDS_AUTOCOMMIT_OFF.
272 */
273 #define CF_POTENTIAL_ATOMIC_DDL (1U << 19)
274 
275 /**
276  Statement is depending on the ACL cache, which can be disabled by the
277  --skip-grant-tables server option.
278 */
279 #define CF_REQUIRE_ACL_CACHE (1U << 20)
280 
281 /**
282  Identifies statements as SHOW commands using INFORMATION_SCHEMA system views.
283 */
284 #define CF_SHOW_USES_SYSTEM_VIEW (1U << 21)
285 
286 /* Bits in server_command_flags */
287 
288 /**
289  Skip the increase of the global query id counter. Commonly set for
290  commands that are stateless (won't cause any change on the server
291  internal states). This is made obsolete as query id is incremented
292  for ping and statistics commands as well because of race condition
293  (Bug#58785).
294 */
295 #define CF_SKIP_QUERY_ID (1U << 0)
296 
297 /**
298  Skip the increase of the number of statements that clients have
299  sent to the server. Commonly used for commands that will cause
300  a statement to be executed but the statement might have not been
301  sent by the user (ie: stored procedure).
302 */
303 #define CF_SKIP_QUESTIONS (1U << 1)
304 
305 /**
306  1U << 16 is reserved for Protocol Plugin statements and commands
307 */
