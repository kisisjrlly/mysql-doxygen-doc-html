62 /** @class Ofstream
63  * @brief interface to std::ofstream and alternative (mock) implementations
64  *
65  * std::ofstream is not mockable, because its methods are not virtual. To work
66  * around this, we create this interface class, which then acts as superclass to
67  * various std::ofstream implementation classes.
68  */
69 class Ofstream : public std::ofstream {
70  public:
71  // disabled copying the ofstream constructurs as sunpro 12.5 says:
72  //
73  // 'ofstream' not in 'std::ofstream'
74  //
75  // If there is a need to have other than the no-param constructor
76  // that is inherited by default, a new solution has to be found.
77  //
78  // using std::ofstream::ofstream;
79  ~Ofstream() override {}
80  virtual void open(const char *filename,
81  std::ios_base::openmode mode = std::ios_base::out) = 0;
82  virtual void open(const std::string &filename,
83  std::ios_base::openmode mode = std::ios_base::out) = 0;
