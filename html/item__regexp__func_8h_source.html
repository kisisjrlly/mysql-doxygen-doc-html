26 /**
27  @file item_regexp_func.h
28 
29  The function classes for regular expression functions. They have a common
30  base class Item_func_regexp, which is also the prefix of their class
31  names. After the %Item_func prefix comes the name of the SQL function,
32  e.g. Item_func_regexp_instr represents the SQL function `REGEXP_INSTR`.
33 
34  Type resolution
35  ===============
36 
37  The type and name resolution procedure is hooked into by the
38  Item_func_regexp class, which implement both
39  Item_result_field::resolve_type() and Item::fix_fields().
40 
41  Collations
42  ==========
43 
44  The regular expression library doesn't deal with collations at all, but we
45  need them because the 'winning' collation of the pattern and the subject
46  strings dictates case-sensitivity. The winning collation is defined by
47  coercion rules, and we don't delve into that here. See
48  Item_func::agg_arg_charsets_for_comparison(). The call to this function is
49  done in resolve_type() as this appears to be an unwritten convention.
50 
51  Implementation
52  ==============
53 
54  All communication with the regular expression library is done through a
55  Regexp_facade object, instantiated in Item_func_regexp::fix_fields().
56 
57  @todo We now clean up ICU heap memory in Item_func_regexp::cleanup. Should
58  it be done more rarely? On session close?
59 */

104  /// The value of the `position` argument, or its default if absent.
105  Mysql::Nullable<int> position() const {
106  int the_index = pos_arg_pos();
107  if (the_index != -1 && arg_count >= static_cast<uint>(the_index) + 1) {
108  int value = args[the_index]->val_int();
109  /*
110  Note: Item::null_value() can't be trusted alone here; there are cases
111  (for the DATE data type in particular) where we can have it set
112  without Item::maybe_null being set! This really should be cleaned up,
113  but until that happens, we need to have a more conservative check.
114  */
115  if (args[the_index]->maybe_null && args[the_index]->null_value)
116  return Mysql::Nullable<int>();
117  else
118  return value;
119  }
120  return 1;
121  }
122 
123  /// The value of the `occurrence` argument, or its default if absent.
124  Mysql::Nullable<int> occurrence() const {
125  int the_index = occ_arg_pos();
126  if (the_index != -1 && arg_count >= static_cast<uint>(the_index) + 1) {
127  int value = args[the_index]->val_int();
128  /*
129  Note: Item::null_value() can't be trusted alone here; there are cases
130  (for the DATE data type in particular) where we can have it set
131  without Item::maybe_null being set! This really should be cleaned up,
132  but until that happens, we need to have a more conservative check.
133  */
134  if (args[the_index]->maybe_null && args[the_index]->null_value)
135  return Mysql::Nullable<int>();
136  else
137  return value;
138  }
139  return 0;
140  }
141 
142  /// The value of the `match_parameter` argument, or an empty string if absent.
143  Mysql::Nullable<std::string> match_parameter() const {
144  int the_index = match_arg_pos();
145  if (the_index != -1 && arg_count >= static_cast<uint>(the_index) + 1) {
146  StringBuffer<5> buf; // Longer match_parameter doesn't make sense.
147  String *s = args[the_index]->val_str(&buf);
148  if (s != nullptr)
149  return to_string(*s);
150  else
151  return Mysql::Nullable<std::string>();
152  }
153  return std::string{};
154  }

180  double convert_str_to_real() {
181  DBUG_ASSERT(fixed == 1);
182  int err_not_used;
183  const char *end_not_used;
184  String *res = val_str(&str_value);
185  if (res == nullptr) return 0.0;
186  return my_strntod(res->charset(), res->ptr(), res->length(), &end_not_used,
187  &err_not_used);

190  longlong convert_str_to_int() {
191  DBUG_ASSERT(fixed == 1);
192  int err;
193  String *res = val_str(&str_value);
194  if (res == nullptr) return 0;
195  return my_strntoll(res->charset(), res->ptr(), res->length(), 10, nullptr,
196  &err);
197  }
