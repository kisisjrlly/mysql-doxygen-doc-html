38 /**
39  * message digest.
40  *
41  * Wrapper around Digest functions of openssl (EVP_MD_...)
42  *
43  * - MD5
44  * - SHA1
45  * - SHA224
46  * - SHA256
47  * - SHA384
48  * - SHA512
49  *
50  * @see openssl's EVP_sha512()

52 class Digest {

54  enum class Type { Md5, Sha1, Sha224, Sha256, Sha384, Sha512 };

56  /**

62 #if defined(OPENSSL_VERSION_NUMBER) && \

65 #else

67 #endif
68  }
69  { init(type_); }

71  /**
72  * initialize the message digest functions.
73  *
74  * Allows reused of the Digest function without reallocating memory
75  *
76  * @pre Digest is not initialized or is finalized.
77  */
78  void init(Type type) {
79  type_ = type;
80  EVP_DigestInit(ctx_.get(), Digest::get_evp_md(type));
81  }

101  /**
102  * finalize the digest and get digest value.
103  *
104  * @param out vector to place the digest value in
105  */
106  void finalize(std::vector<uint8_t> &out) {
107  // if cap is too large, limit it to uint::max and let narrowing handle the
108  // rest
109  unsigned int out_len{static_cast<unsigned int>(std::min(
110  out.capacity(),
111  static_cast<size_t>(std::numeric_limits<unsigned int>::max())))};
112 
113  EVP_DigestFinal_ex(ctx_.get(), out.data(), &out_len);
114  out.resize(out_len);
115  }

123  static size_t digest_size(Type type) {
124  if (auto *dg = Digest::get_evp_md(type)) {
