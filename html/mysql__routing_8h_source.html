81 /**
82  * container of sockets.
83  *
84  * allows to disconnect all of them.
85  *
86  * thread-safe.
87  */
88 template <class Protocol>
89 class SocketContainer {
90  using protocol_type = Protocol;
91  using socket_type = typename protocol_type::socket;
92 
93  // as a ref is returned, we a list to store the sockets
94  using container_type = std::list<socket_type>;
95 
96  public:
97  /**
98  * move ownership of socket_type to the container.
99  *
100  * @return a ref to the stored socket.
101  */
102  socket_type &push_back(socket_type &&sock) {
103  std::lock_guard<std::mutex> lk(mtx_);
104 
105  sockets_.push_back(std::move(sock));
106 
107  return sockets_.back();
108  }
109 
110  /**
111  * release socket from container.
112  *
113  * moves ownership of the socket to the caller.
114  *
115  * @return socket
116  */
117  socket_type release(socket_type &client_sock) {
118  std::lock_guard<std::mutex> lk(mtx_);
119 
120  return release_unlocked(client_sock);
121  }
122 
123  socket_type release_unlocked(socket_type &client_sock) {
124  for (auto cur = sockets_.begin(); cur != sockets_.end(); ++cur) {
125  if (cur->native_handle() == client_sock.native_handle()) {
126  auto sock = std::move(*cur);
127  sockets_.erase(cur);
128  return sock;
129  }
130  }
131 
132  // not found.
133  return socket_type{client_sock.get_executor().context()};
134  }
135 
136  template <class F>
137  auto run(F &&f) {
138  std::lock_guard<std::mutex> lk(mtx_);
139 
140  return f();
141  }
142 
143  /**
144  * disconnect all sockets.
145  */
146 
147  void disconnect_all() {
148  std::lock_guard<std::mutex> lk(mtx_);
149 
150  for (auto &sock : sockets_) {
151  sock.cancel();
152  }
153  }
154 
155  /**
156  * check if the container is empty.
157  */
158  bool empty() const {
159  std::lock_guard<std::mutex> lk(mtx_);
160  return sockets_.empty();
161  }
162 
163  /**
164  * get size of container.
165  */
166  size_t size() const {
167  std::lock_guard<std::mutex> lk(mtx_);
168  return sockets_.size();
169  }
170 
171  private:
172  container_type sockets_;
173 
174  mutable std::mutex mtx_;
175 };

180 /** @class MySQLRoutering
181  * @brief Manage Connections from clients to MySQL servers
182  *
183  * The class MySQLRouter is used to start a service listening on a particular
184  * TCP port for incoming MySQL Client connection and route these to a MySQL
185  * Server.
186  *
187  * Connection routing will not analyze or parse any MySQL package (except from
188  * those in the handshake phase to be able to discover invalid connection
189  * error) nor will it do any authentication. It will not handle errors from the
190  * MySQL server and not automatically recover. The client communicate through
191  * MySQL Router just like it would directly connecting.
192  *
193  * The MySQL Server is chosen from a given list of hosts or IP addresses
194  * (with or without TCP port) based on the the mode. For example, mode
195  * read-only will go through the list of servers in a round-robin way. The
196  * mode read-write will always go through the list from the beginning and
197  * failover to the next available.
198  *
199  *
200  * Example usage: bind to all IP addresses and use TCP Port 7001
201  *
202  * MySQLRouting r(routing::AccessMode::kReadWrite, "0.0.0.0", 7001);
203  * r.destination_connect_timeout = std::chrono::seconds(1);
204  * r.set_destinations_from_csv("10.0.10.5;10.0.11.6");
205  * r.start();
206  *
207  * The above example will, when MySQL running on 10.0.10.5 is not available,
208  * use 10.0.11.6 to setup the connection routing.
209  *
210  */
