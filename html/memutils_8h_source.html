74 /** Primary-template (functor) class for memory-utils. */
75 template <Source>
76 struct Memory {
77  static void *allocate(size_t bytes);
78  static void deallocate(void *ptr, size_t bytes);
79 };

81 /** Template specialization for RAM-based allocation/deallocation. */
82 template <>
83 struct Memory<Source::RAM> {
84  /** Allocates memory from RAM.
85  *
86  * Linux:
87  * - If support for NUMA is compiled in, and NUMA is available on the
88  * platform, NUMA allocation will be used.
89  * - If support for NUMA is compiled in, but NUMA is NOT available on the
90  * platform, non-NUMA allocation will be used (fallback to e.g. malloc).
91  * Windows:
92  * - If support for NUMA is compiled in, NUMA allocation will be used.
93  *
94  * Throws Result::OUT_OF_MEM if allocation was unsuccessful.
95  *
96  * [in] Size of the memory to be allocated.
97  * @return Pointer to allocated memory. */
98  static void *allocate(size_t bytes) {
99  void *memory = fetch(bytes);
100  if (memory == nullptr) {
101  throw Result::OUT_OF_MEM;
102  }
103  return memory;
104  }
105 
106  /** Deallocates memory from RAM.
107  *
108  * [in] Pointer to the memory to be deallocated..
109  * [in] Size of the memory to be deallocated. */
110  static void deallocate(void *ptr, size_t bytes) { drop(ptr, bytes); }
111 
112  private:
113  static void *fetch(size_t bytes);
114  static void drop(void *ptr, size_t bytes);
115 
116  private:
117 #ifdef TEMPTABLE_USE_LINUX_NUMA
118  /** Set to true if Linux's numa_available() reports "available" (!= -1). */
119  static const bool linux_numa_available;
120 #endif /* TEMPTABLE_USE_LINUX_NUMA */
121 };

123 /** Template specialization for MMAP-based allocation/deallocation. */
124 template <>
125 struct Memory<Source::MMAP_FILE> {
126  /** Allocates memory from MMAP-ed file. Throws Result::RECORD_FILE_FULL
127  * if temptable_use_map is not enabled or if allocation was unsuccessful.
128  *
129  * [in] Size of the memory to be allocated.
130  * @return Pointer to allocated memory. */
131  static void *allocate(size_t bytes) {
132  if (!temptable_use_mmap) {
133  throw Result::RECORD_FILE_FULL;
134  }
135  void *memory = fetch(bytes);
136  if (memory == nullptr) {
137  throw Result::RECORD_FILE_FULL;

139  return memory;
140  }

142  /** Deallocates memory from MMAP-ed file.
143  *
144  * [in] Pointer to the memory to be deallocated..
145  * [in] Size of the memory to be deallocated. */
146  static void deallocate(void *ptr, size_t bytes) { drop(ptr, bytes); }
147 
148  private:
149  static void *fetch(size_t bytes);
150  static void drop(void *ptr, size_t bytes);
151 };

153 inline void *Memory<Source::RAM>::fetch(size_t bytes) {
154 #if defined(TEMPTABLE_USE_LINUX_NUMA)
155  if (linux_numa_available) {
156  return numa_alloc_local(bytes);
157  } else {
158  return malloc(bytes);
159  }
160 #elif defined(HAVE_WINNUMA)
161  PROCESSOR_NUMBER processorNumber;
162  USHORT numaNodeId;
163  GetCurrentProcessorNumberEx(&processorNumber);
164  GetNumaProcessorNodeEx(&processorNumber, &numaNodeId);
165  bytes =
166  (bytes + win_page_size - 1) & ~(static_cast<size_t>(win_page_size) - 1);
167  return VirtualAllocExNuma(GetCurrentProcess(), nullptr, bytes,
168  MEM_RESERVE | MEM_COMMIT, PAGE_READWRITE,
169  numaNodeId);
170 #else
171  return malloc(bytes);
172 #endif
173 }
174 
175 inline void Memory<Source::RAM>::drop(void *ptr, size_t bytes) {
176  (void)bytes;
177 #if defined(TEMPTABLE_USE_LINUX_NUMA)
178  if (linux_numa_available) {
179  numa_free(ptr, bytes);
180  } else {
181  free(ptr);
182  }
183 #elif defined(HAVE_WINNUMA)
184  BOOL MY_ATTRIBUTE((unused)) ret = VirtualFree(ptr, 0, MEM_RELEASE);
185  assert(ret != 0);
186 #else
187  free(ptr);
188 #endif
189 }

191 inline void *Memory<Source::MMAP_FILE>::fetch(size_t bytes) {
192  DBUG_EXECUTE_IF("temptable_fetch_from_disk_return_null", return nullptr;);
193 
194 #ifdef _WIN32
195  const int mode = _O_RDWR;
196 #else
197  const int mode = O_RDWR;
198 #endif /* _WIN32 */
199 
200  char file_path[FN_REFLEN];
201  File f = create_temp_file(file_path, mysql_tmpdir, "mysql_temptable.", mode,
202  UNLINK_FILE, MYF(MY_WME));
203  if (f < 0) {
204  return nullptr;
205  }
206 
207  /* This will write `bytes` 0xa bytes to the file on disk. */
208  if (my_fallocator(f, bytes, 0xa, MYF(MY_WME)) != 0 ||
209  my_seek(f, 0, MY_SEEK_SET, MYF(MY_WME)) == MY_FILEPOS_ERROR) {
210  my_close(f, MYF(MY_WME));
211  return nullptr;
212  }
213 
214  void *ptr = my_mmap(nullptr, bytes, PROT_READ | PROT_WRITE, MAP_SHARED, f, 0);
215 
216  // NOTE: Closing the file-descriptor __immediately__ after MMAP-ing it does
217  // not impact the MMAP functionality at all. Both POSIX and Microsoft
218  // implementations keep the reference to the file-descriptor in their internal
219  // structures so we are free to get rid of it.
220  //
221  // What does it mean to us? We don't have to keep the file-descriptor
222  // ourselves in memory and that results with much more simplified
223  // implementation of fetch/drop.
224  //
225  // References:
226  // POSIX (http://pubs.opengroup.org/onlinepubs/7908799/xsh/mmap.html)
227  // The mmap() function adds an extra reference to the file
228  // associated with the file descriptor fildes which is not removed
229  // by a subsequent close() on that file descriptor. This reference
230  // is removed when there are no more mappings to the file.
231  //
232  // From MSDN
233  // (https://docs.microsoft.com/en-us/windows/desktop/api/memoryapi/nf-memoryapi-unmapviewoffile)
234  // Although an application may close the file handle used to create
235  // a file mapping object, the system holds the corresponding file
236  // open until the last view of the file is unmapped.
237  my_close(f, MYF(MY_WME));
238 
239  return (ptr == MAP_FAILED) ? nullptr : ptr;
240 }
