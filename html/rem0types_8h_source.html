46 /* Maximum number of user defined fields/columns. The reserved columns
47 are the ones InnoDB adds internally: DB_ROW_ID, DB_TRX_ID, DB_ROLL_PTR.
48 We need "* 2" because mlog_parse_index() creates a dummy table object
49 possibly, with some of the system columns in it, and then adds the 3
50 system columns (again) using dict_table_add_system_columns(). The problem
51 is that mlog_parse_index() cannot recognize the system columns by
52 just having n_fields, n_uniq and the lengths of the columns. */
53 #define REC_MAX_N_USER_FIELDS (REC_MAX_N_FIELDS - DATA_N_SYS_COLS * 2)
54 
55 /* REC_ANTELOPE_MAX_INDEX_COL_LEN is measured in bytes and is the maximum
56 indexed field length (or indexed prefix length) for indexes on tables of
57 ROW_FORMAT=REDUNDANT and ROW_FORMAT=COMPACT format.
58 Before we support UTF-8 encodings with mbmaxlen = 4, a UTF-8 character
59 may take at most 3 bytes. So the limit was set to 3*256, so that one
60 can create a column prefix index on 256 characters of a TEXT or VARCHAR
61 column also in the UTF-8 charset.
62 This constant MUST NOT BE CHANGED, or the compatibility of InnoDB data
63 files would be at risk! */
64 #define REC_ANTELOPE_MAX_INDEX_COL_LEN 768
65 
66 /** Maximum indexed field length for tables that have atomic BLOBs.
67 This (3072) is the maximum index row length allowed, so we cannot create index
68 prefix column longer than that. */
69 #define REC_VERSION_56_MAX_INDEX_COL_LEN 3072
