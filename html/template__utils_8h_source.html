38  Clears a container, but deletes all objects that the elements point to first.
39  @tparam Container_type Container of pointers.

41 template <typename Container_type>
42 void delete_container_pointers(Container_type &container) {
43  typename Container_type::iterator it1 = container.begin();
44  typename Container_type::iterator it2 = container.end();
45  for (; it1 != it2; ++it1) {
46  delete (*it1);
47  }
48  container.clear();

82 /**
83  Casts from one pointer type to another in a type hierarchy.
84  In debug mode, we verify the cast is indeed legal.
85 
86  @tparam Target The descendent type, must be a pointer type.
87  @tparam Source The parent type.
88 
89  @param arg The pointer to be down-cast.
90 
91  @return A pointer of type Target.
92 */
93 template <typename Target, typename Source>
94 inline Target down_cast(Source *arg) {
95  assert(nullptr != dynamic_cast<Target>(arg));
96  return static_cast<Target>(arg);
97 }

99 /**
100  Casts from one reference type to another in a type hierarchy.
101  In debug mode, we verify the cast is indeed legal.
102 
103  @tparam Target The descendent type, must be a reference type.
104  @tparam Source The parent type.
105 
106  @param arg The reference to be down-cast.
107 
108  @return A reference of type Target.
109 */
110 template <typename Target, typename Source>
111 inline Target down_cast(Source &arg) {
112  // We still use the pointer version of dynamic_cast, as the
113  // reference-accepting version throws exceptions, and we don't want to deal
114  // with that.
115  assert(dynamic_cast<typename std::remove_reference<Target>::type *>(&arg) !=
116  nullptr);
117  return static_cast<Target>(arg);
118 }

120 /**
121  Sometimes the compiler insists that types be the same and does not do any
122  implicit conversion. For example:
123  Derived1 *a;
124  Derived2 *b; // Derived1 and 2 are children classes of Base
125  Base *x= cond ? a : b; // Error, need to force a cast.
126 
127  Use:
128  Base *x= cond ? implicit_cast<Base*>(a) : implicit_cast<Base*>(b);
129  static_cast would work too, but would be less safe (allows any
130  pointer-to-pointer conversion, not only up-casts).
131 */
132 template <typename To>
133 inline To implicit_cast(To x) {
134  return x;
135 }

159 /**
160  Split a range into sub ranges delimited by elements satisfying a predicate.
161  Examines the elements from first to last, exclusive. Each time an element
162  which satisfies the splitting predicate is encountered, the action argument's
163  operator() is invoked with the starting and past-the-end iterators for the
164  current sub-range, even if this is empty. When iteration is complete, action()
165  is called on the range between the start of the last subrange and last.
166 
167  It must be possible to pass a single element with type const
168  InputIt::value_type to is_split_element. It must be possible to pass two
169  InputIt arguments to action.
170 
171  @param first Beginning of the range to split.
172  @param last End of the range to split.
173  @param pred Callable which will be invoked on each element in
174  turn to determine if it is a splitting element.
175  @param action Callable which will be invoked with the beginning
176  and one-past-the-end iterators for each subrange.
177  */
178 template <class InputIt, class Pred, class Action>
179 inline void Split(InputIt first, InputIt last, Pred &&pred, Action &&action) {
180  while (first != last) {
181  InputIt split = std::find_if(first, last, std::forward<Pred>(pred));
182  action(first, split); // Called even for empty subranges, action must
183  // discard if not wanted
184  if (split == last) return;
185  first = split + 1;
186  }
187 }

189 /**
190  Search backwards for the first occurence of an element which does not satisfy
191  the trimming predicate, and return an InputIt to the element after it.
192 
193  @param first Beginning of the range to search.
194  @param last End of the range to search.
195  @param pred Callable which can be applied to a dereferenced InputIt and which
196  returns true if the element should be trimmed.
197 
198  @returns InputIt referencing the first element of sub range satisfying the
199  trimming predicate at the end of the range. last if no elements
200  satisfy the trimming predicate.
201  */
202 template <class InputIt, class Pred>
203 inline InputIt FindTrimmedEnd(InputIt first, InputIt last, Pred &&pred) {
204  return std::find_if_not(std::make_reverse_iterator(last),
205  std::make_reverse_iterator(first),
206  std::forward<Pred>(pred))
207  .base();
208 }
209 
210 /**
211  Searches for a sub range such that no elements before or after fail to
212  satisfy the trimming predicate.
213 
214  @param first Beginning of the range to search.
215  @param last End of the range to search.
216  @param pred Callable which can be applied to a dereferenced InputIt and which
217  returns true if the element should be trimmed.
218 
219  @returns Pair of iterators denoting the sub range which does not include the
220  leading and trailing sub ranges matching the trimming predicate.
221  {last, last} if all elements match the trimming predicate.
222  */
223 template <class InputIt, class Pred>
224 inline std::pair<InputIt, InputIt> FindTrimmedRange(InputIt first, InputIt last,
225  Pred &&pred) {
226  InputIt f = std::find_if_not(first, last, std::forward<Pred>(pred));
227  return {f, FindTrimmedEnd(f, last, std::forward<Pred>(pred))};
228 }
