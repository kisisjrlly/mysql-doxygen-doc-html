190 /*
191  These two definitions contain information on the valid debug options. If
192  one wants to add a new option both definitions must be changed.
193 
194  Any new option must be added before the GCS_DEBUG_ALL and gaps in the
195  numbers are not allowed. Adding a new debug option, e.g. GCS_DEBUG_THREAD,
196  would result in the following definitions:
197 
198  typedef enum
199  {
200  GCS_DEBUG_NONE = 0x00000000,
201  GCS_DEBUG_BASIC = 0x00000001,
202  GCS_DEBUG_TRACE = 0x00000002,
203  XCOM_DEBUG_BASIC = 0x00000004,
204  XCOM_DEBUG_TRACE = 0x00000008,
205  GCS_INVALID_DEBUG = ~(0x7FFFFFFF),
206  GCS_DEBUG_ALL = ~(GCS_DEBUG_NONE)
207  } gcs_xcom_debug_option_t;
208 
209 
210  static const char* const gcs_xcom_debug_strings[]=
211  {
212  "GCS_DEBUG_BASIC",
213  "GCS_DEBUG_TRACE",
214  "XCOM_DEBUG_BASIC",
215  "XCOM_DEBUG_TRACE",
216  "GCS_DEBUG_ALL",
217  "GCS_DEBUG_NONE"
218  };
219 */
220 #if !defined(GCS_XCOM_DEBUG_INFORMATION)
221 #define GCS_XCOM_DEBUG_INFORMATION
222 /*
223 This code is duplicated in the xcom_logger.h file. So if you make any change
224 here, remember to update the aforementioned file. Note that XCOM is written
225 in C which uses a 32 bit integer as the type for the enumeration and although
226 the code is prepared to handle 64 bits, the enumeration in both the GCS (i.e.
227 C++) and XCOM (i.e. C) is restricted to 32 bits.
228 
229 Assuming that we are not using all bits available to define a debug level,
230 GCS_INVALID_DEBUG will give us the last possible entry that is not used.
231 
232 The GCS_DEBUG_NONE, GCS_DEBUG_ALL and GCS_INVALID_DEBUG are options that apply
233 to both GCS and XCOM but we don't prefix it with XCOM to avoid big names.
234 */
235 typedef enum {
236  GCS_DEBUG_NONE = 0x00000000,
237  GCS_DEBUG_BASIC = 0x00000001,
238  GCS_DEBUG_TRACE = 0x00000002,
239  XCOM_DEBUG_BASIC = 0x00000004,
240  XCOM_DEBUG_TRACE = 0x00000008,
241  GCS_DEBUG_MSG_FLOW = 0x00000010,
242  XCOM_DEBUG_MSG_FLOW = 0x00000020,
243  GCS_INVALID_DEBUG = ~(0x7FFFFFFF),
244  GCS_DEBUG_ALL = ~(GCS_DEBUG_NONE)
245 } gcs_xcom_debug_option_t;
