26 /**
27  @file sql/opt_explain.h
28  EXPLAIN @<command@>.
29 
30 Single table UPDATE/DELETE commands are explained by the
31 explain_single_table_modification() function.
32 
33 A query expression (complete SELECT query possibly including
34 subqueries and unions), INSERT...SELECT and multitable UPDATE/DELETE
35 commands are explained like this:
36 
37 (1) explain_query_expression()
38 
39 Is the entry point. Forwards the job to explain_unit().
40 
41 (2) explain_unit()
42 
43 Is for a SELECT_LEX_UNIT, prepares, optimizes, explains one JOIN for
44 each "top-level" SELECT_LEXs of the unit (like: all SELECTs of a
45 UNION; but not subqueries), and one JOIN for the fake SELECT_LEX of
46 UNION); each JOIN explain (JOIN::exec()) calls explain_query_specification()
47 
48 (3) explain_query_specification()
49 
50 Is for a single SELECT_LEX (fake or not). It needs a prepared and
51 optimized JOIN, for which it builds the EXPLAIN rows. But it also
52 launches the EXPLAIN process for "inner units" (==subqueries of this
53 SELECT_LEX), by calling explain_unit() for each of them.
54 */

112 /**
113  EXPLAIN functionality for Query_result_insert, Query_result_update and
114  Query_result_delete.
115 
116  This class objects substitute Query_result_insert, Query_result_update and
117  Query_result_delete data interceptor objects to implement EXPLAIN for INSERT,
118  REPLACE and multi-table UPDATE and DELETE queries.
119  Query_result_explain class object initializes tables like Query_result_insert,
120  Query_result_update or Query_result_delete data interceptor do, but it
121  suppresses table data modification by the underlying interceptor object.
122  Thus, we can use Query_result_explain object in the context of EXPLAIN INSERT/
123  REPLACE/UPDATE/DELETE query like we use Query_result_send in the context of
124  EXPLAIN SELECT command:
125  1) in presence of lex->describe flag, pass Query_result_explain object to
126  execution function,
127  2) it calls prepare(), optimize() and start_execution() functions
128  to mark modified tables etc.
129 */
130

132  protected:
133  /**
134  Pointer to underlying Query_result_insert, Query_result_update or
135  Query_result_delete object.
136  */
137  Query_result *interceptor;

139  public:
140  Query_result_explain(SELECT_LEX_UNIT *unit_arg, Query_result *interceptor_arg)
141  : Query_result_send(), interceptor(interceptor_arg) {
142  unit = unit_arg;
143  }
144 
145  protected:
146  bool prepare(THD *thd, const mem_root_deque<Item *> &list,
147  SELECT_LEX_UNIT *u) override {
148  return Query_result_send::prepare(thd, list, u) ||
149  interceptor->prepare(thd, list, u);
150  }
151 
152  bool start_execution(THD *thd) override {
153  return Query_result_send::start_execution(thd) ||
154  interceptor->start_execution(thd);
155  }
156 
157  bool optimize() override {
158  return Query_result_send::optimize() || interceptor->optimize();
159  }
160 
161  void cleanup(THD *thd) override {
162  Query_result_send::cleanup(thd);
163  interceptor->cleanup(thd);
164  }
165 };
