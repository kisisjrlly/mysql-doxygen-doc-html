433 /** The following function returns the data size of an old-style physical
434  record, that is the sum of field lengths. SQL null fields
435  are counted as length 0 fields. The value returned by the function
436  is the distance from record origin to record end in bytes.
437  @return size */
438 UNIV_INLINE
439 ulint rec_get_data_size_old(const rec_t *rec) /*!< in: physical record */
440  MY_ATTRIBUTE((warn_unused_result));
441 #define rec_offs_init(offsets) \
442  rec_offs_set_n_alloc(offsets, (sizeof offsets) / sizeof *offsets)
443 
444 /**
445 A helper RAII wrapper for otherwise difficult to use sequence of:
446 
447  ulint offsets_[REC_OFFS_NORMAL_SIZE];
448  rec_offs_init(offsets_);
449  mem_heap_t *heap = nullptr;
450 
451  const ulint *offsets =
452  rec_get_offsets(rec, index, offsets_, ULINT_UNDEFINED, &heap);
453 
454  DO_SOMETHING(offsets);
455 
456  if (heap != nullptr) {
457  mem_heap_free(heap);
458  }
459 
460 With this helper you can simply do:
461 
462  DO_SOMETHING(Rec_offsets().compute(rec,index));
463 
464 And if you need to reuse the memory allocated offsets several times you can:
465  Rec_offsets offsets;
466  for(rec: recs) DO_SOMTHING(offsets.compute(rec,index))
467 */
468 class Rec_offsets : private ut::Non_copyable {
469  public:
470  /** Prepares offsets to initially point to the fixed-size buffer, and marks
471  the memory as allocated, but uninitialized. You first need to call compute()
472  to use it */
473  Rec_offsets() { rec_offs_init(m_preallocated_buffer); }
474 
475  /** Computes offsets for given record. Returned array is owned by this
476  instance. You can use its value as long as this object does not go out of
477  scope (which can free the buffer), and you don't call compute() again (which
478  can overwrite the offsets).
479  @param[in] rec The record for which you want to compute the offsets
480  @param[in] index The index which contains the record
481  @return A pointer to offsets array owned by this instance. Valid till next
482  call to compute() or end of this instance lifetime.
483  */
484  const ulint *compute(const rec_t *rec, const dict_index_t *index) {
485  m_offsets =
486  rec_get_offsets(rec, index, m_offsets, ULINT_UNDEFINED, &m_heap);
487  return m_offsets;
488  }
489  /** Deallocated dynamically allocated memory, if any. */
490  ~Rec_offsets() {
491  if (m_heap) {
492  mem_heap_free(m_heap);
493  m_heap = nullptr;
494  }
495  }
496 
497  private:
498  /** Pointer to heap used by rec_get_offsets(). Initially nullptr. If row is
499  really big, rec_get_offsets() may need to allocate new buffer for offsets.
500  At, first, when heap is null, rec_get_offsets() will create new heap, and pass
501  it back via reference. On subsequent calls, we will pass this heap, so it
502  is reused if needed. Therefore all allocated buffers are in this heap, if it
503  is not nullptr */
504  mem_heap_t *m_heap{nullptr};
505 
506  /** Buffer with size large enough to handle common cases without having to use
507  heap. This is the initial value of m_offsets.*/
508  ulint m_preallocated_buffer[REC_OFFS_NORMAL_SIZE];
509 
510  /* Initially points to m_preallocated_buffer (which is uninitialized memory).
511  After each call to compute() contains the pointer to the most recently
512  computed offsets.
513  We pass it back to rec_get_offsets() on subsequent calls to compute() to reuse
514  the same memory if possible. */
515  ulint *m_offsets{m_preallocated_buffer};
516 };
