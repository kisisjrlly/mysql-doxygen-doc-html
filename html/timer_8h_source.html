37 template <class Clock>
38 struct wait_traits {
39  static typename Clock::duration to_wait_duration(
40  const typename Clock::duration &d) {
41  return d;
42  }
43  static typename Clock::duration to_wait_duration(
44  const typename Clock::time_point &tp) {
45  auto diff = tp - Clock::now();
46 
47  if (diff > Clock::duration::max()) return Clock::duration::max();
48  if (diff < Clock::duration::min()) return Clock::duration::min();
49 
50  return diff;
51  }
52 };

54 template <class Clock, class WaitTraits>
55 class basic_waitable_timer {
56  public:
57  using executor_type = io_context::executor_type;
58  using clock_type = Clock;
59  using duration = typename clock_type::duration;
60  using time_point = typename clock_type::time_point;
61  using traits_type = WaitTraits;
62 
63  // 15.4.1 construct/copy/destroy
64  explicit basic_waitable_timer(io_context &io_ctx)
65  : basic_waitable_timer(io_ctx, time_point()) {}
66  basic_waitable_timer(io_context &io_ctx, const time_point &tp)
67  : executor_{io_ctx.get_executor()}, expiry_{tp} {}
68  basic_waitable_timer(io_context &io_ctx, const duration &d)
69  : basic_waitable_timer(io_ctx, Clock::now() + d) {}
70 
71  basic_waitable_timer(const basic_waitable_timer &) = delete;
72  basic_waitable_timer(basic_waitable_timer &&rhs)
73  : executor_{std::move(rhs.executor_)}, expiry_{std::move(rhs.expiry_)} {
74  id_.swap(rhs.id_);
75 
76  rhs.expiry_ = time_point();
77  }
78 
79  ~basic_waitable_timer() { cancel(); }
80 
81  basic_waitable_timer &operator=(const basic_waitable_timer &) = delete;
82  basic_waitable_timer &operator=(basic_waitable_timer &&rhs) {
83  if (this == std::addressof(rhs)) {
84  return *this;
85  }
86  cancel();
87 
88  executor_ = std::move(rhs.executor_);
89  expiry_ = std::move(rhs.expiry_);
90  id_.swap(rhs.id_);
91 
92  rhs.expiry_ = time_point();
93 
94  return *this;
95  }
96 
97  // 15.4.4 ops
98  executor_type get_executor() noexcept { return executor_; }
99  size_t cancel() { return executor_.context().cancel(*this); }
100  size_t cancel_one() { return executor_.context().cancel_one(*this); }
101 
102  time_point expiry() const { return expiry_; }
103 
104  size_t expires_at(const time_point &t) {
105  size_t cancelled = cancel();
106 
107  expiry_ = t;
108 
109  return cancelled;
110  }
111 
112  size_t expires_after(const duration &d) {
113  return expires_at(clock_type::now() + d);
114  }
115 
116  stdx::expected<void, std::error_code> wait() {
117  executor_.dispatch(
118  [this] {
119  while (clock_type::now() < expiry_) {
120  std::this_thread::sleep_for(traits_type::to_wait_duration(expiry_));
121  }
122  },
123  std::allocator<void>{});
124 
125  return {};
126  }

128  template <class CompletionToken>
129  auto async_wait(CompletionToken &&token) {
130  async_completion<CompletionToken, void(std::error_code)> init{token};
131 
132  get_executor().context().async_wait(*this,
133  std::move(init.completion_handler));
134 
135  return init.result.get();
136  }
137 
138  private:
139  executor_type executor_;
140  time_point expiry_;
141 
142  // every timer needs a unique-id to be able to identify it (e.g. to cancel it)
143  //
144  // Note: empty classes like "Id" have a sizeof() > 0. It would perhaps make
145  // sense to use it for something useful.

147

152  friend class io_context;
153 };
