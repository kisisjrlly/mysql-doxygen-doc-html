149 /**
150  Sql_cmd_show represents the SHOW statements that are implemented
151  as SELECT statements internally.
152  Normally, preparation and execution is the same as for regular SELECT
153  statements.
154 */
155 class Sql_cmd_show : public Sql_cmd_select {
156  public:
157  Sql_cmd_show(enum_sql_command sql_command)
158  : Sql_cmd_select(nullptr), m_sql_command(sql_command) {}
159  enum_sql_command sql_command_code() const override { return m_sql_command; }
160  virtual bool check_parameters(THD *) { return false; }
161  /// Generally, the SHOW commands do not distinguish precheck and regular check
162  bool precheck(THD *thd) override { return check_privileges(thd); }
163  bool check_privileges(THD *) override;
164  bool execute(THD *thd) override;
165 
166  protected:
167  // Use for SHOW commands that operate on a single table.
168  bool check_privileges_for_table(THD *thd, bool is_temporary);
169  enum_sql_command m_sql_command;
170 };
171 
172 /**
173  Common base class: Represents commands that are not represented by
174  a plan that iss equivalent to a SELECT statement.
175 
176  This class has a common execution framework with an execute() function
177  that calls check_privileges() and execute_inner().
178 */
179 class Sql_cmd_show_noplan : public Sql_cmd_show {
180  protected:
181  Sql_cmd_show_noplan(enum_sql_command sql_command)
182  : Sql_cmd_show(sql_command) {}
183  bool execute(THD *thd) override {
184  lex = thd->lex;
185  if (check_privileges(thd)) return true;
186  if (execute_inner(thd)) return true;
187  return false;
188  }
189 };
