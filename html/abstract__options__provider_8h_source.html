156  template <typename T_type, typename T_typelib>
157  Enum_option<T_type, T_typelib> *create_new_enum_option(

186  Sets optional option changes listener to which all changes in all options
187  contained in this provider should be reported. This is used when this

189  Part of I_options_provider interface implementation.
190  */
191  void set_option_changed_listener(
192  I_option_changed_listener *listener) override;
193 
194  private:
195  /**
196  Makes sure this provider will be able to watch name and optid usage.
197  */
198  template <typename T_type>
199  T_type *attach_new_option(T_type *option) {
200  // Make this option reporting all name and optid changes to us.
201  option->set_option_changed_listener(this);
202 
203  // Add to list of our own options.
204  this->m_options_created.push_back(option);
205 
206  // Check for name and optid collision.
207  this->notify_option_name_changed(option, "");
208  this->notify_option_optid_changed(option, 0);
209 
210  return option;
211  }
212 
213  /**
214  Called after specified option has name changed.
215  It is also called when new option is added, old_name is empty string in
216  that case.
217  Part of I_option_changed_listener interface implementation.
218  */
219  void notify_option_name_changed(I_option *source,
220  std::string old_name) override;
221  /**
222  Called after specified option has option ID changed.
223  It is also called when new option is added, old_optid is 0 in that case.
224  Part of I_option_changed_listener interface implementation.
225  */
226  void notify_option_optid_changed(I_option *source, uint32 old_optid) override;
227 
228  bool m_are_options_created;
229  std::map<std::string, I_option *> m_name_usage;
230  std::map<uint32, I_option *> m_optid_usage;
231  I_option_changed_listener *m_option_changed_listener;
232  std::vector<I_option *> m_options_created;
233 };
