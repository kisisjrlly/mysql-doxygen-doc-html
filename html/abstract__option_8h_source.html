123 template <typename T_type>
124 T_type *Abstract_option<T_type>::set_short_character(char code) {
125  // Change optid to new one
126  uint32 old_optid = this->m_option_structure.id;
127  this->m_option_structure.id = (int)code;
128 
129  // Inform that it has changed
130  if (this->m_option_changed_listener != nullptr) {
131  this->m_option_changed_listener->notify_option_optid_changed(this,
132  old_optid);
133  }
134 
135  return (T_type *)this;
136 }
