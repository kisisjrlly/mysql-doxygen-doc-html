58  /** @brief Checks if Router with given id is already registered in metadata
59  * database, and belongs to our machine
60  *
61  * @param router_id Router id
62  * @param hostname_override If non-empty, this hostname will be used instead
63  * of getting queried from OS
64  *
65  * @throws LocalHostnameResolutionError(std::runtime_error) on hostname query
66  * failure
67  * @throws std::runtime_error if router_id doesn't exist, or is associated
68  * with a different host
69  * @throws MySQLSession::Error(std::runtime_error) on database error
70  */
71  virtual void verify_router_id_is_ours(
72  const uint32_t router_id, const std::string &hostname_override = "") = 0;
73 
74  /** @brief Registers Router in metadata database
75  *
76  * @param router_name Router name
77  * @param overwrite if Router name is already registered, allow this
78  * registration to be "hijacked" instead of throwing
79  * @param hostname_override If non-empty, this hostname will be used instead
80  * of getting queried from OS
81  *
82  * @returns newly-assigned router_id
83  *
84  * @throws LocalHostnameResolutionError(std::runtime_error) on hostname query
85  * failure, std::runtime_error on other failure
86  */
87  virtual uint32_t register_router(
88  const std::string &router_name, const bool overwrite,
89  const std::string &hostname_override = "") = 0;
90 
91  virtual void update_router_info(const uint32_t router_id,
92  const std::string &cluster_id,
93  const std::string &rw_endpoint,
94  const std::string &ro_endpoint,
95  const std::string &rw_x_endpoint,
96  const std::string &ro_x_endpoint,
97  const std::string &username) = 0;
98 
99  virtual std::vector<std::string> get_routing_mode_queries(
100  const std::string &cluster_name) = 0;
101 
102  /** @brief Verify that host is a valid metadata server
103  *
104  *
105  * @throws MySQLSession::Error
106  * @throws std::runtime_error
107  * @throws std::out_of_range
108  * @throws std::logic_error
109  *
110  * * checks that the server
111  *
112  * - has the metadata in the correct version
113  * - contains metadata for the group it's in (in case of GR cluster)
114  * (metadata server group must be same as managed group currently)
115  */
116  virtual void require_metadata_is_ok();
117 
118  /** @brief Verify that host is a valid cluster member (either Group
119  * Replication or ReplicaSet cluster)
120  *
121  * @throws MySQLSession::Error
122  * @throws std::runtime_error
123  * @throws std::out_of_range
124  * @throws std::logic_error
125  */
126  virtual void require_cluster_is_ok() = 0;
127 
128  virtual std::string get_cluster_type_specific_id() = 0;
129 
130  virtual ClusterInfo fetch_metadata_servers() = 0;
131 
132  virtual std::vector<std::string> get_grant_statements(
133  const std::string &new_accounts) const = 0;
134 
135  virtual std::vector<std::tuple<std::string, unsigned long>>
136  fetch_cluster_hosts() = 0;
137 
138  MySQLSession &get_session() { return *mysql_; }
139 
140  protected:
141  // throws MySQLSession::Error, std::out_of_range, std::logic_error
142  virtual uint64_t query_cluster_count() = 0;
143 
144  MySQLSession *mysql_;
145  mysql_harness::SocketOperationsBase *socket_operations_;
146  mysqlrouter::MetadataSchemaVersion schema_version_;
147 };
