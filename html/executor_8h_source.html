160  // 13.7.3 [async.exec.ctx.ops]
161  void notify_fork(fork_event e) {
162  // prepare is in reverse
163  if (e == fork_event::prepare) {
164  std::for_each(services_.rbegin(), services_.rend(),
165  [e](auto &svc) { svc.ptr_->notify_fork(e); });
166  } else {
167  std::for_each(services_.begin(), services_.end(),
168  [e](auto &svc) { svc.ptr_->notify_fork(e); });
169  }
170  }
171 
172  protected:
173  // 13.7.4 [async.exec.ctx.protected]
174  void shutdown() noexcept {
175  // shutdown in reverse insert-order
176  std::for_each(services_.rbegin(), services_.rend(), [](auto &svc) {
177  if (svc.active_) {
178  svc.ptr_->shutdown();
179  svc.active_ = false;
180  }
181  });
182  }
183 
184  void destroy() noexcept {
185  // destroy in reverse insert-order
186  while (!services_.empty()) services_.pop_back();
187 
188  keys_.clear();
189  }
190 
191  // as service has a protected destructor unique_ptr can't call it itself and
192  // we much provide our own deleter: service_deleter
193  template <class Service>
194  static void service_deleter(service *svc) {
195  delete static_cast<Service *>(svc);
196  }

198  struct ServicePtr {
199  template <class Service>
200  ServicePtr(Service *svc) : ptr_(svc, &service_deleter<Service>) {}
201 
202  // each service is only shutdown once.
203  bool active_{true};
204 
205  std::unique_ptr<service, void (*)(service *)> ptr_;
206  };

247 typename Service::key_type &use_service(execution_context &ctx) {
248  using Key = typename Service::key_type;

252  static_assert(std::is_base_of<Key, Service>::value,

324 // checker for the requirements of a executor
325 //
326 // see 13.2.2 [async.reqmts.executor]
327 //
328 // - copy-constructible
329 // - destructible
330 // - ... no exceptions
331 // - operator==
332 // - operator!=
333 // - .context()
334 // - .on_work_started()
335 // - .on_work_finished()
336 // - .dispatch(void (*)(), allocator)
337 // - .post(void (*)(), allocator)
338 // - .defer(void (*)(), allocator)
339 template <class T, typename U = std::remove_const_t<T>>
340 auto executor_requirements(U *__x = nullptr, const U *__const_x = nullptr,
341  void (*f)() = nullptr,
342  const std::allocator<int> &a = {})
343  -> std::enable_if_t<
344  stdx::conjunction<
345  std::is_copy_constructible<T>,
346  // methods/operators must exist
347  std::is_same<decltype(*__const_x == *__const_x), bool>,
348  std::is_same<decltype(*__const_x != *__const_x), bool>,
349  std::is_void<decltype(__x->on_work_started())>,
350  std::is_void<decltype(__x->on_work_finished())>,
351  std::is_void<decltype(__x->dispatch(std::move(f), a))>,
352  std::is_void<decltype(__x->post(std::move(f), a))>,
353  std::is_void<decltype(__x->defer(std::move(f), a))>>::value,
354 
355  // context() may either return execution_context & or E&
356  stdx::void_t<decltype(__x->context()), void()>>;
357 
358 template <class T>
359 struct is_executor<T, decltype(executor_requirements<T>())> : std::true_type {};
360 } // namespace impl

461 template <class Executor>
462 class executor_work_guard {
463  public:
464  using executor_type = Executor;
465 
466  explicit executor_work_guard(const executor_type &ex) noexcept
467  : ex_{ex}, owns_{true} {
468  ex_.on_work_started();
469  }
470  executor_work_guard(const executor_work_guard &other) noexcept
471  : ex_{other.ex}, owns_{other.owns_} {
472  if (owns_) {
473  ex_.on_work_started();
474  }
475  }
476  executor_work_guard(executor_work_guard &&other) noexcept
477  : ex_{std::move(other.ex_)}, owns_{std::exchange(other.owns_, false)} {}
478 
479  executor_work_guard &operator=(const executor_work_guard &other) = delete;
480 
481  ~executor_work_guard() {
482  if (owns_) {
483  ex_.on_work_finished();
484  }
485  }
486 
487  executor_type get_executor() const noexcept { return ex_; }
488 
489  bool owns_work() const noexcept { return owns_; }
490 
491  void reset() noexcept {
492  if (owns_) {
493  ex_.on_work_finished();
494  }
495  owns_ = false;
496  }
497 
498  private:
499  Executor ex_;
500  bool owns_;
501 };

573 // 13.19 [async.system.context]
574 //
575 // just barely enough of a system_context to run everything in the
576 // main-thread
577 class system_context : public execution_context {
578  public:
579  using executor_type = system_executor;
580 
581  system_context() = delete;
582  system_context(const system_context &) = delete;
583  system_context &operator=(const system_context &) = delete;
584  ~system_context() override {
585  stop();
586  join();
587  }
588 
589  executor_type get_executor() noexcept { return {}; }
590  void stop() {
591  std::lock_guard<std::mutex> lk(mtx_);
592  stopped_ = true;
593 
594  cv_.notify_all();
595  }
596  bool stopped() const noexcept {
597  std::lock_guard<std::mutex> lk(mtx_);
598  return stopped_;
599  }
600  void join() { thread_.join(); }
601 
602  private:
603  struct __tag {};
604  system_context(__tag) {}
605 
606  friend class system_executor;
607 
608  void run_() {
609  for (;;) {
610  std::function<void()> f;
611  {
612  std::unique_lock<std::mutex> lk(mtx_);
613 
614  cv_.wait(lk, [this] { return !stopped_ && !tasks_.empty(); });
615 
616  if (stopped_) return;
617 
618  f = std::move(tasks_.front());
619  tasks_.pop();
620  }
621  f();
622  }
623  }
624 
625  void post_(std::function<void()> f) {
626  std::lock_guard<std::mutex> lk(mtx_);
627  if (stopped_) return;
628 
629  if (!thread_.joinable()) {
630  thread_ = std::thread(&system_context::run_, this);
631  }
632 
633  tasks_.push(std::move(f));
634 
635  cv_.notify_one();
636  }
637 
638  static system_context &get_() noexcept {
639  static system_context sc(__tag{});
640  return sc;
641  }
642 
643  std::thread thread_;
644  mutable std::mutex mtx_;
645  std::condition_variable cv_;
646  std::queue<std::function<void()>> tasks_;
647  bool stopped_{false};
648 };
