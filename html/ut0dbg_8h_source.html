112 /** A "chronometer" used to clock snippets of code.
113 Example usage:
114  ut_chrono_t ch("this loop");
115  for (;;) { ... }
116  ch.show();
117 would print the timings of the for() loop, prefixed with "this loop:" */
118 class ut_chrono_t {
119  public:
120  /** Constructor.
121  @param[in] name chrono's name, used when showing the values */
122  ut_chrono_t(const char *name) : m_name(name), m_show_from_destructor(true) {
123  reset();
124  }
125 
126  /** Resets the chrono (records the current time in it). */
127  void reset() {
128  gettimeofday(&m_tv, nullptr);
129 
130  getrusage(RUSAGE_SELF, &m_ru);
131  }
132 
133  /** Shows the time elapsed and usage statistics since the last reset. */
134  void show() {
135  struct rusage ru_now;
136  struct timeval tv_now;
137  struct timeval tv_diff;
138 
139  getrusage(RUSAGE_SELF, &ru_now);
140 
141  gettimeofday(&tv_now, nullptr);
142 
143 #ifndef timersub
144 #define timersub(a, b, r) \
145  do { \
146  (r)->tv_sec = (a)->tv_sec - (b)->tv_sec; \
147  (r)->tv_usec = (a)->tv_usec - (b)->tv_usec; \
148  if ((r)->tv_usec < 0) { \
149  (r)->tv_sec--; \
150  (r)->tv_usec += 1000000; \
151  } \
152  } while (0)
153 #endif /* timersub */
154 
155 #define CHRONO_PRINT(type, tvp) \
156  fprintf(stderr, "%s: %s% 5ld.%06ld sec\n", m_name, type, \
157  static_cast<long>((tvp)->tv_sec), static_cast<long>((tvp)->tv_usec))
158 
159  timersub(&tv_now, &m_tv, &tv_diff);
160  CHRONO_PRINT("real", &tv_diff);
161 
162  timersub(&ru_now.ru_utime, &m_ru.ru_utime, &tv_diff);
163  CHRONO_PRINT("user", &tv_diff);
164 
165  timersub(&ru_now.ru_stime, &m_ru.ru_stime, &tv_diff);
166  CHRONO_PRINT("sys ", &tv_diff);
167  }
168 
169  /** Cause the timings not to be printed from the destructor. */
170  void end() { m_show_from_destructor = false; }
171 
172  /** Destructor. */
173  ~ut_chrono_t() {
174  if (m_show_from_destructor) {
175  show();
176  }
177  }
178 
179  private:
180  /** Name of this chronometer. */
181  const char *m_name;
182 
183  /** True if the current timings should be printed by the destructor. */
184  bool m_show_from_destructor;
185 
186  /** getrusage() result as of the last reset(). */
187  struct rusage m_ru;
188 
189  /** gettimeofday() result as of the last reset(). */
190  struct timeval m_tv;
191 };
