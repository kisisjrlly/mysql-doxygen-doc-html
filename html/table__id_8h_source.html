33 /**

35 
36  @brief Each table share has a table id, it is mainly used for row based
37  replication. Meanwhile it is used as table's version too.
38 */
39 class Table_id {
40  private:
41  /* In table map event and rows events, table id is 6 bytes.*/
42  static const unsigned long long TABLE_ID_MAX = (~0ULL >> 16);
43  uint64_t m_id;
44 
45  public:
46  Table_id() : m_id(0) {}
47  explicit Table_id(unsigned long long id) : m_id(id) {}
48 
49  unsigned long long id() const { return m_id; }
50  bool is_valid() const { return m_id <= TABLE_ID_MAX; }
51 
52  Table_id &operator=(unsigned long long id) {
53  m_id = id;
54  return *this;
55  }
56 
57  bool operator==(const Table_id &tid) const { return m_id == tid.m_id; }
58  bool operator!=(const Table_id &tid) const { return m_id != tid.m_id; }
59 
60  /* Support implicit type converting from Table_id to unsigned long long */
61  operator unsigned long long() const { return m_id; }
62 
63  Table_id operator++(int) {
64  Table_id id(m_id);
65 
66  /* m_id is reset to 0, when it exceeds the max value. */
67  m_id = (m_id == TABLE_ID_MAX ? 0 : m_id + 1);
68 
69  BAPI_ASSERT(m_id <= TABLE_ID_MAX);
70  return id;
71  }
72 };
