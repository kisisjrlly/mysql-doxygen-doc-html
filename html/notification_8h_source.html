28 /**
29  A convenience context class used to share information between
30  the event handlers and the notifier. It contains flags stating
31  which notifications should be triggered and also the view
32  identifier.
33  */
34 class Notification_context {
35  private:
36  bool m_member_role_changed;
37  bool m_member_state_changed;
38  bool m_quorum_lost;
39  bool m_view_changed;
40  std::string m_view_id;
41 
42  public:
43  Notification_context()
44  : m_member_role_changed(false),
45  m_member_state_changed(false),
46  m_quorum_lost(false),
47  m_view_changed(false),
48  m_view_id("") {}
49 
50  void reset() {
51  m_member_role_changed = false;
52  m_member_state_changed = false;
53  m_view_changed = false;
54  m_quorum_lost = false;
55  }
56  void set_member_role_changed() { m_member_role_changed = true; }
57  void set_member_state_changed() { m_member_state_changed = true; }
58  void set_quorum_lost() { m_quorum_lost = true; }
59  void set_view_changed() { m_view_changed = true; }
60  void set_view_id(const std::string &v) { m_view_id.assign(v); }
61 
62  bool get_member_role_changed() const { return m_member_role_changed; }
63  bool get_member_state_changed() const { return m_member_state_changed; }

66  const std::string &get_view_id() const { return m_view_id; }
67 };

69 /**
70  This function SHALL trigger the notifications based on the
71  notification context provided as an argument. Different
72  notifications SHALL be emitted depending on what is flagged.
73 
74  It calls out into those services that have registered
75  themselves in the server service registry as listeners for
76  the notifications such as view changes, or member state
77  changes.
78 
79  If an error is returned it can be that part of the notifications
80  have succeeded and part have not.
81 
82  @note The assumption is that this function is fast, i.e.,
83  consumers SHALL implement a notification handling system
84  that will release the caller thread immediately after they
85  receive the notification.
86 
87  @param ctx The notification context.
88  */
89 bool notify_and_reset_ctx(Notification_context &ctx);
