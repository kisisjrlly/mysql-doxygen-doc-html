37  Class for storing value or NULL value.
38 */
39 template <typename T_value>
40 class Nullable {
41  public:
42  Nullable() : m_has_value(false) {}
43 
44  Nullable(T_value value) : m_has_value(true), m_value(value) {}
45 
46  Nullable(const Nullable<T_value> &other)
47  : m_has_value(other.m_has_value), m_value(other.m_value) {}
48 
49  /**
50  Assigns value from another Nullable<> object.
51  */
52  Nullable<T_value> &operator=(const Nullable<T_value> &other) {
53  this->m_has_value = other.m_has_value;
54  if (this->m_has_value) {
55  this->m_value = other.m_value;
56  }
57  return *this;
58  }
59 
60  /**
61  Compares two Nullable<> objects for equality.
62  */
63  bool operator==(const Nullable<T_value> &other) const {
64  if (this->has_value() != other.has_value())
65  return false;
66  else if (this->has_value() && this->value() != other.value())
67  return false;
68  return true;
69  }
70 
71  /**
72  Compares two Nullable<> objects for in-equality.
73  */
74  bool operator!=(const Nullable<T_value> &other) const {
75  return !(*this == other);
76  }
77 
78  /**
79  Returns true if object has not-NULL value assigned. If this is false, one
80  should not try to get value by value().
81  */
82  bool has_value() const { return this->m_has_value; }
83 
84  /**

87  */
88  const T_value &value() const {

92

98  bool m_has_value;
99  /**
100  Actual value, if m_has_value is true. Undefined if m_has_value is false.
101  */
102  T_value m_value{};
