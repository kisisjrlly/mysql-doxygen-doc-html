31 /**
32  RAII class for immunizing the THD from kill operations.
33 
34  Interruptions to operations on new Data Dictionary tables due to KILL QUERY,
35  KILL CONNECTION or statement execution timeout would leave DD in inconsistent
36  state. So the operations on the New Data Dictionary tables are made immune to
37  these operations using DD_kill_immunizer.
38 
39  Note:
40  DD operations are made immune to KILL operations till WL#7743 and WL#7016
41  are implemented. So as part of these WL's DD_kill_immunizer should be
42  removed.
43 */
44 
45 class DD_kill_immunizer {
46  public:
47  DD_kill_immunizer(THD *thd)
48  : m_thd(thd), m_killed_state(THD::NOT_KILLED), m_is_active(true) {
49  MUTEX_LOCK(thd_data_lock, &thd->LOCK_thd_data);
50 
51  // If DD_kill_immunizer is initialized as part of nested Transaction_ro's
52  // then store reference to parent kill_immunizer else NULL value is saved in
53  // m_saved_kill_immunizer.
54  m_saved_kill_immunizer = thd->kill_immunizer;
55 
56  // Save either thd->killed value or parent kill_immunizer's m_killed_state
57  // value.
58  m_saved_killed_state = thd->kill_immunizer
59  ? thd->kill_immunizer->m_killed_state
60  : thd->killed.load();
61 
62  // Set current DD_kill_immunizer object to THD.
63  thd->kill_immunizer = this;
64 
65  // Set killed state of THD as NOT_KILLED.
66  thd->killed = THD::NOT_KILLED;
67  }
68 
69  ~DD_kill_immunizer() {
70  MUTEX_LOCK(thd_data_lock, &m_thd->LOCK_thd_data);
71 
72  /*
73  Current instance is of top level kill immunizer, set kill immune mode to
74  inactive(or exiting).
75  */
76  if (m_saved_kill_immunizer == nullptr)
77  m_is_active = false;
78  else
79  // Set kill_immunizer of THD to parent. We must do it before calling awake
80  // to transfer the kill state to parent kill_immunizer.
81  m_thd->kill_immunizer = m_saved_kill_immunizer;
82 
83  // If there were any concurrent kill operations in kill immune mode, call
84  // THD::awake() with the m_killed_state. This will either propagate the
85  // m_killed_state to the parent kill_immunizer and return or assign state to
86  // the THD::killed and complete the THD::awake(). Otherwise, if it is a top
87  // level kill immunizer just reset THD::killed state else we need not have
88  // to do anything.
89  if (m_killed_state)
90  m_thd->awake(m_killed_state);
91  else if (m_saved_kill_immunizer == nullptr)
92  m_thd->killed = m_saved_killed_state;
93 
94  // Reset kill_immunizer of THD.
95  m_thd->kill_immunizer = m_saved_kill_immunizer;
96  }
97 
98  /*
99  Check if thread is in the kill immune mode.
100  */
101  bool is_active() { return m_is_active; }
102 
103  // Save kill state set while kill immune mode is active.
104  void save_killed_state(THD::killed_state state) {
105  mysql_mutex_assert_owner(&m_thd->LOCK_thd_data);
106 
107  if (m_killed_state == THD::NOT_KILLED) m_killed_state = state;
108  }
109 
110  private:
111  THD *m_thd;
112 
113  // When kill_immunizer is set(i.e. operation on DD tables is in progress) to
114  // THD then there might be some concurrent KILL operations. The KILL state
115  // from those operations is stored in the m_killed_state. While exiting from
116  // the kill immune mode THD::awake() is called with this value.
117  THD::killed_state m_killed_state;
118 
119  // In case of nested Transaction_ro, m_saved_kill_immunizer is used to refer
120  // the parent Transaction_ro's kill_immunizer. This is used to propogate the
121  // m_killed_state to the parent kill_immunizer.
122  DD_kill_immunizer *m_saved_kill_immunizer;
123 
124  // THD::killed value is saved before entering the kill immune mode.
125  // If kill_immunizer is of some Transaction_ro in the nested Transaction_ro
126  // then parent kill_immunizer's m_killed_state is saved in the
127  // m_saved_killed_state for reference.
128  THD::killed_state m_saved_killed_state;
129 
130  /*
131  kill immunizer state (active or exiting). The state is set to active when
132  kill immunizer is invoked. Kill_immunizer state is set to inactive(or
133  exiting) while exiting from the kill immune mode (current instance is of top
134  level kill immunizer at this point).
135  */
136  bool m_is_active;
137 };
