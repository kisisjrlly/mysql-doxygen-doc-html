103 inline Cell_calculator::Cell_calculator(const KEY_PART_INFO &mysql_key_part)
104  : m_mysql_field(mysql_key_part.field),
105  m_cs(field_charset(*m_mysql_field)),
106  m_is_space_padded(m_mysql_field->key_type() == HA_KEYTYPE_TEXT),
107  m_char_length(0) {
108  /* Mimic hp_hashnr() from storage/heap/hp_hash.c. */
109 
110  if (m_cs != nullptr) {
111  /* Decide if we should use my_charpos. */
112  bool use_char_length = (m_cs->mbmaxlen > 1) &&
113  (mysql_key_part.key_part_flag & HA_PART_KEY_SEG);
114 
115  DBUG_EXECUTE_IF("temptable_use_char_length", use_char_length = true;);
116 
117  if (use_char_length) {
118  m_char_length = mysql_key_part.length / m_cs->mbmaxlen;
119  m_mode = Mode::CHARSET_AND_CHAR_LENGTH;
120  } else {
121  m_mode = Mode::CHARSET;
122  }
123  } else {
124  m_mode = Mode::BINARY;
125  }
126 }

144 inline const CHARSET_INFO *Cell_calculator::field_charset(const Field &field) {
145  /* Decide if we should use charset+collation for comparisons, or rely on pure
146  * binary data. */
147  switch (field.key_type()) {
148  case HA_KEYTYPE_TEXT:
149  case HA_KEYTYPE_VARTEXT1:
150  case HA_KEYTYPE_VARTEXT2:
151  case HA_KEYTYPE_VARBINARY1:
152  case HA_KEYTYPE_VARBINARY2:
153  if (field.is_flag_set(ENUM_FLAG) || field.is_flag_set(SET_FLAG)) {
154  return &my_charset_bin;
155  } else {
156  return field.charset_for_protocol();
157  }
158  default:
159  return nullptr;
160  }
161 }

163 inline size_t Cell_calculator::hash(const Cell &cell) const {
164  if (cell.is_null()) {
165  return 1;
166  }
167 
168  auto data_length = cell.data_length();
169  if (data_length == 0) {
170  return 0;
171  }
172 
173  auto data = cell.data();
174 
175  size_t length = 0;
176 
177  /*
178  switch (m_mode) {
179  case Mode::CHARSET:
180  length = ...
181  break;
182  case Mode::CHARSET_AND_CHAR_LENGTH:
183  length = ...
184  break;
185  case Mode::BINARY:
186  return ...
187  }
188  code <-- this is executed when
189  indexed_column.cell_hash_function() == Mode::BINARY
190  and compiled with "Studio 12.5 Sun C++ 5.14 SunOS_sparc 2016/05/31" !!!
191  So we use if-else instead of switch below. */
192 
193  if (m_mode == Mode::BINARY) {
194  return murmur3_32(data, data_length, 0);
195  } else if (m_mode == Mode::CHARSET) {
196  length = data_length;
197  } else if (m_mode == Mode::CHARSET_AND_CHAR_LENGTH) {
198  length =
199  std::min(static_cast<size_t>(data_length),
200  my_charpos(m_cs, data, data + data_length, m_char_length));
201  } else {
202  abort();
203  }
204 
205  /* If the field is space padded but collation do not want to use
206  * the padding it is required to strip the spaces from the end. */
207  if (m_is_space_padded && (m_cs->pad_attribute == NO_PAD)) {
208  length = m_cs->cset->lengthsp(m_cs, reinterpret_cast<const char *>(data),
209  length);
210  }
211 
212  uint64 h1 = 1;
213  uint64 h2 = 4;
214  m_cs->coll->hash_sort(m_cs, data, length, &h1, &h2);
215  return h1;
216 }

218 inline int Cell_calculator::compare(const Cell &lhs, const Cell &rhs) const {
219  if (lhs.is_null()) {
220  if (rhs.is_null()) {
221  /* Both are NULL. */
222  return 0;
223  } else {
224  /* NULL < whatever (not NULL). */
225  return -1;
226  }
227  } else {
228  if (rhs.is_null()) {
229  /* whatever (not NULL) > NULL. */
230  return 1;
231  }
232  }
233 
234  /* Both cells are not NULL. */
235  auto lhs_data_length = lhs.data_length();
236  auto rhs_data_length = rhs.data_length();
237 
238  /* If both cells' data is identical, then no need to use the expensive
239  * comparisons below because we know that they will report equality. */
240  if ((lhs_data_length == rhs_data_length) &&
241  ((lhs_data_length == 0) ||
242  (memcmp(lhs.data(), rhs.data(), lhs_data_length) == 0))) {
243  return 0;
244  }
245 
246  auto lhs_data = lhs.data();
247  auto rhs_data = rhs.data();
248 
249  size_t lhs_length = 0;
250  size_t rhs_length = 0;
251 
252  /* Note: Using if-s instead of switch due to bug mentioned in hash(). */
253 
254  if (m_mode == Mode::BINARY) {
255  return const_cast<Field *>(m_mysql_field)->key_cmp(lhs_data, rhs_data);
256  } else if (m_mode == Mode::CHARSET) {
257  lhs_length = lhs_data_length;
258  rhs_length = rhs_data_length;
259  } else if (m_mode == Mode::CHARSET_AND_CHAR_LENGTH) {
260  lhs_length = std::min(
261  static_cast<size_t>(lhs_data_length),
262  my_charpos(m_cs, lhs_data, lhs_data + lhs_data_length, m_char_length));
263  rhs_length = std::min(
264  static_cast<size_t>(rhs_data_length),
265  my_charpos(m_cs, rhs_data, rhs_data + rhs_data_length, m_char_length));
266  } else {
267  abort();
268  }
269 
270  /* If the field is space padded but collation do not want to use
271  * the padding it is required to strip the spaces from the end. */
272  if (m_is_space_padded && (m_cs->pad_attribute == NO_PAD)) {
273  /* Strip trailing spaces. */
274  lhs_length = m_cs->cset->lengthsp(
275  m_cs, reinterpret_cast<const char *>(lhs_data), lhs_length);
276  rhs_length = m_cs->cset->lengthsp(
277  m_cs, reinterpret_cast<const char *>(rhs_data), rhs_length);
278  }
279 
280  return m_cs->coll->strnncollsp(m_cs, lhs_data, lhs_length, rhs_data,
281  rhs_length);
282 }
