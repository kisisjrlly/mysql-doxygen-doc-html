44 /**
45  * Convenience class for handling iterator range.
46  *
47  * This is a template class that accept a pair of iterators and
48  * implement the necessary methods to be able to be used as a sequence
49  * container.
50  */
51 template <class Iter>
52 class Range {
53  public:
54  using value_type = typename Iter::value_type;
55  using reference = typename Iter::reference;
56  using iterator = Iter;
57  using const_iterator = iterator;
58  using difference_type = typename std::iterator_traits<Iter>::difference_type;
59  using size_type = difference_type;
60 
61  Range(Iter start, Iter finish) : begin_(start), end_(finish) {}
62 
63  /** Get iterator to beginning of range. */
64  Iter begin() const { return begin_; }
65 
66  /** Get iterator to end of range. */
67  Iter end() const { return end_; }
68 
69  /** Check if range is empty. */
70  bool empty() const { return begin_ == end_; }
71 
72  /**
73  * Get size of the range, that is, number of elements in the range.
74  *
75  * @note This call `std::distance` internally.
76  *
77  * @returns The number of elements in the range.
78  */
79  size_type size() const { return std::distance(begin_, end_); }
80 
81  private:
82  Iter begin_;
83  Iter end_;
84 };

149  /**

151  *
152  * The configuration section will be updated with the contents of
153  * the other section. For any options that exist in the section, the

294  /**
295  * Read configuration file from file, directory, or input stream.
296  *
297  * If there are conflicting sections (several instance sections with
298  * identical name and key) or conflicting options (several instances
299  * of an option in the same section) in the input, an exception will
300  * be thrown.
301  *
302  * If the input is a stream, the contents of the stream will be
303  * added to the configuration, but if the input is either a file or
304  * a directory, the exisisting contents of the configuration will be
305  * removed.
306  *
307  * If a `pattern` is given, the path is assumed to be a directory
308  * and all files matching the pattern will be read into the the
309  * configuration object. The files together will be treated as if it
310  * is a single file, that is, no conflicting option values or
311  * sections are allowed and will raise an exception.
312  *
313  * @param input Input stream to read from.
314  *
315  * @exception syntax_error Raised if there is a syntax error in the
316  * configuration file and the configuration file have to be corrected.
317  *
318  * @exception bad_section Raised if there is a duplicate section
319  * (section with the same name and key) in the input stream.
320  *
321  * @exception bad_option Raised if there is a duplicate definition
322  * of an option (same option is given twice in a section).
323  */
324  void read(std::istream &input);
