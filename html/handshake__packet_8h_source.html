70 /** @class HandshakeResponsePacket
71  * @brief Creates a MySQL handshake response packet
72  *
73  * This class creates a MySQL handshake response packet which is send by
74  * the MySQL client after receiving the server's handshake packet.
75  *
76  */
77 class MYSQL_PROTOCOL_EXPORT HandshakeResponsePacket final : public Packet {
78  public:
79  /** @brief Constructor
80  *
81  * This version of constructor just creates an uninitialized packet
82  */
83  HandshakeResponsePacket()
84  : Packet(0),
85  username_(""),
86  password_(""),
87  char_set_(8),
88  auth_plugin_("mysql_native_password"),
89  auth_response_({}) {
90  prepare_packet();
91  }
92 
93  /** @overload
94  *
95  * This version of constructor takes in packet bytes, parses it and writes
96  * results in object's fields
97  *
98  * @param buffer Packet payload (including packet header)
99  * @param server_capabilities Capabilities sent by the server in Handshake
100  * Packet; see note in parse_payload()
101  * @param auto_parse_payload Disables automatic parsing of payload if set to
102  * false Note that header is still parsed (sequence_id_ and payload_size_ are
103  * set)
104  *
105  * @throws std::runtime_error on unrecognised or invalid packet, when parsing
106  */
107  HandshakeResponsePacket(
108  std::vector<uint8_t> buffer, bool auto_parse_payload = false,
109  Capabilities::Flags server_capabilities = Capabilities::ALL_ZEROS)
110  : Packet(buffer) {
111  if (auto_parse_payload) parse_payload(server_capabilities);
112  }
113 
114  /** @overload
115  *
116  * This version of constructor takes in fields, and generates packet bytes
117  *
118  * @param sequence_id MySQL Packet number
119  * @param auth_response Authentication data from the MySQL server handshake
120  * @param username MySQL username to use
121  * @param password MySQL password to use
122  * @param database MySQL database to use when connecting (default is empty)
123  * @param char_set MySQL character set code (default 8, latin1)
124  * @param auth_plugin MySQL authentication plugin name (default
125  * 'mysql_native_password')
126  */
127  HandshakeResponsePacket(uint8_t sequence_id,
128  std::vector<unsigned char> auth_response,
129  std::string username, std::string password,
130  std::string database = "", unsigned char char_set = 8,
131  std::string auth_plugin = "mysql_native_password");
132 
133  /** @brief Parses packet payload, results written to object's field
134  *
135  * @param server_capabilities Capabilities sent by the server in Handshake
136  * Packet, see note below
137  *
138  * @throws std::runtime_error on unrecognised or invalid packet
139  *
140  * @note MySQL Protocol has a quirk: In the Handshake Packet, server sends to
141  * client its capability flags, then in Handshake Response Packet, client
142  * sends its own, possibly including some that the server did not advertise.
143  * Despite advertising these flags unique to client, it does not actually use
144  * them. This is vital in understanding packets. If data chunk dataX depeneded
145  * on capability X, then how should a packet be parsed when it comes in?
146  * {data1, data2, dataX, data3, data4}
147  * or
148  * {data1, data2, data3, data4}
149  * Apparently the latter
150  */
151  void parse_payload(Capabilities::Flags server_capabilities) {
152  init_parser_if_not_initialized();
153  parser_->parse(server_capabilities);
154  }
155 
156  /** @brief returns username specified in the packet */
157  const std::string &get_username() const { return username_; }
158 
159  /** @brief returns database name specified in the packet */
160  const std::string &get_database() const { return database_; }
161 
162  /** @brief returns character set specified in the packet */
163  uint8_t get_character_set() const { return char_set_; }
164 
165  /** @brief returns auth-plugin-name specified in the packet */
166  const std::string &get_auth_plugin() const { return auth_plugin_; }
167 
168  /** @brief returns auth-plugin-data specified in the packet */
169  const std::vector<uint8_t> &get_auth_response() const {
170  return auth_response_;
171  }
172 
173  /** @brief returns max packet size specified in the packet */
174  uint32_t get_max_packet_size() const { return max_packet_size_; }
175 
176  /** @brief (debug tool) parse packet contents and print this info on stdout */
177  void debug_dump() {
178  init_parser_if_not_initialized();
179  parser_->debug_dump();
180  }
181 
182  private:
183  class Parser;
184 
185  /** @brief Prepares the packet
186  *
187  * Prepares the actual MySQL Error packet and stores it. The header is
188  * created using the sequence id and the size of the payload.
189  */
190  void prepare_packet();
191 
192  /** @brief Initializes Parser needed to parse the packet payload */
193  void init_parser_if_not_initialized() {
194  if (!parser_) {
195  if (Parser41::is_protocol41(*this)) {
196  parser_.reset(new Parser41(*this));
197  } else if (Parser320::is_protocol320(*this)) {
198  parser_.reset(new Parser320(*this));
199  } else {
200  assert(0);
201  }
202  }
203  }
