34 /**
35  * Key Derivation Function for PBKDF2.
36  *
37  * See: RFC 2898
38  *
39  * while PBKDF2 support several hash-functions, only the most commonly used,
40  * secure variants are exposed:
41  *
42  * - SHA256
43  * - SHA512
44  *
45  * while the insecure ones are not offered:
46  *
47  * - SHA1
48  *
49  * Other HMACs of (https://tools.ietf.org/html/rfc8018#appendix-B.1.2)
50  * may be added:
51  *
52  * - SHA224
53  * - SHA384
54  * - SHA512-224
55  * - SHA512-256
56  */
57 class HTTP_AUTH_BACKEND_LIB_EXPORT Pbkdf2 {
58  public:
59  enum class Type { Sha_256, Sha_512 };
60  static std::vector<uint8_t> salt();
61  static std::vector<uint8_t> derive(Type type, unsigned long rounds,
62  const std::vector<uint8_t> &salt,
63  const std::string &key);
64 };

66 /**
67  * map the MCF name to internal types.
68  *
69  * MCF-name are taking from passlib:
70  *
71  * https://passlib.readthedocs.io/en/stable/modular_crypt_format.html#application-defined-hashes
72  */
73 class Pbkdf2McfType {
74  public:
75  using Type = Pbkdf2::Type;
76  static constexpr char kTypeSha256[] = "pbkdf2-sha256";
77  static constexpr char kTypeSha512[] = "pbkdf2-sha512";
78 
79  static std::pair<bool, std::string> name(Type type) noexcept {
80  switch (type) {
81  case Type::Sha_256:
82  return std::make_pair(true, kTypeSha256);
83  case Type::Sha_512:
84  return std::make_pair(true, kTypeSha512);
85  }
86 
87  return std::make_pair(false, std::string{});
88  }
89 
90  static std::pair<bool, Type> type(const std::string &name) noexcept {
91  if (name == kTypeSha256) {
92  return std::make_pair(true, Type::Sha_256);
93  } else if (name == kTypeSha512) {
94  return std::make_pair(true, Type::Sha_512);
95  }
96 
97  return std::make_pair(false, Type{});
98  }
99 
100  static bool supports_name(const std::string &name) noexcept {
101  if (name == kTypeSha256) {
102  return true;
103  } else if (name == kTypeSha512) {
104  return true;
105  }
106 
107  return false;
108  }
109 };

114 class HTTP_AUTH_BACKEND_LIB_EXPORT Pbkdf2McfAdaptor {
115  public:
116  using mcf_type = Pbkdf2McfType;
117  using kdf_type = Pbkdf2;
118  using Type = mcf_type::Type;
119 
120  /**
121  * rounds if no rounds was specified in from_mcf().
122  */
123  static constexpr unsigned long kDefaultRounds = 1000;
124  /**
125  * minimum rounds.
126  */
127  static constexpr unsigned long kMinRounds = 1;
128  /**
129  * maximum rounds.
130  */
131  static constexpr unsigned long kMaxRounds = 999999999;
132  /**
133  * maximum length of the salt.
134  *
135  * only the first kMaxSaltLength bytes of the salt will be used.
136  */
137  static constexpr size_t kMaxSaltLength = 16;
138 
139  Pbkdf2McfAdaptor(Type type, unsigned long rounds,
140  const std::vector<uint8_t> &salt,
141  const std::vector<uint8_t> &checksum)
142  : type_{type}, rounds_{rounds}, salt_{salt}, checksum_{checksum} {
143  // limit salt
144  if (salt_.size() > kMaxSaltLength) {
145  salt_.resize(kMaxSaltLength);
146  }
147 
148  // limit rounds to allow range
149  if (rounds_ < kMinRounds) rounds_ = kMinRounds;
150  if (rounds_ > kMaxRounds) rounds_ = kMaxRounds;
151  }
152 
153  /**
154  * name of the digest according to MCF.
155  *
156  * - pbkdf2_sha256 for SHA256
157  * - pbkdf2_sha512 for SHA512
158  */
159  std::string mcf_id() const {
160  auto r = mcf_type::name(digest());
161  if (r.first) return r.second;
162 
163  throw std::invalid_argument("failed to map digest to a name");
164  }
165 
166  /**
167  * checkum.
168  *
169  * RFC4648 base64 encoded
170  */
171  std::vector<uint8_t> checksum() const { return checksum_; }
172 
173  /**
174  * salt.
175  *
176  * @pre must be [a-z0-9]*
177  */
178  std::vector<uint8_t> salt() const { return salt_; }
179 
180  /**
181  *
182  */
183  Type digest() const { return type_; }
184 
185  /**
186  * rounds.
187  *
188  * rounds the hash will be applied on itself.
189  */
190  unsigned long rounds() const { return rounds_; }

192  /**
193  * build PBKDF2 from a MCF notation.
194  *
195  * - ${prefix}${rounds}${salt}${checksum}
196  * - ${prefix}${rounds}${salt}
197  *
198  * prefix
199  * : pbkdf2_sha256|pbkdf2_sha512
200  *

203  *

206  *

237  /**
238  * derive a checksum from a key.
239  *
240  * updates checksum
241  */
242  void derive(const std::string &key) {
243  checksum_ = kdf_type::derive(type_, rounds(), salt(), key);
244  }
245 
246  static bool supports_mcf_id(const std::string mcf_id) {
247  return mcf_type::supports_name(mcf_id);
248  }
249 
250  static std::error_code validate(const std::string &mcf_line,
251  const std::string &password) {
252  try {
253  auto mcf = from_mcf(mcf_line);
254  if (kdf_type::derive(mcf.digest(), mcf.rounds(), mcf.salt(), password) ==
255  mcf.checksum()) {
256  return {};
257  } else {
258  return make_error_code(McfErrc::kPasswordNotMatched);
259  }
260  } catch (const std::exception &) {
261  // whatever the exception was, make it a parse-error
262  return make_error_code(McfErrc::kParseError);
263  }
264  }

270  std::vector<uint8_t> checksum_;
271 };
