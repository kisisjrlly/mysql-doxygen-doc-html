81 struct MBR {
82  double xmin, ymin, xmax, ymax;
83 
84  MBR() {
85  xmin = ymin = DBL_MAX;
86  xmax = ymax = -DBL_MAX;
87  }
88 
89  MBR(const double xmin_arg, const double ymin_arg, const double xmax_arg,
90  const double ymax_arg)
91  : xmin(xmin_arg), ymin(ymin_arg), xmax(xmax_arg), ymax(ymax_arg) {}
92 
93  MBR(const point_xy &min, const point_xy &max)
94  : xmin(min.x), ymin(min.y), xmax(max.x), ymax(max.y) {}
95 
96  void add_xy(double x, double y) {
97  /* Not using "else" for proper one point MBR calculation */
98  if (x < xmin) xmin = x;
99  if (x > xmax) xmax = x;
100  if (y < ymin) ymin = y;
101  if (y > ymax) ymax = y;
102  }
103  void add_xy(point_xy p) { add_xy(p.x, p.y); }
104  void add_xy(const char *px, const char *py) {
105  double x = float8get(px);
106  double y = float8get(py);
107  add_xy(x, y);
108  }
109  void add_mbr(const MBR *mbr) {
110  if (mbr->xmin < xmin) xmin = mbr->xmin;
111  if (mbr->xmax > xmax) xmax = mbr->xmax;
112  if (mbr->ymin < ymin) ymin = mbr->ymin;
113  if (mbr->ymax > ymax) ymax = mbr->ymax;
114  }
115 
116  int equals(const MBR *mbr) const {
117  /* The following should be safe, even if we compare doubles */
118  return ((mbr->xmin == xmin) && (mbr->ymin == ymin) && (mbr->xmax == xmax) &&
119  (mbr->ymax == ymax));
120  }
121 
122  int disjoint(const MBR *mbr) const {
123  /* The following should be safe, even if we compare doubles */
124  return ((mbr->xmin > xmax) || (mbr->ymin > ymax) || (mbr->xmax < xmin) ||
125  (mbr->ymax < ymin));
126  }
127 
128  int intersects(const MBR *mbr) const { return !disjoint(mbr); }
129 
130  int touches(const MBR *mbr) const;
131 
132  int within(const MBR *mbr) const;
133 
134  int contains(const MBR *mbr) const { return mbr->within(this); }
135 
136  int covered_by(const MBR *mbr) const {
137  /* The following should be safe, even if we compare doubles */
138  return ((mbr->xmin <= xmin) && (mbr->ymin <= ymin) && (mbr->xmax >= xmax) &&
139  (mbr->ymax >= ymax));
140  }
141 
142  int covers(const MBR *mbr) const { return mbr->covered_by(this); }
143 
144  bool inner_point(double x, double y) const {
145  /* The following should be safe, even if we compare doubles */
146  return (xmin < x) && (xmax > x) && (ymin < y) && (ymax > y);
147  }
148 
149  /**
150  The dimension maps to an integer as:
151  - Polygon -> 2
152  - Horizontal or vertical line -> 1
153  - Point -> 0
154  - Invalid MBR -> -1
155  */
156  int dimension() const {
157  int d = 0;
158 
159  if (xmin > xmax)
160  return -1;
161  else if (xmin < xmax)
162  d++;
163 
164  if (ymin > ymax)
165  return -1;
166  else if (ymin < ymax)
167  d++;
168 
169  return d;
170  }
171 
172  int overlaps(const MBR *mbr) const {
173  /*
174  overlaps() requires that some point inside *this is also inside
175  *mbr, and that both geometries and their intersection are of the
176  same dimension.
177  */
178  int d = dimension();
179  DBUG_ASSERT(d >= 0 && d <= 2);
180 
181  if (d != mbr->dimension() || d == 0 || contains(mbr) || within(mbr))
182  return 0;
183 
184  MBR intersection(std::max(xmin, mbr->xmin), std::max(ymin, mbr->ymin),
185  std::min(xmax, mbr->xmax), std::min(ymax, mbr->ymax));
186 
187  return (d == intersection.dimension());
188  }
