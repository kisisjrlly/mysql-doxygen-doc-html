52 #if !defined(le16toh)
53 /**
54  Converting a 16 bit integer from little-endian byte order to host byteorder.
55 
56  @param x 16-bit integer in little endian byte order
57  @return 16-bit integer in host byte order
58 */
59 
60 uint16_t inline le16toh(uint16_t x) {
61 #ifndef WORDS_BIGENDIAN
62  return x;
63 #else
64  return static_cast<uint16_t>((x >> 8) | (x << 8));
65 #endif
66 }
67 #endif

69 #if !defined(le32toh)
70 /**
71  Converting a 32 bit integer from little-endian byte order to host byteorder.
72 
73  @param x 32-bit integer in little endian byte order
74  @return 32-bit integer in host byte order
75 */
76 
77 uint32_t inline le32toh(uint32_t x) {
78 #ifndef WORDS_BIGENDIAN
79  return x;
80 #else
81  return (((x >> 24) & 0xff) | ((x << 8) & 0xff0000) | ((x >> 8) & 0xff00) |
82  ((x << 24) & 0xff000000));
83 #endif
84 }
85 #endif

87 #if !defined(le64toh)
88 /**
89  Converting a 64 bit integer from little-endian byte order to host byteorder.
90 
91  @param x 64-bit integer in little endian byte order
92  @return 64-bit integer in host byte order
93 */
94 
95 uint64_t inline le64toh(uint64_t x) {
96 #ifndef WORDS_BIGENDIAN
97  return x;
98 #else
99  x = ((x << 8) & 0xff00ff00ff00ff00ULL) | ((x >> 8) & 0x00ff00ff00ff00ffULL);
100  x = ((x << 16) & 0xffff0000ffff0000ULL) | ((x >> 16) & 0x0000ffff0000ffffULL);
101  return (x << 32) | (x >> 32);
102 #endif
103 }
104 #endif
