183  /**
184  Helper class to keep string data in MEM_ROOT before passing to Item_string
185 
186  Since Item_string constructors doesn't copy input string parameter data
187  in the most cases, those input strings must have the same lifetime as
188  Item_string objects, i.e. lifetime of MEM_ROOT.
189  This class allocates input parameters for Item_string objects in MEM_ROOT.
190 
191  @note Call to is_empty() is necessary before the access to "str" and
192  "length" fields, since is_empty() may trigger an evaluation of
193  an associated expression that updates these fields.
194  */
195  struct mem_root_str {
196  const char *str;
197  size_t length;
198  Lazy *
199  deferred; ///< encapsulated expression to evaluate it later (on demand)
200 
201  mem_root_str() { cleanup(); }
202  void cleanup() {
203  str = nullptr;
204  length = 0;
205  deferred = nullptr;
206  }
207  bool is_empty();
208  bool set(const char *str_arg) { return set(str_arg, strlen(str_arg)); }
209  bool set(const String &s) { return set(s.ptr(), s.length()); }
210  /**
211  Make a copy of the string in MEM_ROOT
212 
213  @param str_arg string to copy
214  @param length_arg input string length
215 
216  @return false if success, true if error
217  */
218  bool set(const char *str_arg, size_t length_arg);
219 
220  /**
221  Save expression for further evaluation
222 
223  @param x Expression
224  */
225  void set(Lazy *x) {
226  deferred = x;
227  str = nullptr;
228  length = 0;
229  }
230  /**
231  Make a copy of string constant
232 
233  Variant of set() usable when the str_arg argument lives longer
234  than the mem_root_str instance.
235  */
236  void set_const(const char *str_arg) {
237  return set_const(str_arg, strlen(str_arg));
238  }
239  void set_const(const char *str_arg, size_t length_arg) {
240  deferred = nullptr;
241  str = str_arg;
242  length = length_arg;
243  }
244 
245  static char *strndup_root(MEM_ROOT *root, const char *str, size_t len) {
246  if (len == 0 || str == nullptr) return const_cast<char *>("");
247  if (str[len - 1] == 0)
248  return static_cast<char *>(memdup_root(root, str, len));
249 
250  char *ret = static_cast<char *>(root->Alloc(len + 1));
251  if (ret != nullptr) {
252  memcpy(ret, str, len);
253  ret[len] = 0;
254  }
255  return ret;
256  }

435 /**
436  Bit flags to explain GROUP BY, ORDER BY and DISTINCT clauses
437 */
438 enum Explain_sort_property {
439  ESP_none = 0,
440  ESP_EXISTS = 1 << 0, ///< Original query has this clause
441  ESP_IS_SIMPLE = 1 << 1, ///< Clause is effective for single JOIN_TAB only
442  ESP_USING_FILESORT = 1 << 2, ///< Clause causes a filesort
443  ESP_USING_TMPTABLE = 1 << 3, ///< Clause creates an intermediate table
444  ESP_DUPS_REMOVAL = 1 << 4 ///< Duplicate removal for DISTINCT
445 };

447 class Explain_format_flags {
448  /**
449  Bitmasks of Explain_sort_property flags for Explain_sort_clause clauses
450  */
451  uint8 sorts[ESC_MAX];
452 
453  public:
454  Explain_format_flags() { memset(sorts, 0, sizeof(sorts)); }
455 
456  /**
457  Set property bit flag for the clause
458  */
459  void set(Explain_sort_clause clause, Explain_sort_property property) {
460  sorts[clause] |= property | ESP_EXISTS;
461  }
462 
463  void set(Explain_format_flags &flags) {
464  memcpy(sorts, flags.sorts, sizeof(sorts));
465  }
466 
467  /**
468  Clear property bit flag for the clause
469  */
470  void reset(Explain_sort_clause clause, Explain_sort_property property) {
471  sorts[clause] &= ~property;
472  }
473 
474  /**
475  Return true if property is set for the clause
476  */
477  bool get(Explain_sort_clause clause, Explain_sort_property property) const {
478  return sorts[clause] & property;
479  }
480 
481  /**
482  Return true if any of clauses has this property set
483 
484  @param property Check if this property is present in any of the sorts
485  except clause's sort if specified
486  @param clause Optional. Do not check for the property for this clause. The
487  default is to check all clauses.
488  */
489  bool any(Explain_sort_property property,
490  Explain_sort_clause clause = ESC_none) const {
491  for (size_t i = ESC_none + 1; i <= ESC_MAX - 1; i++) {
492  if (i != clause && (sorts[i] & property)) return true;
493  }
494  return false;
495  }
496 };
