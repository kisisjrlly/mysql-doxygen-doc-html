270 /** We use FlushObserver to track flushing of non-redo logged pages in bulk
271 create index(BtrBulk.cc).Since we disable redo logging during a index build,
272 we need to make sure that all dirty pages modifed by the index build are
273 flushed to disk before any redo logged operations go to the index. */

275 class FlushObserver {
276  public:
277  /** Constructor
278  @param[in] space_id table space id
279  @param[in] trx trx instance
280  @param[in] stage performance schema accounting object,
281  used by ALTER TABLE. It is passed to log_preflush_pool_modified_pages()
282  for accounting. */
283  FlushObserver(space_id_t space_id, trx_t *trx, ut_stage_alter_t *stage);
284 
285  /** Deconstructor */
286  ~FlushObserver();
287 
288  /** Check pages have been flushed and removed from the flush list
289  in a buffer pool instance.
290  @param[in] instance_no buffer pool instance no
291  @return true if the pages were removed from the flush list */
292  bool is_complete(ulint instance_no) {
293  os_rmb;
294  return (m_flushed->at(instance_no) == m_removed->at(instance_no) ||
295  m_interrupted);
296  }
297 
298  /** Interrupt observer not to wait. */
299  void interrupted() { m_interrupted = true; }
300 
301  /** Check whether trx is interrupted
302  @return true if trx is interrupted */
303  bool check_interrupted();
304 
305  /** Flush dirty pages. */
306  void flush();
307 
308  /** Notify observer of flushing a page
309  @param[in] buf_pool buffer pool instance
310  @param[in] bpage buffer page to flush */
311  void notify_flush(buf_pool_t *buf_pool, buf_page_t *bpage);
312 
313  /** Notify observer of removing a page from flush list
314  @param[in] buf_pool buffer pool instance
315  @param[in] bpage buffer page flushed */
316  void notify_remove(buf_pool_t *buf_pool, buf_page_t *bpage);
317 
318  private:
319  /** Table space id */
320  space_id_t m_space_id;
321 
322  /** Trx instance */
323  trx_t *m_trx;
324 
325  /** Performance schema accounting object, used by ALTER TABLE.
326  If not NULL, then stage->begin_phase_flush() will be called initially,
327  specifying the number of pages to be attempted to be flushed and
328  subsequently, stage->inc() will be called for each page we attempt to
329  flush. */
330  ut_stage_alter_t *m_stage;
331 
332  /* Flush request sent */
333  std::vector<ulint> *m_flushed;
334 
335  /* Flush request finished */
336  std::vector<ulint> *m_removed;
337 
338  /* True if the operation was interrupted. */
339  bool m_interrupted;
340 };
