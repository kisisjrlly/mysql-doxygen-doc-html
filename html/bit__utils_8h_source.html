37 // Wraps iteration over interesting states (based on the given policy) over a
38 // single uint64_t into an STL-style adapter.
39 template <class Policy>
40 class BitIteratorAdaptor {
41  public:
42  class iterator {
43  private:
44  uint64_t m_state;
45 
46  public:
47  explicit iterator(uint64_t state) : m_state(state) {}
48  bool operator==(const iterator &other) const {
49  return m_state == other.m_state;
50  }
51  bool operator!=(const iterator &other) const {
52  return m_state != other.m_state;
53  }
54  size_t operator*() const { return Policy::NextValue(m_state); }
55  iterator &operator++() {
56  m_state = Policy::AdvanceState(m_state);
57  return *this;
58  }
59  };
60 
61  explicit BitIteratorAdaptor(uint64_t state) : m_initial_state(state) {}
62 
63  iterator begin() const { return iterator{m_initial_state}; }
64  iterator end() const { return iterator{0}; }
65 
66  private:
67  const uint64_t m_initial_state;
68 };

70 // A policy for BitIteratorAdaptor that gives out the index of each set bit in
71 // the value, ascending.
72 class CountBitsAscending {
73  public:
74  static size_t NextValue(uint64_t state) {
75  // Find the lowest set bit.
76  assert(state != 0);
77 #ifdef _MSC_VER
78  unsigned long idx;
79  _BitScanForward64(&idx, state);
80  return idx;
81 #elif defined(__GNUC__) && defined(__x86_64__)
82  // Using this instead of ffsll() (which maps to the same instruction,
83  // but has an extra zero test and returns an int value) helps
84  // a whopping 10% on some of the microbenchmarks! (GCC 9.2, Skylake.)
85  // Evidently, the test for zero is rewritten into a conditional move,
86  // which turns out to be add a lot of latency into these hot loops.
87  size_t idx;
88  asm("bsfq %1,%q0" : "=r"(idx) : "rm"(state));
89  return idx;
90 #else
91  // The cast to unsigned at least gets rid of the sign extension.
92  return static_cast<unsigned>(ffsll(state)) - 1u;
93 #endif
94  }
95 
96  static uint64_t AdvanceState(uint64_t state) {
97  // Clear the lowest set bit.
98  assert(state != 0);
99  return state & (state - 1);
100  }
101 };

103 // Same as CountBitsAscending, just descending.
104 class CountBitsDescending {
105  public:
106  static size_t NextValue(uint64_t state) {
107  // Find the lowest set bit.
108  assert(state != 0);
109 #ifdef _MSC_VER
110  unsigned long idx;
111  _BitScanReverse64(&idx, state);
112  return idx;
113 #else
114  return __builtin_clzll(state) ^ 63u;
115 #endif
116  }
117 
118  static uint64_t AdvanceState(uint64_t state) {
119  // Clear the highest set bit. (This is fewer operations
120  // then the standard bit-fiddling trick, especially given
121  // that NextValue() is probably already computed.)
122  return state & ~(uint64_t{1} << NextValue(state));
123  }
124 };

129 // An iterator (for range-based for loops) that returns all non-zero subsets of
130 // a given set. This includes the set itself.
131 //
132 // In the database literature, this algorithm is often attributed to
133 // a 1995 paper of Vance and Maier, but it is known to be older than
134 // that. In particular, here is a 1994 reference from Marcel van Kervinck:
135 //
136 // https://groups.google.com/forum/#!msg/rec.games.chess/KnJvBnhgDKU/yCi5yBx18PQJ
137 class NonzeroSubsetsOf {
138  public:
139  class iterator {
140  private:
141  uint64_t m_state;
142  uint64_t m_set;
143 
144  public:
145  iterator(uint64_t state, uint64_t set) : m_state(state), m_set(set) {}
146  bool operator==(const iterator &other) const {
147  assert(m_set == other.m_set);
148  return m_state == other.m_state;
149  }
150  bool operator!=(const iterator &other) const {
151  assert(m_set == other.m_set);
152  return m_state != other.m_state;
153  }
154  size_t operator*() const { return m_state; }
155  iterator &operator++() {
156  m_state = (m_state - m_set) & m_set;
157  return *this;
158  }
159  };
160 
161  explicit NonzeroSubsetsOf(uint64_t set) : m_set(set) {}
162 
163  iterator begin() const { return {(-m_set) & m_set, m_set}; }
164  iterator end() const { return {0, m_set}; }
165 
166  private:
167  const uint64_t m_set;
168 };
