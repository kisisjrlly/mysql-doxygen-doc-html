44 /* Database name and table name for our metadata "system" tables for
45 InnoDB memcache. The table names are the same as those for the
46 NDB memcache, to make the memcache setup compatible between the two.
47 There are 3 "system tables":
48 1) containers - main configure table contains row describing which InnoDB
49  table is used to store/retrieve Memcached key/value if InnoDB
50  Memcached engine is used
51 2) cache_policies - decide whether to use "Memcached Default Engine" or "InnoDB
52  Memcached Engine" to handler the requests
53 3) config_options - for miscellaneous configuration options */
54 #define MCI_CFG_DB_NAME "innodb_memcache"
55 #define MCI_CFG_CONTAINER_TABLE "containers"
56 #define MCI_CFG_CACHE_POLICIES "cache_policies"
57 #define MCI_CFG_CONFIG_OPTIONS "config_options"
