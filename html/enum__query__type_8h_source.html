26 /**
27  Query type constants (usable as bitmap flags).
28 */
29 
30 enum enum_query_type {
31  /// Nothing specific, ordinary SQL query.
32  QT_ORDINARY = 0,
33 
34  /// In utf8.
35  QT_TO_SYSTEM_CHARSET = (1 << 0),
36 
37  /// Without character set introducers.
38  QT_WITHOUT_INTRODUCERS = (1 << 1),
39 
40  /**
41  Causes string literals to always be printed with character set
42  introducers. Takes precedence over QT_WITHOUT_INTRODUCERS.
43  */
44  QT_FORCE_INTRODUCERS = (1 << 2),
45 
46  /// When printing a SELECT, add its number (select_lex->number).
47  QT_SHOW_SELECT_NUMBER = (1 << 3),
48 
49  /// Don't print a database if it's equal to the connection's database.
50  QT_NO_DEFAULT_DB = (1 << 4),
51 
52  /// When printing a derived table, don't print its expression, only alias.
53  QT_DERIVED_TABLE_ONLY_ALIAS = (1 << 5),
54 
55  /// Print in charset of Item::print() argument (typically thd->charset()).
56  QT_TO_ARGUMENT_CHARSET = (1 << 6),
57 
58  /// Print identifiers without database's name.
59  QT_NO_DB = (1 << 7),
60 
61  /// Print identifiers without table's name.
62  QT_NO_TABLE = (1 << 8),
63 
64  /**
65  Change all Item_basic_constant to ? (used by query rewrite to compute
66  digest.) Un-resolved hints will also be printed in this format.
67  */
68  QT_NORMALIZED_FORMAT = (1 << 9),
69 
70  /**
71  If an expression is constant, print the expression, not the value
72  it evaluates to. Should be used for error messages, so that they
73  don't reveal values.
74  */
75  QT_NO_DATA_EXPANSION = (1 << 10),
76 
77  /**
78  Don't print the QB name. Used for the INSERT part of an INSERT...SELECT.
79  */
80  QT_IGNORE_QB_NAME = (1 << 11),
81 
82  /**
83  Print only the QB name. Used for the SELECT part of an INSERT...SELECT.
84  */
85  QT_ONLY_QB_NAME = (1 << 12),
86 
87  /**
88  When printing subselects, print only the select number
89  (like QT_SHOW_SELECT_NUMBER), not the actual subselect text.
90  */
91  QT_SUBSELECT_AS_ONLY_SELECT_NUMBER = (1 << 13),
92 
93  /**
94  Suppress the internal rollup functions used for switching items to NULL
95  or between different sums; these are different from most other internal
96  Items we insert, since they are inserted during resolving and not
97  optimization. Used when getting the canonical representation of a view.
98  */
99  QT_HIDE_ROLLUP_FUNCTIONS = (1 << 14),
100 
101  /**
102  When printing Item_view_ref, print the reference (i.e. column name of the
103  derived table), not the referenced (underlying) expression.
104  */
105  QT_DERIVED_TABLE_ORIG_FIELD_NAMES = (1 << 15),
106 };
107 
108 #endif // ENUM_QUERY_TYPE_INCLUDED
