37 /**
38  A RFC2782 compliant SRV records storage
39 
40  Stores host/port/weight/priority entries
41  into a data structure that then allows retrieving
42  these via the Dns_srv_data::pop_next() method.
43 
44  Entries can be stored by calling Dns_srv_data::add()
45  in any order.
46 
47  This usage pattern is rougly as follows:
48  1. Dns_srv_data construct
49  2. one or more Dns_srv_data::add()
50  3. one or more Dns_srv_data::pop_next()
51  4. Dns_srv_data destructor
52 
53  @sa mysql_real_connect_dns_srv
54 */
55 class Dns_srv_data {
56  class Dns_entry {
57  std::string host_;
58  unsigned port_{0}, weight_{0};
59  unsigned long weight_sum_{0};
60 
61  Dns_entry() = delete; // disable copy constructor
62 
63  public:
64  Dns_entry(const std::string &host, unsigned port, unsigned weight)
65  : host_(host), port_(port), weight_(weight) {}
66 
67  unsigned port() const { return port_; }
68  std::string host() const { return host_; }
69  unsigned long weight_sum() const { return weight_sum_; }
70  void add_weight_sum(unsigned long &weight_sum) {
71  weight_sum_ = (weight_sum += weight_);
72  }
73  };
74  using dns_entry_list_t = std::list<Dns_entry>;
75  using dns_entry_data_t = std::map<unsigned, Dns_srv_data::dns_entry_list_t>;
76  dns_entry_data_t data_;
77 
78  public:
79  void clear() { data_.clear(); }
80  void add(const std::string &host, unsigned port, unsigned priority,
81  unsigned weight) {
82  dns_entry_data_t::iterator list = data_.find(priority);
83  if (list == data_.cend())
84  data_.emplace(priority,
85  dns_entry_list_t(1, Dns_entry(host, port, weight)));
86  else {
87  // RFC2782: put the 0 weight at the front, rest at the back
88  if (weight > 0)
89  list->second.emplace_back(host, port, weight);
90  else
91  list->second.emplace_front(host, port, weight);
92  }
93  }
94  bool pop_next(std::string &host, unsigned &port) {
95  if (data_.empty()) return true;
96 
97  dns_entry_list_t &list = data_.begin()->second;
98  assert(!list.empty());
99 
100  unsigned long sum = 0;
101  for (Dns_entry &elt : list) elt.add_weight_sum(sum);
102 
103  unsigned long draw = (std::rand() * 1UL * sum) / RAND_MAX;
104 
105  dns_entry_list_t::const_iterator iter = list.cbegin();
106  while (iter->weight_sum() < draw) iter++;
107  assert(iter != list.end());
108 
109  host = iter->host();
110  port = iter->port();
111 
112  list.erase(iter);
113  if (list.empty()) data_.erase(data_.begin());
114  return false;
115  }
116 };
