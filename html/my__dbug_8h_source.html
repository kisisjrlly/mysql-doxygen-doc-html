96 /**
97  A RAII helper to do DBUG_ENTER / DBUG_RETURN for you automatically. Use like
98  this:
99 
100  int foo() {
101  DBUG_TRACE;
102  return 42;
103  }
104  */
105 class AutoDebugTrace {
106  public:
107  AutoDebugTrace(const char *function, const char *filename, int line) {
108  // Remove the return type, if it's there.
109  const char *begin = strchr(function, ' ');
110  if (begin != nullptr) {
111  function = begin + 1;
112  }
113 
114  // Cut it off at the first parenthesis; the argument list is
115  // often too long to be interesting.
116  const char *end = strchr(function, '(');
117 
118  if (end == nullptr) {
119  _db_enter_(function, static_cast<int>(strlen(function)), filename, line,
120  &m_stack_frame);
121  } else {
122  _db_enter_(function, static_cast<int>(end - function), filename, line,
123  &m_stack_frame);
124  }
125  }
126 
127  ~AutoDebugTrace() { _db_return_(0, &m_stack_frame); }
128 
129  private:
130  _db_stack_frame_ m_stack_frame;
131 };

306 /*
307  A C++ interface to the DBUG_PRINT macro. The DBUG_LOG macro takes two
308  arguments. The first argument is the keyword, as that of the
309  DBUG_PRINT. The 2nd argument 'v' will be passed to a C++ output stream.
310  This enables the use of C++ style output stream operator. In the code, it
311  will be used as follows:
312 
313  DBUG_LOG("blob", "space: " << space_id);
314 
315  Note: DBUG_PRINT() has a limitation of 1024 bytes for a single
316  print out. So, this limitation is there for DBUG_LOG macro also.
317 */

320  do { \
321  if (_db_enabled_()) { \

325  } \
326  } while (0)

334 /**
335  A type-safe interface to DBUG_EXECUTE_IF, where the debug action to
336  activate when the keyword is provided is given as a callable object
337  (typically a lambda).
338 
339  @note The body of the callable will be checked by the compiler even
340  in optimized mode.
341 
342  @param keyword String literal which will enable this debug action.
343  @param clos Callable object taking no arguments which will be
344  called in debug mode if the keyword is enabled.
345  */
346 template <class DBGCLOS>
347 inline void dbug(const char *keyword MY_ATTRIBUTE((unused)),
348  DBGCLOS &&clos MY_ATTRIBUTE((unused))) {
349  DBUG_EXECUTE_IF(keyword, clos(););
350 }
