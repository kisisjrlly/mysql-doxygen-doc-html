33 /**
34  A typesafe replacement for DYNAMIC_ARRAY.
35  We use MEM_ROOT for allocating storage, rather than the C++ heap.
36  The interface is chosen to be similar to std::vector.
37 
38  @remark
39  Mem_root_array_YY is constructor-less for use in the parser stack of unions.
40  For other needs please use Mem_root_array.
41 
42  @remark
43  Unlike DYNAMIC_ARRAY, elements are properly copied
44  (rather than memcpy()d) if the underlying array needs to be expanded.
45 
46  @remark
47  Unless Element_type's destructor is trivial, we destroy objects when they are
48  removed from the array (including when the array object itself is destroyed).
49 
50  @remark
51  Note that MEM_ROOT has no facility for reusing free space,
52  so don't use this if multiple re-expansions are likely to happen.
53 
54  @tparam Element_type The type of the elements of the container.
55  Elements must be copyable.
56 */
57 template <typename Element_type>
58 class Mem_root_array_YY {
59  /**
60  Is Element_type trivially destructible? If it is, we don't destroy
61  elements when they are removed from the array or when the array is
62  destroyed.
63  */
64  static constexpr bool has_trivial_destructor =
65  std::is_trivially_destructible<Element_type>::value;
66

68  /// Convenience typedef, same typedef name as std::vector
69  typedef Element_type value_type;

127  /**
128  Chops the tail off the array, erasing all tail elements.
129  @param pos Index of first element to erase.
130  */
131  void chop(const size_t pos) {
132  DBUG_ASSERT(pos < m_size);
133  if (!has_trivial_destructor) {
134  for (size_t ix = pos; ix < m_size; ++ix) {
135  Element_type *p = &m_array[ix];
136  p->~Element_type(); // Destroy discarded element.
137  }
138  }
139  m_size = pos;
140  }

142  /**
143  Reserves space for array elements.
144  Copies over existing elements, in case we are re-expanding the array.
145 
146  @param n number of elements.
147  @retval true if out-of-memory, false otherwise.
148  */
149  bool reserve(size_t n) {
150  if (n <= m_capacity) return false;
151 
152  void *mem = m_root->Alloc(n * element_size());
153  if (!mem) return true;
154  Element_type *array = static_cast<Element_type *>(mem);
155 
156  // Copy all the existing elements into the new array.
157  for (size_t ix = 0; ix < m_size; ++ix) {
158  Element_type *new_p = &array[ix];
159  Element_type *old_p = &m_array[ix];
160  ::new (new_p)
161  Element_type(std::move(*old_p)); // Copy or move into new location.
162  if (!has_trivial_destructor)
163  old_p->~Element_type(); // Destroy the old element.
164  }
165 
166  // Forget the old array.
167  m_array = array;
168  m_capacity = n;
169  return false;
170  }
171 
172  /**
173  Adds a new element at the end of the array, after its current last
174  element. The content of this new element is initialized to a copy of
175  the input argument.
176 
177  @param element Object to copy.
178  @retval true if out-of-memory, false otherwise.
179  */
180  bool push_back(const Element_type &element) {
181  const size_t min_capacity = 20;
182  const size_t expansion_factor = 2;
183  if (0 == m_capacity && reserve(min_capacity)) return true;
184  if (m_size == m_capacity && reserve(m_capacity * expansion_factor))
185  return true;
186  Element_type *p = &m_array[m_size++];
187  ::new (p) Element_type(element);
188  return false;
189  }
190 
191  /**
192  Adds a new element at the end of the array, after its current last
193  element. The content of this new element is initialized by moving
194  the input element.
195 
196  @param element Object to move.
197  @retval true if out-of-memory, false otherwise.
198  */
199  bool push_back(Element_type &&element) {
200  const size_t min_capacity = 20;
201  const size_t expansion_factor = 2;
202  if (0 == m_capacity && reserve(min_capacity)) return true;
203  if (m_size == m_capacity && reserve(m_capacity * expansion_factor))
204  return true;
205  Element_type *p = &m_array[m_size++];
206  ::new (p) Element_type(std::move(element));
207  return false;
208  }

210  /**
211  Adds a new element at the beginning of the array.
212  The content of this new element is initialized to a copy of
213  the input argument.
214 
215  @param element Object to copy.
216  @retval true if out-of-memory, false otherwise.
217  */
218  bool push_front(const Element_type &element) {
219  if (push_back(element)) return true;
220  std::rotate(begin(), end() - 1, end());
221  return false;

224  /**
225  Adds a new element at the front of the array.
226  The content of this new element is initialized by moving the input element.
227 
228  @param element Object to move.
229  @retval true if out-of-memory, false otherwise.
230  */
231  bool push_front(Element_type &&element) {
232  if (push_back(std::move(element))) return true;
233  std::rotate(begin(), end() - 1, end());
234  return false;

237  /**
238  Removes the last element in the array, effectively reducing the
239  container size by one. This destroys the removed element.
240  */
241  void pop_back() {
242  DBUG_ASSERT(!empty());
243  if (!has_trivial_destructor) back().~Element_type();
244  m_size -= 1;
245  }

247  /**
248  Resizes the container so that it contains n elements.
249 
250  If n is smaller than the current container size, the content is
251  reduced to its first n elements, removing those beyond (and
252  destroying them).
253 
254  If n is greater than the current container size, the content is
255  expanded by inserting at the end as many elements as needed to
256  reach a size of n. If val is specified, the new elements are
257  initialized as copies of val, otherwise, they are
258  value-initialized.
259 
260  If n is also greater than the current container capacity, an automatic
261  reallocation of the allocated storage space takes place.
262 
263  Notice that this function changes the actual content of the
264  container by inserting or erasing elements from it.
265  */
266  void resize(size_t n, const value_type &val) {
267  if (n == m_size) return;
268  if (n > m_size) {
269  if (!reserve(n)) {
270  while (n != m_size) push_back(val);
271  }
272  return;
273  }
274  if (!has_trivial_destructor) {
275  while (n != m_size) pop_back();
276  }
277  m_size = n;
278  }
279 
280  /**
281  Same as resize(size_t, const value_type &val), but default-constructs
282  the new elements. This allows one to resize containers even if
283  value_type is not copy-constructible.
284  */
285  void resize(size_t n) {
286  if (n == m_size) return;
287  if (n > m_size) {
288  if (!reserve(n)) {
289  while (n != m_size) push_back(value_type());
290  }
291  return;
292  }
293  if (!has_trivial_destructor) {
294  while (n != m_size) pop_back();
295  }
296  m_size = n;
297  }

299  /**
300  Erase all the elements in the specified range.
301 
302  @param first iterator that points to the first element to remove
303  @param last iterator that points to the element after the
304  last one to remove
305  @return an iterator to the first element after the removed range
306  */
307  iterator erase(const_iterator first, const_iterator last) {
308  iterator pos = begin() + (first - cbegin());
309  if (first != last) {
310  iterator new_end = std::move(last, cend(), pos);
311  chop(new_end - begin());
312  }
313  return pos;
314  }

316  /**
317  Removes a single element from the array.
318 
319  @param position iterator that points to the element to remove
320 
321  @return an iterator to the first element after the removed range
322  */
323  iterator erase(const_iterator position) {
324  return erase(position, std::next(position));
325  }

339  /**
340  Insert an element at a given position.
341 
342  @param pos the new element is inserted before the element
343  at this position
344  @param value the value of the new element
345  @return an iterator that points to the inserted element
346  */
347  iterator insert(const_iterator pos, const Element_type &value) {
348  ptrdiff_t idx = pos - cbegin();
349  if (!push_back(value)) std::rotate(begin() + idx, end() - 1, end());
350  return begin() + idx;
351  }

353  /**
354  Removes a single element from the array by value.
355  The removed element is destroyed. This effectively reduces the
356  container size by one.
357  Note that if there are multiple elements having the same
358  value, only the first element is removed.
359 
360  This is generally an inefficient operation, since we need to copy
361  elements to fill the "hole" in the array.
362 
363  We use std::copy to move objects, hence Element_type must be
364  assignable.
365 
366  @retval number of elements removed, 0 or 1.
367  */
368  size_t erase_value(const value_type &val) {
369  iterator position = std::find(begin(), end(), val);
370  if (position != end()) {
371  erase(position);
372  return 1;
373  }
374  return 0; // Not found
375  }
376 
377  /**
378  Removes a single element from the array.
379  The removed element is destroyed.
380  This effectively reduces the container size by one.
381 
382  This is generally an inefficient operation, since we need to copy
383  elements to fill the "hole" in the array.
384 
385  We use std::copy to move objects, hence Element_type must be assignable.
386  */
387  iterator erase(iterator position) {
388  DBUG_ASSERT(position != end());
389  if (position + 1 != end()) std::copy(position + 1, end(), position);
390  this->pop_back();
391  return position;
392  }

395  size_t element_size() const { return sizeof(Element_type); }
396  bool empty() const { return size() == 0; }
397  size_t size() const { return m_size; }
398

410 /**
411  A typesafe replacement for DYNAMIC_ARRAY.
412 
413  @see Mem_root_array_YY.
414 */
415 template <typename Element_type>
416 class Mem_root_array : public Mem_root_array_YY<Element_type> {
417  typedef Mem_root_array_YY<Element_type> super;
418 
419  public:
420  /// Convenience typedef, same typedef name as std::vector
421  typedef Element_type value_type;
422 
423  typedef typename super::const_iterator const_iterator;
424 
425  Mem_root_array() { super::init_empty_const(); }
426 
427  explicit Mem_root_array(MEM_ROOT *root) { super::init(root); }
428 
429  /// Move constructor and assignment.
430  Mem_root_array(Mem_root_array &&other) {
431  this->m_root = other.m_root;
432  this->m_array = other.m_array;
433  this->m_size = other.m_size;
434  this->m_capacity = other.m_capacity;
435  other.init_empty_const();
436  }
437  Mem_root_array &operator=(Mem_root_array &&other) {
438  if (this != &other) {
439  this->~Mem_root_array();
440  new (this) Mem_root_array(std::move(other));
441  }
442  return *this;
443  }
444 
445  Mem_root_array(MEM_ROOT *root, size_t n) {
446  super::init(root);
447  super::resize(n);
448  }
449 
450  Mem_root_array(MEM_ROOT *root, size_t n, const value_type &val) {
451  super::init(root);
452  super::resize(n, val);
453  }
454 
455  /**
456  Range constructor.
457 
458  Constructs a container with as many elements as the range [first,last),
459  with each element constructed from its corresponding element in that range,
460  in the same order.
461 
462  @param root MEM_ROOT to use for memory allocation.
463  @param first iterator that points to the first element to copy
464  @param last iterator that points to the element after the
465  last one to copy
466  */
467  Mem_root_array(MEM_ROOT *root, const_iterator first, const_iterator last) {
468  super::init(root);
469  if (this->reserve(last - first)) return;
470  for (auto it = first; it != last; ++it) this->push_back(*it);
471  }
472 
473  Mem_root_array(MEM_ROOT *root, const Mem_root_array &x)
474  : Mem_root_array(root, x.cbegin(), x.cend()) {}
475 
476  ~Mem_root_array() { super::clear(); }
477 
478  // Not (yet) implemented.
479  Mem_root_array(const Mem_root_array &) = delete;
480  Mem_root_array &operator=(const Mem_root_array &) = delete;
481 };
