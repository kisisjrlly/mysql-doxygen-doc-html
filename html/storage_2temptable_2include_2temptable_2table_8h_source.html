199 inline void Table::row(const Storage::Iterator &pos,
200  unsigned char *mysql_row) const {
201  DBUG_ASSERT(m_mysql_row_length == m_mysql_table_share->rec_buff_length);
202 
203  const Storage::Element *storage_element = *pos;
204 
205  if (m_all_columns_are_fixed_size) {
206  DBUG_ASSERT(m_rows.element_size() == m_mysql_row_length);
207 
208  memcpy(mysql_row, storage_element, m_mysql_row_length);
209  } else {
210  DBUG_ASSERT(m_rows.element_size() == sizeof(Row));
211 
212  const Row *row = static_cast<const Row *>(storage_element);
213  row->copy_to_mysql_row(m_columns, mysql_row, m_mysql_row_length);
214  }
215 }

217 inline void Table::truncate() {
218  if (!m_all_columns_are_fixed_size) {
219  for (auto element : m_rows) {
220  Row *row = static_cast<Row *>(element);
221  row->~Row();
222  }
223  }
224  m_rows.clear();
225 
226  /* Truncate indexes even if `m_indexes_are_enabled` is false. Somebody may
227  * use truncate() before enabling indexes and we don't want an empty m_rows
228  * with some stale data inside the indexes. */
229  for (auto &entry : m_index_entries) {
230  entry.m_index->truncate();
231  }
232 }

252 /** Create index for given key and appends it to indexes table. */
253 template <class T>
254 inline void Table::append_new_index(const KEY &mysql_index) {
255  Index_entry entry;
256 
257  entry.m_alloc_size = sizeof(T);
258 
259  auto mem_ptr = m_allocator.allocate(entry.m_alloc_size);
260  try {
261  entry.m_index = new (mem_ptr) T(*this, mysql_index, m_allocator);
262 
263  m_index_entries.push_back(entry);
264  } catch (...) {
265  m_allocator.deallocate(mem_ptr, entry.m_alloc_size);
266  throw;
267  }
268 }
