32 /**
33  Class that defines basic logging infra-structure to be used in the test
34  cases, for example.
35 */
36 class Gcs_basic_logging {
37  public:
38  /*
39  Pointer to a logger object that is responsible for handling fatal, error,
40  warning and information messages.
41  */
42  Gcs_default_logger *logger;
43 
44  /*
45  Pointer to a debugger object that is responsible for handling debug and
46  trace messages.
47  */
48  Gcs_default_debugger *debugger;
49 
50  /*
51  Pointer to a sink where both the messages produced by a logger or debugger
52  will be written to. This is a simple logging infra-structure and messages
53  are always written to the standard output.
54  */
55  Gcs_async_buffer *sink;
56 
57  /*
58  Save debug options that will be restored when the object is destructed.
59  */
60  int64_t saved_debug_options;
61 
62  /**
63  Constructor that creates the logger, debugger and sink.
64  */
65  Gcs_basic_logging()
66  : logger(nullptr),
67  debugger(nullptr),
68  sink(nullptr),
69  saved_debug_options(GCS_DEBUG_NONE) {
70  saved_debug_options = Gcs_debug_options::get_current_debug_options();
71  Gcs_debug_options::force_debug_options(GCS_DEBUG_ALL);
72 
73  sink = new Gcs_async_buffer(new Gcs_output_sink());
74 
75  logger = new Gcs_default_logger(sink);
76  Gcs_log_manager::initialize(logger);
77 
78  debugger = new Gcs_default_debugger(sink);
79  Gcs_debug_manager::initialize(debugger);
80 
81  ::set_xcom_logger(cb_xcom_logger);
82  ::set_xcom_debugger(cb_xcom_debugger);
83  ::set_xcom_debugger_check(cb_xcom_debugger_check);
84 
85  Gcs_xcom_utils::init_net();
86  }
87 
88  /**
89  Destructor that cleans up and deallocates the logger, debugger and sink.
90  */
91  virtual ~Gcs_basic_logging() {
92  Gcs_log_manager::finalize();
93  logger->finalize();
94  delete logger;
95 
96  Gcs_debug_manager::finalize();
97  debugger->finalize();
98  delete debugger;
99 
100  sink->finalize();
101  delete sink;
102 
103  Gcs_debug_options::force_debug_options(saved_debug_options);
104  }
105 };
