42 /**
43  Bootstrap thread executes SQL statements.
44  Any error in the execution of SQL statements causes call to my_error().
45  At this moment, error handler hook is set to my_message_stderr.
46  my_message_stderr() prints the error messages to standard error stream but
47  it does not follow the standard error format. Further, the error status is
48  not set in Diagnostics Area.
49 
50  This class is to create RAII error handler hooks to be used when executing
51  statements from bootstrap thread.
52 
53  It will print the error in the standard error format.
54  Diagnostics Area error status will be set to avoid asserts.
55  Error will be handler by caller function.
56 */
57 
58 class Bootstrap_error_handler {
59  private:
60  void (*m_old_error_handler_hook)(uint, const char *, myf);
61 
62  // Set the error in DA. Optionally print error in log.
63  static void my_message_bootstrap(uint error, const char *str, myf MyFlags);
64 
65  // Set abort on error flag and enable error logging for certain fatal error.
66  static void set_abort_on_error(uint error);
67 
68  public:
69  Bootstrap_error_handler();
70 
71  // Mark as error is set.
72  void set_log_error(bool log_error);
73 
74  ~Bootstrap_error_handler();
75  static bool m_log_error;
76  static bool abort_on_error;
77 };
