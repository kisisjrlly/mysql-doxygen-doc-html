36 /**
37  Implementation of a map between a key type and an element type.
38 
39  The map supports storing associations between instances of the key
40  type K and pointers to the element type E. Additionally, the map provides
41  support for registering keys that have been searched for in the map
42  without being present. This allows registering a cache miss to avoid
43  situations where several similar cache misses are handled simultaneously.
44  Instead, only the first cache miss needs to be processed, while the
45  subsequent ones can just wait for the first one to complete.
46 
47  There are no expectations as far as the element type E is concerned.
48  Tracking object usage, memory management, loading and evicting elements,
49  locking and synchronization, as well as registering cache misses are all
50  issues that are handled by users of this interface.
51 
52  Basic assumptions regarding correct usage is implemented in terms of
53  asserts to verify that, e.g., a key does not already exist in the map
54  when it is added. There is, however, no assumptions regarding the
55  relationship between the set of missed keys and the state of the map.
56 
57  There is support for representing missed keys, but the usage of this
58  feature is left to the outer layer; e.g., this class does not assert
59  that an element has been registered as missing before it is added to
60  the map.
61 
62  @note The element pointer in a key/value pair may not be NULL.
63 
64  @tparam K Key type.
65  @tparam E Element type (a Cache_element wrapping some dictionary
66  object type).
67 */
68 
69 template <typename K, typename E>
70 class Element_map {
71  public:
72  typedef std::map<K, E *, std::less<K>,
73  Malloc_allocator<std::pair<const K, E *>>>
74  Element_map_type; // Real map type.
75  typedef typename Element_map_type::const_iterator
76  Const_iterator; // Const iterator type.
77  typedef typename Element_map_type::iterator Iterator; // Iterator type.
78 
79  private:
80  Element_map_type m_map; // The real map instance.
81  std::set<K, std::less<K>,
82  Malloc_allocator<K>>
83  m_missed; // Cache misses being handled.
84 
85  public:

140  /**
141  Get the element associated with the given key.
142 
143  If the element is present in the map, return a pointer to it. If the
144  element is not present, return NULL as the element pointer.
145 
146  @param key The key to use for looking up the element.
147  @param [out] element The element associated with the key (if present),
148  or NULL (if not present).
149  */
150 
151  void get(const K &key, E **element) const {
152  DBUG_ASSERT(element);
153  typename Element_map_type::const_iterator it = m_map.find(key);
154  if (it == m_map.end())
155  *element = nullptr;
156  else {
157  DBUG_ASSERT(it->second);
158  *element = it->second;
159  }
160  }
161 
162  /**
163  Put the element into the map and associate it with the given key.
164 
165  The element may not be NULL, and the key may not be already present.
166 
167  @param key The key to be associated with the element.
168  @param element The element to be associated with the key.
169  */
170 
171  void put(const K &key, E *element) {
172  DBUG_ASSERT(element);
173  DBUG_ASSERT(m_map.find(key) == m_map.end());
174  m_map.insert(typename Element_map_type::value_type(key, element));
175  }
176 
177  /**
178  Remove an element from the map.
179 
180  The key/value pair, as indicated by the key, is removed from the map.
181  The element being pointed to is not deleted. The key must be present in
182  the map.
183 
184  @param key The key to be removed.
185  */
186 
187  void remove(const K &key) {
188  DBUG_ASSERT(m_map.find(key) != m_map.end());
189  m_map.erase(key);
190  }
191 
192  /**
193  Check if the given key has been missed in the cache.
194 
195  @param key Key representing element to check for recent cache miss.
196  @return true if the element has been missed, otherwise false.
197  */
198 
199  bool is_missed(const K &key) const {
200  return m_missed.find(key) != m_missed.end();
201  }
202 
203  /**
204  Register the given key as being missed in the cache.
205 
206  The key cannot already be reported as missing.
207 
208  @param key Key representing element missed in the cache.
209  */
210 
211  void set_missed(const K &key) {
212  DBUG_ASSERT(m_missed.find(key) == m_missed.end());
213  m_missed.insert(key);
214  }
215 
216  /**
217  Register that the miss of a key has been handled.
218 
219  The key must have been reported as missing.
220 
221  @param key Key representing element previously missed in the cache.
222  */
223 
224  void set_miss_handled(const K &key) {
225  DBUG_ASSERT(m_missed.find(key) != m_missed.end());
226  m_missed.erase(key);
227  }

229  /**

241 #endif
242  }
243  /* purecov: end */
244 };
