44 /**
45  * @brief Basic Concurrent Map
46  *
47  * The concurrent_map is a hash-map, with fixed number of buckets.
48  * The numer of buckets can be specified in constructor parameter
49  * (num_buckets), by default is set to 23.
50  */
51 template <typename Key, typename Value, typename Hash = std::hash<Key>>
52 class concurrent_map {
53  public:
54  using key_type = Key;
55  using mapped_type = Value;
56  using hash_type = Hash;
57  using value_type = typename std::map<Key, Value>::value_type;
58 
59  concurrent_map(unsigned num_buckets = kDefaultNumberOfBucket,
60  const Hash &hasher = Hash())
61  : buckets_(num_buckets), hasher_(hasher) {}
62 
63  concurrent_map(const concurrent_map &other) = delete;
64  concurrent_map &operator=(const concurrent_map &other) = delete;
65 
66  template <typename Predicate>
67  void for_one(const Key &key, Predicate &p) {
68  get_bucket(key).for_one(key, p);
69  }
70 
71  template <typename Predicate>
72  void for_each(Predicate &p) {
73  for (auto &each_bucket : buckets_) {
74  each_bucket.for_each(p);
75  }
76  }
77 
78  void put(const Key &key, Value &&value) {
79  get_bucket(key).put(key, std::move(value));
80  }
81 
82  void erase(const Key &key) { get_bucket(key).erase(key); }
83 
84  std::size_t size() const {
85  std::size_t result{0};
86  for (auto &each_bucket : buckets_) {
87  result += each_bucket.size();
88  }
89  return result;
90  }
91 
92  bool empty() const {
93  for (const auto &each_bucket : buckets_) {
94  if (!each_bucket.empty()) return false;
95  }
96  return true;
97  }
98 
99  private:
100  static const unsigned kDefaultNumberOfBucket = 127;
101 
102  class Bucket {
103  public:
104  void put(const Key &key, Value &&value) {
105  std::lock_guard<std::mutex> lock(data_mutex_);
106  data_.emplace(key, std::move(value));
107  }
108 
109  void erase(const Key &key) {
110  std::lock_guard<std::mutex> lock(data_mutex_);
111  data_.erase(key);
112  }
113 
114  template <typename Predicate>
115  void for_one(const Key &key, Predicate &p) {
116  std::lock_guard<std::mutex> lock(data_mutex_);
117  const ConstBucketIterator found = data_.find(key);
118  if (found != data_.end()) p(found->second);
119  }
120 
121  template <typename Predicate>
122  void for_each(Predicate &p) {
123  std::lock_guard<std::mutex> lock(data_mutex_);
124  std::for_each(data_.begin(), data_.end(), p);
125  }
126 
127  std::size_t size() const {
128  std::lock_guard<std::mutex> lock(data_mutex_);
129  return data_.size();
130  }
131 
132  bool empty() const {
133  std::lock_guard<std::mutex> lock(data_mutex_);
134  return data_.empty();
135  }
136 
137  private:
138  using BucketData = std::map<Key, Value>;
139  using BucketIterator = typename BucketData::iterator;
140  using ConstBucketIterator = typename BucketData::const_iterator;
141 
142  BucketData data_;
143  mutable std::mutex data_mutex_;
144  };
145 
146  std::vector<Bucket> buckets_;
147  Hash hasher_;
148 
149  Bucket &get_bucket(const Key &key) {
150  const std::size_t bucket_index = hasher_(key) % buckets_.size();
151  return buckets_[bucket_index];
152  }
153 
154  const Bucket &get_bucket(const Key &key) const {
155  const std::size_t bucket_index = hasher_(key) % buckets_.size();
156  return buckets_[bucket_index];
157  }
158 };

160 /**
161  * @brief container for connections to MySQL Server.
162  *
163  * When thread of execution for connection to MySQL Server is completed, it
164  * should call remove_connection to remove itself from connection container.
165  */
166 class ConnectionContainer {
167  concurrent_map<MySQLRoutingConnectionBase *,
168  std::unique_ptr<MySQLRoutingConnectionBase>>
169  connections_;
170 
171  public:
172  using ConnData = MySQLRoutingAPI::ConnData;
173 
174  std::vector<ConnData> get_all_connections_info() {
175  std::vector<ConnData> connection_datas;
176 
177  auto l =
178  [&connection_datas](const decltype(connections_)::value_type &conn) {
179  connection_datas.push_back({
180  conn.second->get_client_address(),
181  conn.second->get_server_address(),
182  conn.second->get_bytes_up(),
183  conn.second->get_bytes_down(),
184  conn.second->get_started(),
185  conn.second->get_connected_to_server(),
186  conn.second->get_last_sent_to_server(),
187  conn.second->get_last_received_from_server(),
188  });
189  };
190  connections_.for_each(l);
191  return connection_datas;
192  }
193  /**
194  * @brief Adds new connection to container.
195  *
196  * @param connection The connection to MySQL server
197  */
198  void add_connection(std::unique_ptr<MySQLRoutingConnectionBase> connection);
199 
200  /**
201  * @brief Disconnects all connections to servers that are not allowed any
202  * longer.
203  *
204  * @param nodes Allowed servers. Connections to servers that are not in nodes
205  * are closed.
206  */
207  void disconnect(const AllowedNodes &nodes);
208 
209  /**
210  * @brief Disconnects all connection in the ConnectionContainer.
211  */
212  void disconnect_all();
213 
214  /**
215  * @brief removes connection from container
216  *
217  * This function should be called by thread of execution when connection
218  * thread completes. Do NOT call this function before connection's thread of
219  * execution completes.
220  *
221  * @param connection The connection to remove from container
222  */
223  void remove_connection(MySQLRoutingConnectionBase *connection);
224 
225  /** number of active client threads. */
226  std::condition_variable connection_removed_cond_;
227  std::mutex connection_removed_cond_m_;
228 
229  /**
230  * check if container is empty.
231  *
232  * as the map is concurrent, empty() only gives a reasonable result if it is
233  * ensured no other thread is currently adding connections.
234  */
235  bool empty() const { return connections_.empty(); }
236 };
