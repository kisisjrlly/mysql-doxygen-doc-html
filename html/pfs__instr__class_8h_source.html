143 /** Information for all instrumentation. */
144 struct PFS_instr_class {
145  /** Class type */
146  PFS_class_type m_type;
147  /** True if this instrument is enabled. */
148  bool m_enabled;
149  /** True if this instrument is timed. */
150  bool m_timed;
151  /** Instrument flags. */
152  uint m_flags;
153  /** Volatility index. */
154  int m_volatility;
155  /**
156  Instrument name index.
157  Self index in:
158  - EVENTS_WAITS_SUMMARY_*_BY_EVENT_NAME for waits
159  - EVENTS_STAGES_SUMMARY_*_BY_EVENT_NAME for stages
160  - EVENTS_STATEMENTS_SUMMARY_*_BY_EVENT_NAME for statements
161  - EVENTS_TRANSACTIONS_SUMMARY_*_BY_EVENT_NAME for transactions
162  */
163  uint m_event_name_index;
164  /** Instrument name. */
165  char m_name[PFS_MAX_INFO_NAME_LENGTH];
166  /** Length in bytes of @c m_name. */
167  uint m_name_length;
168  /** Documentation. */
169  char *m_documentation;
170 
171  bool is_singleton() const { return m_flags & PSI_FLAG_SINGLETON; }
172 
173  bool is_mutable() const { return m_flags & PSI_FLAG_MUTABLE; }
174 
175  bool is_progress() const { return m_flags & PSI_FLAG_STAGE_PROGRESS; }
176 
177  bool is_shared_exclusive() const { return m_flags & PSI_FLAG_RWLOCK_SX; }
178 
179  bool is_priority() const { return m_flags & PSI_FLAG_RWLOCK_PR; }
180 
181  bool is_transferable() const { return m_flags & PSI_FLAG_TRANSFER; }
182 
183  bool is_user() const { return m_flags & PSI_FLAG_USER; }
184 
185  bool is_system_thread() const { return m_flags & PSI_FLAG_THREAD_SYSTEM; }
186 
187  bool is_global() const { return m_flags & PSI_FLAG_ONLY_GLOBAL_STAT; }
188 
189  void enforce_valid_flags(uint allowed_flags) {
190  /* Reserved for future use. */
191  allowed_flags |= PSI_FLAG_THREAD | PSI_FLAG_TRANSFER;
192 
193  uint valid_flags = m_flags & allowed_flags;
194  /*
195  This fails when the instrumented code is providing
196  flags that are not supported for this instrument.
197  To fix it, clean up the instrumented code.
198  */
199  DBUG_ASSERT(valid_flags == m_flags);
200  m_flags = valid_flags;
201  }

204  static void set_timed(PFS_instr_class *pfs, bool timed);

206  bool is_deferred() const {
207  switch (m_type) {
208  case PFS_CLASS_SOCKET:
209  return true;
210  break;
211  default:
212  return false;
213  break;
214  };
215  }
216 
217  bool can_be_timed() const {
218  switch (m_type) {
219  case PFS_CLASS_MEMORY:
220  case PFS_CLASS_ERROR:
221  case PFS_CLASS_THREAD:
222  return false;
223  default:
224  return true;
225  };
226  }
