38 /** Storage container. This mimics std::vector and std::deque with the
39  * difference that the size of the objects that are being stored in it can be
40  * determined at runtime. Elements are stored next to each other in a
41  * pre-allocated pages of fixed size `STORAGE_PAGE_SIZE`. */
42 class Storage {
43  public:
44  /** Type used for elements. We treat elements as black boxes. */
45  typedef void Element;
46  /** Type used for pages. */
47  typedef void Page;

185  private:
186  /** Align elements to this number of bytes. */
187  static constexpr size_t ALIGN_TO = alignof(void *);

189  /** Flag that denotes an element is the first element on a page. */

367 inline Storage::Iterator &Storage::Iterator::operator++() {
368  DBUG_ASSERT(m_storage != nullptr);
369  DBUG_ASSERT(*this != m_storage->end());
370 
371  do {
372  if (m_storage->element_last_on_page(m_element)) {
373  Page *next_page = *m_storage->element_next_page_ptr(m_element);
374  if (next_page == nullptr) {
375  /* Last element on last page, can't go further. */
376  *this = m_storage->end();
377  return *this;
378  }
379  m_element = m_storage->first_possible_element_on_page(next_page);
380  } else {
381  m_element = m_storage->next_element(m_element);
382  }
383  } while (m_storage->element_deleted(m_element));
384 
385  return *this;
386 }
387 
388 inline Storage::Iterator &Storage::Iterator::operator--() {
389  DBUG_ASSERT(m_storage != nullptr);
390  DBUG_ASSERT(*this != m_storage->begin());
391 
392  /* Since *this != m_storage->begin() there is at least one non-deleted element
393  * preceding our position (ie the one pointed to by begin()). */
394 
395  do {
396  if (m_element == nullptr) {
397  m_element = m_storage->back();
398  } else if (m_storage->element_first_on_page(m_element)) {
399  /* Go to the last element on the previous page. */
400  Page *prev_page = *m_storage->element_prev_page_ptr(m_element);
401  DBUG_ASSERT(prev_page != nullptr);
402  m_element = m_storage->last_possible_element_on_page(prev_page);
403  DBUG_ASSERT(m_storage->element_last_on_page(m_element));
404  } else {
405  m_element = m_storage->prev_element(m_element);
406  }
407  } while (m_storage->element_deleted(m_element));
408 
409  return *this;
410 }

461 inline Storage::Iterator Storage::begin() const {
462  if (m_number_of_elements == 0) {
463  return end();
464  }
465 
466  DBUG_ASSERT(m_first_page != nullptr);
467

469

485 inline void Storage::element_size(size_t element_size) {
486  DBUG_ASSERT(m_number_of_elements == 0);
487 
488  m_element_size = element_size;
489 
490  const size_t element_size_plus_meta = element_size + META_BYTES_PER_ELEMENT;
491 
492  /* Confirm that ALIGN_TO is a power of 2 (or zero). */
493  DBUG_ASSERT(ALIGN_TO == 0 || (ALIGN_TO & (ALIGN_TO - 1)) == 0);
494 
495  /* The next multiple of ALIGN_TO from element_size_plus_meta. */
496  m_bytes_used_per_element =
497  (element_size_plus_meta + ALIGN_TO - 1) & ~(ALIGN_TO - 1);
498 
499  m_number_of_elements_per_page =
500  (STORAGE_PAGE_SIZE - META_BYTES_PER_PAGE) / m_bytes_used_per_element;
501 }

509 inline Storage::Element *Storage::allocate_back() {
510  if (m_last_page == nullptr) {
511  /* The storage is empty, create the first page. */
512  DBUG_ASSERT(m_first_page == nullptr);
513 
514  m_first_page = m_allocator->allocate(page_size());
515 
516  *page_prev_page_ptr(m_first_page) = nullptr;
517  *page_next_page_ptr(m_first_page) = nullptr;
518 
519  m_last_page = m_first_page;
520 
521  m_last_element = first_possible_element_on_page(m_first_page);
522 
523 #ifndef DBUG_OFF
524  *element_meta(m_last_element) = 0;
525 #endif /* DBUG_OFF */
526 
527  element_first_on_page(true, m_last_element);
528  } else if (m_last_element == last_possible_element_on_page(m_last_page)) {
529  /* Last page is full, create a new one. */
530  Page *new_page = m_allocator->allocate(page_size());
531 
532  *page_next_page_ptr(m_last_page) = new_page;
533  *page_prev_page_ptr(new_page) = m_last_page;
534  *page_next_page_ptr(new_page) = nullptr;
535 
536  m_last_page = new_page;
537  m_last_element = first_possible_element_on_page(new_page);
538 
539 #ifndef DBUG_OFF
540  *element_meta(m_last_element) = 0;
541 #endif /* DBUG_OFF */
542 
543  element_first_on_page(true, m_last_element);
544  } else {
545  element_last_on_page(false, m_last_element);
546 
547  m_last_element = next_element(m_last_element);
548 
549 #ifndef DBUG_OFF
550  *element_meta(m_last_element) = 0;
551 #endif /* DBUG_OFF */
552 
553  element_first_on_page(false, m_last_element);
554  }
555 
556  ++m_number_of_elements;
557 
558  element_last_on_page(true, m_last_element);
559  element_deleted(false, m_last_element);
560  *element_next_page_ptr(m_last_element) = nullptr;
561 
562  return m_last_element;
563 }

565 inline void Storage::deallocate_back() {
566  DBUG_ASSERT(m_number_of_elements > 0);
567 
568  --m_number_of_elements;
569 
570  do {
571  if (m_last_element != first_possible_element_on_page(m_last_page)) {
572  m_last_element = prev_element(m_last_element);
573  } else if (m_first_page == m_last_page) {
574  DBUG_ASSERT(m_number_of_elements == 0);
575  m_allocator->deallocate(static_cast<uint8_t *>(m_first_page),
576  page_size());
577  m_first_page = nullptr;
578  m_last_page = nullptr;
579  m_last_element = nullptr;
580  return;
581  } else {
582  Page *page_to_free = m_last_page;
583 
584  m_last_page = *page_prev_page_ptr(m_last_page);
585  *page_next_page_ptr(m_last_page) = nullptr;
586 
587  m_last_element = last_possible_element_on_page(m_last_page);
588 
589  m_allocator->deallocate(static_cast<uint8_t *>(page_to_free),
590  page_size());
591  }
592  } while (element_deleted(m_last_element));
593 
594  element_last_on_page(true, m_last_element);
595  *element_next_page_ptr(m_last_element) = nullptr;
596 }

598 inline Storage::Iterator Storage::erase(const Iterator &position) {
599  Iterator next_element_position = position;
600  ++next_element_position;
601 
602  if (*position == m_last_element) {
603  deallocate_back();
604  } else {
605  DBUG_ASSERT(m_number_of_elements > 0);
606  --m_number_of_elements;
607 
608  element_deleted(true, *position);
609  }
610 
611  return next_element_position;
612 }

614 inline void Storage::clear() {
615  if (m_first_page == nullptr) {
616  DBUG_ASSERT(m_number_of_elements == 0);
617  return;
618  }
619 
620  if (m_first_page == m_last_page) {
621  DBUG_ASSERT(m_number_of_elements <= m_number_of_elements_per_page);
622  } else {
623  Page *p = m_first_page;
624  do {
625  p = *page_next_page_ptr(p);
626 
627  m_allocator->deallocate(static_cast<uint8_t *>(*page_prev_page_ptr(p)),
628  page_size());
629  } while (p != m_last_page);
630  }
631 
632  m_allocator->deallocate(static_cast<uint8_t *>(m_last_page), page_size());
633 
634  m_first_page = nullptr;
635  m_last_page = nullptr;
636  m_number_of_elements = 0;
637 }
