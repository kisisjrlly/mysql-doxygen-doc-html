76 /** Memory-block abstraction whose purpose is to serve as a building block
77  * for custom memory-allocator implementations.
78  *
79  * TL;DR How it works:
80  * Instantiation:
81  * - With given size and given memory source, Block will allocate memory
82  * and adjust its Header metadata with the relevant information.
83  * Allocation:
84  * - From allocated memory space, Block finds out what is the next
85  * available slot to fit the new Chunk into.
86  * - Creates a new Chunk with the address pointing to that slot.
87  * - Increments the number of allocated chunks.
88  * - Returns a Chunk.
89  * Deallocation:
90  * - Decrements the number of allocated chunks.
91  * - Returns current number of allocated chunks.
92  * Destruction:
93  * - Simply deallocates the memory.
94  *
95  * Now, more detailed description ...
96  *
97  * Normally, custom memory-allocators will feed clients' memory allocation
98  * and deallocation requests solely through the provided Block interface which
99  * enables allocators not to worry about the whole lot of low-level memory
100  * byte-juggling but to focus on application-level details.
101  *
102  * Block, once created, will occupy at-least (see below why) the specified
103  * amount of memory after which it will be able to serve client-requested
104  * allocations and deallocations in logical units called Chunks. Chunk is
105  * an arbitrarily-sized view over a region of memory allocated during the
106  * Block creation. Block can fit as many Chunks as there is free memory space
107  * left in it. Once there is no free space left, another Block of memory has
108  * to be created. Block is not resizeable. E.g. 4KB-sized Block can feed 1x4KB,
109  * 2x2KB, 1KB+3KB or any other combination of Chunks whose total size does not
110  * exceed the Block size (4KB).
111  *
112  * Organizing Block memory into Chunks is implementation house-keeping detail
113  * stored in its Header metadata region. Block does not maintain the list of
114  * Chunks, it only ever keeps the number of currently allocated Chunks and
115  * the offset to the first memory location available to feed the next
116  * allocation request.
117  *
118  * While still using the same interface, custom memory-allocators are able to
119  * choose where should the Block allocate actual memory from. It could be
120  * anything defined by Source but currently only RAM and MMAP-ed files are
121  * available and implemented as options.
122  *
123  * For the benefit of (amortized) constant-time allocations, Block does not
124  * re-use or do any other special operations over deallocated Chunks so
125  * memory-allocators which will be using it may suffer from block-level
126  * memory-fragmentation and consequently higher memory-consumption. Exceptions
127  * are deallocations of first and last Chunks in a Block when it is possible
128  * to easily re-adjust the offset and therefore be able to re-use that part
129  * of memory.
130  *
131  * Another big advantage, which is very closely related to constant-time
132  * allocations, is that it minimizes the number of system-calls required to
133  * allocate and deallocate the memory which consequently may lower the
134  * process-level memory-fragmentation.
135  *
136  * Block size does not necessarily end-up being the size originally requested by
137  * the client but it will be actually implicitly rounded to the next multiple
138  * of CPU word-size which may result in better memory utilization. Actual block
139  * size can be queried through the Block interface.
140  *
141  * To optimize for the CPU memory-access, but also to enable code not to
142  * segfault on architectures which do not support unaligned-memory-access
143  * (e.g. SPARC), Block will always adjust requested Chunk allocation size to
144  * match the size which is rounded to the next multiple of CPU word-size
145  * (Block::ALIGN_TO constant). End result is that Block might end up allocating
146  * just a few more bytes bigger Chunk than actually requested but that
147  * information, however, does not need to be maintained or cared about by the
148  * client code.
149  *
150  * Along with the small space overhead due to the automatic word-size-adjustment
151  * of Chunk size, each Block allocation will also have a few bytes overhead for
152  * maintaining the Header metadata (Header::SIZE) as well as for maintaining the
153  * Chunk metadata (Chunk::METADATA_SIZE). Implementation and data layout details
154  * can be found at respective header file declarations.
155  *
156  * All dirty-implementation details are hidden in Header implementation
157  * which makes sure that proper care is taken to handle chunk offsets,
158  * available slots, number of present chunks etc. */

276 static inline uint8_t *allocate_from(Source src, size_t size) {
277  uint8_t *ptr = nullptr;
278  if (src == Source::RAM) {
279  ptr = static_cast<uint8_t *>(Memory<Source::RAM>::allocate(size));
280  Block_PSI_track_physical_ram_allocation(size);
281  } else if (src == Source::MMAP_FILE) {
282  ptr = static_cast<uint8_t *>(Memory<Source::MMAP_FILE>::allocate(size));
283  Block_PSI_track_physical_disk_allocation(size);
284  }
285  return ptr;
286 }
287 
288 static inline void deallocate_from(Source src, size_t size,
289  uint8_t *block_address) {
290  if (src == Source::RAM) {
291  Block_PSI_track_physical_ram_deallocation(size);
292  Memory<Source::RAM>::deallocate(block_address, size);
293  } else if (src == Source::MMAP_FILE) {
294  Block_PSI_track_physical_disk_deallocation(size);
295  Memory<Source::MMAP_FILE>::deallocate(block_address, size);
296  }
297 }

311 inline Block::Block(uint8_t *block_memory, Source block_memory_type,
312  size_t block_size) noexcept
313  : Header(block_memory, block_memory_type, block_size) {
314  DBUG_ASSERT(!is_empty());

316  /* Prevent writes to the memory which we took from the OS but still have
317  * not shipped outside of the Allocator. This will also prevent reads, but
318  * reads would have been reported even without this because the memory we
319  * took from the OS is "undefined" by default. */
320  MEM_NOACCESS(Header::block_address() + Header::SIZE,
321  block_size - Header::SIZE);

335 inline Chunk Block::allocate(size_t chunk_size) noexcept {
336  DBUG_ASSERT(!is_empty());
337  DBUG_ASSERT(can_accommodate(chunk_size));
338 
339  const size_t chunk_size_aligned = Block::aligned_size(chunk_size);
340 
341  /* Remove the "no access" flag we set on this memory during block
342  * creation. Relax it to report read+depend_on_contents. */
343  MEM_UNDEFINED(Header::next_available_slot(),
344  Chunk::size_hint(chunk_size_aligned));
345 
346  Chunk chunk{Header::next_available_slot(), Header::first_pristine_offset()};
347  Header::increment_number_of_used_chunks(Chunk::size_hint(chunk_size_aligned));
348 
349  Block_PSI_track_logical_allocation(chunk_size_aligned);
350  DBUG_PRINT("temptable_allocator",
351  ("allocate from block: chunk_size=%zu, from_block=(%s); "
352  "return=%p",
353  chunk_size, to_string().c_str(), chunk.data()));
354 
355  return chunk;
356 }

358 inline size_t Block::deallocate(Chunk chunk, size_t chunk_size) noexcept {
359  DBUG_ASSERT(!is_empty());
360  DBUG_PRINT("temptable_allocator",
361  ("deallocate from block: size=%zu, from_block=(%s), chunk_data=%p",
362  chunk_size, to_string().c_str(), chunk.data()));
363 
364  const size_t chunk_size_aligned = Block::aligned_size(chunk_size);
365  Block_PSI_track_logical_deallocation(chunk_size_aligned);
366 
367  return Header::decrement_number_of_used_chunks(
368  Chunk::size_hint(chunk_size_aligned),
369  is_rightmost_chunk(chunk, Chunk::size_hint(chunk_size_aligned)));
370 }

372 inline void Block::destroy() noexcept {
373  DBUG_ASSERT(!is_empty());
374  DBUG_ASSERT(Header::number_of_used_chunks() == 0);
375  DBUG_PRINT("temptable_allocator",
376  ("destroying the block: (%s)", to_string().c_str()));
377 
378  deallocate_from(Header::memory_source_type(), Header::block_size(),
379  Header::block_address());
380  Header::reset();
381 }

387 inline bool Block::can_accommodate(size_t n_bytes) const {
388  DBUG_ASSERT(!is_empty());
389 
390  const size_t n_bytes_aligned = Block::aligned_size(n_bytes);
391  const size_t block_size = Header::block_size();
392  const size_t first_pristine_offset = Header::first_pristine_offset();
393  DBUG_ASSERT(first_pristine_offset <=
394  std::numeric_limits<decltype(block_size)>::max() -
395  Chunk::size_hint(n_bytes_aligned));
396 
397  return first_pristine_offset + Chunk::size_hint(n_bytes_aligned) <=
398  block_size;
399 }
