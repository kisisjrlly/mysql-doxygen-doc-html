69 /**
70  Public function builder interface.
71  The parser (sql/sql_yacc.yy) uses a factory / builder pattern to
72  construct an <code>Item</code> object for each function call.
73  All the concrete function builders implements this interface,
74  either directly or indirectly with some adapter helpers.
75  Keeping the function creation separated from the bison grammar allows
76  to simplify the parser, and avoid the need to introduce a new token
77  for each function, which has undesirable side effects in the grammar.
78 */
79 
80 class Create_func {
81  public:
82  /**
83  The builder create method.
84  Given the function name and list or arguments, this method creates
85  an <code>Item</code> that represents the function call.
86  In case or errors, a NULL item is returned, and an error is reported.
87  Note that the <code>thd</code> object may be modified by the builder.
88  In particular, the following members/methods can be set/called,
89  depending on the function called and the function possible side effects.
90  <ul>
91  <li><code>thd->lex->binlog_row_based_if_mixed</code></li>
92  <li><code>thd->lex->current_context()</code></li>
93  <li><code>thd->lex->safe_to_cache_query</code></li>
94  <li><code>thd->lex->uncacheable(UNCACHEABLE_SIDEEFFECT)</code></li>
95  <li><code>thd->lex->uncacheable(UNCACHEABLE_RAND)</code></li>
96  <li><code>thd->lex->add_time_zone_tables_to_query_tables(thd)</code></li>
97  </ul>
98  @param thd The current thread
99  @param name The function name
100  @param item_list The list of arguments to the function, can be NULL
101  @return An item representing the parsed function call, or NULL
102  */
103  virtual Item *create_func(THD *thd, LEX_STRING name,
104  PT_item_list *item_list) = 0;
105 
106  protected:
107  Create_func() = default;
108  virtual ~Create_func() {}
109 };

117 class Create_qfunc : public Create_func {

125  @return An item representing the parsed function call
126  */
127  Item *create_func(THD *thd, LEX_STRING name,

129 
130  /**
131  The builder create method, for qualified functions.
132  @param thd The current thread
133  @param db The database name or NULL_STR to use the default db name
134  @param name The function name
135  @param use_explicit_name Should the function be represented as 'db.name'?
136  @param item_list The list of arguments to the function, can be NULL
137  @return An item representing the parsed function call
138  */
139  virtual Item *create(THD *thd, LEX_STRING db, LEX_STRING name,
140  bool use_explicit_name, PT_item_list *item_list) = 0;
141 
142  protected:

144  Create_qfunc() {}

147 };
