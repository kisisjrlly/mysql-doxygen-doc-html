130 /**
131  * @brief MySQLRoutingConnection represents a connection to MYSQL Server.
132  *
133  * The MySQLRoutingConnection wraps thread that handles traffic from one socket
134  * to another.
135  *
136  * When instance of MySQLRoutingConnection object is created, remove callback
137  * is passed, which is called when connection thread completes.
138  */
139 template <class ClientProtocol, class ServerProtocol>
140 class MySQLRoutingConnection : public MySQLRoutingConnectionBase {
141  public:

145  /**
146  * @brief Creates and initializes connection object. It doesn't create
147  * new thread of execution. In order to create new thread of
148  * execution call start().
149  *
150  * @param context wrapper for common data used by all connection threads
151  * @param destination_id identifier of the destination connected to
152  * @param client_socket socket used to send/receive data to/from client
153  * @param client_endpoint address of the socket used to send/receive data
154  * to/from client
155  * @param server_socket socket used to send/receive data to/from server
156  * @param server_endpoint IP address and TCP port of MySQL Server
157  * @param remove_callback called when thread finishes its execution to remove
158  * associated MySQLRoutingConnection from container. It must be called
159  * at the very end of thread execution

161  MySQLRoutingConnection(
162  MySQLRoutingContext &context, std::string destination_id,

237 template <class ClientProtocol, class ServerProtocol>
238 class Splicer : public std::enable_shared_from_this<
239  Splicer<ClientProtocol, ServerProtocol>> {
240  public:
241  enum class State {
242  SPLICE,
243  FINISH,
244  DONE,
245  };
246 
247  using client_protocol = ClientProtocol;
248  using server_protocol = ServerProtocol;
249 
250  Splicer(MySQLRoutingConnection<ClientProtocol, ServerProtocol> *conn,
251  const size_t net_buffer_size)
252  : conn_{conn} {
253  conn_->client_buffer().resize(net_buffer_size);
254  conn_->server_buffer().resize(net_buffer_size);
255  }
256 
257  ~Splicer() {
258  if (state_ != State::DONE) std::terminate();
259 
260  conn_->disassociate();
261  }
262 
263  State copy_client_to_server() {
264  // Handle traffic from Client to Server
265  auto copy_res = conn_->context().get_protocol().copy_packets(
266  conn_->client_socket().native_handle(),
267  conn_->server_socket().native_handle(), true, conn_->server_buffer(),
268  &pktnr_, handshake_done_, false);
269  if (!copy_res) {
270  if (copy_res.error() == std::errc::resource_unavailable_try_again) {
271  return State::SPLICE;
272  } else if (copy_res.error() != net::stream_errc::eof) {
273  extra_msg_ =
274  "Copy client->server failed: " + copy_res.error().message();
275  } else if (!handshake_done_) {
276  extra_msg_ = "Copy client->server failed: unexpected connection close";
277  }
278  // client close on us.
279  return State::FINISH;
280  } else {
281  conn_->transfered_to_server(copy_res.value());
282  }
283 
284  return State::SPLICE;
285  }
286 
287  State copy_server_to_client() {
288  // Handle traffic from Server to Client
289  // Note: In classic protocol Server _always_ talks first
290  const auto copy_res = conn_->context().get_protocol().copy_packets(
291  conn_->server_socket().native_handle(),
292  conn_->client_socket().native_handle(), true, conn_->client_buffer(),
293  &pktnr_, handshake_done_, true);
294 
295  if (!copy_res) {
296  if (copy_res.error() == std::errc::resource_unavailable_try_again) {
297  return State::SPLICE;
298  } else if (copy_res.error() != net::stream_errc::eof) {
299  extra_msg_ =
300  "Copy server->client failed: " + copy_res.error().message();
301  }
302 
303  return State::FINISH;
304  } else {
305  conn_->transfered_to_client(copy_res.value());
306  }
307 
308  // after a successful handshake, we reset client-side connection error
309  // counter, just like the Server
310  if (!error_counter_already_cleared_ && handshake_done_) {
311  conn_->context().template clear_error_counter<ClientProtocol>(
312  conn_->client_endpoint());
313  error_counter_already_cleared_ = true;
314  }
315 
316  return State::SPLICE;
317  }
318 
319  template <bool to_server>
320  void transfer(std::error_code ec) {
321  if (ec == std::errc::operation_canceled) {
322  if (state() != State::DONE) state(finish());
323  return;
324  }
325 
326  state(to_server ? copy_client_to_server() : copy_server_to_client());
327 
328  switch (state()) {
329  case State::SPLICE:
330  if (to_server) {
331  async_wait_client();
332  } else {
333  async_wait_server();
334  }
335  break;
336  case State::FINISH:
337  state(finish());
338  break;
339  case State::DONE:
340  break;
341  }
342  }
343 
344  void async_wait_client() {
345  conn_->client_socket().async_wait(
346  net::socket_base::wait_read,
347  std::bind(&Splicer<ClientProtocol, ServerProtocol>::transfer<true>,
348  this->shared_from_this(), std::placeholders::_1));
349  }
350 
351  void async_wait_server() {
352  conn_->server_socket().async_wait(
353  net::socket_base::wait_read,
354  std::bind(&Splicer<ClientProtocol, ServerProtocol>::transfer<false>,
355  this->shared_from_this(), std::placeholders::_1));
356  }
357 
358  void async_run() {
359  conn_->connected();
360 
361  async_wait_client();
362  async_wait_server();
363  }
364 
365  private:
366  void state(State st) { state_ = st; }
367 
368  State state() { return state_; }
369 
370  State finish() {
371  if (!handshake_done_) {
372  harness_assert(!error_counter_already_cleared_);
373 
374  log_info("[%s] fd=%d Pre-auth socket failure %s: %s",
375  conn_->context().get_name().c_str(),
376  conn_->client_socket().native_handle(),
377  mysqlrouter::to_string(conn_->client_endpoint()).c_str(),
378  extra_msg_.c_str());
379  conn_->context().template block_client_host<ClientProtocol>(
380  conn_->client_endpoint(), conn_->server_socket().native_handle());
381  }
382 
383  // Either client or server terminated
384 
385  log_debug("[%s] fd=%d -- %d: connection closed (up: %zub; down: %zub) %s",
386  conn_->context().get_name().c_str(),
387  conn_->client_socket().native_handle(),
388  conn_->server_socket().native_handle(), conn_->get_bytes_up(),
389  conn_->get_bytes_down(), extra_msg_.c_str());
390 
391  if (conn_->client_socket().is_open()) {
392  conn_->client_socket().shutdown(net::socket_base::shutdown_send);
393  conn_->client_socket().close();
394  }
395 
396  if (conn_->server_socket().is_open()) {
397  conn_->server_socket().shutdown(net::socket_base::shutdown_send);
398 
399  conn_->server_socket().close();
400  }
401 
402  conn_->context().decrease_info_active_routes();
403 
404  return State::DONE;
405  }
406 
407  MySQLRoutingConnection<ClientProtocol, ServerProtocol> *conn_;
408 
409  bool handshake_done_{false};
410  bool error_counter_already_cleared_{false};
411  std::string extra_msg_;
412  int pktnr_{};
413 
414  State state_{State::SPLICE};
415 };
