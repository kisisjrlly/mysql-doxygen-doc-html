46  void report_error(uint code, const char *fmt, ...) override
47  MY_ATTRIBUTE((format(printf, 3, 4))) {
48  va_list args;
49  va_start(args, fmt);
50 
51  if (code == 0)
52  LogEvent()
53  .prio(WARNING_LEVEL)
54  .errcode(ER_SERVER_TABLE_CHECK_FAILED)
55  .subsys(LOG_SUBSYSTEM_TAG)
56  .source_file(MY_BASENAME)
57  .messagev(fmt, args);
58  else if (code == ER_CANNOT_LOAD_FROM_TABLE_V2) {
59  char *db_name, *table_name;
60  db_name = va_arg(args, char *);
61  table_name = va_arg(args, char *);
62  my_error(code, MYF(0), db_name, table_name);
63  LogErr(m_log_level, ER_SERVER_CANNOT_LOAD_FROM_TABLE_V2, db_name,
64  table_name);
65  } else {
66  my_printv_error(code, ER_THD_NONCONST(m_thd, code), MYF(0), args);
67  va_end(args);
68 
69  if (code == ER_COL_COUNT_DOESNT_MATCH_PLEASE_UPDATE_V2)
70  code = ER_SERVER_COL_COUNT_DOESNT_MATCH_PLEASE_UPDATE_V2;
71  else if (code == ER_COL_COUNT_DOESNT_MATCH_CORRUPTED_V2)
72  code = ER_SERVER_COL_COUNT_DOESNT_MATCH_CORRUPTED_V2;
73  else
74  code = ER_SERVER_ACL_TABLE_ERROR;
75 
76  va_start(args, fmt);
77  LogEvent()
78  .prio(m_log_level)
79  .errcode(code)
80  .subsys(LOG_SUBSYSTEM_TAG)
81  .source_file(MY_BASENAME)
82  .messagev(fmt, args);
83  }
84 
85  va_end(args);
86  }
87
