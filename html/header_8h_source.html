37 /** Header is an abstraction with the purpose of holding and maintaining
38  * the Block metadata.
39  *
40  * Block metadata has the following structure:
41  * - N bytes for the block type which indicates where the block
42  * memory was allocated from.
43  * - N bytes for the block size (set at block creation and never
44  * changed later).
45  * - N bytes for the number of used/allocated chunks from this
46  * block (set to 0 at block creation).
47  * - N bytes for the offset of the first byte relative to the
48  * block start that is free and can be used by the next allocation request
49  * (set to 4 * N at block creation). We call this first pristine offset.
50  *
51  * That implies Chunks occupy the following range in memory:
52  * - bytes [4 * N, block size) a sequence of chunks appended to each other.
53  *
54  * 1st byte of [0, N) region is an actual pointer returned by memory
55  * allocation functions (e.g. malloc/new/mmap/etc.). Given that we are working
56  * with contiguous memory, storing that byte (offset) is just enough to be
57  * able to build-up and deduce Header metadata structure.
58  *
59  * Part of the Header contract is to have its metadata properly aligned in
60  * memory. Given that this memory is provided by the Block, Header
61  * implements debug-asserts to actually check if this condition has been met.
62  * If that was not the case, then accessing unaligned memory addresses would:
63  * 1. Incur performance penalty cost on architectures which can
64  * handle misaligned memory access (e.g. x86).
65  * 2. Result with a CPU trap (exception) on architectures which
66  * cannot handle misaligned memory access (e.g. SPARC).
67  *
68  * In order to maintain proper memory alignment of the whole metadata structure,
69  * CPU word-size data-type is used. Our N is defined by the size of that type
70  * (Header::metadata_type).
71  * */
72 class Header {
73  public:
74  /** Type that we will be using for storing metadata information. */
75  using metadata_type = uintptr_t;

77  /** Block header (metadata) size. As described, there are 4 elements. */
78  static constexpr size_t SIZE = 4 * sizeof(Header::metadata_type);

100  size_t first_pristine_offset() const;
101 
102  protected:
103  /** Default constructor which creates an empty Header. */
104  Header() noexcept;

106  /** Constructor which initializes the Header metadata when
107  * constructing fresh Blocks.
108  *
109  * [in] Pointer to the allocated Block memory.
110  * [in] Source where Block has allocated actual memory from.

112  Header(uint8_t *block_memory, Source block_memory_type,

116  * already existing Blocks in memory (e.g. ones that are
117  * deduced from Chunks).

119  * [in] Pointer to the existing Block. */

122  /** Enable Block to get the next available slot that it can use for next
123  * Chunk allocation.
124  *
125  * @return An absolute memory-location offset */

129  *
130  * @return An address where Block was allocated from. */
131  uint8_t *block_address() const;
132 
133  /** Enable Block to increment the reference-count when (logically)
134  * allocating new Chunks.
135  *
136  * [in] Size of the Chunk.
137  * @return New number of Chunks used/allocated by the Block after this
138  * operation. */
139  size_t increment_number_of_used_chunks(size_t chunk_size);
140 
141  /** Enable Block to decrement the reference-count when (logically)
142  * deallocating existing Chunks.
143  *
144  * [in] Size of the Chunk.
145  * [in] Boolean which denotes if Chunk being deallocated is the
146  * last (rightmost) one.
147  * @return New number of Chunks used/allocated by the Block after this
148  * operation. */
149  size_t decrement_number_of_used_chunks(size_t chunk_size,
150  bool rightmost_chunk);
151 
152  /** Enable Block to reset the Header metadata upon Block destruction. */
153  void reset();
154 
155  private:
156  /** Deduce a pointer to the memory type of given Block. */
157  static Header::metadata_type *block_memory_type_ptr(uint8_t *block);
158  /** Deduce a pointer to the size of given Block. */
159  static Header::metadata_type *block_size_ptr(uint8_t *block);
160  /** Deduce a pointer to the number of used/allocated Chunks of given Block. */
161  static Header::metadata_type *block_number_of_used_chunks_ptr(uint8_t *block);
162  /** Deduce a pointer to the first-pristine-offset of given Block. */
163  static Header::metadata_type *block_first_pristine_offset_ptr(uint8_t *block);
164 
165  private:
166  /** A pointer to the allocated Block memory which is used to deduce all
167  * of the other remaining metadata structure. */
168  uint8_t *m_offset;
169 };

211 inline size_t Header::increment_number_of_used_chunks(size_t chunk_size) {
212  *block_first_pristine_offset_ptr(m_offset) += chunk_size;
213  return ++*block_number_of_used_chunks_ptr(m_offset);
214 }
215 
216 inline size_t Header::decrement_number_of_used_chunks(size_t chunk_size,
217  bool rightmost_chunk) {
218  DBUG_ASSERT(*block_number_of_used_chunks_ptr(m_offset) > 0);
219  if (--*block_number_of_used_chunks_ptr(m_offset) == 0) {
220  /* If we are freeing the leftmost chunk in this block, then
221  * first_pristine_offset mark can be reset, so that the memory region
222  * can be reused.
223  */
224  *block_first_pristine_offset_ptr(m_offset) = Header::SIZE;
225  } else if (rightmost_chunk) {
226  /* If we are freeing the rightmost chunk in this block, then lower the
227  * first_pristine_offset mark, so that the memory region can be reused.
228  */
229  *block_first_pristine_offset_ptr(m_offset) -= chunk_size;
230  }
231  return *block_number_of_used_chunks_ptr(m_offset);
232 }
