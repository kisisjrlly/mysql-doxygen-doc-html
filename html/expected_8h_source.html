268 constexpr member_policy operator|(member_policy x, member_policy y) {
269  using int_type = std::underlying_type<member_policy>::type;

403 // enable copy constructor if T and E are copy-constructible or void
404 // enable move constructor if T and E are move-constructible or void
405 template <class T, class E>
406 using select_ctor_base =
407  ctor_base<(and_<or_<std::is_void<T>, std::is_copy_constructible<T>>,
408  or_<std::is_void<E>, std::is_copy_constructible<E>>>::value
409  ? member_policy::copy
410  : member_policy::none) |
411  (and_<or_<std::is_void<T>, std::is_move_constructible<T>>,
412  or_<std::is_void<E>, std::is_move_constructible<E>>>::value
413  ? member_policy::move
414  : member_policy::none)>;

420 // (move-constructible and move-assignable) or void

423  (and_<or_<std::is_void<T>,
