141  /**
142  Do all context-sensitive things and mark the node as contextualized
143 
144  @param pc current parse context
145 
146  @retval false success
147  @retval true syntax/OOM/etc error
148  */
149  virtual bool contextualize(Context *pc) {
150  uchar dummy;
151  if (check_stack_overrun(pc->thd, STACK_MIN_SIZE, &dummy)) return true;
152 
153 #ifndef DBUG_OFF
154  DBUG_ASSERT(!contextualized);
155  contextualized = true;
156 #endif // DBUG_OFF
157 
158  return false;
159  }
