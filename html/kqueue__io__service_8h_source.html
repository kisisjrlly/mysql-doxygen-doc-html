37 class kqueue_io_service : public IoServiceBase {
38  public:
39  using native_handle_type = impl::socket::native_handle_type;

41  ~kqueue_io_service() override { close(); }

43  bool is_open() const noexcept {
44  return epfd_ != impl::file::kInvalidHandle &&
45  wakeup_fds_.first != impl::file::kInvalidHandle &&
46  wakeup_fds_.second != impl::file::kInvalidHandle;
47  }
48 
49  stdx::expected<void, std::error_code> open() override {
50  if (is_open()) {
51  return stdx::make_unexpected(
52  make_error_code(net::socket_errc::already_open));
53  }
54 
55  auto res = impl::kqueue::create();
56  if (!res) return stdx::make_unexpected(res.error());
57 
58  epfd_ = *res;
59 
60  auto pipe_res = impl::file::pipe(O_NONBLOCK);
61  if (!pipe_res) return stdx::make_unexpected(pipe_res.error());
62 
63  wakeup_fds_ = *pipe_res;
64 
65  // set both ends of the pipe non-blocking as
66  //
67  // - read() shouldn't block is pipe is empty
68  // - write() shouldn't block is pipe is full as it only matters there is
69  // something in the pipe to wakeup the poll_one()
70  auto non_block_wakeup_0_res =
71  impl::socket::native_non_blocking(wakeup_fds_.first, true);
72  if (!non_block_wakeup_0_res) return non_block_wakeup_0_res;
73  auto non_block_wakeup_1_res =
74  impl::socket::native_non_blocking(wakeup_fds_.second, true);
75  if (!non_block_wakeup_1_res) return non_block_wakeup_1_res;
76 
77  return {};
78  }
79 
80  void on_notify() {
81  std::array<uint8_t, 256> buf;
82  ssize_t res;
83  do {
84  res = ::read(wakeup_fds_.first, buf.data(), buf.size());
85  } while (res != -1 || errno == EINTR);
86  }
87 
88  void notify() override {
89  ssize_t res;
90  do {
91  res = ::write(wakeup_fds_.second, ".", 1);
92  // retry if interrupted
93  } while ((res == -1) && (errno == EINTR));
94  }
95 
96  stdx::expected<void, std::error_code> close() {
97  if (wakeup_fds_.first != impl::file::kInvalidHandle) {
98  impl::file::close(wakeup_fds_.first);
99  wakeup_fds_.first = impl::file::kInvalidHandle;
100  }
101 
102  if (wakeup_fds_.second != impl::file::kInvalidHandle) {
103  impl::file::close(wakeup_fds_.second);
104  wakeup_fds_.second = impl::file::kInvalidHandle;
105  }
106 
107  if (epfd_ != impl::file::kInvalidHandle) {
108  impl::file::close(epfd_);
109  epfd_ = impl::file::kInvalidHandle;
110  }
111 
112  return {};
113  }

115  stdx::expected<void, std::error_code> add_fd_interest(
116  native_handle_type fd, impl::socket::wait_type wt) override {
117  struct kevent ev;
118 
119  short filter{0};
120 
121  switch (wt) {
122  case impl::socket::wait_type::wait_read:
123  filter = EVFILT_READ;
124  break;
125  case impl::socket::wait_type::wait_write:
126  filter = EVFILT_WRITE;
127  break;
128  default:
129  std::terminate();
130  break;
131  }
132 
133  // edge-triggered: EV_CLEAR
134  EV_SET(&ev, fd, filter, EV_ADD | EV_ONESHOT | EV_CLEAR, 0, 0, NULL);
135 
136  changelist_.push_back(ev);
137 
138  return {};
139  }

173  /**
174  * @returns a fdevent or std::error_code
175  * @retval fd_event on success
176  * @retval std::error_code on failure, std::error_code(success) if no events
177  * where registered.
178  */
179  stdx::expected<fd_event, std::error_code> poll_one(
180  std::chrono::milliseconds timeout) override {
181  if (!is_open()) {
182  return stdx::make_unexpected(
183  make_error_code(std::errc::invalid_argument));
184  }
185 
186  if (fd_events_processed_ == fd_events_size_) {
187  struct timespec ts, *p_ts{};
188 
189  if (timeout.count() != -1) {
190  auto secs = std::chrono::duration_cast<std::chrono::seconds>(timeout);
191  timeout -= secs;
192 
193  ts = {secs.count(),
194  std::chrono::duration_cast<std::chrono::nanoseconds>(timeout)
195  .count()};
196 
197  p_ts = &ts;
198  }
199 
200  auto res =
201  impl::kqueue::kevent(epfd_, changelist_.data(), changelist_.size(),
202  fd_events_.data(), fd_events_.size(), p_ts);
203  if (!res) return stdx::make_unexpected(res.error());
204 
205  changelist_.clear();
206 
207  fd_events_processed_ = 0;
208  fd_events_size_ = *res;
209 
210  if (fd_events_size_ == 0) {
211  return stdx::make_unexpected(make_error_code(std::errc::timed_out));
212  }
213  }
214 
215  const auto ev = fd_events_[fd_events_processed_++];
216 
217  // ev.flags may also set EV_EOF
218 
219  if (ev.flags & EV_ERROR) {
220  if (ev.data == 0) {
221  return stdx::make_unexpected(make_error_code(std::errc::interrupted));
222  } else {
223  // .data is errno
224  //
225  // if EV_RECEIPT is set, ev.data will be 0 in case of OK
226  // eitherwise ...
227  // - ENOENT
228 
229  return fd_event{static_cast<native_handle_type>(ev.ident), POLLERR};
230  }
231  }
232 
233  if (static_cast<impl::file::file_handle_type>(ev.ident) ==
234  wakeup_fds_.first) {
235  // wakeup fd fired
236  //
237  // - don't remove the interest for it
238  // - report to the caller that we don't have an event yet by saying we got
239  // interrupted
240  on_notify();
241 
242  return stdx::make_unexpected(make_error_code(std::errc::interrupted));
243  }
244 
245  after_event_fired(ev);
246 
247  short events{};
248  if (ev.filter == EVFILT_READ) {
249  events = POLLIN;
250  } else if (ev.filter == EVFILT_WRITE) {
251  events = POLLOUT;
252  }
253 
254  // ev.ident is a uintptr_t ... as it supports many kinds of event-sources
255  // ... but we only added a 'int' as file-handle
256  return fd_event{static_cast<native_handle_type>(ev.ident), events};
257  }
258 
259  private:
260  std::array<struct kevent, 16> fd_events_;
261  size_t fd_events_processed_{0};
262  size_t fd_events_size_{0};
263  int epfd_{-1};
264  std::vector<struct kevent> changelist_;
265 
266  std::pair<impl::file::file_handle_type, impl::file::file_handle_type>
267  wakeup_fds_{impl::file::kInvalidHandle, impl::file::kInvalidHandle};
268 };
