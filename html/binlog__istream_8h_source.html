29 /**
30  It defines the error types which could happen when reading binlog files or
31  deserializing binlog events. String error message of the error types are
32  defined as well. It has a member variable to store an error type and
33  provides a few functions to check the error type stored in the member
34  variable.
35  */
36 class Binlog_read_error {
37  public:
38  /**
39  Possible errors which happens when reading an event.
40  */
41  enum Error_type {
42  // No error happens
43  SUCCESS = 0,
44  /*
45  Arrive at the end of the stream. Nothing was read. It is smaller than any
46  other errors. Because READ_EOF is often not an error, and others are
47  usually errors.
48  */
49  READ_EOF = 1,
50  // malformed event
51  BOGUS,
52  // IO error while reading
53  SYSTEM_IO,
54  // Failed to allocate memory
55  MEM_ALLOCATE,
56  // Only a partial event could be read
57  TRUNC_EVENT,
58  // Only a partial Format_description_log_event could be read
59  TRUNC_FD_EVENT,
60  EVENT_TOO_LARGE,
61  CHECKSUM_FAILURE,
62  // Event's is_valid returned false
63  INVALID_EVENT,
64  // Cannot open the binlog file
65  CANNOT_OPEN,
66  // System IO error happened while reading the binlog magic
67  HEADER_IO_FAILURE,
68  // The binlog magic is incorrect
69  BAD_BINLOG_MAGIC,
70  INVALID_ENCRYPTION_HEADER,
71  CANNOT_GET_FILE_PASSWORD,
72  READ_ENCRYPTED_LOG_FILE_IS_NOT_SUPPORTED,
73  ERROR_DECRYPTING_FILE
74  };
75 
76  Binlog_read_error() {}
77  Binlog_read_error(Error_type type) : m_type(type) {}
78 
79  bool has_error() { return m_type != SUCCESS; }
80  bool has_fatal_error() { return m_type > READ_EOF; }
81 
82  /**
83  Return the error encounted when reading events.
84  */
85  Error_type get_type() const { return m_type; }
86 
87  /**
88  Return error message of the error type.
89 
90  @return It will return nullptr if m_type is SUCCESS. In practice, it should
91  never be called if m_type is SUCCESS. So an assertion is added in
92  debug mode which predicts m_type is not SUCCESS.
93  */
94  const char *get_str() const;
95 
96  /**
97  Set m_error to error.
98 
99  @param[in] type The error type will be set
100  @retval false If error is SUCCESS
101  @retval true If error is not SUCCESS.
102  */
103  bool set_type(Error_type type) {
104  m_type = type;
105  return has_error();
106  }
107 
108  private:
109  Error_type m_type = SUCCESS;
110 };

112 /**
113  Seekable_istream with decryption feature. It can be setup into an stream
114  pipeline. In the pipeline, it decrypts the data from down stream and then
115  feeds the decrypted data into up stream.

151 /**
152  Base class of binlog input files. It is a logical binlog file which wraps
153  and hides the detail of lower layer storage implementation. Binlog reader and
154  other binlog code just uses this class to control real storage.
155 */
156 class Basic_binlog_ifile : public Basic_seekable_istream {
157  public:
158  /**
159  @param[in] binlog_read_error Basic_binlog_ifile doesn't own an
160  Binlog_read_error. So the caller should
161  provide one to it. When error happens,
162  the error type will be set into 'error'.
163  */
164  Basic_binlog_ifile(Binlog_read_error *binlog_read_error);
165  Basic_binlog_ifile(const Basic_binlog_ifile &) = delete;
166  Basic_binlog_ifile &operator=(const Basic_binlog_ifile &) = delete;
167  ~Basic_binlog_ifile() override;
168  /**
169  Open a binlog file.
170 
171  @param[in] file_name name of the binlog file which will be opened.
172  */
173  bool open(const char *file_name);
174  /**
175  Close the binlog file it is reading.
176  */
177  void close();
178 
179  ssize_t read(unsigned char *buffer, size_t length) override;
180  bool seek(my_off_t position) override;
181 
182  my_off_t position() const { return m_position; }
183  bool is_open() const { return m_istream != nullptr; }
184 
185  /**
186  Get length of the binlog file. It is not os file length. The content maybe
187  encrypted or compressed. It is the total length of BINLOG_MAGIC and all
188  raw binlog events.
189  */
190  my_off_t length() override;
191 
192  protected:
193  /**
194  Open the system layer file. It is the entry of the stream pipeline.
195  Implementation is delegated to sub-classes. Sub-classes opens system layer
196  files in different way.
197 
198  @param[in] file_name name of the binlog file which will be opened.
199  */
200  virtual std::unique_ptr<Basic_seekable_istream> open_file(
201  const char *file_name) = 0;
202 
203  /**
204  It is convenient for caller to share a Binlog_read_error object between
205  streams. So Binlog_read_error pointer is defined here. It should be
206  initialized in constructor by caller.
207  */
208  Binlog_read_error *m_error;
209 
210  private:
211  /**
212  The binlog's position where it is reading. It is the position in logical
213  binlog file, but not the position of system file.
214  */
215  my_off_t m_position = 0;
216  /** It is the entry of the low level stream pipeline. */
217  std::unique_ptr<Basic_seekable_istream> m_istream;
218 
219  /**
220  Read binlog magic from binlog file and check if it is valid binlog magic.
221 
222  This function also takes care of setting up any other stream layer (i.e.
223  encryption) when needed.
224 
225  @retval false The high level stream layer was recognized as a binary log.
226  @retval true Failure identifying the high level stream layer.
227  */
228  bool read_binlog_magic();
229 };
