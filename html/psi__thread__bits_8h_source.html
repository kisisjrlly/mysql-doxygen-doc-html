344 /**
345  Performance Schema thread type: user/foreground or system/background.
346  @sa get_thread_system_attrs
347 */
348 struct PSI_thread_attrs_v3 {
349  /* PFS internal thread id, unique. */
350  unsigned long long m_thread_internal_id;
351 
352  /* SHOW PROCESSLIST thread id, not unique. */
353  unsigned long m_processlist_id;
354 
355  /* Operating system thread id, if any. */
356  unsigned long long m_thread_os_id;
357 
358  /* User-defined data. */
359  void *m_user_data;
360 
361  /* User name. */
362  char m_username[PSI_USERNAME_LENGTH];
363 
364  /* User name length. */
365  size_t m_username_length;
366 
367  /* Host name. */
368  char m_hostname[PSI_HOSTNAME_LENGTH];
369 
370  /* Host name length. */
371  size_t m_hostname_length;
372 
373  /* Resource group name. */
374  char m_groupname[PSI_NAME_LEN];
375 
376  /* Resource group name length. */
377  size_t m_groupname_length;
378 
379  /** Raw socket address */
380  struct sockaddr_storage m_sock_addr;
381 
382  /** Length of address */
383  socklen_t m_sock_addr_length;
384 
385  /* True if system/background thread, false if user/foreground thread. */
386  bool m_system_thread;
387 };
