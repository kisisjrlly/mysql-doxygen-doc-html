33 /** The LOB data page carrying the user data. */
34 struct data_page_t : public basic_page_t {
35  static const ulint OFFSET_VERSION = FIL_PAGE_DATA;
36  static const ulint OFFSET_DATA_LEN = OFFSET_VERSION + 1;
37  static const ulint OFFSET_TRX_ID = OFFSET_DATA_LEN + 4;
38  static const ulint LOB_PAGE_DATA = OFFSET_TRX_ID + 6;
39 
40  data_page_t() {}
41 
42  /** Contructor. */
43  data_page_t(buf_block_t *block, mtr_t *mtr) : basic_page_t(block, mtr) {
44  page_type_t type = get_page_type();
45  ut_a(type == FIL_PAGE_TYPE_LOB_DATA);
46  }
47 
48  data_page_t(buf_block_t *block, mtr_t *mtr, dict_index_t *index)
49  : basic_page_t(block, mtr, index) {}
50 
51  data_page_t(mtr_t *mtr, dict_index_t *index)
52  : basic_page_t(nullptr, mtr, index) {}
53 
54  /** Constructor.
55  @param[in] block the buffer block.*/
56  data_page_t(buf_block_t *block) : basic_page_t(block, nullptr, nullptr) {}
57 
58  /** Allocate a LOB data page.
59  @param[in] alloc_mtr Mini-transaction to be used for allocation of LOB page.
60  @param[in] is_bulk true if bulk operation.
61  @return allocated buffer block or nullptr */
62  buf_block_t *alloc(mtr_t *alloc_mtr, bool is_bulk);
63 
64  buf_block_t *load_x(page_no_t page_no);
65 
66  void set_version_0() {
67  mlog_write_ulint(frame() + OFFSET_VERSION, 0, MLOG_1BYTE, m_mtr);
68  }
69 
70  void dealloc() {
71  btr_page_free_low(m_index, m_block, ULINT_UNDEFINED, m_mtr);
72  m_block = nullptr;
73  }
74 
75  void set_page_type() {
76  ut_ad(m_mtr != nullptr);
77 
78  mlog_write_ulint(frame() + FIL_PAGE_TYPE, FIL_PAGE_TYPE_LOB_DATA,
79  MLOG_2BYTES, m_mtr);
80  }
81 
82  void set_trx_id(trx_id_t id) {
83  byte *ptr = frame() + OFFSET_TRX_ID;
84  mach_write_to_6(ptr, id);
85  mlog_log_string(ptr, 6, m_mtr);
86  }
87 
88  /** Write the trx identifier to the header, without
89  generating redo log.
90  @param[in] id the transaction identifier.*/
91  void set_trx_id_no_redo(trx_id_t id) {
92  byte *ptr = frame() + OFFSET_TRX_ID;
93  mach_write_to_6(ptr, id);
94  }
95 
96  static ulint payload() {
97  return (UNIV_PAGE_SIZE - LOB_PAGE_DATA - FIL_PAGE_DATA_END);
98  }
99 
100  byte *data_begin() const { return (frame() + LOB_PAGE_DATA); }
101 
102  /** Create a new data page and replace some or all parts of the old data
103  with data.
104  @param[in] trx Current transaction.
105  @param[in] offset Offset where replace begins.
106  @param[in,out] ptr Pointer to new data.
107  @param[in] want Amount of data the caller wants to replace.
108  @param[in] mtr Mini-transaction context.
109  @return the buffer block of the new data page. */
110  buf_block_t *replace(trx_t *trx, ulint offset, const byte *&ptr, ulint &want,
111  mtr_t *mtr);
112 
113  /** Replace some or all parts of the data inline.
114  @param[in] trx Current transaction.
115  @param[in] offset Offset where replace begins.
116  @param[in,out] ptr Pointer to new data.
117  @param[in] want Amount of data the caller wants to replace.
118  @param[in] mtr Mini-transaction context. */
119  void replace_inline(trx_t *trx, ulint offset, const byte *&ptr, ulint &want,
120  mtr_t *mtr);
121 
122  ulint get_data_len() const {
123  return (mach_read_from_4(frame() + OFFSET_DATA_LEN));
124  }
125 
126  void set_data_len(ulint len) {
127  ut_ad(m_mtr != nullptr);
128 
129  mlog_write_ulint(frame() + OFFSET_DATA_LEN, len, MLOG_4BYTES, m_mtr);
130  }
131 
132  /** Read data from the data page.
133  @param[in] offset read begins at this offset.
134  @param[out] ptr the output buffer.
135  @param[in] want bytes to read
136  @return bytes actually read. */
137  ulint read(ulint offset, byte *ptr, ulint want);
138 
139  /** Write data into a data page.
140  @param[in] trxid the transaction identifier of the session writing data.
141  @param[in,out] data the data to be written. it will be updated to
142  point to the byte not yet written.
143  @param[in,out] len length of data to be written.
144  @return amount of data actually written into the page. */
145  ulint write(trx_id_t trxid, const byte *&data, ulint &len);
146 
147  /** Append given data in data page.
148  @param[in] trxid transaction doing append.
149  @param[in,out] data data to be appended.
150  @param[in,out] len length of data.
151  @return number of bytes appended. */
152  ulint append(trx_id_t trxid, byte *&data, ulint &len);
153 
154  std::pair<ulint, byte *> insert_middle(trx_t *trx, ulint offset, byte *&data,
155  ulint &len, buf_block_t *&new_block);
156 
157  buf_block_t *remove_middle(trx_t *trx, ulint offset, ulint &len);
158 
159  ulint max_space_available() const { return (payload()); }
160 
161  ulint space_left() const;
162 };
