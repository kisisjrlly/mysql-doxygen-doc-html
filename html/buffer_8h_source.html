189 // const_buffer_sequence and mutable_buffer_sequence share the same

191 template <class T, class BufferType,
192  class Begin = decltype(net::buffer_sequence_begin(
193  std::declval<typename std::add_lvalue_reference<T>::type>())),
194  class End = decltype(net::buffer_sequence_end(
195  std::declval<typename std::add_lvalue_reference<T>::type>()))>
196 using buffer_sequence_requirements = std::integral_constant<
197  bool,
198  stdx::conjunction<
199  // check if buffer_sequence_begin(T &) and buffer_sequence_end(T &)
200  // exist and return the same type
201  std::is_same<Begin, End>,
202  // check if ::value_type of the retval of buffer_sequence_begin() can be
203  // converted into a BufferType
204  std::is_convertible<typename std::iterator_traits<Begin>::value_type,
205  BufferType>>::value>;

235 /*
236  * helper function for requirement checks.
237  *
238  * - function isn't implemented, just declared to type-check
239  * - function params are default initialized, never have to be specified
240  * - function params exist only to apply type-checks against them and pass them
241  * as parameter to decltype() like checks
242  */
243 template <class T, class U = std::remove_const_t<T>>
244 auto dynamic_buffer_requirements(U *__x = nullptr, const U *__const_x = nullptr,
245  size_t __n = 0)
246  -> std::enable_if_t<stdx::conjunction<
247  // is copy constructible
248  std::is_copy_constructible<U>,
249  // has a const_buffers_type that's a const_buffer_sequence
250  is_const_buffer_sequence<typename T::const_buffers_type>,
251  // has a mutable_buffers_type that's a mutable_buffer_sequence
252  is_mutable_buffer_sequence<typename T::mutable_buffers_type>,
253  // has 'size_t size() const' member
254  std::is_same<decltype(__const_x->size()), size_t>,
255  // has 'size_t max_size() const' member
256  std::is_same<decltype(__const_x->max_size()), size_t>,
257  // has 'size_t capacity() const' member
258  std::is_same<decltype(__const_x->capacity()), size_t>,
259  // has 'const_buffer_type data(size_t, size_t) const' member
260  std::is_same<decltype(__const_x->data(__n, __n)),
261  typename T::const_buffers_type>,
262  // has 'mutable_buffers_type data(size_t, size_t)' member
263  std::is_same<decltype(__x->data(__n, __n)),
264  typename T::mutable_buffers_type>,
265  // has 'void grow(size_t)' member
266  std::is_void<decltype(__x->grow(__n))>,
267  // has 'void shrink(size_t)' member
268  std::is_void<decltype(__x->shrink(__n))>,
269  // has 'void consume(size_t)' member
270  std::is_void<decltype(__x->consume(__n))>>::value>;
271 
272 template <class T>
273 struct is_dynamic_buffer<T, decltype(dynamic_buffer_requirements<T>())>
274  : std::true_type {};
275 } // namespace impl

306 /**
307  * copy bytes from a ConstBufferSequence to a MutableBufferSequence.
308  *
309  * copies min(max_bytes, buffer_size(src), buffer_size(dest)) bytes
310  *
311  * @param dest buffer-sequence to copy to
312  * @param src buffer-sequence to copy from
313  * @param max_size max bytes to copy
314  * @return bytes transfered from src to dest
315  *
316  * see: 16.9 [buffer.copy]
317  */
318 template <class MutableBufferSequence, class ConstBufferSequence>
319 size_t buffer_copy(const MutableBufferSequence &dest,
320  const ConstBufferSequence &src,
321  const size_t max_size) noexcept {
322  size_t transfered{0};
323  auto dest_cur = buffer_sequence_begin(dest);
324  auto const dest_end = buffer_sequence_end(dest);
325  auto src_cur = buffer_sequence_begin(src);
326  auto const src_end = buffer_sequence_end(src);
327 
328  const_buffer src_buf;
329  mutable_buffer dest_buf;
330 
331  while (transfered < max_size) {
332  // if buffer is empty, fetch the next one, if there is one
333  if (src_buf.size() == 0) {
334  if (src_cur == src_end) break;
335  src_buf = const_buffer(*src_cur++);
336  }
337  if (dest_buf.size() == 0) {
338  if (dest_cur == dest_end) break;
339  dest_buf = mutable_buffer(*dest_cur++);
340  }
341 
342  size_t to_copy = std::min(dest_buf.size(), src_buf.size());
343  to_copy = std::min(to_copy, max_size - transfered);
344 
345  // use std::copy() instead of std::memcpy() as it will be constexpr in C++20
346  // and results in the same code as a memcpy() already
347  std::copy(static_cast<const char *>(src_buf.data()),
348  std::next(static_cast<const char *>(src_buf.data()), to_copy),
349  static_cast<char *>(dest_buf.data()));
350 
351  src_buf += to_copy;
352  dest_buf += to_copy;
353 
354  transfered += to_copy;
355  }
356 
357  return transfered;
358 }

360 template <class MutableBufferSequence, class ConstBufferSequence>

362  const ConstBufferSequence &src) noexcept {
363  return buffer_copy(dest, src, std::numeric_limits<size_t>::max());

514 namespace impl {
515 // base for net::dynamic_vector_buffer and net::dynamic_string_buffer who share
516 // the same interface, just differ by storage type
517 template <class T>
518 class dynamic_buffer_base {
519  public:
520  using const_buffers_type = const_buffer;
521  using mutable_buffers_type = mutable_buffer;
522 
523  explicit dynamic_buffer_base(T &v) noexcept
524  : v_{v}, max_size_{v.max_size()} {}
525  dynamic_buffer_base(T &v, size_t max_size) noexcept
526  : v_{v}, max_size_{max_size} {}
527 
528  /**
529  * number of bytes.
530  */
531  size_t size() const noexcept { return std::min(v_.size(), max_size_); }
532 
533  /**
534  * max number of bytes.
535  */
536  size_t max_size() const noexcept { return max_size_; }
537 
538  /**
539  * max number of bytes without requiring reallocation.
540  */
541  size_t capacity() const noexcept {
542  return std::min(v_.capacity(), max_size_);
543  }
544 
545  const_buffers_type data(size_t pos, size_t n) const noexcept {
546  return buffer(buffer(v_, max_size_) + pos, n);
547  }
548 
549  mutable_buffers_type data(size_t pos, size_t n) {
550  return buffer(buffer(v_, max_size_) + pos, n);
551  }
552 
553  /**
554  * append bytes at the end.
555  */
556  void grow(size_t n) {
557  if (size() > max_size() || max_size() - size() < n) {
558  throw std::length_error("overflow");
559  }
560  v_.resize(v_.size() + n);
561  }
562 
563  /**
564  * remove bytes at the end.
565  */
566  void shrink(size_t n) {
567  if (n >= size()) {
568  v_.clear();
569  } else {
570  v_.resize(size() - n);
571  }
572  }
573  /**
574  * remove bytes at the start.
575  */
576  void consume(size_t n) {
577  size_t m = std::min(n, size());
578  if (m == size()) {
579  v_.clear();
580  } else {
581  v_.erase(v_.begin(), std::next(v_.begin(), m));
582  }
583  }
584 
585  private:
586  T &v_;
587  const size_t max_size_;
588 };

721 // get sequence of buffers given byte-size from another sequence of buffers.
722 //
723 // input buffer sequence is unchanged.
724 //
725 // - BufferSequence: ConstBufferSequence or MutableBufferSequence
726 // - BufferType: net::const_buffer or net::mutable_buffer
727 //
728 // Note: mostly used for net::write(), net::read()
729 template <class BufferSequence, class BufferType>
730 class consuming_buffers {
731  public:
732  using prepared_buffer_type = prepared_buffers<BufferType>;
733 
734  consuming_buffers(const BufferSequence &buffers) : buffers_{buffers} {}
735 
736  /**
737  * prepare a buffer sequence, skipping the already consumed bytes.
738  *
739  * @param max_size max bytes to take from the buffer sequence
740  *
741  * note: nax_size may be larger than the size of the buffer-sequence
742  */
743  prepared_buffer_type prepare(size_t max_size) {
744  prepared_buffer_type to_bufs;
745 
746  auto from_cur = buffer_sequence_begin(buffers_);
747  auto const from_end = buffer_sequence_end(buffers_);
748 
749  size_t to_skip = total_consumed();
750 
751  for (; (from_cur != from_end) && to_bufs.size() < to_bufs.max_size() &&
752  max_size > 0;
753  ++from_cur) {
754  if (from_cur->size() > to_skip) {
755  const size_t avail = from_cur->size() - to_skip;
756  const size_t to_use = std::min(avail, max_size);
757  to_bufs.push_back(
758  net::buffer(BufferType(net::buffer(*from_cur)) + to_skip, to_use));
759  to_skip = 0;
760  max_size -= to_use;
761  } else {
762  to_skip -= from_cur->size();
763  }
764  }
765  return to_bufs;
766  }
767 
768  /**
769  * mark bytes as consumed from the beginning of the unconsumed sequence.
770  *
771  * @param n bytes to consume
772  *
773  * note: n may be larger than the size of the buffer-sequence
774  */
775  void consume(size_t n) { total_consumed_ += n; }
776 
777  // sum of all consume()ed bytes
778  size_t total_consumed() const { return total_consumed_; }
779 
780  private:
781  const BufferSequence &buffers_;
782  size_t total_consumed_{0};
783 };

838 
839  while (0 != (to_transfer = cond(ec, transferred)) &&
840  b.size() != b.max_size()) {
841  auto orig_size = b.size();
842  // if there is space available, use that, if not grow by 4k
843  auto avail = b.capacity() - orig_size;
844  size_t grow_size = avail ? avail : 4 * 1024;
845  size_t space_left = b.max_size() - b.size();
846  // limit grow-size by possible space-left
847  grow_size = std::min(grow_size, space_left);
848  // limit grow-size by how much data we still have to read
849  grow_size = std::min(grow_size, to_transfer);
850 
851  b.grow(grow_size);
852  auto res = stream.read_some(b.data(orig_size, grow_size));
853  if (!res) {
854  b.shrink(grow_size);
855

857  // the success

859  transferred != 0) {
860  return transferred;
861  }
862  return res;
863  }
864 
865  transferred += *res;
866 
867  b.shrink(grow_size - *res);
868  }
869
