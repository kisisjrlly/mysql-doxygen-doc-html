35 /**
36  * Automatic cleanup on scope exit utility class.
37  *
38  * Automatic cleanup takes place on AutoCleaner object destruction. It allows
39  * to:
40  * - cleanup files
41  * - remove directories (non-recursive and recursive)
42  * - revert files from auto-managed backup file
43  * - revert files from a user provided backup file
44  * - call user provided callbacks
45  * Callbacks are called in order they were added into AutoCleaner. Files and
46  * directories are being deleted in a reverse order in which they were added to
47  * the AutoCleaner. If automatic cleanup fails (file or directory could not be
48  * deleted, failed to revert a file) a proper error is being logged and
49  * AutoCleaner continues with the next cleanup steps.
50  *
51  * AutoCleaner allows to clear its state so that no action will be taken on
52  * scope exit (auto generated backup files will be cleaned up in such case).
53  *
54  * Adding an action (cleanup, revert) is done once per file. It is not possible
55  * to add a second action for the same file (such add call will fail, initial
56  * action will not be affected). Adding a revert file action may fail if initial
57  * or backup files could not be opened.
58  */
59 class AutoCleaner {
60  public:
61  bool add_file_delete(const std::string &file);
62  bool add_directory_delete(const std::string &d, bool recursive = false);
63  bool add_file_revert(const std::string &file);
64  bool add_file_revert(const std::string &file, const std::string &backup_file);
65  void add_cleanup_callback(std::function<void()> callback) noexcept;
66  void clear_cleanup_callbacks() noexcept;
67  void remove(const std::string &file) noexcept;
68  void clear();
69 
70  AutoCleaner() = default;
71  ~AutoCleaner();
72 
73  AutoCleaner(AutoCleaner &&other) noexcept = default;
74  AutoCleaner &operator=(AutoCleaner &&other) = default;
75 
76  AutoCleaner(const AutoCleaner &) = delete;
77  AutoCleaner &operator=(const AutoCleaner &) = delete;
78 
79  private:
80  enum Type { Directory, DirectoryRecursive, File, FileBackup };
81 
82  /*
83  * The map stores all the files that are scheduled to be auto-removed or
84  * restored from backup if clean() wasn't called.
85  * The key is a name of file to backup, and value is a pair of
86  * backup's type and name of backup file (used only for FileBackup type).
87  */
88  std::map<std::string, std::pair<Type, std::string>> files_;
89 
90  /*
91  * The vector stores callbacks that are scheduled to be called if clean()
92  * wasn't called. Callbacks are not allowed to throw exceptions.
93  */
94  std::vector<std::function<void()>> callbacks_;
95 };
