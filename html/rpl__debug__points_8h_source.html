121 /**
122  Method used to decrease code pollution in slave methods
123 */
124 void rpl_slave_debug_point(enum_rpl_slave_debug_point point_id,
125  THD *thd = nullptr) {
126  if (!thd) thd = current_thd;
127 
128  DBUG_ASSERT(opt_debug_sync_timeout > 0);
129 
130  std::string debug_point_string{""};
131 
132  switch (point_id) {
133  /* stop_slave_cmd */
134  case DBUG_RPL_S_STOP_SLAVE_BACKUP_LOCK: {
135  debug_point_string.assign(
136  "now SIGNAL slave_acquired_backup_lock WAIT_FOR "
137  "tried_to_lock_instance_for_backup");
138  break;
139  }
140  /* terminate_slave_threads */
141  case DBUG_RPL_S_PAUSE_QUEUE_EV: {
142  debug_point_string.assign("now SIGNAL reached_stopping_io_thread");
143  break;
144  }
145  /* terminate_slave_threads */
146  case DBUG_RPL_S_IO_WAIT_FOR_SPACE: {
147  debug_point_string.assign("now SIGNAL reached_stopping_io_thread");
148  break;
149  }
150  /* get_master_uuid */
151  case DBUG_RPL_S_BEFORE_MASTER_UUID: {
152  debug_point_string.assign("now wait_for signal.get_master_uuid");
153  break;
154  }
155  /* get_master_uuid */
156  case DBUG_RPL_S_SIMULATE_BUSY_IO: {
157  debug_point_string.assign(
158  "now signal Reached wait_for signal.got_stop_slave");
159  break;
160  }
161  /* get_master_version_and_clock */
162  case DBUG_RPL_S_BEFORE_UNIX_TIMESTAMP: {
163  debug_point_string.assign("now wait_for signal.get_unix_timestamp");
164  break;
165  }
166  /* get_master_version_and_clock */
167  case DBUG_RPL_S_BEFORE_SERVER_ID: {
168  debug_point_string.assign("now wait_for signal.get_server_id");
169  break;
170  }
171  /* wait_for_relay_log_space */
172  case DBUG_RPL_S_BEFORE_FORCED_ROTATE: {
173  debug_point_string.assign(
174  "now SIGNAL signal.rpl_before_forced_rotate_reached WAIT_FOR "
175  "signal.rpl_before_forced_rotate_continue");
176  break;
177  }
178  /* exec_relay_log_event */
179  case DBUG_RPL_S_SBM_AFTER_PREVIOUS_GTID_EV: {
180  debug_point_string.assign(
181  "now signal signal.reached wait_for signal.done_sbm_calculation");
182  break;
183  }
184  /* exec_relay_log_event */
185  case DBUG_RPL_S_SBM_AFTER_FAKE_ROTATE_EV: {
186  debug_point_string.assign(
187  "now signal signal.reached wait_for signal.done_sbm_calculation");
188  break;
189  }
190  /* exec_relay_log_event */
191  case DBUG_RPL_S_PS_TABLE_WORKER_RETRY: {
192  debug_point_string.assign(
193  "now SIGNAL signal.rpl_ps_tables_worker_retry_pause WAIT_FOR "
194  "signal.rpl_ps_tables_worker_retry_continue");
195  break;
196  }
197  /* handle_slave_io */
198  case DBUG_RPL_S_BEFORE_RUNNING_STATUS: {
199  debug_point_string.assign("now wait_for signal.io_thread_let_running");
200  break;
201  }
202  /* handle_slave_io */
203  case DBUG_RPL_S_PS_TABLE_QUEUE: {
204  debug_point_string.assign(
205  "now SIGNAL signal.rpl_ps_tables_queue_before WAIT_FOR "
206  "signal.rpl_ps_tables_queue_finish");
207  break;
208  }
209  /* handle_slave_io */
210  case DBUG_RPL_S_PS_TABLES: {
211  debug_point_string.assign(
212  "now SIGNAL signal.rpl_ps_tables_queue_after_finish WAIT_FOR "
213  "signal.rpl_ps_tables_queue_continue");
214  break;
215  }
216  /* handle_slave_io */
217  case DBUG_RPL_S_PAUSE_AFTER_QUEUE_EV: {
218  debug_point_string.assign(
219  "now SIGNAL reached_after_queue_event WAIT_FOR "
220  "continue_after_queue_event");
221  break;
222  }
223  /* handle_slave_io */
224  case DBUG_RPL_S_FLUSH_AFTER_USERV_EV: {
225  debug_point_string.assign(
226  "now signal Reached wait_for signal.flush_complete_continue");
227  break;
228  }
229  /* handle_slave_io */
230  case DBUG_RPL_S_PAUSE_AFTER_IO_STOP: {
231  debug_point_string.assign(
232  "now SIGNAL reached_stopping_io_thread WAIT_FOR "
233  "continue_to_stop_io_thread");
234  break;
235  }
236  /* mts_checkpoint_routine */
237  case DBUG_RPL_S_MTS_CHECKPOINT_START: {
238  debug_point_string.assign("now signal mts_checkpoint_start");
239  break;
240  }
241  /* mts_checkpoint_routine */
242  case DBUG_RPL_S_MTS_CHECKPOINT_END: {
243  debug_point_string.assign("now signal mts_checkpoint_end");
244  break;
245  }
246  /* handle_slave_sql */
247  case DBUG_RPL_S_AFTER_SQL_STOP: {
248  debug_point_string.assign(
249  "now SIGNAL reached_stopping_sql_thread WAIT_FOR "
250  "continue_to_stop_sql_thread");
251  break;
252  }
253  /* queue_event */
254  case DBUG_RPL_S_PAUSE_QUEUING: {
255  debug_point_string.assign(
256  "now SIGNAL reached_queuing_event WAIT_FOR continue_queuing_event");
257  break;
258  }
259  /* queue_event */
260  case DBUG_RPL_S_HEARTBEAT_EV: {
261  debug_point_string.assign(
262  "now SIGNAL check_slave_master_info WAIT_FOR proceed_write_rotate");
263  break;
264  }
265  /* handle_slave_io */
266  case DBUG_RPL_S_RETRY_COUNT_EXCEED: {
267  debug_point_string.assign(
268  "now SIGNAL wait_for_retry_count_exceed WAIT_FOR "
269  "continue_retry_count_exceed");
270  break;
271  }
272  }
273 
274  DBUG_ASSERT(!debug_point_string.empty());
275  DBUG_ASSERT(!debug_sync_set_action(thd, debug_point_string.c_str(),
276  debug_point_string.length()));
277 }
