35 /**
36  @class Gcs_interface
37 
38  This interface must be implemented by all specific binding implementations
39  as its entry point.
40 
41  It should afterwards be distributed via a Factory, in order to allow
42  its transparent instantiation.
43 
44  All of the interfaces are group-oriented, meaning that all methods that
45  allow the retrieval of sub-interfaces (control, communication,
46  statistics) are oriented to serve all operations to a single group.
47 
48  It provides two main functionalities:
49  - Binding startup and finish;
50  - Allow access to the control, communication and statistics interface.
51 
52  A typical usage of this interface shall be:
53 
54  @code{.cpp}
55  Gcs_interface *group_if= new My_GCS_Gcs_interface();
56 
57  Gcs_interface_parameters params;
58  params.add_parameter("name1", "value2");
59  if (!group_if->is_initialized())
60  {
61  group_if->initialize(params);
62  }
63 
64  // Inject a logger if wanted
65  Logger_interface *logger= new My_GCS_Logger_interface();
66  group_if->set_logger(logger);
67 
68  Gcs_group_identifier *group_id= new Gcs_group_identifier("my_group");
69 
70  Gcs_control_interface *ctrl_if= group_if->get_control_session(group_id);
71 
72  ctrl_if->join(); // notice here that group id is not used anymore...
73 
74  // Do some operations here, retrieve other interfaces...
75 
76  ctrl_if->leave();
77 
78  group_if->finalize();
79  @endcode
80 */

103  /**
104  Method used by a binding implementation in order to implement any type of
105  necessary dynamic reconfiguration.
106 
107  An example of this could be an underlying GCS that needs to adjust itself
108  to changes in a group. Note, however, that the method must be only used
109  when the system is not running in order to avoid possible concurrency
110  issues. Using cached information by the caller, after this member function
