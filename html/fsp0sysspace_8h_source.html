55 /** Data structure that contains the information about shared tablespaces.
56 Currently this can be the system tablespace or a temporary table tablespace */
57 class SysTablespace : public Tablespace {
58  public:
59  SysTablespace()
60  : m_auto_extend_last_file(),
61  m_last_file_size_max(),
62  m_created_new_raw(),
63  m_is_tablespace_full(false),
64  m_sanity_checks_done(false) {
65  /* No op */
66  }
67 
68  ~SysTablespace() override { shutdown(); }
69 
70  /** Set tablespace full status
71  @param[in] is_full true if full */
72  void set_tablespace_full_status(bool is_full) {
73  m_is_tablespace_full = is_full;
74  }
75 
76  /** Get tablespace full status
77  @return true if table is full */
78  bool get_tablespace_full_status() { return (m_is_tablespace_full); }
79 
80  /** Set sanity check status
81  @param[in] status true if sanity checks are done */
82  void set_sanity_check_status(bool status) { m_sanity_checks_done = status; }
83 
84  /** Get sanity check status
85  @return true if sanity checks are done */
86  bool get_sanity_check_status() { return (m_sanity_checks_done); }
87 
88  /** Parse the input params and populate member variables.
89  @param[in] filepath_spec path to data files
90  @param[in] supports_raw true if the tablespace supports raw devices
91  @return true on success parse */
92  bool parse_params(const char *filepath_spec, bool supports_raw);
93 
94  /** Check the data file specification.
95  @param[in] create_new_db True if a new database is to be created
96  @param[in] min_expected_size Minimum expected tablespace size in bytes
97  @return DB_SUCCESS if all OK else error code */
98  dberr_t check_file_spec(bool create_new_db, ulint min_expected_size);
99 
100  /** Free the memory allocated by parse() */
101  void shutdown();
102 
103  /**
104  @return true if a new raw device was created. */
105  bool created_new_raw() const { return (m_created_new_raw); }
106 
107  /**
108  @return auto_extend value setting */
109  ulint can_auto_extend_last_file() const { return (m_auto_extend_last_file); }
110 
111  /** Set the last file size.
112  @param[in] size the size to set */
113  void set_last_file_size(page_no_t size) {
114  ut_ad(!m_files.empty());
115  m_files.back().m_size = size;
116  }
117 
118  /** Get the number of pages in the last data file in the tablespace
119  @return the size of the last data file in the array */
120  page_no_t last_file_size() const {
121  ut_ad(!m_files.empty());
122  return (m_files.back().m_size);
123  }
124 
125  /**
126  @return the autoextend increment in pages. */
127  page_no_t get_autoextend_increment() const {
128  return (sys_tablespace_auto_extend_increment *
129  ((1024 * 1024) / UNIV_PAGE_SIZE));
130  }
131 
132  /** Round the number of bytes in the file to MegaBytes
133  and then return the number of pages.
134  Note: Only system tablespaces are required to be at least
135  1 megabyte.
136  @return the number of pages in the file. */
137  page_no_t get_pages_from_size(os_offset_t size) {
138  return static_cast<page_no_t>(
139  ((size / (1024 * 1024)) * ((1024 * 1024) / UNIV_PAGE_SIZE)));
140  }
141 
142  /**
143  @return next increment size */
144  page_no_t get_increment() const;
145 
146  /** Open or create the data files
147  @param[in] is_temp whether this is a temporary tablespace
148  @param[in] create_new_db whether we are creating a new database
149  @param[out] sum_new_sizes sum of sizes of the new files added
150  @param[out] flush_lsn FIL_PAGE_FILE_FLUSH_LSN of first file
151  @return DB_SUCCESS or error code */
152  dberr_t open_or_create(bool is_temp, bool create_new_db,
153  page_no_t *sum_new_sizes, lsn_t *flush_lsn)
154  MY_ATTRIBUTE((warn_unused_result));
155 
156  private:
157  /** Check the tablespace header for this tablespace.
158  @param[out] flushed_lsn the value of FIL_PAGE_FILE_FLUSH_LSN
159  @return DB_SUCCESS or error code */
160  dberr_t read_lsn_and_check_flags(lsn_t *flushed_lsn);
161 
162  /** Note that the data file was not found.
163  @param[in] file data file object
164  @param[in] create_new_db true if a new instance to be created
165  @return DB_SUCCESS or error code */
166  dberr_t file_not_found(Datafile &file, bool create_new_db);
167 
168  /** Note that the data file was found.
169  @param[in,out] file data file object */
170  void file_found(Datafile &file);
171 
172  /** Create a data file.
173  @param[in,out] file data file object
174  @return DB_SUCCESS or error code */
175  dberr_t create(Datafile &file);
176 
177  /** Create a data file.
178  @param[in,out] file data file object
179  @return DB_SUCCESS or error code */
180  dberr_t create_file(Datafile &file);
181 
182  /** Open a data file.
183  @param[in,out] file data file object
184  @return DB_SUCCESS or error code */
185  dberr_t open_file(Datafile &file);
186 
187  /** Set the size of the file.
188  @param[in,out] file data file object
189  @return DB_SUCCESS or error code */
190  dberr_t set_size(Datafile &file);
191 
192  private:
193  /* Put the pointer to the next byte after a valid file name.
194  Note that we must step over the ':' in a Windows filepath.
195  A Windows path normally looks like C:\ibdata\ibdata1:1G, but
196  a Windows raw partition may have a specification like
197  \\.\C::1Gnewraw or \\.\PHYSICALDRIVE2:1Gnewraw.
198  @param[in] str system tablespace file path spec
199  @return next character in string after the file name */
200  static char *parse_file_name(char *ptr);
201 
202  /** Convert a numeric string that optionally ends in upper or lower
203  case G, M, or K, rounding off to the nearest number of megabytes.
204  Then return the number of pages in the file.
205  @param[in,out] ptr Pointer to a numeric string
206  @return the number of pages in the file. */
207  page_no_t parse_units(char *&ptr);
208 
209  enum file_status_t {
210  FILE_STATUS_VOID = 0, /** status not set */
211  FILE_STATUS_RW_PERMISSION_ERROR, /** permission error */
212  FILE_STATUS_READ_WRITE_ERROR, /** not readable/writable */
213  FILE_STATUS_NOT_REGULAR_FILE_ERROR /** not a regular file */
214  };
215 
216  /** Verify the size of the physical file
217  @param[in] file data file object
218  @return DB_SUCCESS if OK else error code. */
219  dberr_t check_size(Datafile &file);
220 
221  /** Check if a file can be opened in the correct mode.
222  @param[in,out] file data file object
223  @param[out] reason exact reason if file_status check failed.
224  @return DB_SUCCESS or error code. */
225  dberr_t check_file_status(const Datafile &file, file_status_t &reason);
226 
227  /* DATA MEMBERS */
228 
229  /** if true, then we auto-extend the last data file */
230  bool m_auto_extend_last_file;
231 
232  /** if != 0, this tells the max size auto-extending may increase the
233  last data file size */
234  page_no_t m_last_file_size_max;
235 
236  /** If the following is true we do not allow
237  inserts etc. This protects the user from forgetting
238  the 'newraw' keyword to my.cnf */
239  bool m_created_new_raw;
240 
241  /** Tablespace full status */
242  bool m_is_tablespace_full;
243 
244  /** if false, then sanity checks are still pending */
245  bool m_sanity_checks_done;
246 };
