198  /** @brief Gets an integral from given packet
199  *
200  * Gets an integral from packet buffer at the current position and advances it
201  * by the length of the read. The size of the integral is deduced from the
202  * give type but can be overwritten using the size parameter.
203  *
204  * Supported are integral of 1, 2, 3, 4, or 8 bytes. To retrieve an 24 bit
205  * integral it is necessary to use a 32-bit integral type and
206  * provided the size, for example:
207  *
208  * auto id = Packet::read_int_from<uint32_t>(buffer, 0, 3);
209  *
210  * In MySQL packets, integrals are stored using little-endian format.
211  *
212  * @param length size of the integer to parse
213  * @return integer type
214  *
215  * @throws std::range_error (std::runtime_error) on start or end beyond EOF
216  *
217  * @see read_int_from()
218  */
219  template <typename Type,
220  typename = std::enable_if<std::is_integral<Type>::value>>
221  Type read_int(size_t length = sizeof(Type)) const {
222  Type res = read_int_from<Type>(position_,
223  length); // throws range_error/runtime_error
224  position_ += length;
225  return res;
226  }

301  /** @brief Packs and adds an integral to the buffer
302  *
303  * Packs and adds an integral to the given buffer.
304  *
305  * @param value Integral to add to the packet
306  * @param length Size of the integral (default: size of integral)
307  *
308  */
309  template <class T, typename = std::enable_if<std::is_integral<T>::value>>
310  void write_int(T value, size_t length = sizeof(T)) {
311  msg_.reserve(msg_.size() + length);
312  while (length-- > 0) {
313  // Assignment to temporary variable `b` prevents too aggressive inlining
314  // optimization in some compilers (e.g. GCC 4.9.2 on Solaris, with -O2).
315  // Without it, `value` wasn't getting updated before push_back() under
316  // certain conditions, and resulted in filling packet's buffer with
317  // invalid data.
318  const auto b = static_cast<uint8_t>(value);
319  update_or_append(b);
320  value = static_cast<T>(value >> CHAR_BIT);
321  }
322  }

382  /** @brief Gets an integral from given packet
383  *
384  * Gets an integral from a buffer at the given position. The size of the
385  * integral is deduced from the give type but can be overwritten using
386  * the size parameter.
387  *
388  * Supported are integral of 1, 2, 3, 4, or 8 bytes. To retrieve an 24 bit
389  * integral it is necessary to use a 32-bit integral type and
390  * provided the size, for example:
391  *
392  * auto id = Packet::read_int_from<uint32_t>(buffer, 0, 3);
393  *
394  * In MySQL packets, integrals are stored using little-endian format.
395  *
396  * @param position Position where to start reading
397  * @param length size of the integer to parse
398  * @return integer type
399  *
400  * @throws std::range_error (std::runtime_error) on start or end beyond EOF
401  */
402  template <typename Type,
403  typename = std::enable_if<std::is_integral<Type>::value>>
404  Type read_int_from(size_t position, size_t length = sizeof(Type)) const {
405  harness_assert((length >= 1 && length <= 4) || length == 8);
406  if (position + length > msg_.size())
407  throw std::range_error("start or end beyond EOF");
408 
409  if (length == 1) {
410  return static_cast<Type>(msg_[position]);
411  }
412 
413  uint64_t result = 0;
414  auto it = msg_.begin() + static_cast<long>(position + length);
415  while (length-- > 0) {
416  result <<= 8;
417  result |= *--it;
418  }
419 
420  return static_cast<Type>(result);
421  }
