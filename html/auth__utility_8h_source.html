59 /**
60  Map with RWLock protections
61 */
62 template <typename K, typename V>
63 class Map_with_rw_lock {
64  public:
65  Map_with_rw_lock(PSI_rwlock_key key) { mysql_rwlock_init(key, &m_lock); }
66  Map_with_rw_lock(const Map_with_rw_lock &) = delete;
67  Map_with_rw_lock(Map_with_rw_lock &&) = delete;
68  Map_with_rw_lock &operator=(const Map_with_rw_lock &) = delete;
69  Map_with_rw_lock &operator=(Map_with_rw_lock &&) = delete;
70  ~Map_with_rw_lock() {
71  m_map.clear();
72  mysql_rwlock_destroy(&m_lock);
73  }
74 
75  /**
76  Search an entry
77 
78  @param [in] key Key
79  @param [out] value Value
80 
81  @returns status of search
82  @retval false Entry not found
83  @retval true Entry found. Check value.
84  */
85  bool find(const K &key, V &value) {
86  rwlock_scoped_lock rdlock(&m_lock, false, __FILE__, __LINE__);
87  const auto search_itr = m_map.find(key);
88  if (search_itr != m_map.end()) {
89  value = search_itr->second;
90  return true;
91  }
92  return false;
93  }
94 
95  /**
96  Insert an entry
97 
98  @param [in] key Key
99  @param [in] value Value
100 
101  @returns status of insertion
102  @retval false Error
103  @retval true Success
104  */
105  bool insert(K key, V value) {
106  rwlock_scoped_lock wrlock(&m_lock, true, __FILE__, __LINE__);
107  auto returns = m_map.insert(std::make_pair(key, value));
108  return returns.second;
109  }
110 
111  /**
112  Delete an entry
113 
114  @param [in] key Key to the element to be erased
115  */
116  void erase(K key) {
117  rwlock_scoped_lock wrlock(&m_lock, true, __FILE__, __LINE__);
118  m_map.erase(key);
119  }
120 
121  /**
122  Check limit and clear if needed
123 
124  @param [in] size Limit to check against
125  */
126  void clear_if_greater(size_t size) {
127  rwlock_scoped_lock wrlock(&m_lock, true, __FILE__, __LINE__);
128  if (m_map.size() >= size) m_map.clear();
129  }
130 
131  /** Get the size of the map */
132  size_t size() {
133  rwlock_scoped_lock rdlock(&m_lock, false, __FILE__, __LINE__);
134  return m_map.size();
135  }
136 
137  private:
138  /** Map */
139  std::map<K, V> m_map;
140  /** Lock to protect m_map */
141  mysql_rwlock_t m_lock;
142 };
