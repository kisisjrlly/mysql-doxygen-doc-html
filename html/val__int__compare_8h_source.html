31 /**
32  Holds value/unsigned_flag for the result of val_int(),
33  so that we can compare with operator<(), operator==() and operator<=()
34  */
35 class Integer_value {
36  public:
37  constexpr Integer_value(longlong val, bool unsigned_flag)
38  : m_val(val), m_unsigned_flag(unsigned_flag) {}
39 
40  constexpr longlong val() const { return m_val; }
41  constexpr bool is_unsigned() const { return m_unsigned_flag; }
42 
43  ulonglong val_unsigned() const {
44  DBUG_ASSERT(!is_negative());
45  return static_cast<ulonglong>(m_val);
46  }
47 
48  constexpr bool is_negative() const { return !is_unsigned() && val() < 0; }
49 
50  private:
51  const longlong m_val;
52  const bool m_unsigned_flag;
53 };

55 inline bool operator<(const Integer_value &lhs, const Integer_value &rhs) {
56  const bool lhs_is_neg = lhs.is_negative();
57  const bool rhs_is_neg = rhs.is_negative();
58  if (lhs_is_neg != rhs_is_neg)
59  // Different signs, lhs is smaller if it is negative.
60  return lhs_is_neg;
61  if (lhs_is_neg)
62  // Both are negative, compare as signed.
63  return lhs.val() < rhs.val();
64 
65  // Both are non-negative, compare as unsigned
66  return std::less<ulonglong>()(lhs.val(), rhs.val());
67 }
68 
69 inline bool operator==(const Integer_value &lhs, const Integer_value &rhs) {
70  const bool lhs_is_neg = lhs.is_negative();
71  const bool rhs_is_neg = rhs.is_negative();
72  if (lhs_is_neg != rhs_is_neg)
73  // Different signs, cannot be equal
74  return false;
75 
76  // Same sign, compare values.
77  return lhs.val() == rhs.val();
78 }
79 
80 inline bool operator<=(const Integer_value &lhs, const Integer_value &rhs) {
81  return lhs < rhs || lhs == rhs;
82 }
