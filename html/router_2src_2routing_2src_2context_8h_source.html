71  /** @brief Checks and if needed, blocks a host from using this routing
72  *
73  * Blocks a host from using this routing adding its IP address to the
74  * list of blocked hosts when the maximum client errors has been
75  * reached. Each call of this function will increment the number of
76  * times it was called with the client IP address.
77  *
78  * When a client host is actually blocked, true will be returned,
79  * otherwise false.
80  *
81  * @param endpoint IP address as array[16] of uint8_t
82  * @param server Server file descriptor to wish to send
83  * fake handshake reply (default is not to send anything)
84  * @return bool
85  */
86  template <class Protocol>
87  bool block_client_host(const typename Protocol::endpoint &endpoint,
88  int server = -1);
89 
90  /** @brief Clears error counter (if needed) for particular host
91  *
92  * Resets connection error counter that may eventually lead to blocking host's
93  * IP, if it exceeds a preset threshold.
94  *
95  * @sa block_client_host()
96  *
97  * @param endpoint IP address as array[16] of uint8_t
98  */
99  template <class Protocol>
100  void clear_error_counter(const typename Protocol::endpoint &endpoint);
