44 /**
45  Object registry containing several maps.
46 
47  The registry is mainly a collection of maps for each type supported. The
48  functions dispatch to the appropriate map based on the key and object type
49  parameter. There is no support for locking or thread synchronization. The
50  object registry is kind of the single threaded version of the shared
51  dictionary cache.
52 
53  The object registry is intended to be used as a thread local record of
54  which objects have been used.
55 
56  The individual maps containing DD object pointers are allocated on demand
57  to avoid excessive performance overhead during object instantiation.
58 */
59

64  // selecting map instance.

78  // Not inlined because it is big, and because it takes a lot of time
79  // for the compiler to instantiate. Defined in dd.cc, along the similar

84  /**
85  Create a map instance unless it exists already.
86  */

88  template <typename T>
89  T *create_map_if_needed(std::unique_ptr<T> *map) {
90  if (*map == nullptr) create_map(map);
91  return map->get();

185  /**
186  Template function to get a map instance.
187 
188  To support generic code, the map instances are available through
189  template function instances. This allows looking up the appropriate
190  instance based on the key type. We must use overloading to accomplish
191  this (see above). Const and non-const variants.
192 
193  @note The non-const variant will create a map unless it exists
194  already. The const variant will not create a map, and may
195  thus return a nullptr.
196 
197  @tparam T Dictionary object type.
198 
199  @return The map handling objects of type T.
200  */
201 
202  template <typename T>
203  Local_multi_map<T> *m_map() {
204  return m_map(Type_selector<T>());
205  }
206 
207  template <typename T>
208  const Local_multi_map<T> *m_map() const {
209  return m_map(Type_selector<T>());
210  }
211 
212  public:
213  /**
214  Get an iterator to the beginning of the local reverse map.
215 
216  The reverse map is guaranteed to contain all elements, that why we
217  use it for iteration. The other maps may not contain all elements
218  since keys may be NULL.
219 
220  @tparam T Dictionary object type.
221 
222  @return Iterator to the beginning of the local reverse map.
223  */
224 
225  template <typename T>
226  typename Multi_map_base<T>::Iterator begin() {
227  return m_map<T>()->begin();
228  }
229 
230  /**
231  Get an iterator to one past the end of the local reverse map.
232 
233  The reverse map is guaranteed to contain all elements, that why we
234  use it for iteration. The other maps may not contain all elements
235  since keys may be NULL.
236 
237  @tparam T Dictionary object type.
238 
239  @return Iterator to one past the end of the local reverse map.
240  */
241 
242  template <typename T>
243  typename Multi_map_base<T>::Iterator end() {
244  return m_map<T>()->end();
245  }
246 
247  /**
248  Get the element corresponding to the given key.
249 
250  @note If the map does not exist, we may as well just set the
251  object pointer to NULL right away since we know that the
252  object will not exist either.
253 
254  @tparam K Key type.
255  @tparam T Dictionary object type.
256  @param key Key too lookup.
257  @param [out] element Element, if present, otherwise, NULL.
258  */
259 
260  template <typename K, typename T>
261  void get(const K &key, Cache_element<T> **element) const {
262  const auto map = m_map<T>();
263  if (map)
264  map->get(key, element);
265  else
266  *element = nullptr;
267  }
268 
269  /**
270  Add a new element to the registry.
271 
272  @tparam T Dictionary object type.
273  @param element Element to be added.
274  */
275 
276  template <typename T>
277  void put(Cache_element<T> *element) {
278  // If the map does not exist yet, the call to m_map() will
279  // create it.
280  m_map<T>()->put(element);
281  }
282 
283  /**
284  Remove an element from the registry.
285 
286  @tparam T Dictionary object type.
287  @param element Element to be removed.
288  */
289 
290  template <typename T>
291  void remove(Cache_element<T> *element) {
292  // There should be no need to create a map to remove an element. At the
293  // same time, removing an element from a non-existing map means there
294  // is an error in our bookkeeping.
295  const auto map = m_map<T>();
296  DBUG_ASSERT(map != nullptr);
297  if (map) m_map<T>()->remove(element);
298  }
299 
300  /**
301  Remove and delete all objects of a given type from the registry.
302 
303  @tparam T Dictionary object type.
304  */
305 
306  template <typename T>
307  void erase() {
308  // No need to create a map just to erase it. But in this case,
309  // it's not necessarily an error in our bookkeeping, since this
310  // is done as part of regular clean-up.
311  const auto map = m_map<T>();
312  if (map) m_map<T>()->erase();
313  }
314 
315  /**
316  Remove and delete all objects from the registry.
317  */
318 
319  void erase_all() {
320  erase<Abstract_table>();
321  erase<Charset>();
322  erase<Collation>();
323  erase<Column_statistics>();
324  erase<Event>();
325  erase<Resource_group>();
326  erase<Routine>();
327  erase<Schema>();
328  erase<Spatial_reference_system>();
329  erase<Tablespace>();
330  }

332  /**
333  Get the number of objects of a given type in the registry.
334 
335  @tparam T Dictionary object type.
336  @return Number of objects.
337  */
338 
339  template <typename T>
340  size_t size() const {
341  const auto map = m_map<T>();
342  if (map) return map->size();
343  return 0;
344  }
345 
346  /**
347  Get the total number of objects in the registry.
348 
349  @return Number of objects.
350  */
351 
352  size_t size_all() const {
353  return size<Abstract_table>() + size<Charset>() + size<Collation>() +
354  size<Column_statistics>() + size<Event>() + size<Resource_group>() +
355  size<Routine>() + size<Schema>() + size<Spatial_reference_system>() +
356  size<Tablespace>();
357  }
