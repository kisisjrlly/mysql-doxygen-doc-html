147 /**
148  State data storage for @c start_socket_wait_v1_t.
149  This structure provide temporary storage to a socket locker.
150  The content of this structure is considered opaque,
151  the fields are only hints of what an implementation
152  of the psi interface can use.
153  This memory is provided by the instrumented code for performance reasons.
154  @sa start_socket_wait_v1_t
155 */
156 struct PSI_socket_locker_state_v1 {
157  /** Internal state. */
158  unsigned int m_flags;
159  /** Current socket. */
160  struct PSI_socket *m_socket;
161  /** Current thread. */
162  struct PSI_thread *m_thread;
163  /** Operation number of bytes. */
164  size_t m_number_of_bytes;
165  /** Timer start. */
166  unsigned long long m_timer_start;
167  /** Timer function. */
168  unsigned long long (*m_timer)(void);
169  /** Current operation. */
170  enum PSI_socket_operation m_operation;
171  /** Source file. */
172  const char *m_src_file;
173  /** Source line number. */
174  int m_src_line;
175  /** Internal data. */
176  void *m_wait;
177 };
