92  /** @brief Wrapper around socket library write() with a looping logic
93  * making sure the whole buffer got written
94  */
95  virtual result<size_t> write_all(socket_t fd, const void *buffer,
96  size_t nbyte) {
97  size_t buffer_offset = 0;
98  while (buffer_offset < nbyte) {
99  const auto write_res = this->write(
100  fd, reinterpret_cast<const char *>(buffer) + buffer_offset,
101  nbyte - buffer_offset);
102 
103  if (!write_res) {
104  return write_res;
105  }
106  buffer_offset += write_res.value();
107  }
108  return buffer_offset;
109  }
