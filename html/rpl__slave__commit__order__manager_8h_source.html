118  /* It stores order commit information of all workers. */
119  std::vector<worker_info> m_workers;
120 
121  /*
122  They are used to construct a transaction queue with trx_info::next together.
123  both head and tail point to a slot of m_trx_vector, when the queue is not
124  empty, otherwise their value are QUEUE_EOF.
125  */
126  uint32 queue_head;
127  uint32 queue_tail;
128  static const uint32 QUEUE_EOF = 0xFFFFFFFF;
129  bool queue_empty() {
130  mysql_mutex_assert_owner(&m_mutex);
131  return queue_head == QUEUE_EOF;
132  }
133 
134  void queue_pop() {
135  mysql_mutex_assert_owner(&m_mutex);
136  queue_head = m_workers[queue_head].next;
137  if (queue_head == QUEUE_EOF) queue_tail = QUEUE_EOF;
138  }
139 
140  void queue_push(uint32 index) {
141  mysql_mutex_assert_owner(&m_mutex);
142  if (queue_head == QUEUE_EOF)
143  queue_head = index;
144  else
145  m_workers[queue_tail].next = index;
146  queue_tail = index;
147  m_workers[index].next = QUEUE_EOF;
148  }
149 
150  uint32 queue_front() {
151  mysql_mutex_assert_owner(&m_mutex);
152  return queue_head;
153  }

169  public:
170  /**
171  Check if order commit deadlock happens.
172 
173  Worker1(trx1) Worker2(trx2)
174  ============= =============
175  ... ...
176  Engine acquires lock A
177  ... Engine acquires lock A(waiting for
178  trx1 to release it.
179  COMMIT(waiting for
180  trx2 to commit first).
181 
182  Currently, there are two corner cases can cause the deadlock.
183  - Case 1
184  CREATE TABLE t1(c1 INT PRIMARY KEY, c2 INT, INDEX(c2)) ENGINE = InnoDB;
185  INSERT INTO t1 VALUES(1, NULL),(2, 2), (3, NULL), (4, 4), (5, NULL), (6,
186  6)
187 
188  INSERT INTO t1 VALUES(7, NULL);
189  DELETE FROM t1 WHERE c2 <= 3;
190 
191  - Case 2
192  ANALYZE TABLE t1;
193  INSERT INTO t2 SELECT * FROM mysql.innodb_table_stats
194 
195  Since this is not a real lock deadlock, it could not be handled by engine.
196  slave need to handle it separately.
197  Worker1(trx1) Worker2(trx2)
198  ============= =============
199  ... ...
200  Engine acquires lock A
201  ... Engine acquires lock A.
202  1. found trx1 is holding the lock.
203  2. report the lock wait to server code by
204  calling thd_report_row_lock_wait().
205  Then this function is called to check
206  if it causes a order commit deadlock.
207  Report the deadlock to worker1.
208  3. waiting for trx1 to release it.
209  COMMIT(waiting for
210  trx2 to commit first).
211  Found the deadlock flag set
212  by worker2 and then
213  return with ER_LOCK_DEADLOCK.
214 
215  Rollback the transaction
216  Get lock A and go ahead.
217  ...
218  Retry the transaction
219 
220  To conclude, The transaction A which is waiting for transaction B to commit
221  and is holding a lock which is required by transaction B will be rolled
222  back and try again later.
223 
224  @param[in] thd_self The THD object of self session which is acquiring
225  a lock hold by another session.
226  @param[in] thd_wait_for The THD object of a session which is holding
227  a lock being acquired by current session.
228  */
229  static void check_and_report_deadlock(THD *thd_self, THD *thd_wait_for);
