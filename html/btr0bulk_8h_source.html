345  MY_ATTRIBUTE((warn_unused_result));
346 
347  /** Split a page
348  @param[in] page_bulk page to split
349  @param[in] next_page_bulk next page
350  @return error code */
351  dberr_t pageSplit(PageBulk *page_bulk, PageBulk *next_page_bulk)
352  MY_ATTRIBUTE((warn_unused_result));
353 
354  /** Commit(finish) a page. We set next/prev page no, compress a page of
355  compressed table and split the page if compression fails, insert a node
356  pointer to father page if needed, and commit mini-transaction.
357  @param[in] page_bulk page to commit
358  @param[in] next_page_bulk next page
359  @param[in] insert_father false when page_bulk is a root page and
360  true when it's a non-root page
361  @return error code */
362  dberr_t pageCommit(PageBulk *page_bulk, PageBulk *next_page_bulk,
363  bool insert_father) MY_ATTRIBUTE((warn_unused_result));
364 
365  /** Abort a page when an error occurs
366  @param[in] page_bulk page bulk object
367  @note We should call pageAbort for a PageBulk object, which is not in
368  m_page_bulks after pageCommit, and we will commit or abort PageBulk
369  objects in function "finish". */
370  void pageAbort(PageBulk *page_bulk) { page_bulk->commit(false); }
371 
372  /** Prepare space to insert a tuple.
373  @param[in,out] page_bulk page bulk that will be used to store the record.
374  It may be replaced if there is not enough space
375  to hold the record.
376  @param[in] level B-tree level
377  @param[in] rec_size record size
378  @return error code */
379  dberr_t prepareSpace(PageBulk *&page_bulk, ulint level, ulint rec_size)
380  MY_ATTRIBUTE((warn_unused_result));
381 
382  /** Insert a tuple to a page.
383  @param[in] page_bulk page bulk object
384  @param[in] tuple tuple to insert
385  @param[in] big_rec big record vector, could be nullptr if there is no
386  data to be stored externally.
387  @param[in] rec_size record size
388  @return error code */
389  dberr_t insert(PageBulk *page_bulk, dtuple_t *tuple, big_rec_t *big_rec,
390  ulint rec_size) MY_ATTRIBUTE((warn_unused_result));
391 
392  /** Log free check */
393  void logFreeCheck();
394 
395  /** Btree page bulk load finish. Commits the last page in each level

398  @param[out] last_page_no last page number
399  @return error code */
400  dberr_t finishAllPageBulks(dberr_t err, page_no_t &last_page_no)
401  MY_ATTRIBUTE((warn_unused_result));

423  unsigned m_index_online;
424 #endif // UNIV_DEBUG
