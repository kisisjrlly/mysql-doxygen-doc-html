379 /**
380  Strategy which will be used to handle this subquery: flattening to a
381  semi-join, conversion to a derived table, rewrite of IN to EXISTS...
382  Sometimes the strategy is first only a candidate, then the real decision
383  happens in a second phase. Other times the first decision is final.
384  */
385 enum class Subquery_strategy : int {
386  /// Nothing decided yet
387  UNSPECIFIED,
388  /// Candidate for rewriting IN(subquery) to EXISTS, or subquery
389  /// materialization
390  CANDIDATE_FOR_IN2EXISTS_OR_MAT,
391  /// Candidate for semi-join flattening
392  CANDIDATE_FOR_SEMIJOIN,
393  /// Candidate for rewriting to joined derived table
394  CANDIDATE_FOR_DERIVED_TABLE,
395  /// Semi-join flattening
396  SEMIJOIN,
397  /// Subquery's WHERE is always false, replace predicate with "false"
398  ALWAYS_FALSE,
399  /// Rewrite to joined derived table
400  DERIVED_TABLE,
401  /// Evaluate as EXISTS subquery (possibly after rewriting from another type)
402  SUBQ_EXISTS,
403  /// Subquery materialization (HASH_SJ_ENGINE)
404  SUBQ_MATERIALIZATION,
405 };
