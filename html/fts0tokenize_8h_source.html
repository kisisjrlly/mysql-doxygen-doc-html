66 /** Tokenizer for ngram referring to ft_get_word(ft_parser.c) in MyISAM.
67 Differences: a. code format changed; b. stopword processing removed.
68 @param[in] cs charset
69 @param[in,out] start doc start pointer
70 @param[in,out] end doc end pointer
71 @param[in,out] word token
72 @param[in,out] info token info
73 @retval 0 eof
74 @retval 1 word found
75 @retval 2 left bracket
76 @retval 3 right bracket
77 @retval 4 stopword found */
78 inline uchar fts_get_word(const CHARSET_INFO *cs, uchar **start, uchar *end,
79  FT_WORD *word, MYSQL_FTPARSER_BOOLEAN_INFO *info) {
80  uchar *doc = *start;
81  int ctype;
82  uint mwc;
83  uint length;
84  int mbl;
85 
86  info->yesno = (FTB_YES == ' ') ? 1 : (info->quot != nullptr);
87  info->weight_adjust = info->wasign = 0;
88  info->type = FT_TOKEN_EOF;
89 
90  while (doc < end) {
91  for (; doc < end; doc += (mbl > 0 ? mbl : (mbl < 0 ? -mbl : 1))) {
92  mbl = cs->cset->ctype(cs, &ctype, doc, end);
93 
94  if (true_word_char(ctype, *doc)) {
95  break;
96  }
97 
98  if (*doc == FTB_RQUOT && info->quot) {
99  *start = doc + 1;
100  info->type = FT_TOKEN_RIGHT_PAREN;
101 
102  return (info->type);
103  }
104 
105  if (!info->quot) {
106  if (*doc == FTB_LBR || *doc == FTB_RBR || *doc == FTB_LQUOT) {
107  /* param->prev=' '; */
108  *start = doc + 1;
109  if (*doc == FTB_LQUOT) {
110  info->quot = (char *)1;
111  }
112 
113  info->type =
114  (*doc == FTB_RBR ? FT_TOKEN_RIGHT_PAREN : FT_TOKEN_LEFT_PAREN);
115 
116  return (info->type);
117  }
118 
119  if (info->prev == ' ') {
120  if (*doc == FTB_YES) {
121  info->yesno = +1;
122  continue;
123  } else if (*doc == FTB_EGAL) {
124  info->yesno = 0;
125  continue;
126  } else if (*doc == FTB_NO) {
127  info->yesno = -1;
128  continue;
129  } else if (*doc == FTB_INC) {
130  info->weight_adjust++;
131  continue;
132  } else if (*doc == FTB_DEC) {
133  info->weight_adjust--;
134  continue;
135  } else if (*doc == FTB_NEG) {
136  info->wasign = !info->wasign;
137  continue;
138  }
139  }
140  }
141 
142  info->prev = *doc;
143  info->yesno = (FTB_YES == ' ') ? 1 : (info->quot != nullptr);
144  info->weight_adjust = info->wasign = 0;
145  }
146 
147  mwc = length = 0;
148  for (word->pos = doc; doc < end;
149  length++, doc += (mbl > 0 ? mbl : (mbl < 0 ? -mbl : 1))) {
150  mbl = cs->cset->ctype(cs, &ctype, doc, end);
151 
152  if (true_word_char(ctype, *doc)) {
153  mwc = 0;
154  } else if (!misc_word_char(*doc) || mwc) {
155  break;
156  } else {
157  mwc++;
158  }
159  }
160 
161  /* Be sure *prev is true_word_char. */
162  info->prev = 'A';
163  word->len = (uint)(doc - word->pos) - mwc;
164 
165  if ((info->trunc = (doc < end && *doc == FTB_TRUNC))) {
166  doc++;
167  }
168 
169  /* We don't check stopword here. */
170  *start = doc;
171  info->type = FT_TOKEN_WORD;
172 
173  return (info->type);
174  }
175 
176  if (info->quot) {
177  *start = doc;
178  info->type = FT_TOKEN_RIGHT_PAREN;
179  }
180 
181  return (info->type);
182 }
