45 /**
46  This class represents abstract time zone and provides

53  /**

56  Returns 0 in case of error. Sets in_dst_time_gap to true if date provided
57  falls into spring time-gap (or lefts it untouched otherwise).
58  */
59  virtual my_time_t TIME_to_gmt_sec(const MYSQL_TIME *t,
60  bool *in_dst_time_gap) const = 0;
61  /**
62  Converts time in my_time_t representation to local time in
63  broken down MYSQL_TIME representation.
64  */
65  virtual void gmt_sec_to_TIME(MYSQL_TIME *tmp, my_time_t t) const = 0;
66  /**
67  Comverts "struct timeval" to local time in
68  broken down MYSQL_TIME represendation.
69  */
70  void gmt_sec_to_TIME(MYSQL_TIME *tmp, struct timeval tv) const {
71  gmt_sec_to_TIME(tmp, (my_time_t)tv.tv_sec);
72  tmp->second_part = tv.tv_usec;
73  }
74  /**

79  virtual const String *get_name() const = 0;
