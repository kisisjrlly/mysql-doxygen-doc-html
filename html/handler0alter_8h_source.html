61 /** Generate the next autoinc based on a snapshot of the session
62 auto_increment_increment and auto_increment_offset variables.
63 Assingnment operator would be used during the inplace_alter_table()
64 phase only **/
65 struct ib_sequence_t {
66  /**
67  @param thd the session
68  @param start_value the lower bound
69  @param max_value the upper bound (inclusive) */
70  ib_sequence_t(THD *thd, ulonglong start_value, ulonglong max_value);
71 
72  /** Postfix increment
73  @return the value to insert */
74  ulonglong operator++(int) UNIV_NOTHROW;
75 
76  /** Check if the autoinc "sequence" is exhausted.
77  @return true if the sequence is exhausted */
78  bool eof() const UNIV_NOTHROW { return (m_eof); }
79 
80  /** assignment operator to copy the sequence values
81  @param in sequence to copy from */
82  ib_sequence_t &operator=(const ib_sequence_t &in) {
83  ut_ad(in.m_next_value > 0);
84  ut_ad(in.m_max_value == m_max_value);
85  m_next_value = in.m_next_value;
86  m_increment = in.m_increment;
87  m_offset = in.m_offset;
88  m_eof = in.m_eof;
89  return (*this);
90  }
91  /**
92  @return the next value in the sequence */
93  ulonglong last() const UNIV_NOTHROW {
94  ut_ad(m_next_value > 0);
95 
96  return (m_next_value);
97  }
98 
99  /** Maximum calumn value if adding an AUTOINC column else 0. Once
100  we reach the end of the sequence it will be set to ~0. */
101  const ulonglong m_max_value;
102 
103  /** Value of auto_increment_increment */
104  ulong m_increment;
105 
106  /** Value of auto_increment_offset */
107  ulong m_offset;
108 
109  /** Next value in the sequence */
110  ulonglong m_next_value;
111 
112  /** true if no more values left in the sequence */
113  bool m_eof;
114 };
