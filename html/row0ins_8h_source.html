72 /** Tries to insert an entry into a clustered index, ignoring foreign key

76  updated to an existing record, and we must write an undo log record on
77  the delete marked record.

179  entry should be inserted */

181  after a successful insert of the entry,
182  this should be reset to NULL */
183  UT_LIST_BASE_NODE_T(dtuple_t)

185  byte *row_id_buf; /* buffer for the row id sys field in row */
186  trx_id_t trx_id; /*!< trx id or the last trx which executed the
187  node */
188  byte *trx_id_buf; /* buffer for the trx id sys field in row */
189  mem_heap_t *entry_sys_heap;

192  if this is NULL, entry list should be created
