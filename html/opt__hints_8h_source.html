104 class Opt_hints_map {

110 
111  @param type_arg hint type
112

114  */

116  return hints_specified.is_set(type_arg);
117  }
118  /**
119  Set switch value and set hint into specified state.
120 
121  @param type_arg hint type
122  @param switch_state_arg switch value
123  */

125  if (switch_state_arg)
126  hints.set_bit(type_arg);
127  else
128  hints.clear_bit(type_arg);

132  Get switch value.
133 
134  @param type_arg hint type
135 
136  @return switch value.
137  */
138  bool switch_on(opt_hints_enum type_arg) const {
139  return hints.is_set(type_arg);
140  }

147 /**
148  Opt_hints class is used as ancestor for Opt_hints_global,
149  Opt_hints_qb, Opt_hints_table, Opt_hints_key classes.
150 
151  Opt_hints_global class is hierarchical structure.
152  It contains information about global hints and also
153  conains array of QUERY BLOCK level objects (Opt_hints_qb class).
154  Each QUERY BLOCK level object contains array of TABLE level hints
155  (class Opt_hints_table). Each TABLE level hint contains array of
156  KEY lelev hints (Opt_hints_key class).
157  Hint information(specified, on|off state) is stored in hints_map object.
158 */
159 
160 class Opt_hints {
161  /*
162  Name of object referred by the hint.
163  This name is empty for global level,
164  query block name for query block level,
165  table name for table level and key name
166  for key level.
167  */
168  const LEX_CSTRING *name;
169  /*
170  Parent object. There is no parent for global level,
171  for query block level parent is Opt_hints_global object,
172  for table level parent is Opt_hints_qb object,
173  for key level parent is Opt_hints_key object.
174  */
175  Opt_hints *parent;
176 
177  Opt_hints_map hints_map; // Hint map
178 
179  /* Array of child objects. i.e. array of the lower level objects */
180  Mem_root_array<Opt_hints *> child_array;
181  /* true if hint is connected to the real object */
182  bool resolved;
183  /* Number of resolved children */
184  uint resolved_children;
185 
186  public:
187  Opt_hints(const LEX_CSTRING *name_arg, Opt_hints *parent_arg,
188  MEM_ROOT *mem_root_arg)
189  : name(name_arg),
190  parent(parent_arg),
191  child_array(mem_root_arg),
192  resolved(false),
193  resolved_children(0) {}
194 
195  virtual ~Opt_hints() {}
196 
197  bool is_specified(opt_hints_enum type_arg) const {
198  return hints_map.is_specified(type_arg);
199  }
200 
201  /**
202  Function sets switch hint state.
203 
204  @param switch_state_arg switch hint state
205  @param type_arg hint type
206  @param check_parent true if hint can be on parent level
207 
208  @return true if hint is already specified,
209  false otherwise
210  */
211  bool set_switch(bool switch_state_arg, opt_hints_enum type_arg,
212  bool check_parent) {
213  if (is_specified(type_arg) ||
214  (check_parent && parent->is_specified(type_arg)))
215  return true;
216 
217  hints_map.set_switch(type_arg, switch_state_arg);
218  return false;
219  }
220 
221  /**
222  Function returns switch hint state.
223 
224  @param type_arg hint type
225 
226  @return hint value if hint is specified,
227  false otherwise
228  */
229  bool get_switch(opt_hints_enum type_arg) const;
230 
231  virtual const LEX_CSTRING *get_name() const { return name; }
232  virtual const LEX_CSTRING *get_print_name() { return name; }
233  void set_name(const LEX_CSTRING *name_arg) { name = name_arg; }
234  Opt_hints *get_parent() const { return parent; }
235  virtual void set_resolved() { resolved = true; }
236  /**
237  Returns 'resolved' flag value for depending on hint type.
238 
239  @param type_arg hint type
240 
241  @return true if all hint objects are resolved, false otherwise.
242  */
243  virtual bool is_resolved(opt_hints_enum type_arg MY_ATTRIBUTE((unused))) {
244  return resolved;
245  }
246  /**
247  Set hint to unresolved state.
248 
249  @param type_arg hint type
250  */
251  virtual void set_unresolved(opt_hints_enum type_arg MY_ATTRIBUTE((unused))) {}
252  /**
253  If ignore_print() returns true, hint is not printed
254  in Opt_hints::print() function. Atm used for
255  INDEX_MERGE, SKIP_SCAN, INDEX, JOIN_INDEX, GROUP_INDEX
256  ORDER_INDEX hints.
257 
258  @param type_arg hint type
259 
260  @return true if the hint should not be printed
261  in Opt_hints::print() function, false otherwise.
262  */
263  virtual bool ignore_print(
264  opt_hints_enum type_arg MY_ATTRIBUTE((unused))) const {
265  return false;
266  }
267  void incr_resolved_children() { resolved_children++; }
268  Mem_root_array<Opt_hints *> *child_array_ptr() { return &child_array; }
269 
270  bool is_all_resolved() const {
271  return child_array.size() == resolved_children;
272  }
273 
274  void register_child(Opt_hints *hint_arg) { child_array.push_back(hint_arg); }
275 
276  /**
277  Returns pointer to complex hint for a given type.
278 
279  A complex hint is a hint that has arguments.
280  (It is not just an on/off switch.)
281 
282  @param type hint type
283 
284  @return pointer to complex hint for a given type.
285  */
286  virtual PT_hint *get_complex_hints(
287  opt_hints_enum type MY_ATTRIBUTE((unused))) {

292  /**

299  NULL otherwise
300  */
301  Opt_hints *find_by_name(const LEX_CSTRING *name_arg,
302  const CHARSET_INFO *cs) const;
303  /**
304  Print all hints except of QB_NAME hint.

312  /**
313  Check if there are any unresolved hint objects and

318  void check_unresolved(THD *thd);
319  virtual void append_name(const THD *thd, String *str) = 0;
320 
321  private:
322  /**
323  Append hint type.
324 
325  @param str Pointer to String object
326  @param type Hint type
327  */
328  void append_hint_type(String *str, opt_hints_enum type);

334  void print_warn_unresolved(THD *thd);

339  @param thd pointer to THD object
340  @param str pointer to String object
341  */
342  virtual void print_irregular_hints(const THD *thd MY_ATTRIBUTE((unused)),
343  String *str MY_ATTRIBUTE((unused))) {}
