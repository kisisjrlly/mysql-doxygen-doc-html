42 /** Index interface. */
43 class Index {
44  public:
45  /** Index lookup (search) result. */
46  enum class Lookup {
47  /** The searched for indexed cells were found and the cursor is
48  * positioned
49  * on them. */
50  FOUND,
51 
52  /** The searched for indexed cells were not found and the cursor is
53  * positioned on the next indexed cells in index order. */
54  NOT_FOUND_CURSOR_POSITIONED_ON_NEXT,
55 
56  /** The searched for indexed cells were not found and the cursor
57  * position is undefined. */
58  NOT_FOUND_CURSOR_UNDEFINED,
59  };
60 
61  /** Constructor. */
62  Index(
63  /** [in] Table where this index belongs, used only for fetching metadata
64  * of its columns. */
65  const Table &table,
66  /** [in] MySQL index, for querying metadata. */
67  const KEY &mysql_index);
68 
69  /** Destructor. */
70  virtual ~Index();
71 
72  /** Insert a new entry into the index.
73  * @return Result::OK or another Result::* error code */
74  virtual Result insert(
75  /** [in] Indexed cells to insert. */
76  const Indexed_cells &indexed_cells,
77  /** [out] If insert succeeds (Result::OK) this will be set to the position
78  * inside the index where the new indexed cells where inserted. */
79  Cursor *insert_position) = 0;
80 
81  /** Lookup (search) an indexed cells.
82  * @retval Lookup::FOUND the provided `search_cells` were found and `first`
83  * was positioned on them (the first entry, if there are duplicates).
84  * @retval Lookup::NOT_FOUND_CURSOR_POSITIONED_ON_NEXT the provided
85  * `search_cells` were not found and `first` was positioned on the next
86  * indexed cells in index order.
87  * @retval Lookup::NOT_FOUND_CURSOR_UNDEFINED the provided `search_cells` were
88  * not found and `first` is undefined. */
89  virtual Lookup lookup(
90  /** [in] Indexed cells to search for. */
91  const Indexed_cells &search_cells,
92  /** [out] First indexed cells that were found, see above. */
93  Cursor *first) const = 0;
94 
95  /** Lookup (search) an indexed cells.
96  * @retval Lookup::FOUND the provided `search_cells` were found, `first`
97  * was positioned on them (the first entry, if there are duplicates) and
98  * `after_last` was positioned after the last matching entry.
99  * @retval Lookup::NOT_FOUND_CURSOR_POSITIONED_ON_NEXT the provided
100  * `search_cells` were not found and `first` and `after_last` were
101  * positioned on the next indexed cells in index order.
102  * @retval Lookup::NOT_FOUND_CURSOR_UNDEFINED the provided `search_cells` were
103  * not found and `first` and `after_last` are undefined. */
104  virtual Lookup lookup(const Indexed_cells &search_cells, Cursor *first,
105  Cursor *after_last) const = 0;
106 
107  /** Erase the indexed cells pointer to by a cursor. */
108  virtual void erase(
109  /** [in] Position to erase. */
110  const Cursor &target) = 0;
111 
112  /** Truncate the index, deleting all of its entries. */
113  virtual void truncate() = 0;
114 
115  /** Get a cursor to the first entry.
116  * @return cursor to the first indexed cells in index order */
117  virtual Cursor begin() const = 0;
118 
119  /** Get a cursor after the last entry.
120  * @return a cursor after the last indexed cells in index order */
121  virtual Cursor end() const = 0;
122 
123  /** Get the number of indexed columns by this index.
124  * @return number of indexed columns */
125  size_t number_of_indexed_columns() const;
126 
127  /** Get the Nth indexed column.
128  * @return the Nth indexed column */
129  const Indexed_column &indexed_column(
130  /** [in] Index of the column to fetch,
131  * must be < `number_of_indexed_columns()`. */
132  size_t i) const;
133 
134  /** Get the table of the index.
135  * @return table */
136  const Table &table() const;
137 
138  /** Get the MySQL index structure which corresponds to this index.
139  * @return mysql index */
140  const KEY &mysql_index() const;
141 
142  private:
143  /** Number of indexed columns. */
144  size_t m_number_of_indexed_columns;
145 
146  /** Table of the index. */
147  const Table &m_table;
148 
149  /** Indexed columns metadata, from [0, m_number_of_indexed_columns). */
150  std::array<Indexed_column, MAX_REF_PARTS> m_indexed_columns;
151 
152  /** MySQL index. */
153  const KEY *m_mysql_index;
154 };
