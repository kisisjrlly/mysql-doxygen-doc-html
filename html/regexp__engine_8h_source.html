70 /**
71  This class exposes high-level regular expression operations to the
72  facade. It implements the algorithm for search-and-replace and the various
73  matching options.
74 
75  A buffer is used for search-and-replace, whose initial size is that of the
76  subject string. The buffer uses ICU preflight features to probe the required
77  buffer size within each append operation, and the buffer can grow up until
78  max_allowed_packet, at which case and error will be thrown.
79 */
80 class Regexp_engine {
81  public:
82  /**
83  Compiles the URegularExpression object. If compilation fails, my_error()
84  is called and the IsError() returns true. In this case, all subsequent
85  operations will be no-ops, reporting failure. This follows ICU's chaining
86  conventions, see http://icu-project.org/apiref/icu4c/utypes_8h.html.
87 
88  @param pattern The pattern string in ICU's character set.
89 
90  @param flags ICU flags.
91 
92  @param stack_limit Sets the amount of heap storage, in bytes, that the
93  match backtracking stack is allowed to allocate.
94 
95  @param time_limit Gets set on the URegularExpression. Please refer to the
96  ICU API docs for the definition of time limit.
97  */
98  Regexp_engine(const std::u16string &pattern, uint flags, int stack_limit,
99  int time_limit) {
100  UParseError error;
101  auto upattern = pattern.data();
102  int length = pattern.size();
103  m_re = uregex_open(pointer_cast<const UChar *>(upattern), length, flags,
104  &error, &m_error_code);
105  uregex_setStackLimit(m_re, stack_limit, &m_error_code);
106  uregex_setTimeLimit(m_re, time_limit, &m_error_code);
107  uregex_setMatchCallback(m_re, QueryNotKilled, current_thd, &m_error_code);
108  check_icu_status(m_error_code, &error);
109  }
110 
111  uint flags() {
112  uint flags = uregex_flags(m_re, &m_error_code);
113  check_icu_status(m_error_code);
114  return flags;
115  }
116 
117  /**
118  Resets the engine with a new subject string. This also clears the
119  replacement buffer, see Replace().
120 
121  @param subject The new string to match the regular
122  expression against.
123  */
124  void Reset(const std::u16string &subject);
125 
126  /**
127  Tries to find match number `occurrence` in the string, starting on
128  `start`.
129 
130  @param start Start position, 0-based.
131  @param occurrence Which occurrence to replace. If zero, replace all
132  occurrences.
133  */
134  bool Matches(int start, int occurrence);
135 
136  /**
137  Returns the start position in the input string of the string where
138  Matches() found a match.
139  */
140  int StartOfMatch() {
141  /*
142  The 0 is for capture group number, but we don't deal with those
143  here. Zero means the start of the whole match, which is what's needed.
144  */
145  return uregex_start(m_re, 0, &m_error_code);
146  }
147 
148  /**
149  Returns the position in the input string right after the end of the text
150  where Matches() found a match.
151  */
152  int EndOfMatch() {
153  // The 0 means start of capture group 0, ie., the whole match.
154  return uregex_end(m_re, 0, &m_error_code);
155  }
156 
157  /**
158  Iterates over the subject string, replacing matches.
159 
160  @param replacement The string to replace matches with.
161  @param start Start position, 0-based.
162  @param occurrence Which occurrence to replace. If zero, replace all
163  occurrences.
164 
165  @return Reference to a the result of the operation. It is guaranteed to
166  stay intact until a call is made to Reset().
167  */
168  const std::u16string &Replace(const std::u16string &replacement, int start,
169  int occurrence);
