51 /**
52  struct PSI_xid is binary compatible with the XID structure as
53  in the X/Open CAE Specification, Distributed Transaction Processing:
54  The XA Specification, X/Open Company Ltd., 1991.
55  http://www.opengroup.org/bookstore/catalog/c193.htm
56 
57  A value of -1 in formatID means that the XID is null.
58  Max length for bqual and gtrid is 64 bytes each.
59 
60  @see XID in sql/handler.h
61  @see MYSQL_XID in mysql/plugin.h
62 */
63 struct PSI_xid {
64  /** Format identifier. */
65  long formatID;
66  /** GTRID length, value 1-64. */
67  long gtrid_length;
68  /** BQUAL length, value 1-64. */
69  long bqual_length;
70  /** XID raw data, not \0-terminated */
71  char data[XIDDATASIZE];
72 
73  PSI_xid() { null(); }
74  bool is_null() { return formatID == -1; }
75  void null() {
76  formatID = -1;
77  gtrid_length = 0;
78  bqual_length = 0;
79  }
80 };
