56 /** an alive, spawned process
57  *
58  * @todo
59  * refactor ProcessLauchner and SpawnedProcess into:
60  * - ProcessLauncher having only the spawn/launch() method and no state
61  * - Process as a thin wrapper around 'pid' and operators on it
62  * - SpawnedProcess being a Process with stdin/stdout/stderr
63  * - a way to declare ownership over the 'pid' (if owned, kill pid in
64  * destructor) For now, this mostly exists to make the move-constructor of
65  * ProcessLauncher easier to implement.
66  */
67 class HARNESS_EXPORT SpawnedProcess {
68  public:
69  SpawnedProcess(std::string pexecutable_path, std::vector<std::string> pargs,
70  std::vector<std::pair<std::string, std::string>> penv_vars,
71  bool predirect_stderr = true)
72  : executable_path{std::move(pexecutable_path)},
73  args{std::move(pargs)},
74  env_vars{std::move(penv_vars)},
75 #ifdef _WIN32
76  child_in_rd{INVALID_HANDLE_VALUE},
77  child_in_wr{INVALID_HANDLE_VALUE},
78  child_out_rd{INVALID_HANDLE_VALUE},
79  child_out_wr{INVALID_HANDLE_VALUE},
80  pi{},
81  si{},
82 #else
83  childpid{-1},
84  fd_in{-1, -1},
85  fd_out{-1, -1},
86 #endif
87  redirect_stderr{predirect_stderr} {
88  }
89 
90  SpawnedProcess(const SpawnedProcess &) = default;
91 
92  virtual ~SpawnedProcess() {}
93 
94  std::string get_cmd_line() const;
95 
96  protected:
97  const std::string executable_path;
98  const std::vector<std::string> args;
99  const std::vector<std::pair<std::string, std::string>> env_vars;
100 #ifdef _WIN32

115 // Launches a process as child of current process and exposes the stdin &
116 // stdout of the child process (implemented thru pipelines) so the client of
117 // this class can read from the child's stdout and write to the child's
118 // stdin. For usage, see unit tests.
119 //
120 class HARNESS_EXPORT ProcessLauncher : public SpawnedProcess {
121 #ifdef _WIN32
122  /*
123  * After ProcessLauncher sends all data to remote process, it closes the
124  * handle to notify the remote process that no more data will be sent.
125  *
126  * Since you cannot close the same handle more than once, store
127  * information if handle should be closed in child_in_wr_closed.
128  */
129  bool child_in_wr_closed = false;
130 #endif
131 
132  public:
133  /**
134  * Creates a new process and launch it.
135  * If redirect_stderr is true, the child's stderr is redirected to the
136  * same stream than child's stdout.
137  */
138  ProcessLauncher(std::string pexecutable_path, std::vector<std::string> pargs,
139  std::vector<std::pair<std::string, std::string>> penv_vars,
140  bool predirect_stderr = true)
141  : SpawnedProcess(std::move(pexecutable_path), std::move(pargs),
142  std::move(penv_vars), predirect_stderr),
143  is_alive{false} {}
144 
145  // copying a Process results in multiple destructors trying
146  // to kill the same alive process. Disable it.
147  ProcessLauncher(const ProcessLauncher &) = delete;
148  ProcessLauncher operator=(const ProcessLauncher &) = delete;
149 
150  ProcessLauncher(ProcessLauncher &&rhs)
151  : SpawnedProcess(rhs), is_alive(std::move(rhs.is_alive)) {
152  // make sure destructor on the other object doesn't try to kill
153  // the process-id we just moved
154 
155  rhs.is_alive = false;
156  }
157 
158  ~ProcessLauncher() override;
159 
160  /** Launches the child process, and makes pipes available for read/write.
161  */
162  void start();
163 
164  /**
165  * Read up to a 'count' bytes from the stdout of the child process.
166  * This method blocks until the amount of bytes is read or specified
167  * timeout expires.
168  * @param buf already allocated buffer where the read data will be stored.
169  * @param count the maximum amount of bytes to read.
170  * @param timeout timeout (in milliseconds) for the read to complete
171  * @return the real number of bytes read.
172  * Returns an shcore::Exception in case of error when reading.
173  */
174  int read(char *buf, size_t count, std::chrono::milliseconds timeout);
175 
176  /**
177  * Writes several butes into stdin of child process.
178  * Returns an shcore::Exception in case of error when writing.
179  */
180  int write(const char *buf, size_t count);
181 
182  /**
183  * Kills the child process and returns process' exit code.
184  */
185  int kill();
186 
187  /**
188  * Returns the child process handle.
189  * In Linux this needs to be cast to pid_t, in Windows to cast to HANDLE.
190  */
191  uint64_t get_pid() const;
192 
193  /**
194  * Wait for the child process to exists and returns its exit code.
195  * If the child process is already dead, wait() just returns.
196  * Returns the exit code of the process.
197  */
198  int wait(std::chrono::milliseconds timeout = std::chrono::milliseconds(1000));
199 
200  /**
201  * Returns the file descriptor write handle (to write child's stdin).
202  * In Linux this needs to be cast to int, in Windows to cast to HANDLE.
203  */
204  uint64_t get_fd_write() const;
205 
206  /**
207  * Returns the file descriptor read handle (to read child's stdout).
208  * In Linux this needs to be cast to int, in Windows to cast to HANDLE.
209  */
210  uint64_t get_fd_read() const;
211 
212  /**
213  * Closes pipe to process' STDIN in order to notify the process that all
214  * data was sent.
215  */
216  void end_of_write();

218  enum class ShutdownEvent {
219  TERM, // clean shutdown (ie. SIGTERM on Unix)
220  KILL // immediate (and abrupt) shutdown (ie. SIGKILL on Unix)
221  };
222  /**
223  * Sends a shutdown event to child process (SIGTERM on Unix, Ctrl+C on
224  * Win)
225  *
226  * @param event type of shutdown event
227  * @return std::error_code indicating success/failure
228  */
229  std::error_code send_shutdown_event(
230  ShutdownEvent event = ShutdownEvent::TERM) const noexcept;
231 
232  private:
233  /**
234  * Closes child process and returns process' exit code.
235  *
236  * @throws std::system_error if sending signal to child process fails
237  * @throws std::runtime_error if waiting for process to change state fails
238  *
239  * @return process exit code.
240  */
241  int close();
242 
243  bool is_alive;
244 };
