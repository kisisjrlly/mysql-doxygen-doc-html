182 /** Convert an undo space number (from 1 to 127) into an undo space_id.
183 Use the undo::space_id_bank to return the curent space_id assigned to
184 that undo number.
185 @param[in] space_num undo tablespace number
186 @return space_id of the undo tablespace */
187 inline space_id_t num2id(space_id_t space_num) {
188  ut_ad(space_num > 0);
189  ut_ad(space_num <= FSP_MAX_UNDO_TABLESPACES);
190 
191  size_t slot = space_num - 1;
192 
193  /* The space_id_back is normally protected by undo::spaces::m_latch.
194  But this can only be called on a specific slot when truncation is not
195  happening on that slot, i.e. the undo tablespace is in use. */

198  return (undo::space_id_bank[slot].space_id);

201 /* clang-format off */
202 /** Convert an undo space ID into an undo space number.
203 NOTE: This may be an undo space_id from a pre-exisiting 5.7
204 database which used space_ids from 1 to 127. If so, the
205 space_id is the space_num.
206 The space_ids are assigned to number ranges in reverse from high to low.
207 In addition, the first space IDs for each undo number occur sequentionally
208 and descending before the second space_id.
209 
210 Since s_max_undo_space_id = 0xFFFFFFEF, FSP_MAX_UNDO_TABLESPACES = 127
211 and s_undo_space_id_range = 400,000:
212  Space ID Space Num Space ID Space Num ... Space ID Space Num
213  0xFFFFFFEF 1 0xFFFFFFEe 2 ... 0xFFFFFF71 127
214  0xFFFFFF70 1 0xFFFFFF6F 2 ... 0xFFFFFEF2 127
215  0xFFFFFEF1 1 0xFFFFFEF0 2 ... 0xFFFFFE73 127
216 ...
217 
218 This is done to maintain backward compatibility to when there was only one
219 space_id per undo space number.
220 @param[in] space_id undo tablespace ID
221 @return space number of the undo tablespace */
222 /* clang-format on */
223 inline space_id_t id2num(space_id_t space_id) {
224  if (!is_reserved(space_id)) {
225  return (space_id);
226  }
227 
228  return (((dict_sys_t::s_max_undo_space_id - space_id) %
229  FSP_MAX_UNDO_TABLESPACES) +
230  1);
231 }
232 
233 /* Given a reserved undo space_id, return the next space_id for the associated
234 undo space number. */
235 inline space_id_t id2next_id(space_id_t space_id) {
236  ut_ad(is_reserved(space_id));
237 
238  space_id_t space_num = id2num(space_id);
239  space_id_t first_id = dict_sys_t::s_max_undo_space_id + 1 - space_num;
240  space_id_t last_id = first_id - (FSP_MAX_UNDO_TABLESPACES *
241  (dict_sys_t::s_undo_space_id_range - 1));
242 
243  return (space_id == SPACE_UNKNOWN || space_id == last_id
244  ? first_id
245  : space_id - FSP_MAX_UNDO_TABLESPACES);
246 }
