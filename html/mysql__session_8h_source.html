87  class Transaction {
88  public:
89  Transaction(MySQLSession *session) : session_(session) {
90  session_->execute("START TRANSACTION");
91  }
92 
93  ~Transaction() {
94  if (session_) {
95  try {
96  session_->execute("ROLLBACK");
97  } catch (...) {
98  // ignore errors during rollback on d-tor
99  }
100  }
101  }
102 
103  void commit() {
104  session_->execute("COMMIT");
105  session_ = nullptr;
106  }
107 
108  void rollback() {
109  session_->execute("ROLLBACK");
110  session_ = nullptr;
111  }
112 
113  private:
114  MySQLSession *session_;
115  };
