45 class MetadataCachePluginConfig final : public mysqlrouter::BasePluginConfig {
46  public:
47  /** @brief Constructor
48  *
49  * @param section from configuration file provided as ConfigSection
50  */
51  MetadataCachePluginConfig(const mysql_harness::ConfigSection *section)
52  : BasePluginConfig(section),
53  metadata_cache_dynamic_state(get_dynamic_state(section)),
54  metadata_servers_addresses(get_metadata_servers(
55  section, metadata_cache::kDefaultMetadataPort)),
56  user(get_option_string(section, "user")),
57  ttl(get_option_milliseconds(section, "ttl", 0.0, 3600.0)),
58  auth_cache_ttl(
59  get_option_milliseconds(section, "auth_cache_ttl", -1, 3600.0)),
60  auth_cache_refresh_interval(get_option_milliseconds(
61  section, "auth_cache_refresh_interval", 0.001, 3600.0)),
62  metadata_cluster(get_option_string(section, "metadata_cluster")),
63  connect_timeout(
64  get_uint_option<uint16_t>(section, "connect_timeout", 1)),
65  read_timeout(get_uint_option<uint16_t>(section, "read_timeout", 1)),
66  thread_stack_size(
67  get_uint_option<uint32_t>(section, "thread_stack_size", 1, 65535)),
68  use_gr_notifications(get_uint_option<uint16_t>(
69  section, "use_gr_notifications", 0, 1) == 1),
70  cluster_type(get_cluster_type(section)),
71  router_id(get_uint_option<uint32_t>(section, "router_id")) {
72  if (cluster_type == mysqlrouter::ClusterType::RS_V2 &&
73  section->has("use_gr_notifications")) {
74  throw std::invalid_argument(
75  "option 'use_gr_notifications' is not valid for cluster type 'rs'");
76  }
77  if (auth_cache_ttl > std::chrono::seconds(-1) &&
78  auth_cache_ttl < std::chrono::milliseconds(1)) {
79  throw std::invalid_argument(
80  "'auth_cache_ttl' option value '" +
81  get_option_string(section, "auth_cache_ttl") +
82  "' should be in range 0.001 and 3600 inclusive or -1 for "
83  "auth_cache_ttl disabled");
84  }
85  }
