263  void push_back(element_type &&t) {
264  const auto handle = t->native_handle();
265 
266  std::lock_guard<std::mutex> lk(mtx_);
267 
268  auto it = ops_.find(handle);
269  if (it != ops_.end()) {
270  it->second.push_back(std::move(t));
271  } else {
272  std::vector<element_type> v;
273  v.push_back(std::move(t));
274  ops_.emplace(handle, std::move(v));
275  }
276  }

278  element_type extract_first(native_handle_type fd, short events) {
279  return extract_first(fd, [events](auto const &el) {
280  return static_cast<short>(el->event()) & events;
281  });
282  }
283 
284  element_type extract_first(native_handle_type fd) {
285  return extract_first(fd, [](auto const &) { return true; });
286  }
287 
288  private:
289  template <class Pred>
290  element_type extract_first(native_handle_type fd, Pred &&pred) {
291  std::lock_guard<std::mutex> lk(mtx_);
292 
293  const auto it = ops_.find(fd);
294  if (it != ops_.end()) {
295  auto &async_ops = it->second;
296 
297  const auto end = async_ops.end();
298  for (auto cur = async_ops.begin(); cur != end; ++cur) {
299  auto &el = *cur;
300 
301  if (el->native_handle() == fd && pred(el)) {
302  auto op = std::move(el);
303 
304  if (async_ops.size() == 1) {
305  // remove the current container and with it its only element
306  ops_.erase(it);
307  } else {
308  // remove the current entry
309  async_ops.erase(cur);
310  }
311 
312  return op;
313  }
314  }
315  }
316 
317  return {};
318  }

331  template <class Op>
332  void async_wait(native_handle_type fd, impl::socket::wait_type wt, Op &&op) {
333  // add the socket-wait op to the queue
334  active_ops_.push_back(
335  std::make_unique<async_op_impl<Op>>(std::forward<Op>(op), fd, wt));
336 
337  {
338  auto res = io_service_->add_fd_interest(fd, wt);
339  if (!res) {
340 #if !defined(NDEBUG)
341  // fd may be -1 or so
342  std::cerr << "!! add_fd_interest(" << fd << ", ..."
343  << ") " << res.error() << " " << res.error().message()
344  << std::endl;
345 #endif
346  // adding failed. Cancel it again.
347  //
348  // code should be similar to ::cancel(fd)
349  std::lock_guard<std::mutex> lk(mtx_);
350 
351  if (auto op = active_ops_.extract_first(fd, static_cast<short>(wt))) {
352  op->cancel();
353  cancelled_ops_.push_back(std::move(op));
354  }
355  }
356  }
357 
358  io_service_->notify();
359  }

425  std::chrono::milliseconds next() const override {
426  typename Timer::time_point expiry;
427  {
428  std::lock_guard<std::mutex> lk(queue_mtx_);
429 
430  // no pending timers, return the max-timeout
431  if (cancelled_timers_.empty()) {
432  if (pending_timer_expiries_.empty())
433  return std::chrono::milliseconds::max();
434 
435 #if 0
436  expiry = pending_timers_.front()->expiry();
437 #else
438  expiry = pending_timer_expiries_.begin()->first;
439 #endif
440  } else {
441  // cancelled timers should be executed directly
442  return std::chrono::milliseconds::min();
443  }
444 
445  // the lock isn't needed anymore.
446  }
447 
448  auto duration = Timer::traits_type::to_wait_duration(expiry);
449  if (duration < duration.zero()) {
450  duration = duration.zero();
451  }
452 
453  // round up the next wait-duration to wait /at least/ the expected time.
454  //
455  // In case the expiry is 990us, wait 1ms
456  // If it is 0ns, leave it at 0ms;
457 
458  auto duration_ms =
459  std::chrono::duration_cast<std::chrono::milliseconds>(duration);
460 
461  using namespace std::chrono_literals;
462 
463  // round up to the next millisecond.
464  if ((duration - duration_ms).count() != 0) {
465  duration_ms += 1ms;
466  }
467 
468  return duration_ms;
469  }

471  bool run_one() override {
472  std::unique_ptr<pending_timer> pt;
473 
474  {
475  std::lock_guard<std::mutex> lk(queue_mtx_);
476 
477  // if the pending-timers queue is empty, leave
478  // if the top is cancelled or expired, run it
479  if (cancelled_timers_.empty()) {
480  if (pending_timers_.empty()) return false;
481 
482 #if 0
483  // list
484  if (pending_timers_.front()->expiry() > Timer::clock_type::now()) {
485  return false;
486  }
487  pt = std::move(pending_timers_.front());
488  pending_timers_.pop_front();
489 #else
490  if (pending_timers_.size() != pending_timer_expiries_.size()) abort();
491 
492  auto min = Timer::time_point::min();
493  for (const auto &cur : pending_timer_expiries_) {
494  if (cur.first < min) abort();
495 
496  min = cur.first;
497  }
498 
499  const auto now = Timer::clock_type::now();
500 
501  // multimap
502  auto pending_expiry_it = pending_timer_expiries_.begin();
503  auto timepoint = pending_expiry_it->first;
504 
505  if (timepoint > now) {
506  // not expired yet. leave
507  return false;
508  }
509  typename Timer::Id *timer_id = pending_expiry_it->second;
510 
511  auto pending_it = pending_timers_.find(timer_id);
512  if (pending_it == pending_timers_.end()) {
513  abort();
514  }
515  if (pending_it->second->id() != timer_id) {
516  abort();
517  }
518  if (pending_it->second->expiry() != pending_expiry_it->first) {
519  abort();
520  }
521 
522  pt = std::move(pending_it->second);
523  pending_timer_expiries_.erase(pending_expiry_it);
524  pending_timers_.erase(pending_it);
525 #endif
526  } else {
527  pt = std::move(cancelled_timers_.front());
528  cancelled_timers_.pop_front();
529  }
530  }
531 
532  pt->run();
533 
534  context().get_executor().on_work_finished();
535 
536  return true;
537  }

800 inline stdx::expected<void, std::error_code> io_context::cancel(
801  native_handle_type fd) {
802  bool need_notify{false};
803  {
804  // check all async-ops
805  std::lock_guard<std::mutex> lk(mtx_);
806 
807  while (auto op = active_ops_.extract_first(fd)) {
808  op->cancel();
809 
810  cancelled_ops_.push_back(std::move(op));
811 
812  need_notify = true;
813  }
814  }
815 
816  // wakeup the loop to deliver the cancelled fds
817  if (true || need_notify) {
818  io_service_->remove_fd(fd);
819  io_service_->notify();
820  }
821 
822  return {};
823 }

898 // precond: lk MUST be locked
899 inline io_context::count_type io_context::do_one(
900  std::unique_lock<std::mutex> &lk, std::chrono::milliseconds timeout) {
901  timer_queue_base *timer_q{nullptr};
902 
903  monitor mon(*this);
904 
905  if (!has_outstanding_work()) {
906  wake_one_runner_(lk);
907  return 0;
908  }
909 
910  while (true) {
911  if (timer_q) {
912  if (timer_q->run_one()) {
913  wake_one_runner_(lk);
914  return 1;
915  } else {
916  timer_q = nullptr;
917  }
918  }
919 
920  std::chrono::milliseconds min_duration{0};
921  {
922  std::lock_guard<std::mutex> lk(mtx_);
923  // check the smallest timestamp of all timer-queues
924  for (auto q : timer_queues_) {
925  const auto duration = q->next();
926 
927  if (duration == duration.zero()) {
928  timer_q = q;
929  min_duration = duration;
930  break;
931  } else if ((duration != duration.max()) &&
932  (timeout != timeout.zero()) &&
933  (duration < min_duration || timer_q == nullptr)) {
934  timer_q = q;
935  min_duration = duration;
936  }
937  }
938  }
939 
940  // if we have a timer that has fired, run it right away
941  if (timer_q && min_duration == min_duration.zero()) continue;
942 
943  if (auto op = [this]() -> std::unique_ptr<async_op> {
944  // handle all the cancelled ops without polling first
945  std::lock_guard<std::mutex> lk(mtx_);
946 
947  // ops have all cancelled operators at the front
948  if (!cancelled_ops_.empty() &&
949  cancelled_ops_.front()->is_cancelled()) {
950  auto op = std::move(cancelled_ops_.front());
951 
952  cancelled_ops_.pop_front();
953 
954  return op;
955  }
956 
957  return {};
958  }()) {
959  // before we unlock the concurrent io-context-thread-lock increment the
960  // work-count to ensure the next waiting thread exiting in case:
961  //
962  // - no io-events registered
963  // - no timers registered
964  get_executor().on_work_started();
965  wake_one_runner_(lk);
966  op->run(*this);
967  get_executor().on_work_finished();
968 
969  return 1;
970  }
971 
972  if (stopped() || !io_service_open_res_) {
973  break;
974  }
975 
976  if (timeout != timeout.zero()) {
977  if (timer_q == nullptr) {
978  min_duration = timeout;
979  } else if (timeout.zero() <= timeout && timeout < min_duration) {
980  min_duration = timeout;
981  }
982  }
983 
984  auto res = io_service_->poll_one(min_duration);
985  if (!res) {
986  if (res.error() == std::errc::interrupted) {
987  // poll again as it got interrupted
988  continue;
989  }
990  if (res.error() == std::errc::timed_out && timer_q != nullptr) {
991  // poll() timed out and we have a timer-queue
992  continue;
993  }
994  wake_one_runner_(lk);
995 
996 #if 0
997  // if the poll returns another error, it is a ok that we don't further
998  // check it as we exit cleanly. Still it would be nice to be aware of it
999  // in debug builds.
1000  assert(res.error() == io_service_errc::no_fds ||
1001  res.error() == std::errc::timed_out);
1002 #endif
1003 
1004  // either poll() timed out or there where no file-descriptors that fired
1005  return 0;
1006  }
1007 
1008  // std::cerr << __LINE__ << ": " << res.value().fd << " - "
1009  // << res.value().event << std::endl;
1010 
1011  if (auto op = [this](native_handle_type fd,
1012  short events) -> std::unique_ptr<async_op> {
1013  std::lock_guard<std::mutex> lk(mtx_);
1014 
1015  return active_ops_.extract_first(fd, events);
1016  }(res->fd, res->event)) {
1017  get_executor().on_work_started();
1018  wake_one_runner_(lk);
1019  op->run(*this);
1020  get_executor().on_work_finished();
1021  return 1;
1022  }
1023  // we may not find an async-op for this event if it already has been
1024  // cancelled. Loop around let the "is-cancelled" check handle it.
1025  }
1026 
1027  wake_one_runner_(lk);
1028  return 0;
1029 }
