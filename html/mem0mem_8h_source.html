55 /** Types of allocation for memory heaps: DYNAMIC means allocation from the
56 dynamic memory pool of the C compiler, BUFFER means allocation from the
57 buffer pool; the latter method is used for very big heaps */
58 
59 #define MEM_HEAP_DYNAMIC 0 /* the most common type */
60 #define MEM_HEAP_BUFFER 1
61 #define MEM_HEAP_BTR_SEARCH \
62  2 /* this flag can optionally be \
63  ORed to MEM_HEAP_BUFFER, in which \
64  case heap->free_block is used in \
65  some cases for memory allocations, \
66  and if it's NULL, the memory \
67  allocation functions can return \
68  NULL. */
69 
70 /** Different type of heaps in terms of which data structure is using them */
71 #define MEM_HEAP_FOR_BTR_SEARCH (MEM_HEAP_BTR_SEARCH | MEM_HEAP_BUFFER)
72 #define MEM_HEAP_FOR_PAGE_HASH (MEM_HEAP_DYNAMIC)
73 #define MEM_HEAP_FOR_RECV_SYS (MEM_HEAP_BUFFER)
74 #define MEM_HEAP_FOR_LOCK_HEAP (MEM_HEAP_BUFFER)

76 /** The following start size is used for the first block in the memory heap if
77 the size is not specified, i.e., 0 is given as the parameter in the call of
78 create. The standard size is the maximum (payload) size of the blocks used for
79 allocations of small buffers. */
80 
81 #define MEM_BLOCK_START_SIZE 64
82 #define MEM_BLOCK_STANDARD_SIZE \
83  (UNIV_PAGE_SIZE >= 16384 ? 8000 : MEM_MAX_ALLOC_IN_BUF)
84 
85 /** If a memory heap is allowed to grow into the buffer pool, the following
86 is the maximum size for a single allocated buffer
87 (from UNIV_PAGE_SIZE we subtract MEM_BLOCK_HEADER_SIZE and 2*MEM_NO_MANS_LAND
88 since it's something we always need to put. Since in MEM_SPACE_NEEDED we round
89 n to the next multiple of UNIV_MEM_ALINGMENT, we need to cut from the rest the
90 part that cannot be divided by UNIV_MEM_ALINGMENT): */
91 #define MEM_MAX_ALLOC_IN_BUF \
92  ((UNIV_PAGE_SIZE - MEM_BLOCK_HEADER_SIZE - 2 * MEM_NO_MANS_LAND) & \
93  ~(UNIV_MEM_ALIGNMENT - 1))
94 
95 /* Before and after any allocated object we will put MEM_NO_MANS_LAND bytes of
96 some data (different before and after) which is supposed not to be modified by
97 anyone. This way it would be much easier to determine whether anyone was
98 writing on not his memory, especially that Valgrind can assure there was no
99 reads or writes to this memory. */
100 #ifdef UNIV_DEBUG
101 const int MEM_NO_MANS_LAND = 16;
102 #else
103 const int MEM_NO_MANS_LAND = 0;
104 #endif
105 
106 /* Byte that we would put before allocated object MEM_NO_MANS_LAND times.*/
107 const byte MEM_NO_MANS_LAND_BEFORE_BYTE = 0xCE;
108 /* Byte that we would put after allocated object MEM_NO_MANS_LAND times.*/
109 const byte MEM_NO_MANS_LAND_AFTER_BYTE = 0xDF;
110 
111 /** Space needed when allocating for a user a field of length N.
112 The space is allocated only in multiples of UNIV_MEM_ALIGNMENT. In debug mode
113 contains two areas of no mans lands before and after the buffer requested. */
114 #define MEM_SPACE_NEEDED(N) \
115  ut_calc_align(N + 2 * MEM_NO_MANS_LAND, UNIV_MEM_ALIGNMENT)

348 /** The info structure stored at the beginning of a heap block */
349 struct mem_block_info_t {
350  uint64_t magic_n; /* magic number for debugging */
351 #ifdef UNIV_DEBUG
352  char file_name[16]; /* file name where the mem heap was created */
353  ulint line; /*!< line number where the mem heap was created */
354 #endif /* UNIV_DEBUG */
355  UT_LIST_BASE_NODE_T(mem_block_t)
356  base; /* In the first block in the
357 the list this is the base node of the list of blocks;
358 in subsequent blocks this is undefined */
359  UT_LIST_NODE_T(mem_block_t)
360  list; /* This contains pointers to next
361  and prev in the list. The first block allocated
362  to the heap is also the first block in this list,
363  though it also contains the base node of the list. */
364  ulint len; /*!< physical length of this block in bytes */
365  ulint total_size; /*!< physical length in bytes of all blocks
366  in the heap. This is defined only in the base
367  node and is set to ULINT_UNDEFINED in others. */
368  ulint type; /*!< type of heap: MEM_HEAP_DYNAMIC, or
369  MEM_HEAP_BUF possibly ORed to MEM_HEAP_BTR_SEARCH */
370  ulint free; /*!< offset in bytes of the first free position for
371  user data in the block */
372  ulint start; /*!< the value of the struct field 'free' at the
373  creation of the block */
374  void *free_block;
375  /* if the MEM_HEAP_BTR_SEARCH bit is set in type,
376  and this is the heap root, this can contain an
377  allocated buffer frame, which can be appended as a
378  free block to the heap, if we need more space;
379  otherwise, this is NULL */
380  void *buf_block;
381  /* if this block has been allocated from the buffer
382  pool, this contains the buf_block_t handle;
383  otherwise, this is NULL */
384 };

395 /** A C++ wrapper class to the mem_heap_t routines, so that it can be used
396 as an STL allocator */
397 template <typename T>
398 class mem_heap_allocator {
399  public:
400  typedef T value_type;
401  typedef size_t size_type;
402  typedef ptrdiff_t difference_type;
403  typedef T *pointer;
404  typedef const T *const_pointer;
405  typedef T &reference;
406  typedef const T &const_reference;
407 
408  mem_heap_allocator(mem_heap_t *heap) : m_heap(heap) {}
409 
410  mem_heap_allocator(const mem_heap_allocator &other) : m_heap(other.m_heap) {
411  // Do nothing
412  }
413 
414  template <typename U>
415  mem_heap_allocator(const mem_heap_allocator<U> &other)
416  : m_heap(other.m_heap) {
417  // Do nothing
418  }
419 
420  ~mem_heap_allocator() { m_heap = nullptr; }
421 
422  size_type max_size() const { return (ULONG_MAX / sizeof(T)); }
423 
424  /** This function returns a pointer to the first element of a newly
425  allocated array large enough to contain n objects of type T; only the
426  memory is allocated, and the objects are not constructed. Moreover,
427  an optional pointer argument (that points to an object already
428  allocated by mem_heap_allocator) can be used as a hint to the
429  implementation about where the new memory should be allocated in
430  order to improve locality. */
431  pointer allocate(size_type n, const_pointer hint = nullptr) {
432  return (reinterpret_cast<pointer>(mem_heap_alloc(m_heap, n * sizeof(T))));
433  }
434 
435  void deallocate(pointer p, size_type n) {}
436 
437  pointer address(reference r) const { return (&r); }
438 
439  const_pointer address(const_reference r) const { return (&r); }
440 
441  void construct(pointer p, const_reference t) {
442  new (reinterpret_cast<void *>(p)) T(t);
443  }
444 
445  void destroy(pointer p) { (reinterpret_cast<T *>(p))->~T(); }
446 
447  /** Allocators are required to supply the below template class member
448  which enables the possibility of obtaining a related allocator,
449  parametrized in terms of a different type. For example, given an
450  allocator type IntAllocator for objects of type int, a related
451  allocator type for objects of type long could be obtained using
452  IntAllocator::rebind<long>::other */
453  template <typename U>
454  struct rebind {
455  typedef mem_heap_allocator<U> other;
456  };
457 
458  /** Get the underlying memory heap object.
459  @return the underlying memory heap object. */
460  mem_heap_t *get_mem_heap() const { return (m_heap); }
461 
462  private:
463  mem_heap_t *m_heap;
464  template <typename U>
465  friend class mem_heap_allocator;
466 };
467 
468 template <class T>
469 bool operator==(const mem_heap_allocator<T> &left,
470  const mem_heap_allocator<T> &right) {
471  return (left.get_mem_heap() == right.get_mem_heap());
472 }
473 
474 template <class T>
475 bool operator!=(const mem_heap_allocator<T> &left,
476  const mem_heap_allocator<T> &right) {
477  return (left.get_mem_heap() != right.get_mem_heap());
478 }
