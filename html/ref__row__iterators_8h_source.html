163 /*
164  This is for QS_DYNAMIC_RANGE, i.e., "Range checked for each
165  record". The trace for the range analysis below this point will
166  be printed with different ranges for every record to the left of
167  this table in the join; the range optimizer can either select any
168  QUICK_SELECT_I (aka IndexRangeScanIterator) or a full table
169  scan, and any Read() is just proxied over to that.
170 
171  Note in particular that this means the range optimizer will be
172  executed anew on every single call to Init(), and modify the
173  query plan accordingly! It is not clear whether this is an actual
174  win in a typical query.
175  */

221 /**
222  Read a table *assumed* to be included in execution of a pushed join.
223  This is the counterpart of RefIterator / EQRefIterator for child
224  tables in a pushed join. As the underlying handler interface for
225  pushed joins are the same for Ref / EQRef operations, we implement
226  both in the same PushedJoinRefIterator class.
227 
228  In order to differentiate between a 'range' and 'single-row lookup'
229  in the DebugString(), the class takes a 'bool Unique' C'tor argument.
230  This also offers some optimizations in implementation of ::Read().
231 
232  When the table access is performed as part of the pushed join,
233  all 'linked' child colums are prefetched together with the parent row.
234  The handler will then only format the row as required by MySQL and set
235  table status accordingly.
236 
237  However, there may be situations where the prepared pushed join was not
238  executed as assumed. It is the responsibility of the handler to handle
239  these situation by letting @c ha_index_read_pushed() then effectively do a
240  plain old' index_read_map(..., HA_READ_KEY_EXACT);
241  */
242 class PushedJoinRefIterator final : public TableRowIterator {
243  public:
244  // "examined_rows", if not nullptr, is incremented for each successful Read().
245  PushedJoinRefIterator(THD *thd, TABLE *table, TABLE_REF *ref, bool use_order,
246  bool is_unique, ha_rows *examined_rows);
247 
248  bool Init() override;
249  int Read() override;

253  const bool m_use_order;
254  const bool m_is_unique;

259 /**

261  or similar) and a TableScanIterator.
262 
263  This is used when predicates have been pushed down into an IN subquery
264  and then created ref accesses, but said predicates should not be checked for
265  a NULL value (so we need to revert to table scans). See
266  QEP_TAB::pick_table_access_method() for a more thorough explanation.
267  */
268 class AlternativeIterator final : public RowIterator {
269  public:
270  // Takes ownership of "source", and is responsible for
271  // calling Init() on it, but does not hold the memory.
272  AlternativeIterator(THD *thd, TABLE *table,
273  unique_ptr_destroy_only<RowIterator> source,
274  unique_ptr_destroy_only<RowIterator> table_scan_iterator,
275  TABLE_REF *ref);
276 
277  bool Init() override;
278 
279  int Read() override { return m_iterator->Read(); }
280 
281  void SetNullRowFlag(bool is_null_row) override {
282  // Init() may not have been called yet, so just forward to both iterators.
283  m_source_iterator->SetNullRowFlag(is_null_row);
284  m_table_scan_iterator->SetNullRowFlag(is_null_row);
285  }
286 
287  void UnlockRow() override { m_iterator->UnlockRow(); }
288 
289  private:
290  // If any of these are false during Init(), we are having a NULL IN ( ... ),
291  // and need to fall back to table scan. Extracted from m_ref.
292  std::vector<bool *> m_applicable_cond_guards;
293 
294  // Points to either m_source_iterator or m_table_scan_iterator,
295  // depending on the value of applicable_cond_guards. Set up during Init().
296  RowIterator *m_iterator = nullptr;
297 
298  // Points to the last iterator that was Init()-ed. Used to reset the handler
299  // when switching from one iterator to the other.
300  RowIterator *m_last_iterator_inited = nullptr;
301 
302  // The iterator we are normally reading records from (a RefIterator or
303  // similar).
304  unique_ptr_destroy_only<RowIterator> m_source_iterator;
305 
306  // Our fallback iterator (possibly wrapped in a TimingIterator).
307  unique_ptr_destroy_only<RowIterator> m_table_scan_iterator;
308 
309  // The underlying table.
310  TABLE *const m_table;
311 
312  /**
313  A read set we can use when we fall back to table scans,
314  to get the base columns we need for virtual generated columns.
315  See add_virtual_gcol_base_cols().
316  */
317  MY_BITMAP m_table_scan_read_set;
318 
319  /// The original value of table->read_set.
