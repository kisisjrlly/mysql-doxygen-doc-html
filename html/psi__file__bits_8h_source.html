62 /** Operation performed on an instrumented file. */
63 enum PSI_file_operation {
64  /** File creation, as in @c create(). */
65  PSI_FILE_CREATE = 0,
66  /** Temporary file creation, as in @c create_temp_file(). */
67  PSI_FILE_CREATE_TMP = 1,
68  /** File open, as in @c open(). */
69  PSI_FILE_OPEN = 2,
70  /** File open, as in @c fopen(). */
71  PSI_FILE_STREAM_OPEN = 3,
72  /** File close, as in @c close(). */
73  PSI_FILE_CLOSE = 4,
74  /** File close, as in @c fclose(). */
75  PSI_FILE_STREAM_CLOSE = 5,
76  /**
77  Generic file read, such as @c fgets(), @c fgetc(), @c fread(), @c read(),
78  @c pread().
79  */
80  PSI_FILE_READ = 6,
81  /**
82  Generic file write, such as @c fputs(), @c fputc(), @c fprintf(),
83  @c vfprintf(), @c fwrite(), @c write(), @c pwrite().
84  */
85  PSI_FILE_WRITE = 7,
86  /** Generic file seek, such as @c fseek() or @c seek(). */
87  PSI_FILE_SEEK = 8,
88  /** Generic file tell, such as @c ftell() or @c tell(). */
89  PSI_FILE_TELL = 9,
90  /** File flush, as in @c fflush(). */
91  PSI_FILE_FLUSH = 10,
92  /** File stat, as in @c stat(). */
93  PSI_FILE_STAT = 11,
94  /** File stat, as in @c fstat(). */
95  PSI_FILE_FSTAT = 12,
96  /** File chsize, as in @c my_chsize(). */
97  PSI_FILE_CHSIZE = 13,
98  /** File delete, such as @c my_delete() or @c my_delete_with_symlink(). */
99  PSI_FILE_DELETE = 14,
100  /** File rename, such as @c my_rename() or @c my_rename_with_symlink(). */
101  PSI_FILE_RENAME = 15,
102  /** File sync, as in @c fsync() or @c my_sync(). */
103  PSI_FILE_SYNC = 16
104 };
105 typedef enum PSI_file_operation PSI_file_operation;

133 /**
134  State data storage for @c get_thread_file_name_locker_v1_t.
135  This structure provide temporary storage to a file locker.
136  The content of this structure is considered opaque,
137  the fields are only hints of what an implementation
138  of the psi interface can use.
139  This memory is provided by the instrumented code for performance reasons.
140  @sa get_thread_file_name_locker_v1_t
141  @sa get_thread_file_stream_locker_v1_t
142  @sa get_thread_file_descriptor_locker_v1_t
143 */
144 struct PSI_file_locker_state_v1 {
145  /** Internal state. */
146  unsigned int m_flags;
147  /** Current operation. */
148  enum PSI_file_operation m_operation;
149  /** Current file. */
150  struct PSI_file *m_file;
151  /** Current file name. */
152  const char *m_name;
153  /** Current file class. */
154  void *m_class;
155  /** Current thread. */
156  struct PSI_thread *m_thread;
157  /** Operation number of bytes. */
158  size_t m_number_of_bytes;
159  /** Timer start. */
160  unsigned long long m_timer_start;
161  /** Timer function. */
162  unsigned long long (*m_timer)(void);
163  /** Internal data. */
164  void *m_wait;
165 };
166 typedef struct PSI_file_locker_state_v1 PSI_file_locker_state_v1;
