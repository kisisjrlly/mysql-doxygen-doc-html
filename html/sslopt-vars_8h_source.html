74 static inline int set_client_ssl_options(MYSQL *mysql) {
75  /*
76  Print a warning if explicitly defined combination of --ssl-mode other than
77  VERIFY_CA or VERIFY_IDENTITY with explicit --ssl-ca or --ssl-capath values.
78  */
79  if (ssl_mode_set_explicitly && opt_ssl_mode < SSL_MODE_VERIFY_CA &&
80  (opt_ssl_ca || opt_ssl_capath)) {
81  fprintf(stderr,
82  "WARNING: no verification of server certificate will be done. "
83  "Use --ssl-mode=VERIFY_CA or VERIFY_IDENTITY.\n");
84  }
85 
86  /* Set SSL parameters: key, cert, ca, capath, cipher, clr, clrpath. */
87  if (opt_ssl_mode >= SSL_MODE_VERIFY_CA)
88  mysql_ssl_set(mysql, opt_ssl_key, opt_ssl_cert, opt_ssl_ca, opt_ssl_capath,
89  opt_ssl_cipher);
90  else
91  mysql_ssl_set(mysql, opt_ssl_key, opt_ssl_cert, nullptr, nullptr,
92  opt_ssl_cipher);
93  mysql_options(mysql, MYSQL_OPT_SSL_CRL, opt_ssl_crl);
94  mysql_options(mysql, MYSQL_OPT_SSL_CRLPATH, opt_ssl_crlpath);
95  mysql_options(mysql, MYSQL_OPT_TLS_VERSION, opt_tls_version);
96  mysql_options(mysql, MYSQL_OPT_SSL_MODE, &opt_ssl_mode);
97  mysql_options(mysql, MYSQL_OPT_SSL_FIPS_MODE, &opt_ssl_fips_mode);
98  if (opt_ssl_fips_mode > 0 && mysql_errno(mysql) == CR_SSL_FIPS_MODE_ERR)
99  return 1;
100  mysql_options(mysql, MYSQL_OPT_TLS_CIPHERSUITES, opt_tls_ciphersuites);
101 
102  return 0;
103 }
