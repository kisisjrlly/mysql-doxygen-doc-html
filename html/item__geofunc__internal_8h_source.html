89 inline void make_bg_box(const Geometry *g, BG_box *box) {
90  MBR mbr;
91  g->envelope(&mbr);
92  box->min_corner().set<0>(mbr.xmin);
93  box->min_corner().set<1>(mbr.ymin);
94  box->max_corner().set<0>(mbr.xmax);

98 inline bool is_box_valid(const BG_box &box) {
99  return !(!std::isfinite(box.min_corner().get<0>()) ||
100  !std::isfinite(box.min_corner().get<1>()) ||
101  !std::isfinite(box.max_corner().get<0>()) ||
102  !std::isfinite(box.max_corner().get<1>()) ||
103  box.max_corner().get<0>() < box.min_corner().get<0>() ||
104  box.max_corner().get<1>() < box.min_corner().get<1>());
105 }

155 struct bgpt_lt {
156  template <typename Point>
157  bool operator()(const Point &p1, const Point &p2) const {
158  if (p1.template get<0>() != p2.template get<0>())
159  return p1.template get<0>() < p2.template get<0>();
