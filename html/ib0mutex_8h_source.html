44 /** OS mutex for tracking lock/unlock for debugging */
45 template <template <typename> class Policy = NoPolicy>
46 struct OSTrackMutex {
47  typedef Policy<OSTrackMutex> MutexPolicy;
48 
49  explicit OSTrackMutex(bool destroy_mutex_at_exit = true) UNIV_NOTHROW {
50  ut_d(m_freed = true);
51  ut_d(m_locked = false);
52  ut_d(m_destroy_at_exit = destroy_mutex_at_exit);
53  }
54 
55  ~OSTrackMutex() UNIV_NOTHROW { ut_ad(!m_destroy_at_exit || !m_locked); }
56 
57  /** Initialise the mutex.
58  @param[in] id Mutex ID
59  @param[in] filename File where mutex was created
60  @param[in] line Line in filename */
61  void init(latch_id_t id, const char *filename, uint32_t line) UNIV_NOTHROW {
62  ut_ad(m_freed);
63  ut_ad(!m_locked);
64 
65  m_mutex.init();
66 
67  ut_d(m_freed = false);
68 
69  m_policy.init(*this, id, filename, line);
70  }
71 
72  /** Destroy the mutex */
73  void destroy() UNIV_NOTHROW {
74  ut_ad(!m_locked);
75  ut_ad(innodb_calling_exit || !m_freed);
76 
77  m_mutex.destroy();
78 
79  ut_d(m_freed = true);
80 
81  m_policy.destroy();
82  }
83 
84  /** Release the mutex. */
85  void exit() UNIV_NOTHROW {
86  ut_ad(m_locked);
87  ut_d(m_locked = false);
88  ut_ad(innodb_calling_exit || !m_freed);
89 
90  m_mutex.exit();
91  }
92 
93  /** Acquire the mutex.
94  @param[in] max_spins max number of spins
95  @param[in] max_delay max delay per spin
96  @param[in] filename from where called
97  @param[in] line within filename */
98  void enter(uint32_t max_spins, uint32_t max_delay, const char *filename,
99  uint32_t line) UNIV_NOTHROW {
100  ut_ad(innodb_calling_exit || !m_freed);
101 
102  m_mutex.enter();
103 
104  ut_ad(!m_locked);
105  ut_d(m_locked = true);
106  }
107 
108  /** @return true if locking succeeded */
109  bool try_lock() UNIV_NOTHROW {
110  ut_ad(innodb_calling_exit || !m_freed);
111 
112  bool locked = m_mutex.try_lock();
113 
114  if (locked) {
115  ut_ad(!m_locked);
116  ut_d(m_locked = locked);
117  }
118 
119  return (locked);
120  }
121 
122 #ifdef UNIV_DEBUG
123  /** @return true if the thread owns the mutex. */
124  bool is_owned() const UNIV_NOTHROW {
125  return (m_locked && m_policy.is_owned());
126  }
127 #endif /* UNIV_DEBUG */
128 
129  /** @return non-const version of the policy */
130  MutexPolicy &policy() UNIV_NOTHROW { return (m_policy); }
131 
132  /** @return the const version of the policy */
133  const MutexPolicy &policy() const UNIV_NOTHROW { return (m_policy); }
134 
135  private:
136 #ifdef UNIV_DEBUG
137  /** true if the mutex has not be initialized */
138  bool m_freed;
139 
140  /** true if the mutex has been locked. */
141  bool m_locked;
142 
143  /** Do/Dont destroy mutex at exit */
144  bool m_destroy_at_exit;
145 #endif /* UNIV_DEBUG */
146 
147  /** OS Mutex instance */
148  OSMutex m_mutex;
149 
150  /** Policy data */
151  MutexPolicy m_policy;
152 };

159 /** Mutex implementation that used the Linux futex. */
160 template <template <typename> class Policy = NoPolicy>
161 struct TTASFutexMutex {
162  typedef Policy<TTASFutexMutex> MutexPolicy;
163 
164  TTASFutexMutex() UNIV_NOTHROW : m_lock_word(MUTEX_STATE_UNLOCKED) {
165  /* Check that lock_word is aligned. */
166  ut_ad(!((ulint)&m_lock_word % sizeof(ulint)));
167  }
168 
169  ~TTASFutexMutex() { ut_a(m_lock_word == MUTEX_STATE_UNLOCKED); }
170 
171  /** Called when the mutex is "created". Note: Not from the constructor
172  but when the mutex is initialised.
173  @param[in] id Mutex ID
174  @param[in] filename File where mutex was created
175  @param[in] line Line in filename */
176  void init(latch_id_t id, const char *filename, uint32_t line) UNIV_NOTHROW {
177  ut_a(m_lock_word == MUTEX_STATE_UNLOCKED);
178  m_policy.init(*this, id, filename, line);
179  }
180 
181  /** Destroy the mutex. */
182  void destroy() UNIV_NOTHROW {
183  /* The destructor can be called at shutdown. */
184  ut_a(m_lock_word == MUTEX_STATE_UNLOCKED);
185  m_policy.destroy();
186  }
187 
188  /** Acquire the mutex.
189  @param[in] max_spins max number of spins
190  @param[in] max_delay max delay per spin
191  @param[in] filename from where called
192  @param[in] line within filename */
193  void enter(uint32_t max_spins, uint32_t max_delay, const char *filename,
194  uint32_t line) UNIV_NOTHROW {
195  uint32_t n_spins;
196  lock_word_t lock = ttas(max_spins, max_delay, n_spins);
197 
198  /* If there were no waiters when this thread tried
199  to acquire the mutex then set the waiters flag now.
200  Additionally, when this thread set the waiters flag it is
201  possible that the mutex had already been released
202  by then. In this case the thread can assume it
203  was granted the mutex. */
204 
205  uint32_t n_waits;
206 
207  if (lock != MUTEX_STATE_UNLOCKED) {
208  if (lock != MUTEX_STATE_LOCKED || !set_waiters()) {
209  n_waits = wait();
210  } else {
211  n_waits = 0;
212  }
213 
214  } else {
215  n_waits = 0;
216  }
217 
218  m_policy.add(n_spins, n_waits);
219  }
220 
221  /** Release the mutex. */
222  void exit() UNIV_NOTHROW {
223  /* If there are threads waiting then we have to wake
224  them up. Reset the lock state to unlocked so that waiting
225  threads can test for success. */
226 
227  std::atomic_thread_fence(std::memory_order_acquire);
228 
229  if (state() == MUTEX_STATE_WAITERS) {
230  m_lock_word = MUTEX_STATE_UNLOCKED;
231 
232  } else if (unlock() == MUTEX_STATE_LOCKED) {
233  /* No threads waiting, no need to signal a wakeup. */
234  return;
235  }
236 
237  signal();
238  }
239 
240  /** Try and lock the mutex.
241  @return the old state of the mutex */
242  lock_word_t trylock() UNIV_NOTHROW {
243  return (CAS(&m_lock_word, MUTEX_STATE_UNLOCKED, MUTEX_STATE_LOCKED));
244  }
245 
246  /** Try and lock the mutex.
247  @return true if successful */
248  bool try_lock() UNIV_NOTHROW { return (trylock() == MUTEX_STATE_UNLOCKED); }
249 
250  /** @return true if mutex is unlocked */
251  bool is_locked() const UNIV_NOTHROW {
252  return (state() != MUTEX_STATE_UNLOCKED);
253  }
254 
255 #ifdef UNIV_DEBUG
256  /** @return true if the thread owns the mutex. */
257  bool is_owned() const UNIV_NOTHROW {
258  return (is_locked() && m_policy.is_owned());
259  }
260 #endif /* UNIV_DEBUG */
261 
262  /** @return non-const version of the policy */
263  MutexPolicy &policy() UNIV_NOTHROW { return (m_policy); }
264 
265  /** @return const version of the policy */
266  const MutexPolicy &policy() const UNIV_NOTHROW { return (m_policy); }
267 
268  private:
269  /** @return the lock state. */
270  lock_word_t state() const UNIV_NOTHROW { return (m_lock_word); }
271 
272  /** Release the mutex.
273  @return the new state of the mutex */
274  lock_word_t unlock() UNIV_NOTHROW {
275  return (TAS(&m_lock_word, MUTEX_STATE_UNLOCKED));
276  }
277 
278  /** Note that there are threads waiting and need to be woken up.
279  @return true if state was MUTEX_STATE_UNLOCKED (ie. granted) */
280  bool set_waiters() UNIV_NOTHROW {
281  return (TAS(&m_lock_word, MUTEX_STATE_WAITERS) == MUTEX_STATE_UNLOCKED);
282  }
283 
284  /** Set the waiters flag, only if the mutex is locked
285  @return true if succesful. */
286  bool try_set_waiters() UNIV_NOTHROW {
287  return (CAS(&m_lock_word, MUTEX_STATE_LOCKED, MUTEX_STATE_WAITERS) !=
288  MUTEX_STATE_UNLOCKED);
289  }
290 
291  /** Wait if the lock is contended.
292  @return the number of waits */
293  uint32_t wait() UNIV_NOTHROW {
294  uint32_t n_waits = 0;
295 
296  /* Use FUTEX_WAIT_PRIVATE because our mutexes are
297  not shared between processes. */
298 
299  do {
300  ++n_waits;
301 
302  syscall(SYS_futex, &m_lock_word, FUTEX_WAIT_PRIVATE, MUTEX_STATE_WAITERS,
303  0, 0, 0);
304 
305  // Since we are retrying the operation the return
306  // value doesn't matter.
307 
308  } while (!set_waiters());
309 
310  return (n_waits);
311  }
312 
313  /** Wakeup a waiting thread */
314  void signal() UNIV_NOTHROW {
315  syscall(SYS_futex, &m_lock_word, FUTEX_WAKE_PRIVATE, 1, 0, 0, 0);
316  }
317 
318  /** Poll waiting for mutex to be unlocked.
319  @param[in] max_spins max spins
320  @param[in] max_delay max delay per spin
321  @param[out] n_spins retries before acquire
322  @return value of lock word before locking. */
323  lock_word_t ttas(uint32_t max_spins, uint32_t max_delay,
324  uint32_t &n_spins) UNIV_NOTHROW {
325  std::atomic_thread_fence(std::memory_order_acquire);
326 
327  for (n_spins = 0; n_spins < max_spins; ++n_spins) {
328  if (!is_locked()) {
329  lock_word_t lock = trylock();
330 
331  if (lock == MUTEX_STATE_UNLOCKED) {
332  /* Lock successful */
333  return (lock);
334  }
335  }
336 
337  ut_delay(ut_rnd_interval(0, max_delay));
338  }
339 
340  return (trylock());
341  }
342 
343  private:
344  /** Policy data */
345  MutexPolicy m_policy;
346 
347  /** lock_word is the target of the atomic test-and-set instruction
348  when atomic operations are enabled. */
349  alignas(ulint) lock_word_t m_lock_word;
350 };

355 struct TTASMutex {
356  typedef Policy<TTASMutex> MutexPolicy;
357 
358  TTASMutex() UNIV_NOTHROW : m_lock_word(MUTEX_STATE_UNLOCKED) {
359  /* Check that lock_word is aligned. */
360  ut_ad(!((ulint)&m_lock_word % sizeof(ulint)));
361  }
362 
363  ~TTASMutex() { ut_ad(m_lock_word == MUTEX_STATE_UNLOCKED); }
364 
365  /** Called when the mutex is "created". Note: Not from the constructor
366  but when the mutex is initialised.
367  @param[in] id Mutex ID
368  @param[in] filename File where mutex was created
369  @param[in] line Line in filename */
370  void init(latch_id_t id, const char *filename, uint32_t line) UNIV_NOTHROW {
371  ut_ad(m_lock_word == MUTEX_STATE_UNLOCKED);
372  m_policy.init(*this, id, filename, line);
373  }
374 
375  /** Destroy the mutex. */
376  void destroy() UNIV_NOTHROW {
377  /* The destructor can be called at shutdown. */
378  ut_ad(m_lock_word == MUTEX_STATE_UNLOCKED);
379  m_policy.destroy();
380  }
381 
382  /**
383  Try and acquire the lock using TestAndSet.
384  @return true if lock succeeded */
385  bool tas_lock() UNIV_NOTHROW {
386  return (TAS(&m_lock_word, MUTEX_STATE_LOCKED) == MUTEX_STATE_UNLOCKED);
387  }
388 
389  /** In theory __sync_lock_release should be used to release the lock.
390  Unfortunately, it does not work properly alone. The workaround is
391  that more conservative __sync_lock_test_and_set is used instead. */
392  void tas_unlock() UNIV_NOTHROW {
393 #ifdef UNIV_DEBUG
394  ut_ad(state() == MUTEX_STATE_LOCKED);
395 
396  lock_word_t lock =
397 #endif /* UNIV_DEBUG */
398 
399  TAS(&m_lock_word, MUTEX_STATE_UNLOCKED);
400 
401  ut_ad(lock == MUTEX_STATE_LOCKED);
402  }
403 
404  /** Try and lock the mutex.
405  @return true on success */
406  bool try_lock() UNIV_NOTHROW { return (tas_lock()); }
407 
408  /** Release the mutex. */
409  void exit() UNIV_NOTHROW { tas_unlock(); }
410 
411  /** Acquire the mutex.
412  @param max_spins max number of spins
413  @param max_delay max delay per spin
414  @param filename from where called
415  @param line within filename */
416  void enter(uint32_t max_spins, uint32_t max_delay, const char *filename,
417  uint32_t line) UNIV_NOTHROW {
418  if (!try_lock()) {
419  uint32_t n_spins = ttas(max_spins, max_delay);
420 
421  /* No OS waits for spin mutexes */
422  m_policy.add(n_spins, 0);
423  }
424  }
425 
426  /** @return the lock state. */
427  lock_word_t state() const UNIV_NOTHROW { return (m_lock_word); }
428 
429  /** @return true if locked by some thread */
430  bool is_locked() const UNIV_NOTHROW {
431  return (m_lock_word != MUTEX_STATE_UNLOCKED);
432  }
433 
434 #ifdef UNIV_DEBUG
435  /** @return true if the calling thread owns the mutex. */
436  bool is_owned() const UNIV_NOTHROW {
437  return (is_locked() && m_policy.is_owned());
438  }
439 #endif /* UNIV_DEBUG */
440 
441  /** @return non-const version of the policy */
442  MutexPolicy &policy() UNIV_NOTHROW { return (m_policy); }
443 
444  /** @return const version of the policy */
445  const MutexPolicy &policy() const UNIV_NOTHROW { return (m_policy); }
446 
447  private:
448  /** Spin and try to acquire the lock.
449  @param[in] max_spins max spins
450  @param[in] max_delay max delay per spin
451  @return number spins before acquire */
452  uint32_t ttas(uint32_t max_spins, uint32_t max_delay) UNIV_NOTHROW {
453  uint32_t i = 0;
454  const uint32_t step = max_spins;
455 
456  std::atomic_thread_fence(std::memory_order_acquire);
457 
458  do {
459  while (is_locked()) {
460  ut_delay(ut_rnd_interval(0, max_delay));
461 
462  ++i;
463 
464  if (i >= max_spins) {
465  max_spins += step;
466 
467  os_thread_yield();
468 
469  break;
470  }
471  }
472 
473  } while (!try_lock());
474 
475  return (i);
476  }
477 
478  private:
479  // Disable copying
480  TTASMutex(const TTASMutex &);
481  TTASMutex &operator=(const TTASMutex &);
482 
483  /** Policy data */
484  MutexPolicy m_policy;
485 
486  /** lock_word is the target of the atomic test-and-set instruction
487  when atomic operations are enabled. */
488  lock_word_t m_lock_word;
489 };

495  TTASEventMutex() UNIV_NOTHROW : m_lock_word(MUTEX_STATE_UNLOCKED),

504  /** Called when the mutex is "created". Note: Not from the constructor

506  @param[in] id Mutex ID
507  @param[in] filename File where mutex was created
508  @param[in] line Line in filename */

511  ut_a(!m_lock_word.load(std::memory_order_relaxed));

514 
515  m_policy.init(*this, id, filename, line);
516  }

519  its desctructor will be called on exit(). We can't call
520  os_event_destroy() at that stage. */

522  ut_ad(!m_lock_word.load(std::memory_order_relaxed));

531  /** Try and lock the mutex. Note: POSIX returns 0 on success.
532  @return true on success */
533  bool try_lock() UNIV_NOTHROW {
534  bool expected = false;
535  return (m_lock_word.compare_exchange_strong(expected, true));
536  }
537 
538  /** Release the mutex. */
539  void exit() UNIV_NOTHROW {
540  m_lock_word.store(false);
541 
542  if (m_waiters.load()) {
543  signal();
544  }
545  }

547  /** Acquire the mutex.
548  @param[in] max_spins max number of spins
549  @param[in] max_delay max delay per spin
550  @param[in] filename from where called
551  @param[in] line within filename */
552  void enter(uint32_t max_spins, uint32_t max_delay, const char *filename,
553  uint32_t line) UNIV_NOTHROW {
554  if (!try_lock()) {
555  spin_and_try_lock(max_spins, max_delay, filename, line);

586  private:
587  /** Wait in the sync array.
588  @param[in] filename from where it was called
589  @param[in] line line number in file
590  @param[in] spin retry this many times again
591  @return true if the mutex acquisition was successful. */
592  bool wait(const char *filename, uint32_t line, uint32_t spin) UNIV_NOTHROW;
593 
594  /** Spin and wait for the mutex to become free.
595  @param[in] max_spins max spins
596  @param[in] max_delay max delay per spin
597  @param[in,out] n_spins spin start index
598  @return true if unlocked */
599  bool is_free(uint32_t max_spins, uint32_t max_delay,
600  uint32_t &n_spins) const UNIV_NOTHROW {
601  ut_ad(n_spins <= max_spins);
602 
603  /* Spin waiting for the lock word to become zero. Note
604  that we do not have to assume that the read access to
605  the lock word is atomic, as the actual locking is always
606  committed with atomic test-and-set. In reality, however,
607  all processors probably have an atomic read of a memory word. */
608 
609  do {
610  if (!is_locked()) {
611  return (true);
612  }
613 
614  ut_delay(ut_rnd_interval(0, max_delay));
615 
616  ++n_spins;
617 
618  } while (n_spins < max_spins);
619 
620  return (false);
621  }
622 
623  /** Spin while trying to acquire the mutex
624  @param[in] max_spins max number of spins
625  @param[in] max_delay max delay per spin
626  @param[in] filename from where called
627  @param[in] line within filename */
628  void spin_and_try_lock(uint32_t max_spins, uint32_t max_delay,
629  const char *filename, uint32_t line) UNIV_NOTHROW {
630  uint32_t n_spins = 0;
631  uint32_t n_waits = 0;
632  const uint32_t step = max_spins;
633 
634  for (;;) {
635  /* If the lock was free then try and acquire it. */
636 
637  if (is_free(max_spins, max_delay, n_spins)) {
638  if (try_lock()) {
639  break;
640  } else {
641  continue;
642  }
643 
644  } else {
645  max_spins = n_spins + step;
646  }
647 
648  ++n_waits;
649 
650  os_thread_yield();
651 
652  /* The 4 below is a heuristic that has existed for a
653  very long time now. It is unclear if changing this
654  value will make a difference.
655 
656  NOTE: There is a delay that happens before the retry,
657  finding a free slot in the sync arary and the yield
658  above. Otherwise we could have simply done the extra
659  spin above. */
660 
661  if (wait(filename, line, 4)) {
662  n_spins += 4;
663 
664  break;
665  }
666  }
667 
668  /* Waits and yields will be the same number in our
669  mutex design */
670 
671  m_policy.add(n_spins, n_waits);
672  }

674  /** Note that there are threads waiting on the mutex */
675  void set_waiters() UNIV_NOTHROW { m_waiters.store(true); }
676 
677  /** Note that there are no threads waiting on the mutex */
678  void clear_waiters() UNIV_NOTHROW { m_waiters.store(false); }
679 
680  /** Wakeup any waiting thread(s). */
681  void signal() UNIV_NOTHROW;

765  @param name filename where locked
766  @param line line number where locked */

773  PSI_mutex_locker *locker;

776 #endif /* UNIV_PFS_MUTEX */

779  after locking here. This is only done to avoid add and
780  then remove if the trylock was unsuccesful. */
781

792 #endif /* UNIV_PFS_MUTEX */
