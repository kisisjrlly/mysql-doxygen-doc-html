28 class Gcs_group_management_interface {
29  public:
30  /**
31  Method that allows sending of a new group configuration. This is
32  to be used when the group is blocked due to a loss of majority and
33  only on one node from the new group.
34 
35  Implementations in each binding might have there specificities but some
36  rules must be followed:
37  - This should be a non-blocking call;
38  - Typically, it shall cause a View Change, since we are forcing a new
39  configuration.
40 
41  @param[in] reconfigured_group a list containing the new nodes
42  */
43 
44  virtual enum_gcs_error modify_configuration(
45  const Gcs_interface_parameters &reconfigured_group) = 0;
46 
47  /**
48  Retrieves the minimum supported "write concurrency" value.
49  */
50  virtual uint32_t get_minimum_write_concurrency() const = 0;
51 
52  /**
53  Retrieves the maximum supported "write concurrency" value.
54  */
55  virtual uint32_t get_maximum_write_concurrency() const = 0;
56 
57  /**
58  Retrieves the group's "write concurrency" value.
59 
60  @param[out] write_concurrency A reference to where the group's
61  "write concurrency" value will be written to
62 
63  @retval GCS_OK if successful and write_concurrency has been written to
64  @retval GCS_NOK if GCS was unable to request the current "write concurrency"
65  value from the underlying implementation
66  */
67  virtual enum_gcs_error get_write_concurrency(
68  uint32_t &write_concurrency) const = 0;
69 
70  /**
71  Reconfigures the group's "write concurrency" value.
72 
73  The caller should ensure that the supplied value is between @c
74  minimum_write_concurrency and @c maximum_write_concurrency.
75 
76  The method is non-blocking, meaning that it shall only send the request to
77  an underlying GCS. The final result can be polled via @c
78  get_write_concurrency.
79 
80  @param write_concurrency The desired "write concurrency" value.
81 
82  @retval GCS_OK if successful
83  @retval GCS_NOK if unsuccessful
84  */
85  virtual enum_gcs_error set_write_concurrency(uint32_t write_concurrency) = 0;
86 
87  virtual ~Gcs_group_management_interface() {}
88 };
