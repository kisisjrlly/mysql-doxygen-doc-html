57 /**

59 
60  @param ctx Plugin's context
61  @param num_cols Number of fields being sent
62  @param flags Flags to alter the metadata sending
63  @param resultcs Charset of the result set
64 
65  @note resultcs is the charset in which the data should be encoded before
66  sent to the client. This is the value of the session variable
67  character_set_results. The implementor most probably will need to save
68  this value in the context and use it as "to" charset in get_string().
69

317 struct st_command_service_cbs {
318  /*
319  For a statement that returns a result, the flow of called callbacks will be:
320 
321  start_result_metadata()
322  field_metadata()
323  ....
324  end_result_metadata() (in the classic protocol this generates an EOF packet)
325  start_row()
326  get_xxx()
327  ...
328  end_row()
329  start_row()
330  get_xxx()
331  ...
332  end_row()
333  handle_ok() (with data for an EOF packet)
334 
335  For a statement that does NOT return a result, but only status, like
336  INSERT, UPDATE, DELETE, REPLACE, TRUNCATE, CREATE, DROP, ALTER, etc. only
337  handle_ok() will be invoked, in case of success.
338 
339  All statements that result in an error will invoke handle_error().
340 
341  For statements that return a result set, handle_error() might be invoked
342  even after metadata was sent. This will indicate an error during the
343  execution of the statement.
344  */

395 /**
396  Executes a server command in a session.
397 
398 
399  There are two cases. Execution in a physical thread :
400  1. initialized by the srv_session service
401  2. NOT initialized by the srv_session service
402 
403  In case of 1, if there is currently attached session, and it is
404  different from the passed one, the former will be automatically
405  detached. The session to be used for the execution will then be
406  attached. After the command is executed, the attached session will
407  not be detached. It will be detached by a next call to run_command()
408  with another session as parameter. In other words, for all sessions
409  used in a physical thread, there will be at most one in attached
410  state.
411 
412  In case of 2, the current state (current_thd) will be
413  preserved. Then the given session will move to attached state and
414  the command will be executed. After the execution the state of the
415  session will be changed to detached and the preserved state
416  (current_thd) will be restored.
417 
418  The client charset is used for commands like COM_QUERY and
419  COM_STMT_PREPARE to know how to threat the char* fields. This
420  charset will be used until the next call of run_command when it may
421  be changed again.
422 
423  @param session The session
424  @param command The command to be executed.
425  @param data The data needed for the command to be executed
426  @param client_cs The charset for the string data input(COM_QUERY for example)
427  @param callbacks Callbacks to be used by the server to encode data and
428  to communicate with the client (plugin) side.
429  @param text_or_binary Select which representation the server will use for the
430  data passed to the callbacks. For more information
431  @see cs_text_or_binary enum
432  @param service_callbacks_ctx Context passed to the command service callbacks
433 
434  @return
435  0 success
436  1 failure
437 */
438 int command_service_run_command(MYSQL_SESSION session,
439  enum enum_server_command command,
440  const union COM_DATA *data,
441  const CHARSET_INFO *client_cs,
442  const struct st_command_service_cbs *callbacks,
443  enum cs_text_or_binary text_or_binary,
444  void *service_callbacks_ctx);
