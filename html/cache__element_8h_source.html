45 /**
46  Implementation of a cache element.
47 
48  This template class implements a wrapper to support caching of
49  arbitrary objects. The wrapper provides support for reference counting,
50  but does not make any assumptions regarding the semantics of this
51  functionality. The enforcement of such assumptions must be built into
52  the layer using the cache element implementation.
53 
54  The cache element stores copies of the keys that are used for looking
55  up the object in the cache. This is needed to support fast reverse lookup
56  of keys, given the object instance, e.g. to enable removing old keys when
57  new keys must be created. The keys are stored in pre-allocated memory.
58 
59  @note The usage of the reference counter is not implemented by means of
60  atomic operations. Locking at an outer level takes care of race
61  conditions.
62 
63  @tparam T Dictionary object type being wrapped.
64 */
65 
66 template <typename T>
67 class Cache_element {
68  friend class Storage_adapter; // Unit test access.
69  friend class dd_cache_unittest::CacheTestHelper; // Unit test access.
70  friend class Shared_multi_map<T>; // Access to changing data.
71  friend class Dictionary_client; // Access to changing data.
72 
73  private:
74  const T *m_object; // Pointer to the actual object.
75  uint m_ref_counter; // Number of concurrent object usages.
76 
77  /**
78  Helper class to represent a key instance. We also need to
79  represent whether the instance is NULL.
80 
81  @tparam K Key type.
82  */
83 
84  template <typename K>
85  class Key_wrapper {
86  public:
87  bool is_null;
88  K key;
89  Key_wrapper() : is_null(true), key() {}
90  };
91 
92  Key_wrapper<typename T::Id_key> m_id_key; // The id key for the object.
93  Key_wrapper<typename T::Name_key> m_name_key; // The name key for the object.
94  Key_wrapper<typename T::Aux_key> m_aux_key; // The aux key for the object.
95 
96  // Helper functions using overloading to get keys using a template.
97  template <typename K>
98  struct Type_selector {};
99 
100  const T *const *get_key(Type_selector<const T *>) const {
101  return m_object ? &m_object : nullptr;
102  }
103 
104  const typename T::Id_key *get_key(Type_selector<typename T::Id_key>) const {
105  return id_key();
106  }
107 
108  const typename T::Name_key *get_key(
109  Type_selector<typename T::Name_key>) const {
110  return name_key();
111  }
112 
113  const typename T::Aux_key *get_key(Type_selector<typename T::Aux_key>) const {
114  return aux_key();
115  }
116 
117  // Delete all keys.
118  void delete_keys() {
119  m_id_key.is_null = true;
120  m_name_key.is_null = true;
121  m_aux_key.is_null = true;
122  }
123 
124  // Increment the reference counter associated with the object.
125  void use() { m_ref_counter++; }
126 
127  // Let the cache element point to another object.
128  void set_object(const T *replacement_object) {
129  m_object = replacement_object;
130  }
131 
132  // Update the keys based on the object pointed to.
133  void recreate_keys() {
134  DBUG_ASSERT(m_object);
135  m_id_key.is_null = m_object->update_id_key(&m_id_key.key);
136  m_name_key.is_null = m_object->update_name_key(&m_name_key.key);
137  m_aux_key.is_null = m_object->update_aux_key(&m_aux_key.key);
138  }
139 
140  public:
141  // Initialize an instance to having NULL pointers and 0 count.
142  Cache_element()
143  : m_object(nullptr),
144  m_ref_counter(0),
145  m_id_key(),
146  m_name_key(),
147  m_aux_key() {} /* purecov: tested */
148 
149  // Note that the object being pointed to is not deleted implicitly.
150  ~Cache_element() { delete_keys(); }
151 
152  // Initialize an existing instance.
153  void init() {
154  m_object = nullptr;
155  m_ref_counter = 0;
156  delete_keys();
157  }
158 
159  // Decrement the reference counter associated with the object.
160  void release() {
161  DBUG_ASSERT(m_ref_counter > 0);
162  m_ref_counter--;
163  }
164 
165  // Return current number of usages of the object.
166  uint usage() const { return m_ref_counter; }
167 
168  // Return the object pointer.
169  const T *object() const { return m_object; }
170 
171  // Get the id key.
172  const typename T::Id_key *id_key() const {
173  return m_id_key.is_null ? nullptr : &m_id_key.key;
174  }
175 
176  // Get the name key.
177  const typename T::Name_key *name_key() const {
178  return m_name_key.is_null ? nullptr : &m_name_key.key;
179  }
180 
181  // Get the aux key.
182  const typename T::Aux_key *aux_key() const {
183  return m_aux_key.is_null ? nullptr : &m_aux_key.key;
184  }
185 
186  /**
187  Template function to get a pointer to a key based on the type.
188 
189  @tparam K Key type.
190  */
191 
192  template <typename K>
193  const K *get_key() const {
194  return get_key(Type_selector<K>());
195  }
196 
197  // Debug dump of the element to stderr.
198  /* purecov: begin inspected */
199  void dump(const String_type &prefix MY_ATTRIBUTE((unused)) = " ") const {
200 #ifndef DBUG_OFF
201  fprintf(stderr, "%sobj: %p, id: %llu, cnt: %u", prefix.c_str(), m_object,
202  m_object ? m_object->id() : 0, m_ref_counter);
203  fprintf(stderr, ", id_k: %s",
204  m_id_key.is_null ? "NULL" : m_id_key.key.str().c_str());
205  fprintf(stderr, ", name_k: %s",
206  m_name_key.is_null ? "NULL" : m_name_key.key.str().c_str());
207  fprintf(stderr, ", aux_k: %s\n",
208  m_aux_key.is_null ? "NULL" : m_aux_key.key.str().c_str());
209 #endif
210  }
211  /* purecov: end */
212 };
