68 /** Wrapper for a callable, it will count the number of registered
69 Runnable instances and will register the thread executing the callable
70 with the PFS and the Server threading infrastructure. */
71 class Runnable {
72  public:
73 #ifdef UNIV_PFS_THREAD
74  /** Constructor for the Runnable object.
75  @param[in] pfs_key Performance schema key */
76  explicit Runnable(mysql_pfs_key_t pfs_key) : m_pfs_key(pfs_key) { init(); }
77 #else
78  /** Constructor for the Runnable object.
79  @param[in] pfs_key Performance schema key (ignored) */
80  explicit Runnable(mysql_pfs_key_t) { init(); }
81 #endif /* UNIV_PFS_THREAD */
82 
83  public:
84  /** Method to execute the callable
85  @param[in] f Callable object
86  @param[in] args Variable number of args to F */
87  template <typename F, typename... Args>
88  void operator()(F &&f, Args &&... args) {
89  while (m_thread.state() == IB_thread::State::NOT_STARTED) {
90  UT_RELAX_CPU();
91  }
92 
93  ut_a(m_thread.state() == IB_thread::State::ALLOWED_TO_START);
94 
95  preamble();
96 
97  m_thread.set_state(IB_thread::State::STARTED);
98 
99  auto task = std::bind(std::forward<F>(f), std::forward<Args>(args)...);
100 
101  task();
102 
103  epilogue();
104 
105  m_thread.set_state(IB_thread::State::STOPPED);
106  }
107 
108  IB_thread thread() const { return (m_thread); }
109 
110  private:
111  /** Register the thread with the server */
112  void preamble() {
113  my_thread_init();
114 
115 #if defined(UNIV_PFS_THREAD) && !defined(UNIV_HOTBACKUP)
116  if (m_pfs_key.m_value != PFS_NOT_INSTRUMENTED.m_value) {
117  PSI_thread *psi;
118 
119  psi = PSI_THREAD_CALL(new_thread)(m_pfs_key.m_value, nullptr, 0);
120 
121  PSI_THREAD_CALL(set_thread_os_id)(psi);
122  PSI_THREAD_CALL(set_thread)(psi);
123  }
124 #endif /* UNIV_PFS_THREAD && !UNIV_HOTBACKUP */
125 
126  std::atomic_thread_fence(std::memory_order_release);
127 
128  int old;
129 
130  old = os_thread_count.fetch_add(1, std::memory_order_relaxed);
131 
132  ut_a(old <= static_cast<int>(srv_max_n_threads) - 1);
133  }
134 
135  /** Deregister the thread */
136  void epilogue() {
137  std::atomic_thread_fence(std::memory_order_release);
138 
139  int old;
140 
141  old = os_thread_count.fetch_sub(1, std::memory_order_relaxed);
142  ut_a(old > 0);
143 
144  my_thread_end();
145 
146 #if defined(UNIV_PFS_THREAD) && !defined(UNIV_HOTBACKUP)
147  if (m_pfs_key.m_value != PFS_NOT_INSTRUMENTED.m_value) {
148  PSI_THREAD_CALL(delete_current_thread)();
149  }
150 #endif /* UNIV_PFS_THREAD && !UNIV_HOTBACKUP */
151 
152  m_promise.set_value();
153  }
154 
155  private:
156 #ifdef UNIV_PFS_THREAD
157  /** Performance schema key */
158  const mysql_pfs_key_t m_pfs_key;
159 #endif /* UNIV_PFS_THREAD */
160 
161  /** Promise which is set when task is done. */
162  std::promise<void> m_promise;
163 
164  /** Future object which keeps the ref counter >= 1 at least
165  as long as the Runnable is non-destroyed. */
166  mutable IB_thread m_thread;
167 
168  /** Initializes the m_shared_future, uses the m_promise's get_future,
169  which cannot be used since then, according to its documentation. */
170  void init() { m_thread.init(m_promise); }
171 };

180 /** Check if thread is active
181 @param[in] thread Thread handle.
182 @return true if the thread is active. */
183 inline bool thread_is_active(const IB_thread &thread) {
184  switch (thread.state()) {
185  case IB_thread::State::NOT_STARTED:
186  /* Not yet started. */
187  return (false);
188 
189  case IB_thread::State::ALLOWED_TO_START:
190  /* Thread "thread" is already active, but start() has not been called.
191  Note that when start() is called, the thread's routine may decide to
192  check if it is active or trigger other thread to do similar check
193  regarding "thread". That could happen faster than thread's state
194  is advanced from ALLOWED_TO_START to STARTED. Therefore we must
195  already consider such thread as "active". */
196  return (true);
197 
198  case IB_thread::State::STARTED:
199  /* Note, that potentially the thread might be doing its cleanup after
200  it has already ended its task. We still consider it active, until the
201  cleanup is finished. */
202  return (true);
203 
204  case IB_thread::State::STOPPED:
205  /* Ended its task and became marked as STOPPED (cleanup finished) */
206  return (false);
207 
208  case IB_thread::State::INVALID:
209  default:
210  /* The thread object has not been assigned yet. */
211  return (false);
212  }
213 
214  /* Note that similar goal was achieved by the usage of shared_future:
215  return (shared_future.valid() && shared_future.wait_for(std::chrono::seconds(
216  0)) != std::future_status::ready);
217  However this resulted in longer execution of mtr tests (63minutes ->
218  75minutes). You could try `mtr --mem collations.esperanto` (cmake
219  WITH_DEBUG=1) */
220 }

222 /** Create a detached non-started thread. After thread is created, you should
223 assign the received object to any of variables/fields which you later could
224 access to check thread's state. You are allowed to either move or copy that
225 object (any number of copies is allowed). After assigning you are allowed to
226 start the thread by calling start() on any of those objects.
227 @param[in] pfs_key Performance schema thread key
228 @param[in] f Callable instance
229 @param[in] args Zero or more args
230 @return Object which allows to start the created thread, monitor its state and
231  wait until the thread is finished. */
232 template <typename F, typename... Args>
233 IB_thread create_detached_thread(mysql_pfs_key_t pfs_key, F &&f,
234  Args &&... args) {
235  Runnable runnable{pfs_key};
236 
237  auto thread = runnable.thread();
238 
239  std::thread t(std::move(runnable), f, args...);
240  t.detach();
241 
242  /* Thread t is doing busy waiting until the state is changed
243  from NOT_STARTED to ALLOWED_TO_START. That will happen when
244  thread.start() will be called. */
245  ut_a(thread.state() == IB_thread::State::NOT_STARTED);
246 
247  return (thread);
248 }

256 /** Parallel for loop over a container.
257 @param[in] pfs_key Performance schema thread key
258 @param[in] c Container to iterate over in parallel
259 @param[in] n Number of threads to create
260 @param[in] f Callable instance
261 @param[in] args Zero or more args */
262 template <typename Container, typename F, typename... Args>
263 void par_for(mysql_pfs_key_t pfs_key, const Container &c, size_t n, F &&f,
264  Args &&... args) {
265  if (c.empty()) {
266  return;
267  }
268 
269  size_t slice = (n > 0) ? c.size() / n : 0;
270 
271  using Workers = std::vector<IB_thread>;
272 
273  Workers workers;
274 
275  workers.reserve(n);
276 
277  for (size_t i = 0; i < n; ++i) {
278  auto b = c.begin() + (i * slice);
279  auto e = b + slice;
280 
281  auto worker = os_thread_create(pfs_key, f, b, e, i, args...);
282  worker.start();
283 
284  workers.push_back(std::move(worker));
285  }
286 
287  f(c.begin() + (n * slice), c.end(), n, args...);
288 
289  for (auto &worker : workers) {
290  worker.join();
291  }
292 }

295 #define par_for(...) par_for(__VA_ARGS__)
296 #else
297 #define par_for(k, ...) par_for(0, __VA_ARGS__)
298 #endif /* UNIV_PFS_THREAD */
