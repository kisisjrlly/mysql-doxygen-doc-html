49 /**
50  Ack_receiver is responsible to control ack receive thread and maintain
51  slave information used by ack receive thread.
52 
53  There are mainly four operations on ack receive thread:
54  start: start ack receive thread
55  stop: stop ack receive thread
56  add_slave: maintain a new semisync slave's information
57  remove_slave: remove a semisync slave's information
58  */
59 class Ack_receiver : public ReplSemiSyncBase {
60  public:
61  Ack_receiver();
62  ~Ack_receiver();
63 
64  /**
65  Notify ack receiver to receive acks on the dump session.
66 
67  It adds the given dump thread into the slave list and wakes
68  up ack thread if it is waiting for any slave coming.
69 
70  @param[in] thd THD of a dump thread.
71 
72  @return it return false if succeeds, otherwise true is returned.
73  */
74  bool add_slave(THD *thd);
75 
76  /**
77  Notify ack receiver not to receive ack on the dump session.
78 
79  it removes the given dump thread from slave list.
80 
81  @param[in] thd THD of a dump thread.
82  */
83  void remove_slave(THD *thd);
84 
85  /**
86  Start ack receive thread
87 
88  @return it return false if succeeds, otherwise true is returned.
89  */
90  bool start();
91 
92  /**
93  Stop ack receive thread
94  */
95  void stop();
96 
97  /**
98  The core of ack receive thread.
99 
100  It monitors all slaves' sockets and receives acks when they come.
101  */
102  void run();
103 
104  void setTraceLevel(unsigned long trace_level) { trace_level_ = trace_level; }
105 
106  bool init() {
107  setTraceLevel(rpl_semi_sync_master_trace_level);
108  if (rpl_semi_sync_master_enabled) return start();
109  return false;
110  }
111 
112  private:
113  enum status { ST_UP, ST_DOWN, ST_STOPPING };
114  uint8 m_status;
115  /*
116  Protect m_status, m_slaves_changed and m_slaves. ack thread and other
117  session may access the variables at the same time.
118  */
119  mysql_mutex_t m_mutex;
120  mysql_cond_t m_cond;
121  /* If slave list is updated(add or remove). */
122  bool m_slaves_changed;
123  Slave_vector m_slaves;
124  my_thread_handle m_pid;
125 
126  /* Declare them private, so no one can copy the object. */
127  Ack_receiver(const Ack_receiver &ack_receiver);
128  Ack_receiver &operator=(const Ack_receiver &ack_receiver);
129 
130  void set_stage_info(const PSI_stage_info &stage);
131  void wait_for_slave_connection();
132 };
