42 /**
43  Descriptor for a merge chunk to be sort-merged.
44  A merge chunk is a sequence of pre-sorted records, written to a
45  temporary file. A Merge_chunk instance describes where this chunk is stored
46  in the file, and where it is located when it is in memory.
47 
48  It is a POD because we read/write them from/to files (but note,
49  only m_file_position and m_rowcount are actually used in that

51

53  */

99  /**
100  Tries to merge *this with *mc, returns true if successful.
101  The assumption is that *this is no longer in use,
102  and the space it has been allocated can be handed over to a
103  buffer which is adjacent to it.
104  */
105  bool merge_freed_buff(Merge_chunk *mc) const {
106  if (mc->m_buffer_end == m_buffer_start) {
107  mc->m_buffer_end = m_buffer_end;
108  mc->m_max_keys += m_max_keys;
109  return true;
110  } else if (mc->m_buffer_start == m_buffer_end) {
111  mc->m_buffer_start = m_buffer_start;
112  mc->m_max_keys += m_max_keys;
113  return true;
114  }
115  return false;
116  }
117 
118  private:
119  uchar *m_current_key; ///< The current key for this chunk.
120  my_off_t m_file_position; ///< Current position in the file to be sorted.
121  uchar *m_buffer_start; ///< Start of main-memory buffer for this chunk.
122  uchar *m_buffer_end; ///< End of main-memory buffer for this chunk.
123  ha_rows m_rowcount; ///< Number of unread rows in this chunk.
124  ha_rows m_mem_count; ///< Number of rows in the main-memory buffer.
125  ha_rows m_max_keys; ///< If we have fixed-size rows:
126  /// max number of rows in buffer.

270 /**
271  Put all room used by freed buffer to use in adjacent buffer.
272 
273  Note, that we can't simply distribute memory evenly between all buffers,
274  because new areas must not overlap with old ones.
275 */
276 template <typename Heap_type>
277 void reuse_freed_buff(Merge_chunk *old_top, Heap_type *heap) {
278  typename Heap_type::iterator it = heap->begin();
279  typename Heap_type::iterator end = heap->end();
280  for (; it != end; ++it) {
281  if (old_top->merge_freed_buff(*it)) return;
282  }
283  DBUG_ASSERT(0);
284 }
