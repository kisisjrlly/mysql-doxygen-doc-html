74 /**
75  Implementation of a shared set of maps for a given object type.
76 
77  The implementation is an extension of the Multi_map_base, and adds support
78  for:
79 
80  - Locking and thread synchronization.
81  - Broadcasting a miss which has been handled.
82  - Tracking object usage.
83  - Managing the free list.
84  - Tracking cache misses.
85  - Maintaining a pool of allocated cache elements.
86  - Deleting objects and elements after releasing the mutex.
87 
88  The pool of vacant cache elements is used to avoid deleting and allocating
89  memory. We keep a pool of up to 'max_connections' elements. When a new
90  element is needed, we first try to get one from the pool. If that does
91  not succeed, a new element is allocated dynamically. In this case, the
92  allocation will happen while the mutex is locked.
93 
94  When an element and object are to be removed from the shared map and
95  deleted, the size of the pool decides whether the element should be
96  added to the pool or not. If it is not added to the pool, it is signed
97  up for being deleted by the Autolocker. The element's object is always
98  signed up for being deleted, it is never reused even if the corresponding
99  element is added to the pool. The Autolocker is used for two purposes:
100 
101  1. To ensure that the mutex is locked and unlocked correctly.
102  2. To delete unused elements and objects outside the scope
103  where the mutex is locked.
104 
105  When the Autolocker is deleted, it will first unlock the mutex, then
106  it will iterate over the elements and objects and delete them.
107 
108  @note The multi map provides support for tracking cache misses, but does
109  not handle the miss. The miss must be handled by the user of the
110  multi map.
111 
112  @tparam T Dictionary object type.
113 */
114 
115 template <typename T>
116 class Shared_multi_map : public Multi_map_base<T> {
117  private:
118  static const size_t initial_capacity = 256;
119 
120  // Inner class to lock and unlock the multi map instance.
121  class Autolocker {
122  private:
123  // Vector containing objects to be deleted unconditionally.
124  typedef std::vector<const T *, Malloc_allocator<const T *>>
125  Object_list_type;
126  Object_list_type m_objects_to_delete;
127 
128  // Vector containing elements to be deleted unconditionally, which
129  // happens when elements are evicted while the pool is already full.
130  typedef std::vector<const Cache_element<T> *,
131  Malloc_allocator<const Cache_element<T> *>>
132  Element_list_type;
133  Element_list_type m_elements_to_delete;
134 
135  Shared_multi_map<T> *m_map; // Pointer to the enclosing multi map.
136 
137  public:
138  // Lock the multi map on instantiation.
139  explicit Autolocker(Shared_multi_map<T> *map)
140  : m_objects_to_delete(Malloc_allocator<const T *>(PSI_INSTRUMENT_ME)),
141  m_elements_to_delete(
142  Malloc_allocator<const Cache_element<T> *>(PSI_INSTRUMENT_ME)),
143  m_map(map) {
144  mysql_mutex_lock(&m_map->m_lock);
145  }
146 
147  // Add object to be deleted.
148  void auto_delete(const T *object) { m_objects_to_delete.push_back(object); }
149 
150  // Add element to be deleted.
151  void auto_delete(const Cache_element<T> *element) {
152  m_elements_to_delete.push_back(element);
153  }
154 
155  // Unlock the multi map when being deleted (e.g. going out of scope)
156  // and delete the objects and elements.
157  ~Autolocker() {
158  mysql_mutex_unlock(&m_map->m_lock);
159  // Delete the objects.
160  for (typename Object_list_type::const_iterator it =
161  m_objects_to_delete.begin();
162  it != m_objects_to_delete.end(); ++it)
163  delete *it;
164 
165  // Delete the elements.
166  for (typename Element_list_type::const_iterator it =
167  m_elements_to_delete.begin();
168  it != m_elements_to_delete.end(); ++it)
169  delete *it;
170  }
171  };

380  /**
381  Put a new object and element wrapper into the map.
382 
383  The new object may be NULL if it does not exist in the DD tables.
384  In that case, we must still update the missing set and broadcast,
385  if necessary, because other threads may be waiting. Then, the
386  element returned is also NULL.
387 
388  Otherwise, we check that the actual object instance is not present
389  in the map, then all the keys are generated based on the new object.
390  The keys that are missed are removed from the missed set, and then
391  we verify that all keys are either present or absent.
392 
393  If no keys are present, it means we must add the new object. We
394  remove the least recently used objects, if cache capacity is exceeded,
395  and add the new object to the reverse map. We also mark the object
396  as being used. The element wrapper of the new object is returned.
397 
398  If all keys are present, we return the wrapper element already present
399  and delete the newly read object.
400 
401  Finally, if at least one key was missed, we broadcast that the miss
402  has been handled.
403 
404  @note This function will delete the newly read object if an object with
405  the same keys is already present.
406 
407  @note The function may add a new object which is not registered as being
408  missed, i.e., without a preceding cache miss In this case, the
409  submitted key is NULL.
410 
411  @tparam K Key type.
412  @param key Key used when loading the object.
413  @param object New object to be added.
414  @param [out] element Element wrapper: NULL if the new object is
415  NULL, otherwise the wrapper of the newly
416  read object or the existing object with
417  the same keys.
418  */
