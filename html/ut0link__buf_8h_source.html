55 /** Concurrent data structure, which allows to track concurrently
56 performed operations which locally might be dis-ordered.
57 
58 This data structure is informed about finished concurrent operations
59 and tracks up to which point in a total order all operations have
60 been finished (there are no holes).
61 
62 It also allows to limit the last period in which there might be holes.
63 These holes refer to unfinished concurrent operations, which preceed
64 in the total order some operations that are already finished.
65 
66 Threads might concurrently report finished operations (lock-free).
67 
68 Threads might ask for maximum currently known position in total order,
69 up to which all operations are already finished (lock-free).
70 
71 Single thread might track the reported finished operations and update
72 maximum position in total order, up to which all operations are done.
73 
74 You might look at current usages of this data structure in log0buf.cc.
75 */

196 template <typename Position>
197 Link_buf<Position>::Link_buf(size_t capacity)
198  : m_capacity(capacity), m_tail(0) {
199  if (capacity == 0) {
200  m_links = nullptr;
201  return;
202  }
203 
204  ut_a((capacity & (capacity - 1)) == 0);
205 
206  m_links = UT_NEW_ARRAY_NOKEY(std::atomic<Distance>, capacity);
207 
208  for (size_t i = 0; i < capacity; ++i) {
209  m_links[i].store(0);
210  }
211 }

250 template <typename Position>
251 inline void Link_buf<Position>::add_link(Position from, Position to) {
252  ut_ad(to > from);
253  ut_ad(to - from <= std::numeric_limits<Distance>::max());
254 
255  const auto index = slot_index(from);
256 
257  auto &slot = m_links[index];
258 
259  slot.store(to);
260 }

264  Position &next) {

274 template <typename Position>
275 inline void Link_buf<Position>::add_link_advance_tail(Position from,
276  Position to) {
277  ut_ad(to > from);
278  ut_ad(to - from <= std::numeric_limits<Distance>::max());
279 
280  auto position = m_tail.load(std::memory_order_acquire);
281 
282  ut_ad(position <= from);
283 
284  if (position == from) {
285  /* can advance m_tail directly and exclusively, and it is unlock */
286  m_tail.store(to, std::memory_order_release);
287  } else {
288  auto index = slot_index(from);
289  auto &slot = m_links[index];
290 
291  /* add link */
292  slot.store(to, std::memory_order_release);
293 
294  auto stop_condition = [&](Position prev_pos, Position next_pos) {
295  return (prev_pos > from);
296  };
297 
298  advance_tail_until(stop_condition);
299  }
300 }

302 template <typename Position>
303 template <typename Stop_condition>
304 bool Link_buf<Position>::advance_tail_until(Stop_condition stop_condition,
305  uint32_t max_retry) {
306  /* multi threaded aware */
307  auto position = m_tail.load(std::memory_order_acquire);
308  auto from = position;
309 
310  uint32_t retry = 0;
311  while (true) {
312  auto index = slot_index(position);
313  auto &slot = m_links[index];
314 
315  auto next_load = slot.load(std::memory_order_acquire);
316 
317  if (next_load >= position + m_capacity) {
318  /* either we wrapped and tail was advanced mean while,
319  or there is link start_lsn -> end_lsn of length >= m_capacity */
320  position = m_tail.load(std::memory_order_acquire);
321  if (position != from) {
322  from = position;
323  continue;
324  }
325  }
326 
327  if (next_load <= position || stop_condition(position, next_load)) {
328  /* nothing to advance for now */
329  return false;
330  }
331 
332  /* try to lock as storing the end */
333  if (slot.compare_exchange_strong(next_load, position,
334  std::memory_order_acq_rel)) {
335  /* it could happen, that after thread read position = m_tail.load(),
336  it got scheduled out for longer; when it comes back it might still
337  see the link going forward in that slot but m_tail could have been
338  already advanced forward (as we do not reset slots when traversing
339  them); thread needs to re-check if m_tail is still behind the slot. */
340  position = m_tail.load(std::memory_order_acquire);
341  if (position == from) {
342  /* confirmed. can advance m_tail exclusively */
343  position = next_load;
344  break;
345  }
346  }
347 
348  retry++;
349  if (retry > max_retry) {
350  /* give up */
351  return false;
352  }
353 
354  UT_RELAX_CPU();
355  position = m_tail.load(std::memory_order_acquire);
356  if (position == from) {
357  /* no progress? */
358  return false;
359  }
360  from = position;
361  }
362 
363  while (true) {
364  Position next;
365 
366  bool stop = next_position(position, next);
367 
368  if (stop || stop_condition(position, next)) {
369  break;
370  }
371 
372  position = next;
373  }
374 
375  ut_a(from == m_tail.load(std::memory_order_acquire));
376 
377  /* unlock */
378  m_tail.store(position, std::memory_order_release);
379 
380  if (position == from) {
381  return false;
382  }
383 
384  return true;
385 }

404 template <typename Position>
405 inline bool Link_buf<Position>::has_space(Position position) {
406  auto tail = m_tail.load(std::memory_order_acquire);
407  if (tail + m_capacity > position) {
408  return true;
409  }
410 
411  auto stop_condition = [](Position from, Position to) { return false; };
412  advance_tail_until(stop_condition, 0);
413 
414  tail = m_tail.load(std::memory_order_acquire);
415  return tail + m_capacity > position;
416 }

423 template <typename Position>
424 void Link_buf<Position>::validate_no_links(Position begin, Position end) {
425  const auto tail = m_tail.load();
426 
427  /* After m_capacity iterations we would have all slots tested. */
428 
429  end = std::min(end, begin + m_capacity);
430 
431  for (; begin < end; ++begin) {
432  const size_t index = slot_index(begin);
433 
434  const auto &slot = m_links[index];
435 
436  ut_a(slot.load() <= tail);
437  }
438 }
