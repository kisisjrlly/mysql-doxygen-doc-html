39 struct pk_pos_data_lock {
40  pk_pos_data_lock() { reset(); }
41 
42  void reset() {
43  memset(m_engine_lock_id, 0, sizeof(m_engine_lock_id));
44  m_engine_lock_id_length = 0;
45  }
46 
47  void set(const pk_pos_data_lock *other) {
48  memcpy(m_engine_lock_id, other->m_engine_lock_id, sizeof(m_engine_lock_id));
49  m_engine_lock_id_length = other->m_engine_lock_id_length;
50  }
51 
52  /** Column ENGINE_LOCK_ID */
53  char m_engine_lock_id[128];
54  size_t m_engine_lock_id_length;
55 };

92 struct pk_pos_data_lock_wait {
93  pk_pos_data_lock_wait() { reset(); }
94 
95  void reset() {
96  // POT type, must initialize every byte for memcmp()
97  memset(m_requesting_engine_lock_id, 0, sizeof(m_requesting_engine_lock_id));
98  m_requesting_engine_lock_id_length = 0;
99  memset(m_blocking_engine_lock_id, 0, sizeof(m_blocking_engine_lock_id));
100  m_blocking_engine_lock_id_length = 0;
101  }
102 
103  void set(const pk_pos_data_lock_wait *other) {
104  memcpy(m_requesting_engine_lock_id, other->m_requesting_engine_lock_id,
105  sizeof(m_requesting_engine_lock_id));
106  m_requesting_engine_lock_id_length =
107  other->m_requesting_engine_lock_id_length;
108  memcpy(m_blocking_engine_lock_id, other->m_blocking_engine_lock_id,
109  sizeof(m_blocking_engine_lock_id));
110  m_blocking_engine_lock_id_length = other->m_blocking_engine_lock_id_length;
111  }
112 
113  /** Column REQUESTING_ENGINE_LOCK_ID */
114  char m_requesting_engine_lock_id[128];
115  size_t m_requesting_engine_lock_id_length;
116  /** Column BLOCKING_ENGINE_LOCK_ID */
117  char m_blocking_engine_lock_id[128];
118  size_t m_blocking_engine_lock_id_length;
119 };
120 
121 // This structure is memcmp-ed, so we need to have no padding.
122 static_assert(sizeof(pk_pos_data_lock_wait) == 2 * (128 + sizeof(size_t)), "");
