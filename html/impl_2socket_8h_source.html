100  native_handle_type native_handle) {

114 inline stdx::expected<void, error_type> native_non_blocking(
115  native_handle_type native_handle, bool on) {
116 #ifdef _WIN32
117  unsigned long nonblocking{on};
118 
119  return ioctl(native_handle, FIONBIO, &nonblocking);
120 #else
121  auto res = impl::file::fcntl(native_handle, impl::file::get_file_status());
122  if (!res) return stdx::make_unexpected(res.error());
123 
124  int flags = *res;
125 
126  if (on) {
127  if (flags & O_NONBLOCK) return {};
128  flags |= O_NONBLOCK;
129  } else {
130  if (!(flags & O_NONBLOCK)) return {};
131  flags &= ~O_NONBLOCK;
132  }
133 
134  auto set_res =
135  impl::file::fcntl(native_handle, impl::file::set_file_status(flags));
136  if (!set_res) return stdx::make_unexpected(set_res.error());
137 
138  return {};

183 /**
184  * wrap ::recv() in a portable way.
185  *
186  * @param native_handle socket handle
187  * @param buf pointer to a mutable buffer of size 'buf_len'
188  * @param buf_len size of 'buf'
189  * @param flags message flags
190  * @returns bytes transferred on success, std::error_code otherwise
191  */
192 inline stdx::expected<size_t, error_type> recv(native_handle_type native_handle,
193  void *buf, size_t buf_len,
194  message_flags flags) {
195 #if defined(_WIN32)
196  // recv() expects an 'int' instead of a 'size_t'.
197  // Ensure, buf_len is properly narrowed instead of wrapped around
198  auto bytes_transferred = ::recv(
199  native_handle, static_cast<char *>(buf),
200  std::min(static_cast<size_t>(std::numeric_limits<int>::max()), buf_len),
201  flags.to_ulong());
202 #else
203  auto bytes_transferred =
204  ::recv(native_handle, buf, buf_len, static_cast<int>(flags.to_ulong()));
205 #endif
206  if (kSocketError == bytes_transferred) {
207  return stdx::make_unexpected(last_error_code());
208  }
209 
210  return {static_cast<size_t>(bytes_transferred)};
211 }

438 /**
439  * socketpair().
440  *
441  * - wraps socketpair() on POSIX
442  * - emualated on windows via a AF_INET + SOCK_STREAM socket on localhost on a
443  * system assigned port.
444  */
445 inline stdx::expected<std::pair<native_handle_type, native_handle_type>,
446  error_type>
447 socketpair(int family, int sock_type, int protocol) {
448 #if !defined(_WIN32)
449  std::array<native_handle_type, 2> fds;
450 
451  if (0 != ::socketpair(family, sock_type, protocol, fds.data())) {
452  return stdx::make_unexpected(last_error_code());
453  }
454 
455  return std::make_pair(fds[0], fds[1]);
456 #else
457  if (family != AF_INET) {
458  return stdx::make_unexpected(
459  std::error_code(WSAEAFNOSUPPORT, std::system_category()));
460  }
461  if (sock_type != SOCK_STREAM) {
462  return stdx::make_unexpected(
463  std::error_code(WSAESOCKTNOSUPPORT, std::system_category()));
464  }
465  // we don't have socketpair on windows, build one ourselves
466  auto listener_res = impl::socket::socket(family, sock_type, protocol);
467  if (!listener_res) return stdx::make_unexpected(listener_res.error());
468  auto listener = listener_res.value();
469 
470  int reuse = 1;
471  impl::socket::setsockopt(listener, SOL_SOCKET, SO_REUSEADDR, &reuse,
472  sizeof(reuse));
473 
474  struct sockaddr_in sa {};
475  sa.sin_family = family;
476  sa.sin_addr.s_addr = htonl(INADDR_LOOPBACK);
477  sa.sin_port = 0; // pick a random port
478 
479  size_t sa_len = sizeof(sa);
480  auto bind_res =
481  impl::socket::bind(listener, reinterpret_cast<sockaddr *>(&sa), sa_len);
482  if (!bind_res) {
483  impl::socket::close(listener);
484  return stdx::make_unexpected(bind_res.error());
485  }
486  auto listen_res = impl::socket::listen(listener, 128);
487  if (!listen_res) {
488  impl::socket::close(listener);
489  return stdx::make_unexpected(listen_res.error());
490  }
491  auto first_res = impl::socket::socket(AF_INET, SOCK_STREAM, 0);
492  if (!first_res) {
493  impl::socket::close(listener);
494  return stdx::make_unexpected(first_res.error());
495  }
496 
497  memset(&sa, 0, sizeof(sa));
498  impl::socket::getsockname(listener, reinterpret_cast<sockaddr *>(&sa),
499  &sa_len);
500  sa.sin_addr.s_addr = htonl(INADDR_LOOPBACK);
501  sa.sin_family = AF_INET;
502 
503  auto first_fd = first_res.value();
504 
505  auto connect_res = impl::socket::connect(
506  first_fd, reinterpret_cast<sockaddr *>(&sa), sa_len);
507  if (!connect_res) {
508  impl::socket::close(listener);
509  impl::socket::close(first_fd);
510  return stdx::make_unexpected(connect_res.error());
511  }
512 
513  auto second_res = impl::socket::accept(listener, nullptr, nullptr);
514  if (!second_res) {
515  impl::socket::close(listener);
516  impl::socket::close(first_fd);
517  return stdx::make_unexpected(second_res.error());
518  }
519 
520  auto second_fd = second_res.value();
521 
522  impl::socket::close(listener);
523 
524  return std::make_pair(first_fd, second_fd);
525 #endif
526 }

580 inline stdx::expected<void, error_type> wait(native_handle_type fd,
581  wait_type wt) {
582  short events{};
583 
584  switch (wt) {
585  case wait_type::wait_read:
586  events |= POLLIN;
587  break;
588  case wait_type::wait_write:
589  events |= POLLOUT;
590  break;
591  case wait_type::wait_error:
592  events |= POLLERR;
593  break;
594  }
595  std::array<impl::poll::poll_fd, 1> fds{{
596  {fd, events, 0},
597  }};
598 
599  const auto res =
600  impl::poll::poll(fds.data(), fds.size(), std::chrono::milliseconds{-1});
601 
602  if (!res) return stdx::make_unexpected(res.error());
603 
604  return {};
