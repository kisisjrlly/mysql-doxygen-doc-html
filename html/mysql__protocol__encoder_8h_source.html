175  template <class T, typename = std::enable_if<std::is_integral<T>::value>>
176  void append_int(MsgBuffer &buffer, T value, size_t len = sizeof(T)) {
177  buffer.reserve(buffer.size() + len);
178  while (len-- > 0) {
179  byte b = static_cast<byte>(value);
180  buffer.push_back(b);
181  value = static_cast<T>(value >> 8);
182  }
183  }
