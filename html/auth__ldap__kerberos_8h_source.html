32 /**
33  Kerberos class is built around kerberos library.
34  This class should/can be used for different part of code as standalone
35  class.
36  This class performs following operations:
37  1. Authentication with kerberos server and store the credentials in cache.
38  2. Get the default configured kerberos user in the OS, from default principal.
39 
40  Credentials:
41  A ticket plus the secret session key necessary to use that ticket successfully
42  in an authentication exchange.
43 
44  Principal:
45  A named client or server entity that participates in a network communication,
46  with one name that is considered canonical
47 
48  Credential cache:
49  A credential cache (or “ccache”) holds Kerberos credentials while they remain
50  valid and, generally, while the user’s session lasts, so that authenticating
51  to a service multiple times (e.g., connecting to a web or mail server more
52  than once) doesn’t require contacting the KDC every time.
53 */
54 namespace auth_ldap_client_kerberos_context {
55 class Kerberos {
56  public:
57  Kerberos(const char *user, const char *password);
58  ~Kerberos();
59  /**
60  1. This function authenticates with kerberos server.
61  2. If TGT destroy is false, this function stores the TGT in Kerberos cache
62  for subsequent usage.
63  3. If user credentials already exist in the cache, it doesn't attempt to get
64  it again.
65 
66  @return
67  @retval true, If function is successfully able to obtain and store
68  credentials.
69  @retval false, If function is failed to obtain and store credentials.
70  */
71  bool obtain_store_credentials();
72  /**
73  This function retrieves default principle from kerberos configuration and
74  parses the user name from it. If user name has not been provided in the
75  MySQL client, This method can be used to get the user name and use for
76  authentication.
77  @return
78  @retval true, If function is successfully able to get user name.
79  @retval false, If function is failed to get user name.
80  */
81  bool get_user_name(std::string *name);
82  void destroy_credentials();
83  /**
84  This function gets LDAP host from krb5.conf file.
85  */
86  void get_ldap_host(std::string &host);

88  private:
