41 /**
42  Representation of an SQL command.
43 
44  This class is an interface between the parser and the runtime.
45  The parser builds the appropriate derived classes of Sql_cmd
46  to represent a SQL statement in the parsed tree.
47  The execute() method in the derived classes of Sql_cmd contain the runtime
48  implementation.
49  Note that this interface is used for SQL statements recently implemented,
50  the code for older statements tend to load the LEX structure with more
51  attributes instead.
52  Implement new statements by sub-classing Sql_cmd, as this improves
53  code modularity (see the 'big switch' in dispatch_command()), and decreases
54  the total size of the LEX structure (therefore saving memory in stored
55  programs).
56  The recommended name of a derived class of Sql_cmd is Sql_cmd_<derived>.
57 
58  Notice that the Sql_cmd class should not be confused with the Statement class.
59  Statement is a class that is used to manage an SQL command or a set
60  of SQL commands. When the SQL statement text is analyzed, the parser will
61  create one or more Sql_cmd objects to represent the actual SQL commands.
62 */
63 class Sql_cmd {
64  private:
65  Sql_cmd(const Sql_cmd &); // No copy constructor wanted
66  void operator=(Sql_cmd &); // No assignment operator wanted
67 
68  public:
69  /**
70  @brief Return the command code for this statement
71  */
72  virtual enum_sql_command sql_command_code() const = 0;
73 
74  /**
75  @return true if object represents a preparable statement, ie. a query
76  that is prepared with a PREPARE statement and executed with an EXECUTE
77  statement. False is returned for regular statements (non-preparable
78  statements) that are executed directly. Also false if statement is part
79  of a stored procedure.
80  */
81  bool needs_explicit_preparation() const {
82  return m_owner != nullptr && !m_part_of_sp;
83  }
84  /**
85  @return true if statement is regular, ie not prepared statement and not
86  part of stored procedure.
87  */
88  bool is_regular() const { return m_owner == nullptr && !m_part_of_sp; }
89 
90  /// @return true if this statement is prepared
91  bool is_prepared() const { return m_prepared; }
92 
93  /**
94  Prepare this SQL statement.
95 
96  param thd the current thread
97 
98  @returns false if success, true if error
99  */
100  virtual bool prepare(THD *) {
101  // Default behavior for a statement is to have no preparation code.
102  /* purecov: begin inspected */
103  DBUG_ASSERT(!is_prepared());
104  set_prepared();
105  return false;
106  /* purecov: end */
107  }
108 
109  /**
110  Execute this SQL statement.
111  @param thd the current thread.
112  @returns false if success, true if error
113  */
114  virtual bool execute(THD *thd) = 0;
115 
116  /**
117  Command-specific reinitialization before execution of prepared statement
118 
119  param thd Current THD.
120  */
121  virtual void cleanup(THD *) { m_secondary_engine = nullptr; }
122 
123  /// Set the owning prepared statement
124  void set_owner(Prepared_statement *stmt) {
125  DBUG_ASSERT(!m_part_of_sp);
126  m_owner = stmt;
127  }
128 
129  /// Get the owning prepared statement
130  Prepared_statement *owner() const { return m_owner; }
131 
132  /**
133  Mark statement as part of procedure. Such statements can be executed
134  multiple times, the first execute() call will also prepare it.
135  */
136  void set_as_part_of_sp() {
137  DBUG_ASSERT(!m_part_of_sp && m_owner == nullptr);
138  m_part_of_sp = true;
139  }
140  /// @returns true if statement is part of a stored procedure
141  bool is_part_of_sp() const { return m_part_of_sp; }
142 
143  /// @return true if SQL command is a DML statement
144  virtual bool is_dml() const { return false; }
145 
146  /// @return true if implemented as single table plan, DML statement only
147  virtual bool is_single_table_plan() const {
148  /* purecov: begin inspected */
149  DBUG_ASSERT(is_dml());
150  return false;
151  /* purecov: end */
152  }
153 
154  virtual bool accept(THD *, Select_lex_visitor *) { return false; }
155 
156  /**
157  Is this statement of a type and on a form that makes it eligible
158  for execution in a secondary storage engine?
159 
160  @return the name of the secondary storage engine, or nullptr if
161  the statement is not eligible for execution in a secondary storage
162  engine
163  */
164  virtual const MYSQL_LEX_CSTRING *eligible_secondary_storage_engine() const {
165  return nullptr;
166  }
167 
168  /**
169  Disable use of secondary storage engines in this statement. After
170  a call to this function, the statement will not try to use a
171  secondary storage engine until it is reprepared.
172  */
173  void disable_secondary_storage_engine() {
174  DBUG_ASSERT(m_secondary_engine == nullptr);
175  m_secondary_engine_enabled = false;
176  }
177 
178  /**
179  Has use of secondary storage engines been disabled for this statement?
180  */
181  bool secondary_storage_engine_disabled() const {
182  return !m_secondary_engine_enabled;
183  }
184 
185  /**
186  Mark the current statement as using a secondary storage engine.
187  This function must be called before the statement starts opening
188  tables in a secondary engine.
189  */
190  void use_secondary_storage_engine(const handlerton *hton) {
191  DBUG_ASSERT(m_secondary_engine_enabled);
192  m_secondary_engine = hton;
193  }
194 
195  /**
196  Is this statement using a secondary storage engine?
197  @note that this is reliable during optimization and afterwards; during
198  preparation, if this is an explicit preparation (SQL PREPARE, C API
199  PREPARE, and automatic repreparation), it may be false as RAPID tables have
200  not yet been opened. Therefore, during preparation, it is safer to test
201  THD::secondary_engine_optimization().
202  */
203  bool using_secondary_storage_engine() const {
204  return m_secondary_engine != nullptr;
205  }
206 
207  /**
208  Get the handlerton of the secondary engine that is used for
209  executing this statement, or nullptr if a secondary engine is not
210  used.
211  */
212  const handlerton *secondary_engine() const { return m_secondary_engine; }
213 
214  void set_optional_transform_prepared(bool value) {
215  m_prepared_with_optional_transform = value;
216  }
217 
218  bool is_optional_transform_prepared() {
219  return m_prepared_with_optional_transform;
220  }
221 
222  protected:
223  Sql_cmd() : m_owner(nullptr), m_part_of_sp(false), m_prepared(false) {}
224 
225  virtual ~Sql_cmd() {
226  /*
227  Sql_cmd objects are allocated in thd->mem_root.
228  In MySQL, the C++ destructor is never called, the underlying MEM_ROOT is
229  simply destroyed instead.
230  Do not rely on the destructor for any cleanup.
231  */
232  DBUG_ASSERT(false);
233  }
234 
235  /// Set this statement as prepared
236  void set_prepared() { m_prepared = true; }
237 
238  private:
239  Prepared_statement *m_owner; /// Owning prepared statement, NULL if non-prep.
240  bool m_part_of_sp; /// True when statement is part of stored proc.
241  bool m_prepared; /// True when statement has been prepared
242 
243  /**
244  Tells if a secondary storage engine can be used for this
245  statement. If it is false, use of a secondary storage engine will
246  not be considered for executing this statement.
247  */
248  bool m_secondary_engine_enabled{true};
249 
250  /**
251  Keeps track of whether the statement was prepared optional
252  transformation.
253  */
254  bool m_prepared_with_optional_transform{false};
255 
256  /**
257  The secondary storage engine to use for execution of this
258  statement, if any, or nullptr if the primary engine is used.
259  This property is reset at the start of each execution.
260  */
261  const handlerton *m_secondary_engine{nullptr};
262 };
