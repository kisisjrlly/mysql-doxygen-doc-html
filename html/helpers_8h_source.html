60 template <typename SeqCont1, typename SeqCont2>
61 ::testing::AssertionResult AssertSetEqual(const char *seq1_expr,
62  const char *seq2_expr,
63  const SeqCont1 &seq1,
64  const SeqCont2 &seq2) {
65  std::vector<typename SeqCont1::value_type> c1(seq1.begin(), seq1.end());
66  std::vector<typename SeqCont2::value_type> c2(seq2.begin(), seq2.end());
67  std::sort(c1.begin(), c1.end());
68  std::sort(c2.begin(), c2.end());
69 
70  // Check for elements that are in the first range but not in the second.
71  std::vector<typename SeqCont2::value_type> c1_not_c2;
72  std::set_difference(c1.begin(), c1.end(), c2.begin(), c2.end(),
73  std::back_inserter(c1_not_c2));
74  if (c1_not_c2.size() > 0) {
75  auto result = ::testing::AssertionFailure();
76  result << seq1_expr << " had elements not in " << seq2_expr << ": ";
77  for (auto elem : c1_not_c2) result << elem << " ";
78  return result;
79  }
80 
81  // Check for elements that are in the second range but not in the first.
82  std::vector<typename SeqCont2::value_type> c2_not_c1;
83  std::set_difference(c2.begin(), c2.end(), c1.begin(), c1.end(),
84  std::back_inserter(c2_not_c1));
85  if (c2_not_c1.size() > 0) {
86  auto result = ::testing::AssertionFailure();
87  result << seq2_expr << " had elements not in " << seq1_expr << ": ";
88  for (auto elem : c2_not_c1) result << elem << " ";
89  return result;
90  }
91 
92  return ::testing::AssertionSuccess();
93 }

122 /**
123  * Temporary Directory object using RAII semantics
124  *
125  * This class creates a temporary directory with (partially) random name, which
126  * it removes on destruction along with its contents. It exposes an API to
127  * conveniently work with this directory.
128  */
129 class TmpDir {
130  public:
131  // RAII
132  TmpDir() : tmp_dir_(mysql_harness::get_tmp_dir()) {}
133  TmpDir(const std::string &dir_name_prefix)
134  : tmp_dir_(mysql_harness::get_tmp_dir(dir_name_prefix)) {}
135  ~TmpDir() { mysql_harness::delete_dir(tmp_dir_); }
136 
137  // return path to tmp dir
138  std::string operator()() { return tmp_dir_; }
139 
140  // return path/to/file in tmp dir
141  std::string file(const std::string &fname) { return tmp_dir_ + "/" + fname; }
142 
143  private:
144  std::string tmp_dir_;
145 };
