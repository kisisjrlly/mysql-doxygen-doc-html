200 inline Cursor &Cursor::operator=(const Cursor &rhs) {
201  m_is_positioned = rhs.m_is_positioned;
202 
203  m_type = rhs.m_type;
204 
205  if (m_is_positioned) {
206  if (m_type == Type::HASH) {
207  m_hash_iterator = rhs.m_hash_iterator;
208  } else {
209  DBUG_ASSERT(m_type == Type::TREE);
210  m_tree_iterator = rhs.m_tree_iterator;
211  }
212  }
213 
214  return *this;
215 }

230 inline Cursor &Cursor::operator--() {
231  DBUG_ASSERT(m_is_positioned);
232 
233  if (m_type == Type::HASH) {
234  /* We don't support decrement on a hash and it shouldn't be called. */
235  abort();
236  } else {
237  DBUG_ASSERT(m_type == Type::TREE);
238  --m_tree_iterator;
239  }
240 
241  return *this;
242 }
