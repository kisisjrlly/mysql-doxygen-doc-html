270 /**
271  @class Gcs_message
272 
273  Class that represents the data that is exchanged within a group. It is sent
274  by a member having the group as destination. It shall be received by all
275  members that pertain to the group in that moment.
276 
277  It is built using two major blocks: the header and the payload of the
278  message. A user of Gcs_message can freely add data to the header and to the
279  payload.
280 
281  Each binding implementation might use these two fields at its own
282  discretion but that data should be removed from the header/payload
283  before its delivery to the client layer.
284 
285  When the message is built, none of the data is passed unto it. One has to
286  use the append_* methods in order to append data. Calling encode() at the end
287  will provide a ready-to-send message.
288 
289  On the receiver side, one shall use decode() in order to have the header and
290  the payload restored in the original message fields.
291 
292  A typical use case of sending a message is:
293 
294  @code{.cpp}
295 
296  Gcs_control_interface *gcs_ctrl_interface_instance;
297  Gcs_communication_interface *gcs_comm_interface_instance; // obtained
298  // previously
299 
300  Gcs_member_identifier *myself=
301  gcs_ctrl_interface_instance->get_local_information();
302 
303  Gcs_group_identifier destination("the_group");
304 
305  Gcs_message *msg= new Gcs_message(&myself, new Gcs_message_data(0, 3));
306 
307  uchar[] some_data= {(uchar)0x12,
308  (uchar)0x24,
309  (uchar)0x00};
310 
311  msg->get_message_data()->append_to_payload(&some_data,
312  strlen(some_data));
313 
314  gcs_comm_interface_instance->send_message(msg);
315  @endcode
316 
317  A typical use case of receiving a message is:
318 
319  @code{.cpp}
320  class my_Gcs_communication_event_listener : Gcs_communication_event_listener
321  {
322  my_Gcs_communication_event_listener(my_Message_delegator *delegator)
323  {
324  this->delegator= delegator;
325  }
326 
327  void on_message_received(Gcs_message &message)
328  {
329  // Do something when message arrives...
330  delegator->process_gcs_message(message);
331  }
332 
333  private:
334  my_Message_delegator *delegator;
335  }
336  @endcode
337 */
