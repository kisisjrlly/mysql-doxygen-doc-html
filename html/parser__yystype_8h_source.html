263 /**
264  std::optional-like wrapper for simple bitmaps (usually enums of binary flags)
265 
266  This template wraps trivial bitmap implementations to add two features:
267 
268  * std::optional-like behavior -- the "unset" flag, so we don't have
269  to inject a special "invalid" value into existent enum types, this
270  wrapper class does that for us.
271 
272  * the merge() function to merge two bitmap values in a type-safe way.
273 
274  @tparam Enum Usually a enum type which simulates a bit set.
275  @tparam Default_value A default Enum value for "unset" variables.
276 
277 */
278 template <typename Enum, Enum Default_value>
279 class Enum_parser {
280  public:
281  /// Constructor-like function
282  ///
283  /// The Enum_parser<> class is designed for use as a field of restricted
284  /// unions, so it can't have C++ constructors.
285  void init() { m_is_set = false; }
286 
287  /// False if the wrapped Enum value is not assigned.
288  bool is_set() const { return m_is_set; }
289 
290  /// Return the wrapped Enum value.
291  ///
292  /// @note The wrapped value must be assigned.
293  Enum get() const {
294  DBUG_ASSERT(is_set());
295  return m_enum;
296  }
297 
298  /// Return the wrapped Enum value (if any) or the Default_value.
299  Enum get_or_default() const { return is_set() ? get() : Default_value; }
300 
301  /// Assign the wrapped Enum value.
302  void set(Enum value) {
303  m_is_set = true;
304  m_enum = value;
305  }
306 
307  /// Merge the x bit set into the wrapped Enum value (if any), or replace it
308  void merge(const Enum_parser &x) {
309  if (x.is_set()) set(x.get());
310  }
311 
312  private:
313  bool m_is_set; ///< True if m_enum is assigned with some value
314  Enum m_enum; ///< The wrapped Enum value.
315 };
