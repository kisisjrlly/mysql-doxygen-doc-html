38 /* This module gives a macro definition of the body of
39 a standard sort function for an array of elements of any
40 type. The comparison function is given as a parameter to
41 the macro. The sort algorithm is mergesort which has logarithmic
42 worst case.
43 */
44 
45 /** This macro expands to the body of a standard sort function.
46  The sort function uses mergesort and must be defined separately
47  for each type of array.
48  Also the comparison function has to be defined individually
49  for each array cell type. SORT_FUN is the sort function name.
50  The function takes the array to be sorted (ARR),
51  the array of auxiliary space (AUX_ARR) of same size,
52  and the low (LOW), inclusive, and high (HIGH), noninclusive,
53  limits for the sort interval as arguments.
54  CMP_FUN is the comparison function name. It takes as arguments
55  two elements from the array and returns 1, if the first is bigger,
56  0 if equal, and -1 if the second bigger. */
57 
58 #define UT_SORT_FUNCTION_BODY(SORT_FUN, ARR, AUX_ARR, LOW, HIGH, CMP_FUN) \
59  { \
60  ulint ut_sort_mid77; \
61  ulint ut_sort_i77; \
62  ulint ut_sort_low77; \
63  ulint ut_sort_high77; \
64  \
65  ut_ad((LOW) < (HIGH)); \
66  ut_ad(ARR); \
67  ut_ad(AUX_ARR); \
68  \
69  if ((LOW) == (HIGH)-1) { \
70  return; \
71  } else if ((LOW) == (HIGH)-2) { \
72  if (CMP_FUN((ARR)[LOW], (ARR)[(HIGH)-1]) > 0) { \
73  (AUX_ARR)[LOW] = (ARR)[LOW]; \
74  (ARR)[LOW] = (ARR)[(HIGH)-1]; \
75  (ARR)[(HIGH)-1] = (AUX_ARR)[LOW]; \
76  } \
77  return; \
78  } \
79  \
80  ut_sort_mid77 = ((LOW) + (HIGH)) / 2; \
81  \
82  SORT_FUN((ARR), (AUX_ARR), (LOW), ut_sort_mid77); \
83  SORT_FUN((ARR), (AUX_ARR), ut_sort_mid77, (HIGH)); \
84  \
85  ut_sort_low77 = (LOW); \
86  ut_sort_high77 = ut_sort_mid77; \
87  \
88  for (ut_sort_i77 = (LOW); ut_sort_i77 < (HIGH); ut_sort_i77++) { \
89  if (ut_sort_low77 >= ut_sort_mid77) { \
90  (AUX_ARR)[ut_sort_i77] = (ARR)[ut_sort_high77]; \
91  ut_sort_high77++; \
92  } else if (ut_sort_high77 >= (HIGH)) { \
93  (AUX_ARR)[ut_sort_i77] = (ARR)[ut_sort_low77]; \
94  ut_sort_low77++; \
95  } else if (CMP_FUN((ARR)[ut_sort_low77], (ARR)[ut_sort_high77]) > 0) { \
96  (AUX_ARR)[ut_sort_i77] = (ARR)[ut_sort_high77]; \
97  ut_sort_high77++; \
98  } else { \
99  (AUX_ARR)[ut_sort_i77] = (ARR)[ut_sort_low77]; \
100  ut_sort_low77++; \
101  } \
102  } \
103  \
104  memcpy((void *)((ARR) + (LOW)), (AUX_ARR) + (LOW), \
105  ((HIGH) - (LOW)) * sizeof *(ARR)); \
106  }
107
