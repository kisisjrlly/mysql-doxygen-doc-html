172  /** @brief Gets an unsigned integer using the given option
173  *
174  * Gets an unsigned integer using the given option. The type can be
175  * any unsigned integer type such as uint16_t.
176  *
177  * The min_value argument can be used to set a minimum value for
178  * the option. For example, when 0 (zero) is not allowed, min_value
179  * can be set to 0. The maximum value is whatever the maximum of the
180  * use type is.
181  *
182  * Throws std::invalid_argument on errors.
183  *
184  * @param section Instance of ConfigSection
185  * @param option Option name in section
186  * @param min_value Minimum value
187  * @param max_value Maximum value
188  * @return value read from the configuration
189  */
190  template <typename T>
191  T get_uint_option(const mysql_harness::ConfigSection *section,
192  const std::string &option, T min_value = 0,
193  T max_value = std::numeric_limits<T>::max()) {
194  std::string value = get_option_string(section, option);
195 
196  static_assert(
197  std::numeric_limits<T>::max() <= std::numeric_limits<long long>::max(),
198  "");
199 
200  char *rest;
201  errno = 0;
202  long long tol = std::strtoll(value.c_str(), &rest, 0);
203  T result = static_cast<T>(tol);
204 
205  if (tol < 0 || errno > 0 || *rest != '\0' || result > max_value ||
206  result < min_value ||
207  result != tol || // if casting lost high-order bytes
208  (max_value > 0 && result > max_value)) {
209  std::ostringstream os;
210  os << get_log_prefix(option, section) << " needs value between "
211  << min_value << " and " << to_string(max_value) << " inclusive";
212  if (!value.empty()) {
213  os << ", was '" << value << "'";
214  }
215  throw std::invalid_argument(os.str());
216  }
217  return result;
218  }
