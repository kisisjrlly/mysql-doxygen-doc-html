46 /* Thin abstraction which enables logging of memory operations.
47  *
48  * Used by the Allocator to implement switching from RAM to MMAP-backed
49  * allocations and vice-versa. E.g. Allocator will switch to MMAP-backed
50  * allocation strategy once temptable RAM-consumption threshold, which is
51  * defined by temptable_max_ram user-modifiable variable, is reached.
52  **/
53 struct MemoryMonitor {
54  protected:
55  /** Log increments of heap-memory consumption.
56  *
57  * [in] Number of bytes.
58  * @return Heap-memory consumption after increase. */
59  static size_t ram_increase(size_t bytes) {
60  DBUG_ASSERT(ram <= std::numeric_limits<decltype(bytes)>::max() - bytes);
61  return ram.fetch_add(bytes) + bytes;
62  }
63  /** Log decrements of heap-memory consumption.
64  *
65  * [in] Number of bytes.
66  * @return Heap-memory consumption after decrease. */
67  static size_t ram_decrease(size_t bytes) {
68  DBUG_ASSERT(ram >= bytes);
69  return ram.fetch_sub(bytes) - bytes;
70  }
71  /** Get heap-memory threshold level. Level is defined by this Allocator.
72  *
73  * @return Heap-memory threshold. */
74  static size_t ram_threshold() { return temptable_max_ram; }
75  /** Get current level of heap-memory consumption.
76  *
77  * @return Current level of heap-memory consumption (in bytes). */
78  static size_t ram_consumption() { return ram; }
79 
80  private:
81  /** Total bytes allocated so far by all threads in RAM. */
82  static std::atomic<size_t> ram;
83 };

86 /* Block allocation scheme which grows the block-size at exponential scale with
87  * upper limit of ALLOCATOR_MAX_BLOCK_BYTES (2 ^ ALLOCATOR_MAX_BLOCK_MB_EXP):
88  * 1 MiB,
89  * 2 MiB,
90  * 4 MiB,
91  * 8 MiB,
92  * 16 MiB,
93  * 32 MiB,
94  * ...,
95  * ALLOCATOR_MAX_BLOCK_BYTES,
96  * ALLOCATOR_MAX_BLOCK_BYTES,
97  *
98  * In case the user requested block-size is bigger than the one calculated by
99  * the scheme, bigger one will be returned.
100  *
101  * This scheme is a default one for temptable::Allocator but just one of the
102  * many block-size allocation schemes we can come up with and plug it in without
103  * changing the underlying temptable::Allocator implementation.
104  * */
105 struct Exponential {
106  /** Given the current number of allocated blocks by the allocator, and number
107  * of bytes actually requested by the client code, calculate the new block
108  * size.
109  *
110  * [in] Current number of allocated blocks.
111  * [in] Number of bytes requested by the client code.
112  * @return New block size. */
113  static size_t block_size(size_t number_of_blocks, size_t n_bytes_requested) {
114  size_t block_size_hint;
115  if (number_of_blocks < ALLOCATOR_MAX_BLOCK_MB_EXP) {
116  block_size_hint = (1ULL << number_of_blocks) * 1_MiB;
117  } else {
118  block_size_hint = ALLOCATOR_MAX_BLOCK_BYTES;
119  }
120  return std::max(block_size_hint, Block::size_hint(n_bytes_requested));
121  }
122 };
123 } // namespace AllocationScheme

125 /**
126  Shared state between all instances of a given allocator.
127 
128  STL allocators can (since C++11) carry state; however, that state should
129  never be mutable, as the allocator can be copy-constructed and rebound
130  without further notice, so e.g. deallocating memory in one allocator could
131  mean freeing a block that an earlier copy of the allocator still thinks is
132  valid.
133 
134  Usually, mutable state will be external to the allocator (e.g.
135  Mem_root_allocator will point to a MEM_ROOT, but it won't own the MEM_ROOT);
136  however, TempTable was never written this way, and doesn't have a natural
137  place to stick the allocator state. Thus, we need a kludge where the
138  allocator's state is held in a shared_ptr, owned by all the instances
139  together. This is suboptimal for performance, and also is against the style
140  guide's recommendation to have clear ownership of objects, but at least it
141  avoids the use-after-free.
142  */
143 struct AllocatorState {
144  /** Current not-yet-full block to feed allocations from. */
145  Block current_block;
146 
147  /**
148  * Number of created blocks so far (by this Allocator object).
149  * We use this number only as a hint as to how big block to create when a
150  * new block needs to be created.
151  */
152  size_t number_of_blocks = 0;
153 };

155 /** Custom memory allocator. All dynamic memory used by the TempTable engine
156  * is allocated through this allocator.
157  *
158  * The purpose of this allocator is to minimize the number of calls to the OS
159  * for allocating new memory (e.g. malloc()) and to improve the spatial
160  * locality of reference. It is able to do so quite easily thanks to the
161  * Block/Chunk entities it is implemented in terms of. Due to the design of
162  * these entities, it is also able to feed allocations and deallocations in
163  * (amortized) constant-time and keep being CPU memory-access friendly because
164  * of the internal self-adjustment to word-size memory alignment. To learn even
165  * more about specifics and more properties please have a look at the respective
166  * header files of Header/Block/Chunk class declarations.
167  *
168  * The most common use case, for which it is optimized,
169  * is to have the following performed by a single thread:
170  * - allocate many times (creation of a temp table and inserting data into it).
171  * - use the allocated memory (selects on the temp table).
172  * - free all the pieces (drop of the temp table).
173  *
174  * The allocator allocates memory from the OS in large blocks (e.g. a few MiB)
175  * whose size also increases progressively by the increasing number of
176  * allocation requests. Exact block-size increase progress is defined by the
177  * block allocation scheme which, by default, is set to
178  * AllocationScheme::Exponential.
179  *
180  * Allocator does not store a list of all allocated blocks but only keeps track
181  * of the current block which has not yet been entirely filled up and the
182  * overall number of allocated blocks. When current block gets filled up, new
183  * one is created and immediately made current.
184  *
185  * Furthermore, it always keeps the last block alive. It cannot be deallocated
186  * by the user. Last block is automatically deallocated at the thread exit.
187  *
188  * Allocator will also keep track of RAM-consumption and in case it reaches the
189  * threshold defined by temptable_max_ram, it will switch to MMAP-backed block
190  * allocations. It will switch back once RAM consumption is again below the
191  * threshold. */

336 template <class T, class AllocationScheme>
337 inline T *Allocator<T, AllocationScheme>::allocate(size_t n_elements) {
338  DBUG_ASSERT(n_elements <= std::numeric_limits<size_type>::max() / sizeof(T));
339  DBUG_EXECUTE_IF("temptable_allocator_oom", throw Result::OUT_OF_MEM;);
340  DBUG_EXECUTE_IF("temptable_allocator_record_file_full",
341  throw Result::RECORD_FILE_FULL;);
342 
343  const size_t n_bytes_requested = n_elements * sizeof(T);
344  if (n_bytes_requested == 0) {
345  return nullptr;
346  }
347 
348  Block *block;
349 
350  if (m_shared_block && m_shared_block->is_empty()) {
351  *m_shared_block = Block(AllocationScheme::block_size(
352  m_state->number_of_blocks, n_bytes_requested),
353  Source::RAM);
354  block = m_shared_block;
355  ++m_state->number_of_blocks;
356  } else if (m_shared_block &&
357  m_shared_block->can_accommodate(n_bytes_requested)) {
358  block = m_shared_block;
359  } else if (m_state->current_block.is_empty() ||
360  !m_state->current_block.can_accommodate(n_bytes_requested)) {
361  const size_t block_size = AllocationScheme::block_size(
362  m_state->number_of_blocks, n_bytes_requested);
363  const Source block_source = [block_size]() {
364  // Decide whether to switch between RAM and MMAP-backed allocations.
365  if (MemoryMonitor::ram_consumption() >= MemoryMonitor::ram_threshold()) {
366  return Source::MMAP_FILE;
367  } else {
368  if (MemoryMonitor::ram_increase(block_size) <=
369  MemoryMonitor::ram_threshold()) {
370  return Source::RAM;
371  } else {
372  MemoryMonitor::ram_decrease(block_size);
373  return Source::MMAP_FILE;
374  }
375  }
376  }();
377  m_state->current_block = Block(block_size, block_source);
378  block = &m_state->current_block;
379  ++m_state->number_of_blocks;
380  } else {
381  block = &m_state->current_block;
382  }
383 
384  T *chunk_data =
385  reinterpret_cast<T *>(block->allocate(n_bytes_requested).data());
386  DBUG_ASSERT(reinterpret_cast<uintptr_t>(chunk_data) % alignof(T) == 0);
387  return chunk_data;
388 }

390 template <class T, class AllocationScheme>
391 inline void Allocator<T, AllocationScheme>::deallocate(T *chunk_data,
392  size_t n_elements) {
393  DBUG_ASSERT(reinterpret_cast<uintptr_t>(chunk_data) % alignof(T) == 0);
394 
395  if (chunk_data == nullptr) {
396  return;
397  }
398 
399  const size_t n_bytes_requested = n_elements * sizeof(T);
400 
401  Block block = Block(Chunk(chunk_data));
402  const auto remaining_chunks =
403  block.deallocate(Chunk(chunk_data), n_bytes_requested);
404  if (remaining_chunks == 0) {
405  if (m_shared_block && (block == *m_shared_block)) {
406  // Do nothing. Keep the last block alive.
407  } else {
408  DBUG_ASSERT(m_state->number_of_blocks > 0);
409  if (block.type() == Source::RAM) {
410  MemoryMonitor::ram_decrease(block.size());
411  }
412  if (block == m_state->current_block) {
413  m_state->current_block.destroy();
414  } else {
415  block.destroy();
416  }
417  --m_state->number_of_blocks;
418  }
419  }
420 }
