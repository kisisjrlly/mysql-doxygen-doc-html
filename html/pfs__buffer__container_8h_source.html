86 class PFS_buffer_default_array {
87  public:
88  typedef T value_type;
89 
90  value_type *allocate(pfs_dirty_state *dirty_state) {
91  size_t index;
92  size_t monotonic;
93  size_t monotonic_max;
94  value_type *pfs;
95 
96  if (m_full) {
97  return nullptr;
98  }
99 
100  monotonic = m_monotonic.m_size_t++;
101  monotonic_max = monotonic + m_max;
102 
103  if (unlikely(monotonic >= monotonic_max)) {
104  /*
105  This will happen once every 2^64 - m_max calls.
106  Computation of monotonic_max just overflowed,
107  so reset monotonic counters and start again from the beginning.
108  */
109  m_monotonic.m_size_t.store(0);
110  monotonic = 0;
111  monotonic_max = m_max;
112  }
113 
114  while (monotonic < monotonic_max) {
115  index = monotonic % m_max;
116  pfs = m_ptr + index;
117 
118  if (pfs->m_lock.free_to_dirty(dirty_state)) {
119  return pfs;
120  }
121  monotonic = m_monotonic.m_size_t++;
122  }
123 
124  m_full = true;
125  return nullptr;
126  }
127 
128  void deallocate(value_type *pfs) {
129  pfs->m_lock.allocated_to_free();
130  m_full = false;
131  }
132 
133  T *get_first() { return m_ptr; }
134 
135  T *get_last() { return m_ptr + m_max; }
136 
137  bool m_full;
138  PFS_cacheline_atomic_size_t m_monotonic;
139  T *m_ptr;
140  size_t m_max;
141  /** Container. */
142  PFS_opaque_container *m_container;

393  int init(long max_size) {
394  int i;
395 
396  m_initialized = true;
397  m_full = true;
398  m_max = PFS_PAGE_COUNT * PFS_PAGE_SIZE;
399  m_max_page_count = PFS_PAGE_COUNT;
400  m_last_page_size = PFS_PAGE_SIZE;
401  m_lost = 0;
402  m_monotonic.m_size_t.store(0);
403  m_max_page_index.m_size_t.store(0);
404 
405  for (i = 0; i < PFS_PAGE_COUNT; i++) {
406  m_pages[i] = nullptr;
407  }
408 
409  if (max_size == 0) {
410  /* No allocation. */
411  m_max_page_count = 0;
412  } else if (max_size > 0) {
413  if (max_size % PFS_PAGE_SIZE == 0) {

415  } else {
416  m_max_page_count = max_size / PFS_PAGE_SIZE + 1;
417  m_last_page_size = max_size % PFS_PAGE_SIZE;
418  }
419  /* Bounded allocation. */
420  m_full = false;
421 
422  if (m_max_page_count > PFS_PAGE_COUNT) {
423  m_max_page_count = PFS_PAGE_COUNT;
424  m_last_page_size = PFS_PAGE_SIZE;
425  }
426  } else {
427  /* max_size = -1 means unbounded allocation */
428  m_full = false;
429  }
430 
431  DBUG_ASSERT(m_max_page_count <= PFS_PAGE_COUNT);
432  DBUG_ASSERT(0 < m_last_page_size);
433  DBUG_ASSERT(m_last_page_size <= PFS_PAGE_SIZE);
434 
435  native_mutex_init(&m_critical_section, nullptr);
436  return 0;
437  }

474  value_type *allocate(pfs_dirty_state *dirty_state) {
475  if (m_full) {
476  m_lost++;
477  return nullptr;
478  }
479 
480  size_t index;
481  size_t monotonic;
482  size_t monotonic_max;
483  size_t current_page_count;
484  value_type *pfs;
485  array_type *array;
486 
487  /*
488  1: Try to find an available record within the existing pages
489  */
490  current_page_count = m_max_page_index.m_size_t.load();
491 
492  if (current_page_count != 0) {
493  monotonic = m_monotonic.m_size_t.load();
494  monotonic_max = monotonic + current_page_count;
495 
496  if (unlikely(monotonic >= monotonic_max)) {
497  /*
498  This will happen once every 2^64 - current_page_count calls.
499  Computation of monotonic_max just overflowed,
500  so reset monotonic counters and start again from the beginning.
501  */
502  m_monotonic.m_size_t.store(0);
503  monotonic = 0;
504  monotonic_max = current_page_count;
505  }
506 
507  while (monotonic < monotonic_max) {
508  /*
509  Scan in the [0 .. current_page_count - 1] range,
510  in parallel with m_monotonic (see below)
511  */
512  index = monotonic % current_page_count;
513 
514  /* Atomic Load, array= m_pages[index] */
515  array = m_pages[index].load();
516 
517  if (array != nullptr) {
518  pfs = array->allocate(dirty_state);
519  if (pfs != nullptr) {
520  /* Keep a pointer to the parent page, for deallocate(). */
521  pfs->m_page = reinterpret_cast<PFS_opaque_container_page *>(array);
522  return pfs;
523  }
524  }
525 
526  /*
527  Parallel scans collaborate to increase
528  the common monotonic scan counter.
529 
530  Note that when all the existing page are full,
531  one thread will eventually add a new page,
532  and cause m_max_page_index to increase,
533  which fools all the modulo logic for scans already in progress,
534  because the monotonic counter is not folded to the same place
535  (sometime modulo N, sometime modulo N+1).
536 
537  This is actually ok: since all the pages are full anyway,
538  there is nothing to miss, so better increase the monotonic
539  counter faster and then move on to the detection of new pages,
540  in part 2: below.
541  */
542  monotonic = m_monotonic.m_size_t++;
543  };
544  }
545 
546  /*
547  2: Try to add a new page, beyond the m_max_page_index limit
548  */
549  while (current_page_count < m_max_page_count) {
550  /* Peek for pages added by collaborating threads */
551 
552  /* (2-a) Atomic Load, array= m_pages[current_page_count] */
553  array = m_pages[current_page_count].load();
554 
555  if (array == nullptr) {
556  // ==================================================================
557  // BEGIN CRITICAL SECTION -- buffer expand
558  // ==================================================================
559 
560  /*
561  On a fresh started server, buffers are typically empty.
562  When a sudden load spike is seen by the server,
563  multiple threads may want to expand the buffer at the same time.
564 
565  Using a compare and swap to allow multiple pages to be added,
566  possibly freeing duplicate pages on collisions,
567  does not work well because the amount of code involved
568  when creating a new page can be significant (PFS_thread),
569  causing MANY collisions between (2-b) and (2-d).
570 
571  A huge number of collisions (which can happen when thousands
572  of new connections hits the server after a restart)
573  leads to a huge memory consumption, and to OOM.
574 
575  To mitigate this, we use here a mutex,
576  to enforce that only ONE page is added at a time,
577  so that scaling the buffer happens in a predictable
578  and controlled manner.
579  */
580  native_mutex_lock(&m_critical_section);
581 
582  /*
583  Peek again for pages added by collaborating threads,
584  this time as the only thread allowed to expand the buffer
585  */
586 
587  /* (2-b) Atomic Load, array= m_pages[current_page_count] */
588 
589  array = m_pages[current_page_count].load();
590 
591  if (array == nullptr) {
592  /* (2-c) Found no page, allocate a new one */
593  array = new array_type();
594  builtin_memory_scalable_buffer.count_alloc(sizeof(array_type));
595 
596  array->m_max = get_page_logical_size(current_page_count);
597  int rc = m_allocator->alloc_array(array);
598  if (rc != 0) {
599  m_allocator->free_array(array);
600  delete array;
601  builtin_memory_scalable_buffer.count_free(sizeof(array_type));
602  m_lost++;
603  native_mutex_unlock(&m_critical_section);
604  return nullptr;
605  }
606 
607  /* Keep a pointer to this container, for static_deallocate(). */
608  array->m_container = reinterpret_cast<PFS_opaque_container *>(this);
609 
610  /* (2-d) Atomic STORE, m_pages[current_page_count] = array */
611  m_pages[current_page_count].store(array);
612 
613  /* Advertise the new page */
614  ++m_max_page_index.m_size_t;
615  }
616 
617  native_mutex_unlock(&m_critical_section);
618 
619  // ==================================================================
620  // END CRITICAL SECTION -- buffer expand
621  // ==================================================================
622  }
623 
624  DBUG_ASSERT(array != nullptr);
625  pfs = array->allocate(dirty_state);
626  if (pfs != nullptr) {
627  /* Keep a pointer to the parent page, for deallocate(). */
628  pfs->m_page = reinterpret_cast<PFS_opaque_container_page *>(array);
629  return pfs;
630  }
631 
632  current_page_count++;
633  }
634 
635  m_lost++;
636  m_full = true;
637  return nullptr;
638  }

859  value_type *scan_next(uint &index, uint *found_index) {
860  DBUG_ASSERT(index <= m_max);
861 
862  uint index_1 = index / PFS_PAGE_SIZE;
863  uint index_2 = index % PFS_PAGE_SIZE;
864  array_type *page;
865  value_type *pfs_first;
866  value_type *pfs;
867  value_type *pfs_last;
868 
869  while (index_1 < PFS_PAGE_COUNT) {
870  page = m_pages[index_1];
871 
872  if (page == nullptr) {
873  index = static_cast<uint>(m_max);
874  return nullptr;
875  }
876 
877  pfs_first = page->get_first();
878  pfs = pfs_first + index_2;
879  pfs_last = page->get_last();
880 
881  while (pfs < pfs_last) {
882  if (pfs->m_lock.is_populated()) {
883  uint found =
884  index_1 * PFS_PAGE_SIZE + static_cast<uint>(pfs - pfs_first);
885  *found_index = found;
886  index = found + 1;
887  return pfs;
888  }
889  pfs++;
890  }
891 
892  index_1++;
893  index_2 = 0;
894  }
895 
896  index = static_cast<uint>(m_max);
897  return nullptr;
898  }
